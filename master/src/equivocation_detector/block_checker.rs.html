27/// First step in the block checking state machine.
28///
29/// Getting the finality info associated to the source headers synced with the target chain
30/// at the specified block.
31#[cfg_attr(test, derive(Debug, PartialEq))]
32pub struct ReadSyncedHeaders<P: EquivocationDetectionPipeline> {
33	pub target_block_num: P::TargetNumber,
34}
35
36impl<P: EquivocationDetectionPipeline> ReadSyncedHeaders<P> {
37	pub async fn next<TC: TargetClient<P>>(
38		self,
39		target_client: &mut TC,
40	) -> Result<ReadContext<P>, Self> {
41		match target_client.synced_headers_finality_info(self.target_block_num).await {
42			Ok(synced_headers) =>
43				Ok(ReadContext { target_block_num: self.target_block_num, synced_headers }),
44			Err(e) => {
45				tracing::error!(
46					target: "bridge",
47					error=?e,
48					source=%P::SOURCE_NAME,
49					target=%P::TARGET_NAME,
50					block=%self.target_block_num,
51					"Could not get headers synced at block"
52				);
53
54				// Reconnect target client in case of a connection error.
55				handle_client_error(target_client, e).await;
56
57				Err(self)
58			},
59		}
60	}
61}

63/// Second step in the block checking state machine.
64///
65/// Reading the equivocation reporting context from the target chain.
66#[cfg_attr(test, derive(Debug))]
67pub struct ReadContext<P: EquivocationDetectionPipeline> {
68	target_block_num: P::TargetNumber,
69	synced_headers: Vec<HeaderFinalityInfo<P>>,
70}
71
72impl<P: EquivocationDetectionPipeline> ReadContext<P> {
73	pub async fn next<TC: TargetClient<P>>(
74		self,
75		target_client: &mut TC,
76	) -> Result<Option<FindEquivocations<P>>, Self> {
77		match EquivocationReportingContext::try_read_from_target::<TC>(
78			target_client,
79			self.target_block_num.saturating_sub(1.into()),
80		)
81		.await
82		{
83			Ok(Some(context)) => Ok(Some(FindEquivocations {
84				target_block_num: self.target_block_num,
85				synced_headers: self.synced_headers,
86				context,
87			})),
88			Ok(None) => Ok(None),
89			Err(e) => {
90				tracing::error!(
91					target: "bridge",
92					error=?e,
93					source=%P::SOURCE_NAME,
94					target=%P::TARGET_NAME,
95					block=%self.target_block_num.saturating_sub(1.into()),
96					"Could not read `EquivocationReportingContext` at block",
97				);
98
99				// Reconnect target client in case of a connection error.
100				handle_client_error(target_client, e).await;
101
102				Err(self)
103			},
104		}
105	}
106}

108/// Third step in the block checking state machine.
109///
110/// Searching for equivocations in the source headers synced with the target chain.
111#[cfg_attr(test, derive(Debug))]
112pub struct FindEquivocations<P: EquivocationDetectionPipeline> {
113	target_block_num: P::TargetNumber,
114	synced_headers: Vec<HeaderFinalityInfo<P>>,
115	context: EquivocationReportingContext<P>,
116}
117
118impl<P: EquivocationDetectionPipeline> FindEquivocations<P> {
119	pub fn next(
120		mut self,
121		finality_proofs_buf: &mut FinalityProofsBuf<P>,
122	) -> Vec<ReportEquivocations<P>> {
123		let mut result = vec![];
124		for synced_header in self.synced_headers {
125			match P::EquivocationsFinder::find_equivocations(
126				&self.context.synced_verification_context,
127				&synced_header.finality_proof,
128				finality_proofs_buf.buf().as_slice(),
129			) {
130				Ok(equivocations) =>
131					if !equivocations.is_empty() {
132						result.push(ReportEquivocations {
133							source_block_hash: self.context.synced_header_hash,
134							equivocations,
135						})
136					},
137				Err(e) => {
138					tracing::error!(
139						target: "bridge",
140						error=?e,
141						source_header=?synced_header.finality_proof.target_header_hash(),
142						block=%self.target_block_num,
143						"Could not search for equivocations in the finality proof \
144						for source header synced at target block"
145					);
146				},
147			};
148
149			finality_proofs_buf.prune(synced_header.finality_proof.target_header_number(), None);
150			self.context.update(synced_header);
151		}
152
153		result
154	}
155}

157/// Fourth step in the block checking state machine.
158///
159/// Reporting the detected equivocations (if any).
160#[cfg_attr(test, derive(Debug))]
161pub struct ReportEquivocations<P: EquivocationDetectionPipeline> {
162	source_block_hash: P::Hash,
163	equivocations: Vec<P::EquivocationProof>,
164}
165
166impl<P: EquivocationDetectionPipeline> ReportEquivocations<P> {
167	pub async fn next<SC: SourceClient<P>>(
168		mut self,
169		source_client: &mut SC,
170		reporter: &mut EquivocationsReporter<'_, P, SC>,
171	) -> Result<(), Self> {
172		let mut unprocessed_equivocations = vec![];
173		for equivocation in self.equivocations {
174			match reporter
175				.submit_report(source_client, self.source_block_hash, equivocation.clone())
176				.await
177			{
178				Ok(_) => {},
179				Err(e) => {
180					tracing::error!(
181						target: "bridge",
182						error=?e,
183						source=%P::SOURCE_NAME,
184						?equivocation,
185						"Could not submit equivocation report"
186					);
187
188					// Mark the equivocation as unprocessed
189					unprocessed_equivocations.push(equivocation);
190					// Reconnect source client in case of a connection error.
191					handle_client_error(source_client, e).await;
192				},
193			}
194		}
195
196		self.equivocations = unprocessed_equivocations;
197		if !self.equivocations.is_empty() {
198			return Err(self)
199		}
200
201		Ok(())
202	}
203}

205/// Block checking state machine.
206#[cfg_attr(test, derive(Debug))]
207pub enum BlockChecker<P: EquivocationDetectionPipeline> {
208	ReadSyncedHeaders(ReadSyncedHeaders<P>),
209	ReadContext(ReadContext<P>),
210	ReportEquivocations(Vec<ReportEquivocations<P>>),
211}
212
213impl<P: EquivocationDetectionPipeline> BlockChecker<P> {
214	pub fn new(target_block_num: P::TargetNumber) -> Self {
215		Self::ReadSyncedHeaders(ReadSyncedHeaders { target_block_num })
216	}
217
218	pub fn run<'a, SC: SourceClient<P>, TC: TargetClient<P>>(
219		self,
220		source_client: &'a mut SC,
221		target_client: &'a mut TC,
222		finality_proofs_buf: &'a mut FinalityProofsBuf<P>,
223		reporter: &'a mut EquivocationsReporter<P, SC>,
224	) -> BoxFuture<'a, Result<(), Self>> {
225		async move {
226			match self {
227				Self::ReadSyncedHeaders(state) => {
228					let read_context =
229						state.next(target_client).await.map_err(Self::ReadSyncedHeaders)?;
230					Self::ReadContext(read_context)
231						.run(source_client, target_client, finality_proofs_buf, reporter)
232						.await
233				},
234				Self::ReadContext(state) => {
235					let maybe_find_equivocations =
236						state.next(target_client).await.map_err(Self::ReadContext)?;
237					let find_equivocations = match maybe_find_equivocations {
238						Some(find_equivocations) => find_equivocations,
239						None => return Ok(()),
240					};
241					Self::ReportEquivocations(find_equivocations.next(finality_proofs_buf))
242						.run(source_client, target_client, finality_proofs_buf, reporter)
243						.await
244				},
245				Self::ReportEquivocations(state) => {
246					let mut failures = vec![];
247					for report_equivocations in state {
248						if let Err(failure) =
249							report_equivocations.next(source_client, reporter).await
250						{
251							failures.push(failure);
252						}
253					}
254
255					if !failures.is_empty() {
256						return Err(Self::ReportEquivocations(failures))
257					}
258
259					Ok(())
260				},
261			}
262		}
263		.boxed()
264	}
265}

301	#[async_std::test]
302	async fn block_checker_works() {
303		let mut source_client = TestSourceClient { ..Default::default() };
304		let mut target_client = TestTargetClient {
305			best_synced_header_hash: HashMap::from([(9, Ok(Some(5)))]),
306			finality_verification_context: HashMap::from([(
307				9,
308				Ok(TestFinalityVerificationContext { check_equivocations: true }),
309			)]),
310			synced_headers_finality_info: HashMap::from([(
311				10,
312				Ok(vec![
313					new_header_finality_info(6, None),
314					new_header_finality_info(7, Some(false)),
315					new_header_finality_info(8, None),
316					new_header_finality_info(9, Some(true)),
317					new_header_finality_info(10, None),
318					new_header_finality_info(11, None),
319					new_header_finality_info(12, None),
320				]),
321			)]),
322			..Default::default()
323		};
324		let mut reporter =
325			EquivocationsReporter::<TestEquivocationDetectionPipeline, TestSourceClient>::new();
326
327		let block_checker = BlockChecker::new(10);
328		assert!(block_checker
329			.run(
330				&mut source_client,
331				&mut target_client,
332				&mut FinalityProofsBuf::new(vec![
333					TestFinalityProof(6, vec!["6-1"]),
334					TestFinalityProof(7, vec![]),
335					TestFinalityProof(8, vec!["8-1"]),
336					TestFinalityProof(9, vec!["9-1"]),
337					TestFinalityProof(10, vec![]),
338					TestFinalityProof(11, vec!["11-1", "11-2"]),
339					TestFinalityProof(12, vec!["12-1"])
340				]),
341				&mut reporter
342			)
343			.await
344			.is_ok());
345		assert_eq!(
346			*source_client.reported_equivocations.lock().unwrap(),
347			HashMap::from([(5, vec!["6-1"]), (9, vec!["11-1", "11-2", "12-1"])])
348		);
349	}

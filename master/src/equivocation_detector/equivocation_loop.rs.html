29/// Equivocations detection loop state.
30struct EquivocationDetectionLoop<
31	P: EquivocationDetectionPipeline,
32	SC: SourceClient<P>,
33	TC: TargetClient<P>,
34> {
35	source_client: SC,
36	target_client: TC,
37
38	from_block_num: Option<P::TargetNumber>,
39	until_block_num: Option<P::TargetNumber>,
40
41	reporter: EquivocationsReporter<'static, P, SC>,
42
43	finality_proofs_stream: FinalityProofsStream<P, SC>,
44	finality_proofs_buf: FinalityProofsBuf<P>,
45}
46
47impl<P: EquivocationDetectionPipeline, SC: SourceClient<P>, TC: TargetClient<P>>
48	EquivocationDetectionLoop<P, SC, TC>
49{
50	async fn ensure_finality_proofs_stream(&mut self) {
51		match self.finality_proofs_stream.ensure_stream(&self.source_client).await {
52			Ok(_) => {},
53			Err(e) => {
54				tracing::error!(
55					target: "bridge",
56					error=?e,
57					source=%P::SOURCE_NAME,
58					"Could not connect to `FinalityProofsStream`"
59				);
60
61				// Reconnect to the source client if needed
62				handle_client_error(&mut self.source_client, e).await;
63			},
64		}
65	}
66
67	async fn best_finalized_target_block_number(&mut self) -> Option<P::TargetNumber> {
68		match self.target_client.best_finalized_header_number().await {
69			Ok(block_num) => Some(block_num),
70			Err(e) => {
71				tracing::error!(
72					target: "bridge",
73					error=?e,
74					target=%P::TARGET_NAME,
75					"Could not read best finalized header number"
76				);
77
78				// Reconnect target client and move on
79				handle_client_error(&mut self.target_client, e).await;
80
81				None
82			},
83		}
84	}
85
86	async fn do_run(&mut self, tick: Duration, exit_signal: impl Future<Output = ()>) {
87		let exit_signal = exit_signal.fuse();
88		futures::pin_mut!(exit_signal);
89
90		loop {
91			// Make sure that we are connected to the source finality proofs stream.
92			self.ensure_finality_proofs_stream().await;
93			// Check the status of the pending equivocation reports
94			self.reporter.process_pending_reports().await;
95
96			// Update blocks range.
97			if let Some(block_number) = self.best_finalized_target_block_number().await {
98				self.from_block_num.get_or_insert(block_number);
99				self.until_block_num = Some(block_number);
100			}
101			let (from, until) = match (self.from_block_num, self.until_block_num) {
102				(Some(from), Some(until)) => (from, until),
103				_ => continue,
104			};
105
106			// Check the available blocks
107			let mut current_block_number = from;
108			while current_block_number <= until {
109				self.finality_proofs_buf.fill(&mut self.finality_proofs_stream);
110				let block_checker = BlockChecker::new(current_block_number);
111				let _ = block_checker
112					.run(
113						&mut self.source_client,
114						&mut self.target_client,
115						&mut self.finality_proofs_buf,
116						&mut self.reporter,
117					)
118					.await;
119				current_block_number = current_block_number.saturating_add(1.into());
120			}
121			self.from_block_num = Some(current_block_number);
122
123			select_biased! {
124				_ = exit_signal => return,
125				_ = async_std::task::sleep(tick).fuse() => {},
126			}
127		}
128	}
129
130	pub async fn run(
131		source_client: SC,
132		target_client: TC,
133		tick: Duration,
134		exit_signal: impl Future<Output = ()>,
135	) -> Result<(), FailedClient> {
136		let mut equivocation_detection_loop = Self {
137			source_client,
138			target_client,
139			from_block_num: None,
140			until_block_num: None,
141			reporter: EquivocationsReporter::<P, SC>::new(),
142			finality_proofs_stream: FinalityProofsStream::new(),
143			finality_proofs_buf: FinalityProofsBuf::new(vec![]),
144		};
145
146		equivocation_detection_loop.do_run(tick, exit_signal).await;
147		Ok(())
148	}
149}
150
151/// Spawn the equivocations detection loop.
152pub async fn run<P: EquivocationDetectionPipeline>(
153	source_client: impl SourceClient<P>,
154	target_client: impl TargetClient<P>,
155	tick: Duration,
156	metrics_params: MetricsParams,
157	exit_signal: impl Future<Output = ()> + 'static + Send,
158) -> Result<(), relay_utils::Error> {
159	let exit_signal = exit_signal.shared();
160	relay_utils::relay_loop(source_client, target_client)
161		.with_metrics(metrics_params)
162		.expose()
163		.await?
164		.run(
165			format!("{}_to_{}_EquivocationDetection", P::SOURCE_NAME, P::TARGET_NAME),
166			move |source_client, target_client, _metrics| {
167				EquivocationDetectionLoop::run(
168					source_client,
169					target_client,
170					tick,
171					exit_signal.clone(),
172				)
173			},
174		)
175		.await
176}

188	fn best_finalized_header_number(
189		best_finalized_headers: &Mutex<VecDeque<Result<TestTargetNumber, TestClientError>>>,
190		exit_sender: &UnboundedSender<()>,
191	) -> Result<TestTargetNumber, TestClientError> {
192		let mut best_finalized_headers = best_finalized_headers.lock().unwrap();
193		let result = best_finalized_headers.pop_front().unwrap();
194		if best_finalized_headers.is_empty() {
195			exit_sender.unbounded_send(()).unwrap();
196		}
197		result
198	}

200	#[async_std::test]
201	async fn multiple_blocks_are_checked_correctly() {
202		let best_finalized_headers = Arc::new(Mutex::new(VecDeque::from([Ok(10), Ok(12), Ok(13)])));
203		let (exit_sender, exit_receiver) = futures::channel::mpsc::unbounded();
204
205		let source_client = TestSourceClient {
206			finality_proofs: Arc::new(Mutex::new(vec![
207				TestFinalityProof(2, vec!["2-1"]),
208				TestFinalityProof(3, vec!["3-1", "3-2"]),
209				TestFinalityProof(4, vec!["4-1"]),
210				TestFinalityProof(5, vec!["5-1"]),
211				TestFinalityProof(6, vec!["6-1", "6-2"]),
212				TestFinalityProof(7, vec!["7-1", "7-2"]),
213			])),
214			..Default::default()
215		};
216		let reported_equivocations = source_client.reported_equivocations.clone();
217		let target_client = TestTargetClient {
218			best_finalized_header_number: Arc::new(move || {
219				best_finalized_header_number(&best_finalized_headers, &exit_sender)
220			}),
221			best_synced_header_hash: HashMap::from([
222				(9, Ok(Some(1))),
223				(10, Ok(Some(3))),
224				(11, Ok(Some(5))),
225				(12, Ok(Some(6))),
226			]),
227			finality_verification_context: HashMap::from([
228				(9, Ok(TestFinalityVerificationContext { check_equivocations: true })),
229				(10, Ok(TestFinalityVerificationContext { check_equivocations: true })),
230				(11, Ok(TestFinalityVerificationContext { check_equivocations: false })),
231				(12, Ok(TestFinalityVerificationContext { check_equivocations: true })),
232			]),
233			synced_headers_finality_info: HashMap::from([
234				(
235					10,
236					Ok(vec![new_header_finality_info(2, None), new_header_finality_info(3, None)]),
237				),
238				(
239					11,
240					Ok(vec![
241						new_header_finality_info(4, None),
242						new_header_finality_info(5, Some(false)),
243					]),
244				),
245				(12, Ok(vec![new_header_finality_info(6, None)])),
246				(13, Ok(vec![new_header_finality_info(7, None)])),
247			]),
248			..Default::default()
249		};
250
251		assert!(run::<TestEquivocationDetectionPipeline>(
252			source_client,
253			target_client,
254			Duration::from_secs(0),
255			MetricsParams { address: None, registry: Default::default() },
256			exit_receiver.into_future().map(|(_, _)| ()),
257		)
258		.await
259		.is_ok());
260		assert_eq!(
261			*reported_equivocations.lock().unwrap(),
262			HashMap::from([
263				(1, vec!["2-1", "3-1", "3-2"]),
264				(3, vec!["4-1", "5-1"]),
265				(6, vec!["7-1", "7-2"])
266			])
267		);
268	}

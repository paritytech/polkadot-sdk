14//! Loop that is serving single race within message lane. This could be
15//! message delivery race, receiving confirmations race or processing
16//! confirmations race.
17//!
18//! The idea of the race is simple - we have `nonce`-s on source and target
19//! nodes. We're trying to prove that the source node has this nonce (and
20//! associated data - like messages, lane state, etc) to the target node by
21//! generating and submitting proof.
22

41/// One of races within lane.
42pub trait MessageRace {
43	/// Header id of the race source.
44	type SourceHeaderId: Debug + Clone + PartialEq + Send + Sync;
45	/// Header id of the race source.
46	type TargetHeaderId: Debug + Clone + PartialEq + Send + Sync;
47
48	/// Message nonce used in the race.
49	type MessageNonce: Debug + Clone;
50	/// Proof that is generated and delivered in this race.
51	type Proof: Debug + Clone + Send + Sync;
52
53	/// Name of the race source.
54	fn source_name() -> String;
55	/// Name of the race target.
56	fn target_name() -> String;
57}
58
59/// State of race source client.
60type SourceClientState<P> =
61	ClientState<<P as MessageRace>::SourceHeaderId, <P as MessageRace>::TargetHeaderId>;
62
63/// State of race target client.
64type TargetClientState<P> =
65	ClientState<<P as MessageRace>::TargetHeaderId, <P as MessageRace>::SourceHeaderId>;
66
67/// Inclusive nonces range.
68pub trait NoncesRange: Debug + Sized {
69	/// Get begin of the range.
70	fn begin(&self) -> MessageNonce;
71	/// Get end of the range.
72	fn end(&self) -> MessageNonce;
73	/// Returns new range with current range nonces that are greater than the passed `nonce`.
74	/// If there are no such nonces, `None` is returned.
75	fn greater_than(self, nonce: MessageNonce) -> Option<Self>;
76}
77
78/// Nonces on the race source client.
79#[derive(Debug, Clone)]
80pub struct SourceClientNonces<NoncesRange> {
81	/// New nonces range known to the client. `New` here means all nonces generated after
82	/// `prev_latest_nonce` passed to the `SourceClient::nonces` method.
83	pub new_nonces: NoncesRange,
84	/// The latest nonce that is confirmed to the bridged client. This nonce only makes
85	/// sense in some races. In other races it is `None`.
86	pub confirmed_nonce: Option<MessageNonce>,
87}
88
89/// Nonces on the race target client.
90#[derive(Debug, Clone)]
91pub struct TargetClientNonces<TargetNoncesData> {
92	/// The latest nonce that is known to the target client.
93	pub latest_nonce: MessageNonce,
94	/// Additional data from target node that may be used by the race.
95	pub nonces_data: TargetNoncesData,
96}

98/// One of message lane clients, which is source client for the race.
99#[async_trait]
100pub trait SourceClient<P: MessageRace> {
101	/// Type of error these clients returns.
102	type Error: std::fmt::Debug + MaybeConnectionError;
103	/// Type of nonces range returned by the source client.
104	type NoncesRange: NoncesRange;
105	/// Additional proof parameters required to generate proof.
106	type ProofParameters;
107
108	/// Return nonces that are known to the source client.
109	async fn nonces(
110		&self,
111		at_block: P::SourceHeaderId,
112		prev_latest_nonce: MessageNonce,
113	) -> Result<(P::SourceHeaderId, SourceClientNonces<Self::NoncesRange>), Self::Error>;
114	/// Generate proof for delivering to the target client.
115	async fn generate_proof(
116		&self,
117		at_block: P::SourceHeaderId,
118		nonces: RangeInclusive<MessageNonce>,
119		proof_parameters: Self::ProofParameters,
120	) -> Result<(P::SourceHeaderId, RangeInclusive<MessageNonce>, P::Proof), Self::Error>;
121}
122
123/// One of message lane clients, which is target client for the race.
124#[async_trait]
125pub trait TargetClient<P: MessageRace> {
126	/// Type of error these clients returns.
127	type Error: std::fmt::Debug + MaybeConnectionError;
128	/// Type of the additional data from the target client, used by the race.
129	type TargetNoncesData: std::fmt::Debug;
130	/// Type of batch transaction that submits finality and proof to the target node.
131	type BatchTransaction: BatchTransaction<P::SourceHeaderId> + Clone;
132	/// Transaction tracker to track submitted transactions.
133	type TransactionTracker: TransactionTracker<HeaderId = P::TargetHeaderId>;
134
135	/// Ask headers relay to relay finalized headers up to (and including) given header
136	/// from race source to race target.
137	///
138	/// The client may return `Some(_)`, which means that nothing has happened yet and
139	/// the caller must generate and append proof to the batch transaction
140	/// to actually send it (along with required header) to the node.
141	///
142	/// If function has returned `None`, it means that the caller now must wait for the
143	/// appearance of the required header `id` at the target client.
144	async fn require_source_header(
145		&self,
146		id: P::SourceHeaderId,
147	) -> Result<Option<Self::BatchTransaction>, Self::Error>;
148
149	/// Return nonces that are known to the target client.
150	async fn nonces(
151		&self,
152		at_block: P::TargetHeaderId,
153		update_metrics: bool,
154	) -> Result<(P::TargetHeaderId, TargetClientNonces<Self::TargetNoncesData>), Self::Error>;
155	/// Submit proof to the target client.
156	async fn submit_proof(
157		&self,
158		maybe_batch_tx: Option<Self::BatchTransaction>,
159		generated_at_block: P::SourceHeaderId,
160		nonces: RangeInclusive<MessageNonce>,
161		proof: P::Proof,
162	) -> Result<NoncesSubmitArtifacts<Self::TransactionTracker>, Self::Error>;
163}
164
165/// Race strategy.
166#[async_trait]
167pub trait RaceStrategy<SourceHeaderId, TargetHeaderId, Proof>: Debug {
168	/// Type of nonces range expected from the source client.
169	type SourceNoncesRange: NoncesRange;
170	/// Additional proof parameters required to generate proof.
171	type ProofParameters;
172	/// Additional data expected from the target client.
173	type TargetNoncesData;
174
175	/// Return id of source header that is required to be on target to continue synchronization.
176	async fn required_source_header_at_target<RS: RaceState<SourceHeaderId, TargetHeaderId>>(
177		&self,
178		race_state: RS,
179	) -> Option<SourceHeaderId>;
180	/// Return the best nonce at source node.
181	///
182	/// `Some` is returned only if we are sure that the value is greater or equal
183	/// than the result of `best_at_target`.
184	fn best_at_source(&self) -> Option<MessageNonce>;
185	/// Return the best nonce at target node.
186	///
187	/// May return `None` if value is yet unknown.
188	fn best_at_target(&self) -> Option<MessageNonce>;
189
190	/// Called when nonces are updated at source node of the race.
191	fn source_nonces_updated(
192		&mut self,
193		at_block: SourceHeaderId,
194		nonces: SourceClientNonces<Self::SourceNoncesRange>,
195	);
196	/// Called when we want to wait until next `best_target_nonces_updated` before selecting
197	/// any nonces for delivery.
198	fn reset_best_target_nonces(&mut self);
199	/// Called when best nonces are updated at target node of the race.
200	fn best_target_nonces_updated<RS: RaceState<SourceHeaderId, TargetHeaderId>>(
201		&mut self,
202		nonces: TargetClientNonces<Self::TargetNoncesData>,
203		race_state: &mut RS,
204	);
205	/// Called when finalized nonces are updated at target node of the race.
206	fn finalized_target_nonces_updated<RS: RaceState<SourceHeaderId, TargetHeaderId>>(
207		&mut self,
208		nonces: TargetClientNonces<Self::TargetNoncesData>,
209		race_state: &mut RS,
210	);
211	/// Should return `Some(nonces)` if we need to deliver proof of `nonces` (and associated
212	/// data) from source to target node.
213	/// Additionally, parameters required to generate proof are returned.
214	async fn select_nonces_to_deliver<RS: RaceState<SourceHeaderId, TargetHeaderId>>(
215		&self,
216		race_state: RS,
217	) -> Option<(RangeInclusive<MessageNonce>, Self::ProofParameters)>;
218}
219
220/// State of the race.
221pub trait RaceState<SourceHeaderId, TargetHeaderId>: Clone + Send + Sync {
222	/// Set best finalized source header id at the best block on the target
223	/// client (at the `best_finalized_source_header_id_at_best_target`).
224	fn set_best_finalized_source_header_id_at_best_target(&mut self, id: SourceHeaderId);
225
226	/// Best finalized source header id at the best block on the target
227	/// client (at the `best_finalized_source_header_id_at_best_target`).
228	fn best_finalized_source_header_id_at_best_target(&self) -> Option<SourceHeaderId>;
229
230	/// Returns `true` if we have selected nonces to submit to the target node.
231	fn nonces_to_submit(&self) -> Option<RangeInclusive<MessageNonce>>;
232	/// Reset our nonces selection.
233	fn reset_nonces_to_submit(&mut self);
234
235	/// Returns `true` if we have submitted some nonces to the target node and are
236	/// waiting for them to appear there.
237	fn nonces_submitted(&self) -> Option<RangeInclusive<MessageNonce>>;
238	/// Reset our nonces submission.
239	fn reset_nonces_submitted(&mut self);
240}

242/// State of the race and prepared batch transaction (if available).
243#[derive(Debug, Clone)]
244pub(crate) struct RaceStateImpl<SourceHeaderId, TargetHeaderId, Proof, BatchTx> {
245	/// Best finalized source header id at the source client.
246	pub best_finalized_source_header_id_at_source: Option<SourceHeaderId>,
247	/// Best finalized source header id at the best block on the target
248	/// client (at the `best_finalized_source_header_id_at_best_target`).
249	pub best_finalized_source_header_id_at_best_target: Option<SourceHeaderId>,
250	/// The best header id at the target client.
251	pub best_target_header_id: Option<TargetHeaderId>,
252	/// Best finalized header id at the target client.
253	pub best_finalized_target_header_id: Option<TargetHeaderId>,
254	/// Range of nonces that we have selected to submit.
255	pub nonces_to_submit: Option<(SourceHeaderId, RangeInclusive<MessageNonce>, Proof)>,
256	/// Batch transaction ready to include and deliver selected `nonces_to_submit` from the
257	/// `state`.
258	pub nonces_to_submit_batch: Option<BatchTx>,
259	/// Range of nonces that is currently submitted.
260	pub nonces_submitted: Option<RangeInclusive<MessageNonce>>,
261}
262
263impl<SourceHeaderId, TargetHeaderId, Proof, BatchTx> Default
264	for RaceStateImpl<SourceHeaderId, TargetHeaderId, Proof, BatchTx>
265{
266	fn default() -> Self {
267		RaceStateImpl {
268			best_finalized_source_header_id_at_source: None,
269			best_finalized_source_header_id_at_best_target: None,
270			best_target_header_id: None,
271			best_finalized_target_header_id: None,
272			nonces_to_submit: None,
273			nonces_to_submit_batch: None,
274			nonces_submitted: None,
275		}
276	}
277}

279impl<SourceHeaderId, TargetHeaderId, Proof, BatchTx> RaceState<SourceHeaderId, TargetHeaderId>
280	for RaceStateImpl<SourceHeaderId, TargetHeaderId, Proof, BatchTx>
281where
282	SourceHeaderId: Clone + Send + Sync,
283	TargetHeaderId: Clone + Send + Sync,
284	Proof: Clone + Send + Sync,
285	BatchTx: Clone + Send + Sync,
286{
287	fn set_best_finalized_source_header_id_at_best_target(&mut self, id: SourceHeaderId) {
288		self.best_finalized_source_header_id_at_best_target = Some(id);
289	}
290
291	fn best_finalized_source_header_id_at_best_target(&self) -> Option<SourceHeaderId> {
292		self.best_finalized_source_header_id_at_best_target.clone()
293	}
294
295	fn nonces_to_submit(&self) -> Option<RangeInclusive<MessageNonce>> {
296		self.nonces_to_submit.clone().map(|(_, nonces, _)| nonces)
297	}
298
299	fn reset_nonces_to_submit(&mut self) {
300		self.nonces_to_submit = None;
301		self.nonces_to_submit_batch = None;
302	}
303
304	fn nonces_submitted(&self) -> Option<RangeInclusive<MessageNonce>> {
305		self.nonces_submitted.clone()
306	}
307
308	fn reset_nonces_submitted(&mut self) {
309		self.nonces_submitted = None;
310	}
311}

313/// Run race loop until connection with target or source node is lost.
314pub async fn run<P: MessageRace, SC: SourceClient<P>, TC: TargetClient<P>>(
315	race_source: SC,
316	race_source_updated: impl FusedStream<Item = SourceClientState<P>>,
317	race_target: TC,
318	race_target_updated: impl FusedStream<Item = TargetClientState<P>>,
319	mut strategy: impl RaceStrategy<
320		P::SourceHeaderId,
321		P::TargetHeaderId,
322		P::Proof,
323		SourceNoncesRange = SC::NoncesRange,
324		ProofParameters = SC::ProofParameters,
325		TargetNoncesData = TC::TargetNoncesData,
326	>,
327) -> Result<(), FailedClient> {
328	let mut progress_context = Instant::now();
329	let mut race_state = RaceStateImpl::default();
330
331	let mut source_retry_backoff = retry_backoff();
332	let mut source_client_is_online = true;
333	let mut source_nonces_required = false;
334	let mut source_required_header = None;
335	let source_nonces = futures::future::Fuse::terminated();
336	let source_generate_proof = futures::future::Fuse::terminated();
337	let source_go_offline_future = futures::future::Fuse::terminated();
338
339	let mut target_retry_backoff = retry_backoff();
340	let mut target_client_is_online = true;
341	let mut target_best_nonces_required = false;
342	let mut target_finalized_nonces_required = false;
343	let mut target_batch_transaction = None;
344	let target_require_source_header = futures::future::Fuse::terminated();
345	let target_best_nonces = futures::future::Fuse::terminated();
346	let target_finalized_nonces = futures::future::Fuse::terminated();
347	let target_submit_proof = futures::future::Fuse::terminated();
348	let target_tx_tracker = futures::future::Fuse::terminated();
349	let target_go_offline_future = futures::future::Fuse::terminated();
350
351	futures::pin_mut!(
352		race_source_updated,
353		source_nonces,
354		source_generate_proof,
355		source_go_offline_future,
356		race_target_updated,
357		target_require_source_header,
358		target_best_nonces,
359		target_finalized_nonces,
360		target_submit_proof,
361		target_tx_tracker,
362		target_go_offline_future,
363	);
364
365	loop {
366		futures::select! {
367			// when headers ids are updated
368			source_state = race_source_updated.next() => {
369				if let Some(source_state) = source_state {
370					let is_source_state_updated = race_state.best_finalized_source_header_id_at_source.as_ref()
371						!= Some(&source_state.best_finalized_self);
372					if is_source_state_updated {
373						source_nonces_required = true;
374						race_state.best_finalized_source_header_id_at_source
375							= Some(source_state.best_finalized_self);
376					}
377				}
378			},
379			target_state = race_target_updated.next() => {
380				if let Some(target_state) = target_state {
381					let is_target_best_state_updated = race_state.best_target_header_id.as_ref()
382						!= Some(&target_state.best_self);
383
384					if is_target_best_state_updated {
385						target_best_nonces_required = true;
386						race_state.best_target_header_id = Some(target_state.best_self);
387						race_state.best_finalized_source_header_id_at_best_target
388							= target_state.best_finalized_peer_at_best_self;
389					}
390
391					let is_target_finalized_state_updated = race_state.best_finalized_target_header_id.as_ref()
392						!= Some(&target_state.best_finalized_self);
393					if  is_target_finalized_state_updated {
394						target_finalized_nonces_required = true;
395						race_state.best_finalized_target_header_id = Some(target_state.best_finalized_self);
396					}
397				}
398			},
399
400			// when nonces are updated
401			nonces = source_nonces => {
402				source_nonces_required = false;
403
404				source_client_is_online = process_future_result(
405					nonces,
406					&mut source_retry_backoff,
407					|(at_block, nonces)| {
408						log::debug!(
409							target: "bridge",
410							"Received nonces from {}: {:?}",
411							P::source_name(),
412							nonces,
413						);
414
415						strategy.source_nonces_updated(at_block, nonces);
416					},
417					&mut source_go_offline_future,
418					async_std::task::sleep,
419					|| format!("Error retrieving nonces from {}", P::source_name()),
420				).fail_if_connection_error(FailedClient::Source)?;
421
422				// ask for more headers if we have nonces to deliver and required headers are missing
423				source_required_header = strategy
424					.required_source_header_at_target(race_state.clone())
425					.await;
426			},
427			nonces = target_best_nonces => {
428				target_best_nonces_required = false;
429
430				target_client_is_online = process_future_result(
431					nonces,
432					&mut target_retry_backoff,
433					|(_, nonces)| {
434						log::debug!(
435							target: "bridge",
436							"Received best nonces from {}: {:?}",
437							P::target_name(),
438							nonces,
439						);
440
441						strategy.best_target_nonces_updated(nonces, &mut race_state);
442					},
443					&mut target_go_offline_future,
444					async_std::task::sleep,
445					|| format!("Error retrieving best nonces from {}", P::target_name()),
446				).fail_if_connection_error(FailedClient::Target)?;
447			},
448			nonces = target_finalized_nonces => {
449				target_finalized_nonces_required = false;
450
451				target_client_is_online = process_future_result(
452					nonces,
453					&mut target_retry_backoff,
454					|(_, nonces)| {
455						log::debug!(
456							target: "bridge",
457							"Received finalized nonces from {}: {:?}",
458							P::target_name(),
459							nonces,
460						);
461
462						strategy.finalized_target_nonces_updated(nonces, &mut race_state);
463					},
464					&mut target_go_offline_future,
465					async_std::task::sleep,
466					|| format!("Error retrieving finalized nonces from {}", P::target_name()),
467				).fail_if_connection_error(FailedClient::Target)?;
468			},
469
470			// proof generation and submission
471			maybe_batch_transaction = target_require_source_header => {
472				source_required_header = None;
473
474				target_client_is_online = process_future_result(
475					maybe_batch_transaction,
476					&mut target_retry_backoff,
477					|maybe_batch_transaction: Option<TC::BatchTransaction>| {
478						log::debug!(
479							target: "bridge",
480							"Target {} client has been asked for more {} headers. Batch tx: {}",
481							P::target_name(),
482							P::source_name(),
483							maybe_batch_transaction
484								.as_ref()
485								.map(|bt| format!("yes ({:?})", bt.required_header_id()))
486								.unwrap_or_else(|| "no".into()),
487						);
488
489						target_batch_transaction = maybe_batch_transaction;
490					},
491					&mut target_go_offline_future,
492					async_std::task::sleep,
493					|| format!("Error asking for source headers at {}", P::target_name()),
494				).fail_if_connection_error(FailedClient::Target)?;
495			},
496			proof = source_generate_proof => {
497				source_client_is_online = process_future_result(
498					proof,
499					&mut source_retry_backoff,
500					|(at_block, nonces_range, proof, batch_transaction)| {
501						log::debug!(
502							target: "bridge",
503							"Received proof for nonces in range {:?} from {}",
504							nonces_range,
505							P::source_name(),
506						);
507
508						race_state.nonces_to_submit = Some((at_block, nonces_range, proof));
509						race_state.nonces_to_submit_batch = batch_transaction;
510					},
511					&mut source_go_offline_future,
512					async_std::task::sleep,
513					|| format!("Error generating proof at {}", P::source_name()),
514				).fail_if_error(FailedClient::Source).map(|_| true)?;
515			},
516			proof_submit_result = target_submit_proof => {
517				target_client_is_online = process_future_result(
518					proof_submit_result,
519					&mut target_retry_backoff,
520					|artifacts: NoncesSubmitArtifacts<TC::TransactionTracker>| {
521						log::debug!(
522							target: "bridge",
523							"Successfully submitted proof of nonces {:?} to {}",
524							artifacts.nonces,
525							P::target_name(),
526						);
527
528						race_state.nonces_submitted = Some(artifacts.nonces);
529						target_tx_tracker.set(artifacts.tx_tracker.wait().fuse());
530					},
531					&mut target_go_offline_future,
532					async_std::task::sleep,
533					|| format!("Error submitting proof {}", P::target_name()),
534				).fail_if_connection_error(FailedClient::Target)?;
535
536				// in any case - we don't need to retry submitting the same nonces again until
537				// we read nonces from the target client
538				race_state.reset_nonces_to_submit();
539				// if we have failed to submit transaction AND that is not the connection issue,
540				// then we need to read best target nonces before selecting nonces again
541				if !target_client_is_online {
542					strategy.reset_best_target_nonces();
543				}
544			},
545			target_transaction_status = target_tx_tracker => {
546				match (target_transaction_status, race_state.nonces_submitted.as_ref()) {
547					(TrackedTransactionStatus::Finalized(at_block), Some(nonces_submitted)) => {
548						// our transaction has been mined, but was it successful or not? let's check the best
549						// nonce at the target node.
550						let _ = race_target.nonces(at_block, false)
551							.await
552							.map_err(|e| format!("failed to read nonces from target node: {e:?}"))
553							.and_then(|(_, nonces_at_target)| {
554								if nonces_at_target.latest_nonce < *nonces_submitted.end() {
555									Err(format!(
556										"best nonce at target after tx is {:?} and we've submitted {:?}",
557										nonces_at_target.latest_nonce,
558										nonces_submitted.end(),
559									))
560								} else {
561									Ok(())
562								}
563							})
564							.map_err(|e| {
565								log::error!(
566									target: "bridge",
567									"{} -> {} race transaction failed: {}",
568									P::source_name(),
569									P::target_name(),
570									e,
571								);
572
573								race_state.reset_nonces_submitted();
574							});
575					},
576					(TrackedTransactionStatus::Lost, _) => {
577						log::warn!(
578							target: "bridge",
579							"{} -> {} race transaction has been lost. State: {:?}. Strategy: {:?}",
580							P::source_name(),
581							P::target_name(),
582							race_state,
583							strategy,
584						);
585
586						race_state.reset_nonces_submitted();
587					},
588					_ => (),
589				}
590			},
591
592			// when we're ready to retry request
593			_ = source_go_offline_future => {
594				source_client_is_online = true;
595			},
596			_ = target_go_offline_future => {
597				target_client_is_online = true;
598			},
599		}
600
601		progress_context = print_race_progress::<P, _>(progress_context, &strategy);
602
603		if source_client_is_online {
604			source_client_is_online = false;
605
606			// if we've started to submit batch transaction, let's prioritize it
607			//
608			// we're using `take` here, because we don't need batch transaction (i.e. some
609			// underlying finality proof) anymore for our future calls - we were unable to
610			// use it for our current state, so why would we need to keep an obsolete proof
611			// for the future?
612			let target_batch_transaction = target_batch_transaction.take();
613			let expected_race_state =
614				if let Some(ref target_batch_transaction) = target_batch_transaction {
615					// when selecting nonces for the batch transaction, we assume that the required
616					// source header is already at the target chain
617					let required_source_header_at_target =
618						target_batch_transaction.required_header_id();
619					let mut expected_race_state = race_state.clone();
620					expected_race_state.best_finalized_source_header_id_at_best_target =
621						Some(required_source_header_at_target);
622					expected_race_state
623				} else {
624					race_state.clone()
625				};
626
627			let nonces_to_deliver = select_nonces_to_deliver(expected_race_state, &strategy).await;
628			let best_at_source = strategy.best_at_source();
629
630			if let Some((at_block, nonces_range, proof_parameters)) = nonces_to_deliver {
631				log::debug!(
632					target: "bridge",
633					"Asking {} to prove nonces in range {:?} at block {:?}",
634					P::source_name(),
635					nonces_range,
636					at_block,
637				);
638
639				source_generate_proof.set(
640					race_source
641						.generate_proof(at_block, nonces_range, proof_parameters)
642						.and_then(|(at_source_block, nonces, proof)| async {
643							Ok((at_source_block, nonces, proof, target_batch_transaction))
644						})
645						.fuse(),
646				);
647			} else if let (true, Some(best_at_source)) = (source_nonces_required, best_at_source) {
648				log::debug!(target: "bridge", "Asking {} about message nonces", P::source_name());
649				let at_block = race_state
650					.best_finalized_source_header_id_at_source
651					.as_ref()
652					.expect(
653						"source_nonces_required is only true when\
654						best_finalized_source_header_id_at_source is Some; qed",
655					)
656					.clone();
657				source_nonces.set(race_source.nonces(at_block, best_at_source).fuse());
658			} else {
659				source_client_is_online = true;
660			}
661		}
662
663		if target_client_is_online {
664			target_client_is_online = false;
665
666			if let Some((at_block, nonces_range, proof)) = race_state.nonces_to_submit.as_ref() {
667				log::debug!(
668					target: "bridge",
669					"Going to submit proof of messages in range {:?} to {} node{}",
670					nonces_range,
671					P::target_name(),
672					race_state.nonces_to_submit_batch.as_ref().map(|tx| format!(
673						". This transaction is batched with sending the proof for header {:?}.",
674						tx.required_header_id())
675					).unwrap_or_default(),
676				);
677
678				target_submit_proof.set(
679					race_target
680						.submit_proof(
681							race_state.nonces_to_submit_batch.clone(),
682							at_block.clone(),
683							nonces_range.clone(),
684							proof.clone(),
685						)
686						.fuse(),
687				);
688			} else if let Some(source_required_header) = source_required_header.clone() {
689				log::debug!(
690					target: "bridge",
691					"Going to require {} header {:?} at {}",
692					P::source_name(),
693					source_required_header,
694					P::target_name(),
695				);
696				target_require_source_header
697					.set(race_target.require_source_header(source_required_header).fuse());
698			} else if target_best_nonces_required {
699				log::debug!(target: "bridge", "Asking {} about best message nonces", P::target_name());
700				let at_block = race_state
701					.best_target_header_id
702					.as_ref()
703					.expect("target_best_nonces_required is only true when best_target_header_id is Some; qed")
704					.clone();
705				target_best_nonces.set(race_target.nonces(at_block, false).fuse());
706			} else if target_finalized_nonces_required {
707				log::debug!(target: "bridge", "Asking {} about finalized message nonces", P::target_name());
708				let at_block = race_state
709					.best_finalized_target_header_id
710					.as_ref()
711					.expect(
712						"target_finalized_nonces_required is only true when\
713						best_finalized_target_header_id is Some; qed",
714					)
715					.clone();
716				target_finalized_nonces.set(race_target.nonces(at_block, true).fuse());
717			} else {
718				target_client_is_online = true;
719			}
720		}
721	}
722}

724/// Print race progress.
725fn print_race_progress<P, S>(prev_time: Instant, strategy: &S) -> Instant
726where
727	P: MessageRace,
728	S: RaceStrategy<P::SourceHeaderId, P::TargetHeaderId, P::Proof>,
729{
730	let now_time = Instant::now();
731
732	let need_update = now_time.saturating_duration_since(prev_time) > Duration::from_secs(10);
733	if !need_update {
734		return prev_time
735	}
736
737	let now_best_nonce_at_source = strategy.best_at_source();
738	let now_best_nonce_at_target = strategy.best_at_target();
739	log::info!(
740		target: "bridge",
741		"Synced {:?} of {:?} nonces in {} -> {} race",
742		now_best_nonce_at_target,
743		now_best_nonce_at_source,
744		P::source_name(),
745		P::target_name(),
746	);
747	now_time
748}
749
750async fn select_nonces_to_deliver<SourceHeaderId, TargetHeaderId, Proof, Strategy>(
751	race_state: impl RaceState<SourceHeaderId, TargetHeaderId>,
752	strategy: &Strategy,
753) -> Option<(SourceHeaderId, RangeInclusive<MessageNonce>, Strategy::ProofParameters)>
754where
755	SourceHeaderId: Clone,
756	Strategy: RaceStrategy<SourceHeaderId, TargetHeaderId, Proof>,
757{
758	let best_finalized_source_header_id_at_best_target =
759		race_state.best_finalized_source_header_id_at_best_target()?;
760	strategy
761		.select_nonces_to_deliver(race_state)
762		.await
763		.map(|(nonces_range, proof_parameters)| {
764			(best_finalized_source_header_id_at_best_target, nonces_range, proof_parameters)
765		})
766}

771	use crate::message_race_strategy::BasicStrategy;
772	use relay_utils::HeaderId;
773
774	#[async_std::test]
775	async fn proof_is_generated_at_best_block_known_to_target_node() {
776		const GENERATED_AT: u64 = 6;
777		const BEST_AT_SOURCE: u64 = 10;
778		const BEST_AT_TARGET: u64 = 8;
779
780		// target node only knows about source' BEST_AT_TARGET block
781		// source node has BEST_AT_SOURCE > BEST_AT_TARGET block
782		let mut race_state = RaceStateImpl::<_, _, (), ()> {
783			best_finalized_source_header_id_at_source: Some(HeaderId(
784				BEST_AT_SOURCE,
785				BEST_AT_SOURCE,
786			)),
787			best_finalized_source_header_id_at_best_target: Some(HeaderId(
788				BEST_AT_TARGET,
789				BEST_AT_TARGET,
790			)),
791			best_target_header_id: Some(HeaderId(0, 0)),
792			best_finalized_target_header_id: Some(HeaderId(0, 0)),
793			nonces_to_submit: None,
794			nonces_to_submit_batch: None,
795			nonces_submitted: None,
796		};
797
798		// we have some nonces to deliver and they're generated at GENERATED_AT < BEST_AT_SOURCE
799		let mut strategy = BasicStrategy::<_, _, _, _, _, ()>::new();
800		strategy.source_nonces_updated(
801			HeaderId(GENERATED_AT, GENERATED_AT),
802			SourceClientNonces { new_nonces: 0..=10, confirmed_nonce: None },
803		);
804		strategy.best_target_nonces_updated(
805			TargetClientNonces { latest_nonce: 5u64, nonces_data: () },
806			&mut race_state,
807		);
808
809		// the proof will be generated on source, but using BEST_AT_TARGET block
810		assert_eq!(
811			select_nonces_to_deliver(race_state, &strategy).await,
812			Some((HeaderId(BEST_AT_TARGET, BEST_AT_TARGET), 6..=10, (),))
813		);
814	}

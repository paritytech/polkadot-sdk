14//! Basic delivery strategy. The strategy selects nonces if:
15//!
16//! 1) there are more nonces on the source side than on the target side;
17//! 2) new nonces may be proved to target node (i.e. they have appeared at the block, which is known
18//!    to the target node).

29/// Queue of nonces known to the source node.
30pub type SourceRangesQueue<SourceHeaderHash, SourceHeaderNumber, SourceNoncesRange> =
31	VecDeque<(HeaderId<SourceHeaderHash, SourceHeaderNumber>, SourceNoncesRange)>;
32
33/// Nonces delivery strategy.
34#[derive(Debug)]
35pub struct BasicStrategy<
36	SourceHeaderNumber,
37	SourceHeaderHash,
38	TargetHeaderNumber,
39	TargetHeaderHash,
40	SourceNoncesRange,
41	Proof,
42> {
43	/// All queued nonces.
44	///
45	/// The queue may contain already delivered nonces. We only remove entries from this
46	/// queue after corresponding nonces are finalized by the target chain.
47	source_queue: SourceRangesQueue<SourceHeaderHash, SourceHeaderNumber, SourceNoncesRange>,
48	/// The best nonce known to target node at its best block. `None` if it has not been received
49	/// yet.
50	best_target_nonce: Option<MessageNonce>,
51	/// Unused generic types dump.
52	_phantom: PhantomData<(TargetHeaderNumber, TargetHeaderHash, Proof)>,
53}

100	/// Returns indices of source queue entries, which may be delivered to the target node.
101	///
102	/// The function may skip some nonces from the queue front if nonces from this entry are
103	/// already available at the **best** target block. After this block is finalized, the entry
104	/// will be removed from the queue.
105	///
106	/// All entries before and including the range end index, are guaranteed to be witnessed
107	/// at source blocks that are known to be finalized at the target node.
108	///
109	/// Returns `None` if no entries may be delivered.
110	pub fn available_source_queue_indices<
111		RS: RaceState<
112			HeaderId<SourceHeaderHash, SourceHeaderNumber>,
113			HeaderId<TargetHeaderHash, TargetHeaderNumber>,
114		>,
115	>(
116		&self,
117		race_state: RS,
118	) -> Option<RangeInclusive<usize>> {
119		// if we do not know best nonce at target node, we can't select anything
120		let best_target_nonce = self.best_target_nonce?;
121
122		// if we have already selected nonces that we want to submit, do nothing
123		if race_state.nonces_to_submit().is_some() {
124			return None
125		}
126
127		// if we already submitted some nonces, do nothing
128		if race_state.nonces_submitted().is_some() {
129			return None
130		}
131
132		// find first entry that may be delivered to the target node
133		let begin_index = self
134			.source_queue
135			.iter()
136			.enumerate()
137			.skip_while(|(_, (_, nonces))| nonces.end() <= best_target_nonce)
138			.map(|(index, _)| index)
139			.next()?;
140
141		// 1) we want to deliver all nonces, starting from `target_nonce + 1`
142		// 2) we can't deliver new nonce until header, that has emitted this nonce, is finalized
143		// by target client
144		// 3) selector is used for more complicated logic
145		//
146		// => let's first select range of entries inside deque that are already finalized at
147		// the target client and pass this range to the selector
148		let best_header_at_target = race_state.best_finalized_source_header_id_at_best_target()?;
149		let end_index = self
150			.source_queue
151			.iter()
152			.enumerate()
153			.skip(begin_index)
154			.take_while(|(_, (queued_at, _))| queued_at.0 <= best_header_at_target.0)
155			.map(|(index, _)| index)
156			.last()?;
157
158		Some(begin_index..=end_index)
159	}

161	/// Remove all nonces that are less than or equal to given nonce from the source queue.
162	fn remove_le_nonces_from_source_queue(&mut self, nonce: MessageNonce) {
163		while let Some((queued_at, queued_range)) = self.source_queue.pop_front() {
164			if let Some(range_to_requeue) = queued_range.greater_than(nonce) {
165				self.source_queue.push_front((queued_at, range_to_requeue));
166				break
167			}
168		}
169	}
170}

206	async fn required_source_header_at_target<
207		RS: RaceState<
208			HeaderId<SourceHeaderHash, SourceHeaderNumber>,
209			HeaderId<TargetHeaderHash, TargetHeaderNumber>,
210		>,
211	>(
212		&self,
213		race_state: RS,
214	) -> Option<HeaderId<SourceHeaderHash, SourceHeaderNumber>> {
215		let current_best = race_state.best_finalized_source_header_id_at_best_target()?;
216		self.source_queue
217			.back()
218			.and_then(|(h, _)| if h.0 > current_best.0 { Some(h.clone()) } else { None })
219	}

221	fn best_at_source(&self) -> Option<MessageNonce> {
222		let best_in_queue = self.source_queue.back().map(|(_, range)| range.end());
223		match (best_in_queue, self.best_target_nonce) {
224			(Some(best_in_queue), Some(best_target_nonce)) if best_in_queue > best_target_nonce =>
225				Some(best_in_queue),
226			(_, Some(best_target_nonce)) => Some(best_target_nonce),
227			(_, None) => None,
228		}
229	}
230
231	fn best_at_target(&self) -> Option<MessageNonce> {
232		self.best_target_nonce
233	}

235	fn source_nonces_updated(
236		&mut self,
237		at_block: HeaderId<SourceHeaderHash, SourceHeaderNumber>,
238		nonces: SourceClientNonces<SourceNoncesRange>,
239	) {
240		let best_in_queue = self
241			.source_queue
242			.back()
243			.map(|(_, range)| range.end())
244			.or(self.best_target_nonce)
245			.unwrap_or_default();
246		self.source_queue.extend(
247			nonces
248				.new_nonces
249				.greater_than(best_in_queue)
250				.into_iter()
251				.map(move |range| (at_block.clone(), range)),
252		)
253	}
254
255	fn reset_best_target_nonces(&mut self) {
256		self.best_target_nonce = None;
257	}
258
259	fn best_target_nonces_updated<
260		RS: RaceState<
261			HeaderId<SourceHeaderHash, SourceHeaderNumber>,
262			HeaderId<TargetHeaderHash, TargetHeaderNumber>,
263		>,
264	>(
265		&mut self,
266		nonces: TargetClientNonces<()>,
267		race_state: &mut RS,
268	) {
269		let nonce = nonces.latest_nonce;
270
271		// if **some** of nonces that we have selected to submit already present at the
272		// target chain => select new nonces
273		let need_to_select_new_nonces = race_state
274			.nonces_to_submit()
275			.map(|nonces| nonce >= *nonces.start())
276			.unwrap_or(false);
277		if need_to_select_new_nonces {
278			tracing::trace!(
279				target: "bridge",
280				%nonce,
281				nonces_to_submit=?race_state.nonces_to_submit(),
282				"Latest nonce at target. Clearing nonces to submit"
283			);
284
285			race_state.reset_nonces_to_submit();
286		}
287
288		// if **some** of nonces that we have submitted already present at the
289		// target chain => select new nonces
290		let need_new_nonces_to_submit = race_state
291			.nonces_submitted()
292			.map(|nonces| nonce >= *nonces.start())
293			.unwrap_or(false);
294		if need_new_nonces_to_submit {
295			tracing::trace!(
296				target: "bridge",
297				%nonce,
298				nonces_submitted=?race_state.nonces_submitted(),
299				"Latest nonce at target. Clearing submitted nonces"
300			);
301
302			race_state.reset_nonces_submitted();
303		}
304
305		self.best_target_nonce = Some(nonce);
306	}
307
308	fn finalized_target_nonces_updated<
309		RS: RaceState<
310			HeaderId<SourceHeaderHash, SourceHeaderNumber>,
311			HeaderId<TargetHeaderHash, TargetHeaderNumber>,
312		>,
313	>(
314		&mut self,
315		nonces: TargetClientNonces<()>,
316		_race_state: &mut RS,
317	) {
318		self.remove_le_nonces_from_source_queue(nonces.latest_nonce);
319		self.best_target_nonce = Some(std::cmp::max(
320			self.best_target_nonce.unwrap_or(nonces.latest_nonce),
321			nonces.latest_nonce,
322		));
323	}
324
325	async fn select_nonces_to_deliver<
326		RS: RaceState<
327			HeaderId<SourceHeaderHash, SourceHeaderNumber>,
328			HeaderId<TargetHeaderHash, TargetHeaderNumber>,
329		>,
330	>(
331		&self,
332		race_state: RS,
333	) -> Option<(RangeInclusive<MessageNonce>, Self::ProofParameters)> {
334		let available_indices = self.available_source_queue_indices(race_state)?;
335		let range_begin = std::cmp::max(
336			self.best_target_nonce? + 1,
337			self.source_queue[*available_indices.start()].1.begin(),
338		);
339		let range_end = self.source_queue[*available_indices.end()].1.end();
340		Some((range_begin..=range_end, ()))
341	}
342}

558		let target_header_1 = header_id(1);
559
560		// we start in perfect sync state - all headers are synced and finalized on both ends
561		let mut state = TestRaceStateImpl {
562			best_finalized_source_header_id_at_source: Some(source_header_1),
563			best_finalized_source_header_id_at_best_target: Some(source_header_1),
564			best_target_header_id: Some(target_header_1),
565			best_finalized_target_header_id: Some(target_header_1),
566			nonces_to_submit: None,
567			nonces_to_submit_batch: None,
568			nonces_submitted: None,
569		};
570
571		// in this state we have 1 available nonce for delivery
572		let mut strategy = BasicStrategy::<TestMessageLane> {
573			source_queue: vec![(header_id(1), 1..=1)].into_iter().collect(),
574			best_target_nonce: Some(0),
575			_phantom: PhantomData,
576		};
577		assert_eq!(strategy.select_nonces_to_deliver(state.clone()).await, Some((1..=1, ())),);
578
579		// let's say we have submitted 1..=1
580		state.nonces_submitted = Some(1..=1);
581
582		// then new nonce 2 appear at the source block 2
583		let source_header_2 = header_id(2);
584		state.best_finalized_source_header_id_at_source = Some(source_header_2);
585		strategy.source_nonces_updated(
586			source_header_2,
587			SourceClientNonces { new_nonces: 2..=2, confirmed_nonce: None },
588		);
589		// and nonce 1 appear at the best block of the target node (best finalized still has 0
590		// nonces)
591		let target_header_2 = header_id(2);
592		state.best_target_header_id = Some(target_header_2);
593		strategy.best_target_nonces_updated(
594			TargetClientNonces { latest_nonce: 1, nonces_data: () },
595			&mut state,
596		);
597
598		// then best target header is retracted
599		strategy.best_target_nonces_updated(
600			TargetClientNonces { latest_nonce: 0, nonces_data: () },
601			&mut state,
602		);
603
604		// ... and some fork with zero delivered nonces is finalized
605		let target_header_2_fork = header_id(2_1);
606		state.best_finalized_source_header_id_at_source = Some(source_header_2);
607		state.best_finalized_source_header_id_at_best_target = Some(source_header_2);
608		state.best_target_header_id = Some(target_header_2_fork);
609		state.best_finalized_target_header_id = Some(target_header_2_fork);
610		strategy.finalized_target_nonces_updated(
611			TargetClientNonces { latest_nonce: 0, nonces_data: () },
612			&mut state,
613		);
614
615		// now we have to select nonce 1 for delivery again
616		assert_eq!(strategy.select_nonces_to_deliver(state.clone()).await, Some((1..=2, ())),);
617	}

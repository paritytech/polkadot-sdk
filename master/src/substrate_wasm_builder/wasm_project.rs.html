84/// Helper struct for managing blob file paths.
85struct BlobPaths {
86	/// The base name of the blob (without extension).
87	blob_name: String,
88	/// The project directory where blobs are stored.
89	project: PathBuf,
90}
91
92impl BlobPaths {
93	fn new(blob_name: String, project: PathBuf) -> Self {
94		Self { blob_name, project }
95	}
96
97	/// Returns the path to the bloaty wasm file.
98	fn bloaty(&self) -> PathBuf {
99		self.project.join(format!("{}.wasm", self.blob_name))
100	}
101
102	/// Returns the path to the compact wasm file.
103	fn compact(&self) -> PathBuf {
104		self.project.join(format!("{}.compact.wasm", self.blob_name))
105	}
106
107	/// Returns the path to the compact compressed wasm file.
108	fn compact_compressed(&self) -> PathBuf {
109		self.project.join(format!("{}.compact.compressed.wasm", self.blob_name))
110	}
111
112	/// Returns the blob name.
113	fn name(&self) -> &str {
114		&self.blob_name
115	}
116}

118fn crate_metadata(cargo_manifest: &Path) -> Metadata {
119	let mut cargo_lock = cargo_manifest.to_path_buf();
120	cargo_lock.set_file_name("Cargo.lock");
121
122	let cargo_lock_existed = cargo_lock.exists();
123
124	// If we can find a `Cargo.lock`, we assume that this is the workspace root and there exists a
125	// `Cargo.toml` that we can use for getting the metadata.
126	let cargo_manifest = if let Some(mut cargo_lock) = find_cargo_lock(cargo_manifest) {
127		cargo_lock.set_file_name("Cargo.toml");
128		cargo_lock
129	} else {
130		cargo_manifest.to_path_buf()
131	};
132
133	let crate_metadata_command = create_metadata_command(cargo_manifest);
134
135	let crate_metadata = crate_metadata_command
136		.exec()
137		.expect("`cargo metadata` can not fail on project `Cargo.toml`; qed");
138	// If the `Cargo.lock` didn't exist, we need to remove it after
139	// calling `cargo metadata`. This is required to ensure that we don't change
140	// the build directory outside of the `target` folder. Commands like
141	// `cargo publish` require this.
142	if !cargo_lock_existed {
143		let _ = fs::remove_file(&cargo_lock);
144	}
145
146	crate_metadata
147}

149/// Keep the build directories separate so that when switching between the
150/// targets we won't trigger unnecessary rebuilds.
151fn build_subdirectory(target: RuntimeTarget) -> &'static str {
152	match target {
153		RuntimeTarget::Wasm => "wbuild",
154		RuntimeTarget::Riscv => "rbuild",
155	}

158/// Creates the WASM project, compiles the WASM binary and compacts the WASM binary.
159///
160/// # Returns
161///
162/// The path to the compact runtime binary and the bloaty runtime binary.
163pub(crate) fn create_and_compile(
164	target: RuntimeTarget,
165	orig_project_cargo_toml: &Path,
166	default_rustflags: &str,
167	cargo_cmd: CargoCommandVersioned,
168	features_to_enable: Vec<String>,
169	blob_out_name_override: Option<String>,
170	check_for_runtime_version_section: bool,
171	#[cfg(feature = "metadata-hash")] enable_metadata_hash: Option<MetadataExtraInfo>,
172) -> (Option<WasmBinary>, WasmBinaryBloaty) {
173	let runtime_workspace_root = get_wasm_workspace_root();
174	let runtime_workspace = runtime_workspace_root.join(build_subdirectory(target));
175
176	let crate_metadata = crate_metadata(orig_project_cargo_toml);
177
178	let project = create_project(
179		target,
180		orig_project_cargo_toml,
181		&runtime_workspace,
182		&crate_metadata,
183		crate_metadata.workspace_root.as_ref(),
184		features_to_enable,
185	);
186	let wasm_project_cargo_toml = project.join("Cargo.toml");
187
188	let build_config = BuildConfiguration::detect(target, &project);
189
190	#[cfg(feature = "metadata-hash")]
191	let raw_blob_path = match enable_metadata_hash {
192		Some(extra_info) => {
193			// When the metadata hash is enabled we need to build the runtime twice.
194			let raw_blob_path = build_bloaty_blob(
195				target,
196				&build_config.blob_build_profile,
197				&project,
198				default_rustflags,
199				cargo_cmd.clone(),
200				None,
201			);
202
203			let hash = crate::metadata_hash::generate_metadata_hash(&raw_blob_path, extra_info);
204
205			build_bloaty_blob(
206				target,
207				&build_config.blob_build_profile,
208				&project,
209				default_rustflags,
210				cargo_cmd,
211				Some(hash),
212			)
213		},
214		None => build_bloaty_blob(
215			target,
216			&build_config.blob_build_profile,
217			&project,
218			default_rustflags,
219			cargo_cmd,
220			None,
221		),
222	};
223
224	// If the feature is not enabled, we only need to do it once.
225	#[cfg(not(feature = "metadata-hash"))]
226	let raw_blob_path = {
227		build_bloaty_blob(
228			target,
229			&build_config.blob_build_profile,
230			&project,
231			default_rustflags,
232			cargo_cmd,
233		)
234	};
235
236	let blob_name =
237		blob_out_name_override.unwrap_or_else(|| get_blob_name(target, &wasm_project_cargo_toml));
238	let blob_paths = BlobPaths::new(blob_name, project.clone());
239
240	let (final_blob_binary, bloaty_blob_binary, any_changed) = match target {
241		RuntimeTarget::Wasm => {
242			let out_path = blob_paths.bloaty();
243			let bloaty_changed = copy_file_if_changed(&raw_blob_path, &out_path);
244
245			let (final_binary, bloaty_binary, did_compact) = maybe_compact_and_compress_wasm(
246				&wasm_project_cargo_toml,
247				WasmBinaryBloaty(out_path),
248				&blob_paths,
249				check_for_runtime_version_section,
250				&build_config,
251				bloaty_changed,
252			);
253			(final_binary, bloaty_binary, bloaty_changed || did_compact)
254		},
255		RuntimeTarget::Riscv => {
256			let out_path = project.join(format!("{}.polkavm", blob_paths.name()));
257			let changed = copy_file_if_changed(&raw_blob_path, &out_path);
258			(None, WasmBinaryBloaty(out_path), changed)
259		},
260	};
261
262	generate_rerun_if_changed_instructions(
263		orig_project_cargo_toml,
264		&project,
265		&runtime_workspace,
266		final_blob_binary.as_ref(),
267		&bloaty_blob_binary,
268	);
269
270	if any_changed {
271		if let Err(err) = adjust_mtime(&bloaty_blob_binary, final_blob_binary.as_ref()) {
272			build_helper::warning!("Error while adjusting the mtime of the blob binaries: {}", err)
273		}
274	}
275
276	(final_blob_binary, bloaty_blob_binary)
277}

279fn maybe_compact_and_compress_wasm(
280	wasm_project_cargo_toml: &Path,
281	bloaty_blob_binary: WasmBinaryBloaty,
282	blob_paths: &BlobPaths,
283	check_for_runtime_version_section: bool,
284	build_config: &BuildConfiguration,
285	bloaty_changed: bool,
286) -> (Option<WasmBinary>, WasmBinaryBloaty, bool) {
287	let needs_compact = build_config.outer_build_profile.wants_compact();
288	let compact_path = blob_paths.compact();
289	let compressed_path = blob_paths.compact_compressed();
290	let compact_or_compressed_exists = compact_path.exists() || compressed_path.exists();
291	let should_regenerate = bloaty_changed || (needs_compact && !compact_or_compressed_exists);
292
293	if !should_regenerate {
294		let final_blob = if compressed_path.exists() {
295			Some(WasmBinary(compressed_path))
296		} else if compact_path.exists() {
297			Some(WasmBinary(compact_path))
298		} else {
299			None
300		};
301
302		return (final_blob, bloaty_blob_binary, false);
303	}
304
305	// Try to compact and compress the bloaty blob, if the *outer* profile wants it.
306	//
307	// This is because, by default the inner profile will be set to `Release` even when the outer
308	// profile is `Debug`, because the blob built in `Debug` profile is too slow for normal
309	// development activities.
310	let (compact_blob_path, compact_compressed_blob_path) = if needs_compact {
311		let compact_blob_path = compact_wasm(blob_paths, &bloaty_blob_binary);
312		let compact_compressed_blob_path =
313			compact_blob_path.as_ref().and_then(|p| try_compress_blob(blob_paths, p));
314		(compact_blob_path, compact_compressed_blob_path)
315	} else {
316		// We at least want to lower the `sign-ext` code to `mvp`.
317		wasm_opt::OptimizationOptions::new_opt_level_0()
318			.add_pass(wasm_opt::Pass::SignextLowering)
319			.debug_info(true)
320			.run(bloaty_blob_binary.bloaty_path(), bloaty_blob_binary.bloaty_path())
321			.expect("Failed to lower sign-ext in WASM binary.");
322
323		(None, None)
324	};
325
326	if check_for_runtime_version_section {
327		ensure_runtime_version_wasm_section_exists(bloaty_blob_binary.bloaty_path());
328	}
329
330	let final_blob_binary = compact_compressed_blob_path.or(compact_blob_path);
331
332	final_blob_binary
333		.as_ref()
334		.map(|binary| copy_blob_to_target_directory(wasm_project_cargo_toml, binary));
335
336	(final_blob_binary, bloaty_blob_binary, true)
337}

339/// Ensures that the `runtime_version` section exists in the given blob.
340///
341/// If the section can not be found, it will print an error and exit the builder.
342fn ensure_runtime_version_wasm_section_exists(blob_path: &Path) {
343	let blob = fs::read(blob_path).expect("`{blob_path}` was just written and should exist; qed");
344
345	let module: Module = match deserialize_buffer(&blob) {
346		Ok(m) => m,
347		Err(e) => {
348			println!("Failed to deserialize `{}`: {e:?}", blob_path.display());
349			process::exit(1);
350		},
351	};
352
353	if !module.custom_sections().any(|cs| cs.name() == "runtime_version") {
354		println!(
355			"Couldn't find the `runtime_version` section. \
356				  Please ensure that you are using the `sp_version::runtime_version` attribute macro!"
357		);
358		process::exit(1);
359	}
360}

362/// Adjust the mtime of the bloaty and compressed/compact wasm files.
363///
364/// We add the bloaty and the compressed/compact wasm file to the `rerun-if-changed` files.
365/// Cargo/Rustc determines based on the timestamp of the `invoked.timestamp` file that can be found
366/// in the `OUT_DIR/..`, if it needs to rerun a `build.rs` script. The problem is that this
367/// `invoked.timestamp` is created when the `build.rs` is executed and the wasm binaries are created
368/// later. This leads to them having a later mtime than the `invoked.timestamp` file and thus,
369/// cargo/rustc always re-executes the `build.rs` script. To hack around this, we copy the mtime of
370/// the `invoked.timestamp` to the wasm binaries.
371fn adjust_mtime(
372	bloaty_wasm: &WasmBinaryBloaty,
373	compressed_or_compact_wasm: Option<&WasmBinary>,
374) -> std::io::Result<()> {
375	let out_dir = build_helper::out_dir();
376	let invoked_timestamp = out_dir.join("../invoked.timestamp");
377
378	// Get the mtime of the `invoked.timestamp`
379	let metadata = fs::metadata(invoked_timestamp)?;
380	let mtime = filetime::FileTime::from_last_modification_time(&metadata);
381
382	filetime::set_file_mtime(bloaty_wasm.bloaty_path(), mtime)?;
383	if let Some(binary) = compressed_or_compact_wasm.as_ref() {
384		filetime::set_file_mtime(binary.wasm_binary_path(), mtime)?;
385	}
386
387	Ok(())
388}

390/// Find the `Cargo.lock` relative to the `OUT_DIR` environment variable.
391///
392/// If the `Cargo.lock` cannot be found, we emit a warning and return `None`.
393fn find_cargo_lock(cargo_manifest: &Path) -> Option<PathBuf> {
394	fn find_impl(mut path: PathBuf) -> Option<PathBuf> {
395		loop {
396			if path.join("Cargo.lock").exists() {
397				return Some(path.join("Cargo.lock"))
398			}
399
400			if !path.pop() {
401				return None
402			}
403		}
404	}
405
406	if let Ok(workspace) = env::var(crate::WASM_BUILD_WORKSPACE_HINT) {
407		let path = PathBuf::from(workspace);
408
409		if path.join("Cargo.lock").exists() {
410			return Some(path.join("Cargo.lock"))
411		} else {
412			build_helper::warning!(
413				"`{}` env variable doesn't point to a directory that contains a `Cargo.lock`.",
414				crate::WASM_BUILD_WORKSPACE_HINT,
415			);
416		}
417	}
418
419	if let Some(path) = find_impl(build_helper::out_dir()) {
420		return Some(path)
421	}
422
423	build_helper::warning!(
424		"Could not find `Cargo.lock` for `{}`, while searching from `{}`. \
425		 To fix this, point the `{}` env variable to the directory of the workspace being compiled.",
426		cargo_manifest.display(),
427		build_helper::out_dir().display(),
428		crate::WASM_BUILD_WORKSPACE_HINT,
429	);
430
431	None
432}

434/// Extract the crate name from the given `Cargo.toml`.
435fn get_crate_name(cargo_manifest: &Path) -> String {
436	let cargo_toml: Table = toml::from_str(
437		&fs::read_to_string(cargo_manifest).expect("File exists as checked before; qed"),
438	)
439	.expect("Cargo manifest is a valid toml file; qed");
440
441	let package = cargo_toml
442		.get("package")
443		.and_then(|t| t.as_table())
444		.expect("`package` key exists in valid `Cargo.toml`; qed");
445
446	package
447		.get("name")
448		.and_then(|p| p.as_str())
449		.map(ToOwned::to_owned)
450		.expect("Package name exists; qed")
451}

453/// Extract the `lib.name` from the given `Cargo.toml`.
454fn get_lib_name(cargo_manifest: &Path) -> Option<String> {
455	let cargo_toml: Table = toml::from_str(
456		&fs::read_to_string(cargo_manifest).expect("File exists as checked before; qed"),
457	)
458	.expect("Cargo manifest is a valid toml file; qed");
459
460	let lib = cargo_toml.get("lib").and_then(|t| t.as_table())?;
461
462	lib.get("name").and_then(|p| p.as_str()).map(ToOwned::to_owned)
463}

465/// Returns the name for the blob binary.
466fn get_blob_name(target: RuntimeTarget, cargo_manifest: &Path) -> String {
467	match target {
468		RuntimeTarget::Wasm => get_lib_name(cargo_manifest)
469			.expect("The wasm project should have a `lib.name`; qed")
470			.replace('-', "_"),
471		RuntimeTarget::Riscv => get_crate_name(cargo_manifest),
472	}
473}

475/// Returns the root path of the wasm workspace.
476fn get_wasm_workspace_root() -> PathBuf {
477	let mut out_dir = build_helper::out_dir();
478
479	loop {
480		match out_dir.parent() {
481			Some(parent) if out_dir.ends_with("build") => return parent.to_path_buf(),
482			_ =>
483				if !out_dir.pop() {
484					break
485				},
486		}
487	}
488
489	panic!("Could not find target dir in: {}", build_helper::out_dir().display())
490}

492fn create_project_cargo_toml(
493	target: RuntimeTarget,
494	wasm_workspace: &Path,
495	workspace_root_path: &Path,
496	crate_name: &str,
497	crate_path: &Path,
498	enabled_features: impl Iterator<Item = String>,
499) {
500	let mut workspace_toml: Table = toml::from_str(
501		&fs::read_to_string(workspace_root_path.join("Cargo.toml"))
502			.expect("Workspace root `Cargo.toml` exists; qed"),
503	)
504	.expect("Workspace root `Cargo.toml` is a valid toml file; qed");
505
506	let mut wasm_workspace_toml = Table::new();
507
508	// Add different profiles which are selected by setting `WASM_BUILD_TYPE`.
509	let mut release_profile = Table::new();
510	release_profile.insert("panic".into(), "abort".into());
511	release_profile.insert("lto".into(), "thin".into());
512
513	let mut production_profile = Table::new();
514	production_profile.insert("inherits".into(), "release".into());
515	production_profile.insert("lto".into(), "fat".into());
516	production_profile.insert("codegen-units".into(), 1.into());
517
518	let mut dev_profile = Table::new();
519	dev_profile.insert("panic".into(), "abort".into());
520
521	let mut profile = Table::new();
522	profile.insert("release".into(), release_profile.into());
523	profile.insert("production".into(), production_profile.into());
524	profile.insert("dev".into(), dev_profile.into());
525
526	wasm_workspace_toml.insert("profile".into(), profile.into());
527
528	// Add patch section from the project root `Cargo.toml`
529	while let Some(mut patch) =
530		workspace_toml.remove("patch").and_then(|p| p.try_into::<Table>().ok())
531	{
532		// Iterate over all patches and make the patch path absolute from the workspace root path.
533		patch
534			.iter_mut()
535			.filter_map(|p| {
536				p.1.as_table_mut().map(|t| t.iter_mut().filter_map(|t| t.1.as_table_mut()))
537			})
538			.flatten()
539			.for_each(|p| {
540				p.iter_mut().filter(|(k, _)| k == &"path").for_each(|(_, v)| {
541					if let Some(path) = v.as_str().map(PathBuf::from) {
542						if path.is_relative() {
543							*v = workspace_root_path.join(path).display().to_string().into();
544						}
545					}
546				})
547			});
548
549		wasm_workspace_toml.insert("patch".into(), patch.into());
550	}
551
552	let mut package = Table::new();
553	package.insert("name".into(), format!("{}-blob", crate_name).into());
554	package.insert("version".into(), "1.0.0".into());
555	package.insert("edition".into(), "2021".into());
556
557	wasm_workspace_toml.insert("package".into(), package.into());
558
559	if target == RuntimeTarget::Wasm {
560		let mut lib = Table::new();
561		lib.insert("name".into(), crate_name.replace("-", "_").into());
562		lib.insert("crate-type".into(), vec!["cdylib".to_string()].into());
563		wasm_workspace_toml.insert("lib".into(), lib.into());
564	}
565
566	let mut dependencies = Table::new();
567
568	let mut wasm_project = Table::new();
569	wasm_project.insert("package".into(), crate_name.into());
570	wasm_project.insert("path".into(), crate_path.display().to_string().into());
571	wasm_project.insert("default-features".into(), false.into());
572	wasm_project.insert("features".into(), enabled_features.collect::<Vec<_>>().into());
573
574	dependencies.insert("wasm-project".into(), wasm_project.into());
575
576	wasm_workspace_toml.insert("dependencies".into(), dependencies.into());
577
578	let mut workspace = Table::new();
579	workspace.insert("resolver".into(), "2".into());
580
581	wasm_workspace_toml.insert("workspace".into(), workspace.into());
582
583	if target == RuntimeTarget::Riscv {
584		// This dependency currently doesn't compile under RISC-V, so patch it with our own fork.
585		//
586		// TODO: Remove this once a new version of `bitvec` (which uses a new version of `radium`
587		//       which doesn't have this problem) is released on crates.io.
588		let radium_patch = toml::toml! {
589			radium = { git = "https://github.com/paritytech/radium-0.7-fork.git", rev = "a5da15a15c90fd169d661d206cf0db592487f52b" }
590		};
591
592		let mut patch = wasm_workspace_toml
593			.get("patch")
594			.and_then(|p| p.as_table().cloned())
595			.unwrap_or_default();
596
597		if let Some(existing_crates_io) = patch.get_mut("crates-io").and_then(|t| t.as_table_mut())
598		{
599			existing_crates_io.extend(radium_patch);
600		} else {
601			patch.insert("crates-io".into(), radium_patch.into());
602		}
603
604		wasm_workspace_toml.insert("patch".into(), patch.into());
605	}
606
607	write_file_if_changed(
608		wasm_workspace.join("Cargo.toml"),
609		toml::to_string_pretty(&wasm_workspace_toml).expect("Wasm workspace toml is valid; qed"),
610	);
611}

613/// Find a package by the given `manifest_path` in the metadata. In case it can't be found by its
614/// manifest_path, fallback to finding it by name; this is necessary during publish because the
615/// package's manifest path will be *generated* within a specific packaging directory, thus it won't
616/// be found by its original path anymore.
617///
618/// Panics if the package could not be found.
619fn find_package_by_manifest_path<'a>(
620	pkg_name: &str,
621	manifest_path: &Path,
622	crate_metadata: &'a cargo_metadata::Metadata,
623) -> &'a cargo_metadata::Package {
624	if let Some(pkg) = crate_metadata.packages.iter().find(|p| p.manifest_path == manifest_path) {
625		return pkg
626	}
627
628	let pkgs_by_name = crate_metadata
629		.packages
630		.iter()
631		.filter(|p| p.name == pkg_name)
632		.collect::<Vec<_>>();
633
634	if let Some(pkg) = pkgs_by_name.first() {
635		if pkgs_by_name.len() > 1 {
636			panic!(
637				"Found multiple packages matching the name {pkg_name} ({manifest_path:?}): {:?}",
638				pkgs_by_name
639			);
640		} else {
641			return pkg
642		}
643	} else {
644		panic!("Failed to find entry for package {pkg_name} ({manifest_path:?}).");
645	}
646}

648/// Get a list of enabled features for the project.
649fn project_enabled_features(
650	pkg_name: &str,
651	cargo_manifest: &Path,
652	crate_metadata: &cargo_metadata::Metadata,
653) -> Vec<String> {
654	let package = find_package_by_manifest_path(pkg_name, cargo_manifest, crate_metadata);
655
656	let std_enabled = package.features.get("std");
657
658	let mut enabled_features = package
659		.features
660		.iter()
661		.filter(|(f, v)| {
662			let mut feature_env = f.replace("-", "_");
663			feature_env.make_ascii_uppercase();
664
665			// If this is a feature that corresponds only to an optional dependency
666			// and this feature is enabled by the `std` feature, we assume that this
667			// is only done through the `std` feature. This is a bad heuristic and should
668			// be removed after namespaced features are landed:
669			// https://doc.rust-lang.org/cargo/reference/unstable.html#namespaced-features
670			// Then we can just express this directly in the `Cargo.toml` and do not require
671			// this heuristic anymore. However, for the transition phase between now and namespaced
672			// features already being present in nightly, we need this code to make
673			// runtimes compile with all the possible rustc versions.
674			if v.len() == 1 &&
675				v.get(0).map_or(false, |v| *v == format!("dep:{}", f)) &&
676				std_enabled.as_ref().map(|e| e.iter().any(|ef| ef == *f)).unwrap_or(false)
677			{
678				return false
679			}
680
681			// We don't want to enable the `std`/`default` feature for the wasm build and
682			// we need to check if the feature is enabled by checking the env variable.
683			*f != "std" &&
684				*f != "default" &&
685				env::var(format!("CARGO_FEATURE_{feature_env}"))
686					.map(|v| v == "1")
687					.unwrap_or_default()
688		})
689		.map(|d| d.0.clone())
690		.collect::<Vec<_>>();
691
692	enabled_features.sort();
693	enabled_features
694}

696/// Returns if the project has the `runtime-wasm` feature
697fn has_runtime_wasm_feature_declared(
698	pkg_name: &str,
699	cargo_manifest: &Path,
700	crate_metadata: &cargo_metadata::Metadata,
701) -> bool {
702	let package = find_package_by_manifest_path(pkg_name, cargo_manifest, crate_metadata);
703
704	package.features.keys().any(|k| k == "runtime-wasm")
705}

707/// Create the project used to build the wasm binary.
708///
709/// # Returns
710///
711/// The path to the created wasm project.
712fn create_project(
713	target: RuntimeTarget,
714	project_cargo_toml: &Path,
715	wasm_workspace: &Path,
716	crate_metadata: &Metadata,
717	workspace_root_path: &Path,
718	features_to_enable: Vec<String>,
719) -> PathBuf {
720	let crate_name = get_crate_name(project_cargo_toml);
721	let crate_path = project_cargo_toml.parent().expect("Parent path exists; qed");
722	let wasm_project_folder = wasm_workspace.join(&crate_name);
723
724	fs::create_dir_all(wasm_project_folder.join("src"))
725		.expect("Wasm project dir create can not fail; qed");
726
727	let mut enabled_features =
728		project_enabled_features(&crate_name, project_cargo_toml, crate_metadata);
729
730	if has_runtime_wasm_feature_declared(&crate_name, project_cargo_toml, crate_metadata) {
731		enabled_features.push("runtime-wasm".into());
732	}
733
734	let mut enabled_features = enabled_features.into_iter().collect::<HashSet<_>>();
735	enabled_features.extend(features_to_enable.into_iter());
736
737	create_project_cargo_toml(
738		target,
739		&wasm_project_folder,
740		workspace_root_path,
741		&crate_name,
742		crate_path,
743		enabled_features.into_iter(),
744	);
745
746	match target {
747		RuntimeTarget::Wasm => {
748			write_file_if_changed(
749				wasm_project_folder.join("src/lib.rs"),
750				"#![no_std] #![allow(unused_imports)] pub use wasm_project::*;",
751			);
752		},
753		RuntimeTarget::Riscv => {
754			write_file_if_changed(
755				wasm_project_folder.join("src/main.rs"),
756				"#![no_std] #![no_main] #![allow(unused_imports)] pub use wasm_project::*;",
757			);
758		},
759	}
760
761	if let Some(crate_lock_file) = find_cargo_lock(project_cargo_toml) {
762		// Use the `Cargo.lock` of the main project.
763		copy_file_if_changed(&crate_lock_file, &wasm_project_folder.join("Cargo.lock"));
764	}
765
766	wasm_project_folder
767}

769/// A rustc profile.
770#[derive(Clone, Debug, EnumIter)]
771enum Profile {
772	/// The `--profile dev` profile.
773	Debug,
774	/// The `--profile release` profile.
775	Release,
776	/// The `--profile production` profile.
777	Production,
778}
779
780impl Profile {
781	/// The name of the profile as supplied to the cargo `--profile` cli option.
782	fn name(&self) -> &'static str {
783		match self {
784			Self::Debug => "dev",
785			Self::Release => "release",
786			Self::Production => "production",
787		}
788	}
789
790	/// The sub directory within `target` where cargo places the build output.
791	///
792	/// # Note
793	///
794	/// Usually this is the same as [`Self::name`] with the exception of the debug
795	/// profile which is called `dev`.
796	fn directory(&self) -> &'static str {
797		match self {
798			Self::Debug => "debug",
799			_ => self.name(),
800		}
801	}
802
803	/// Whether the resulting binary should be compacted and compressed.
804	fn wants_compact(&self) -> bool {
805		!matches!(self, Self::Debug)
806	}
807}
808
809/// The build configuration for this build.
810#[derive(Debug)]
811struct BuildConfiguration {
812	/// The profile that is used to build the outer project.
813	pub outer_build_profile: Profile,
814	/// The profile to use to build the runtime blob.
815	pub blob_build_profile: Profile,
816}
817
818impl BuildConfiguration {
819	/// Create a [`BuildConfiguration`] by detecting which profile is used for the main build and
820	/// checking any env var overrides.
821	///
822	/// We cannot easily determine the profile that is used by the main cargo invocation
823	/// because the `PROFILE` environment variable won't contain any custom profiles like
824	/// "production". It would only contain the builtin profile where the custom profile
825	/// inherits from. This is why we inspect the build path to learn which profile is used.
826	///
827	/// When not overridden by a env variable we always default to building wasm with the `Release`
828	/// profile even when the main build uses the debug build. This is because wasm built with the
829	/// `Debug` profile is too slow for normal development activities and almost never intended.
830	///
831	/// When cargo is building in `--profile dev`, user likely intends to compile fast, so we don't
832	/// bother producing compact or compressed blobs.
833	///
834	/// # Note
835	///
836	/// Can be overridden by setting [`crate::WASM_BUILD_TYPE_ENV`].
837	fn detect(target: RuntimeTarget, wasm_project: &Path) -> Self {
838		let (name, overridden) = if let Ok(name) = env::var(crate::WASM_BUILD_TYPE_ENV) {
839			(name, true)
840		} else {
841			// First go backwards to the beginning of the target directory.
842			// Then go forwards to find the build subdirectory.
843			// We need to go backwards first because when starting from the root there
844			// might be a chance that someone has a directory somewhere in the path with the same
845			// name.
846			let name = wasm_project
847				.components()
848				.rev()
849				.take_while(|c| c.as_os_str() != "target")
850				.collect::<Vec<_>>()
851				.iter()
852				.rev()
853				.take_while(|c| c.as_os_str() != build_subdirectory(target))
854				.last()
855				.expect("We put the runtime project within a `target/.../[rw]build` path; qed")
856				.as_os_str()
857				.to_str()
858				.expect("All our profile directory names are ascii; qed")
859				.to_string();
860			(name, false)
861		};
862		let outer_build_profile = Profile::iter().find(|p| p.directory() == name);
863		let blob_build_profile = match (outer_build_profile.clone(), overridden) {
864			// When not overridden by a env variable we default to using the `Release` profile
865			// for the wasm build even when the main build uses the debug build. This
866			// is because the `Debug` profile is too slow for normal development activities.
867			(Some(Profile::Debug), false) => Profile::Release,
868			// For any other profile or when overridden we take it at face value.
869			(Some(profile), _) => profile,
870			// For non overridden unknown profiles we fall back to `Release`.
871			// This allows us to continue building when a custom profile is used for the
872			// main builds cargo. When explicitly passing a profile via env variable we are
873			// not doing a fallback.
874			(None, false) => {
875				let profile = Profile::Release;
876				build_helper::warning!(
877					"Unknown cargo profile `{name}`. Defaulted to `{profile:?}` for the runtime build.",
878				);
879				profile
880			},
881			// Invalid profile specified.
882			(None, true) => {
883				// We use println! + exit instead of a panic in order to have a cleaner output.
884				println!(
885					"Unexpected profile name: `{name}`. One of the following is expected: {:?}",
886					Profile::iter().map(|p| p.directory()).collect::<Vec<_>>(),
887				);
888				process::exit(1);
889			},
890		};
891		BuildConfiguration {
892			outer_build_profile: outer_build_profile.unwrap_or(Profile::Release),
893			blob_build_profile,
894		}
895	}
896}

903/// Build the project and create the bloaty runtime blob.
904///
905/// Returns the path to the generated bloaty runtime blob.
906fn build_bloaty_blob(
907	target: RuntimeTarget,
908	blob_build_profile: &Profile,
909	project: &Path,
910	default_rustflags: &str,
911	cargo_cmd: CargoCommandVersioned,
912	#[cfg(feature = "metadata-hash")] metadata_hash: Option<[u8; 32]>,
913) -> PathBuf {
914	let manifest_path = project.join("Cargo.toml");
915	let mut build_cmd = cargo_cmd.command();
916
917	let mut rustflags = String::new();
918	match target {
919		RuntimeTarget::Wasm => {
920			// For Rust >= 1.70 and Rust < 1.84 with `wasm32-unknown-unknown` target,
921			// it's required to disable default WASM features:
922			// - `sign-ext` (since Rust 1.70)
923			// - `multivalue` and `reference-types` (since Rust 1.82)
924			//
925			// For Rust >= 1.84, we use `wasm32v1-none` target
926			// (disables all "post-MVP" WASM features except `mutable-globals`):
927			// - https://doc.rust-lang.org/beta/rustc/platform-support/wasm32v1-none.html
928			//
929			// Also see:
930			// https://blog.rust-lang.org/2024/09/24/webassembly-targets-change-in-default-target-features.html#disabling-on-by-default-webassembly-proposals
931
932			if !cargo_cmd.is_wasm32v1_none_target_available() {
933				rustflags.push_str("-C target-cpu=mvp ");
934			}
935
936			rustflags.push_str("-C link-arg=--export-table ");
937		},
938		RuntimeTarget::Riscv => (),
939	}
940
941	rustflags.push_str(default_rustflags);
942	rustflags.push_str(" --cfg substrate_runtime ");
943	rustflags.push_str(&env::var(crate::WASM_BUILD_RUSTFLAGS_ENV).unwrap_or_default());
944
945	build_cmd
946		.arg("rustc")
947		.arg(format!("--target={}", target.rustc_target(&cargo_cmd)))
948		.arg(format!("--manifest-path={}", manifest_path.display()))
949		.env("RUSTFLAGS", rustflags)
950		// Manually set the `CARGO_TARGET_DIR` to prevent a cargo deadlock (cargo locks a target dir
951		// exclusive). The runner project is created in `CARGO_TARGET_DIR` and executing it will
952		// create a sub target directory inside of `CARGO_TARGET_DIR`.
953		.env("CARGO_TARGET_DIR", &project.join("target").display().to_string())
954		// As we are being called inside a build-script, this env variable is set. However, we set
955		// our own `RUSTFLAGS` and thus, we need to remove this. Otherwise cargo favors this
956		// env variable.
957		.env_remove("CARGO_ENCODED_RUSTFLAGS")
958		// Make sure if we're called from within a `build.rs` the host toolchain won't override a
959		// rustup toolchain we've picked.
960		.env_remove("RUSTC")
961		// We don't want to call ourselves recursively
962		.env(crate::SKIP_BUILD_ENV, "");
963
964	let cargo_args = env::var(crate::WASM_BUILD_CARGO_ARGS).unwrap_or_default();
965	if !cargo_args.is_empty() {
966		let Some(args) = shlex::split(&cargo_args) else {
967			build_helper::warning(format!(
968				"the {} environment variable is not a valid shell string",
969				crate::WASM_BUILD_CARGO_ARGS
970			));
971			std::process::exit(1);
972		};
973		build_cmd.args(args);
974	}
975
976	#[cfg(feature = "metadata-hash")]
977	if let Some(hash) = metadata_hash {
978		build_cmd.env("RUNTIME_METADATA_HASH", array_bytes::bytes2hex("0x", &hash));
979	}
980
981	if super::color_output_enabled() {
982		build_cmd.arg("--color=always");
983	}
984
985	build_cmd.arg("--profile");
986	build_cmd.arg(blob_build_profile.name());
987
988	if offline_build() {
989		build_cmd.arg("--offline");
990	}
991
992	// For Rust >= 1.70 and Rust < 1.84 with `wasm32-unknown-unknown` target,
993	// it's required to disable default WASM features:
994	// - `sign-ext` (since Rust 1.70)
995	// - `multivalue` and `reference-types` (since Rust 1.82)
996	//
997	// For Rust >= 1.84, we use `wasm32v1-none` target
998	// (disables all "post-MVP" WASM features except `mutable-globals`):
999	// - https://doc.rust-lang.org/beta/rustc/platform-support/wasm32v1-none.html
1000	//
1001	// Our executor currently only supports the WASM MVP feature set, however nowadays
1002	// when compiling WASM the Rust compiler has more features enabled by default.
1003	//
1004	// We do set the `-C target-cpu=mvp` flag to make sure that *our* code gets compiled
1005	// in a way that is compatible with our executor, however this doesn't affect Rust's
1006	// standard library crates (`std`, `core` and `alloc`) which are by default precompiled
1007	// and still can make use of these extra features.
1008	//
1009	// So here we force the compiler to also compile the standard library crates for us
1010	// to make sure that they also only use the MVP features.
1011	//
1012	// So the `-Zbuild-std` and `RUSTC_BOOTSTRAP=1` hacks are only used for Rust < 1.84.
1013	//
1014	// Also see:
1015	// https://blog.rust-lang.org/2024/09/24/webassembly-targets-change-in-default-target-features.html#disabling-on-by-default-webassembly-proposals
1016	if let Some(arg) = target.rustc_target_build_std(&cargo_cmd) {
1017		build_cmd.arg("-Z").arg(arg);
1018
1019		if !cargo_cmd.supports_nightly_features() {
1020			build_cmd.env("RUSTC_BOOTSTRAP", "1");
1021		}
1022	}
1023
1024	// Inherit jobserver in child cargo command to ensure we don't try to use more concurrency than
1025	// available
1026	if let Some(c) = get_jobserver() {
1027		c.configure(&mut build_cmd);
1028	}
1029
1030	println!("{}", colorize_info_message("Information that should be included in a bug report."));
1031	println!("{} {:?}", colorize_info_message("Executing build command:"), build_cmd);
1032	println!("{} {}", colorize_info_message("Using rustc version:"), cargo_cmd.rustc_version());
1033
1034	// Use `process::exit(1)` to have a clean error output.
1035	if !build_cmd.status().map_or(false, |s| s.success()) {
1036		process::exit(1);
1037	}
1038
1039	let blob_name = get_blob_name(target, &manifest_path);
1040	let target_directory = project
1041		.join("target")
1042		.join(target.rustc_target_dir(&cargo_cmd))
1043		.join(blob_build_profile.directory());
1044	match target {
1045		RuntimeTarget::Riscv => {
1046			let elf_path = target_directory.join(&blob_name);
1047			let elf_metadata = match elf_path.metadata() {
1048				Ok(path) => path,
1049				Err(error) =>
1050					panic!("internal error: couldn't read the metadata of {elf_path:?}: {error}"),
1051			};
1052
1053			let polkavm_path = target_directory.join(format!("{}.polkavm", blob_name));
1054			if polkavm_path
1055				.metadata()
1056				.map(|polkavm_metadata| {
1057					polkavm_metadata.modified().unwrap() < elf_metadata.modified().unwrap()
1058				})
1059				.unwrap_or(true)
1060			{
1061				let blob_bytes =
1062					std::fs::read(elf_path).expect("binary always exists after its built");
1063
1064				let mut config = polkavm_linker::Config::default();
1065				config.set_strip(true); // TODO: This shouldn't always be done.
1066
1067				let program = match polkavm_linker::program_from_elf(
1068					config,
1069					TargetInstructionSet::Latest,
1070					&blob_bytes,
1071				) {
1072					Ok(program) => program,
1073					Err(error) => {
1074						println!("Failed to link the runtime blob; this is probably a bug!");
1075						println!("Linking error: {error}");
1076						process::exit(1);
1077					},
1078				};
1079
1080				std::fs::write(&polkavm_path, program)
1081					.expect("writing the blob to a file always works");
1082			}
1083
1084			polkavm_path
1085		},
1086		RuntimeTarget::Wasm => target_directory.join(format!("{}.wasm", blob_name)),
1087	}
1088}

1090fn compact_wasm(blob_paths: &BlobPaths, bloaty_binary: &WasmBinaryBloaty) -> Option<WasmBinary> {
1091	let wasm_compact_path = blob_paths.compact();
1092	let start = std::time::Instant::now();
1093	wasm_opt::OptimizationOptions::new_opt_level_0()
1094		.mvp_features_only()
1095		.debug_info(true)
1096		.add_pass(wasm_opt::Pass::StripDwarf)
1097		.add_pass(wasm_opt::Pass::SignextLowering)
1098		.run(bloaty_binary.bloaty_path(), &wasm_compact_path)
1099		.expect("Failed to compact generated WASM binary.");
1100
1101	println!(
1102		"{} {}",
1103		colorize_info_message("Compacted wasm in"),
1104		colorize_info_message(format!("{:?}", start.elapsed()).as_str())
1105	);
1106
1107	Some(WasmBinary(wasm_compact_path))
1108}

1110fn try_compress_blob(blob_paths: &BlobPaths, compact_blob: &WasmBinary) -> Option<WasmBinary> {
1111	use sp_maybe_compressed_blob::CODE_BLOB_BOMB_LIMIT;
1112
1113	let compact_compressed_blob_path = blob_paths.compact_compressed();
1114
1115	let start = std::time::Instant::now();
1116	let data = fs::read(compact_blob.wasm_binary_path()).expect("Failed to read WASM binary");
1117	if let Some(compressed) =
1118		sp_maybe_compressed_blob::compress_strongly(&data, CODE_BLOB_BOMB_LIMIT)
1119	{
1120		fs::write(&compact_compressed_blob_path, &compressed[..])
1121			.expect("Failed to write WASM binary");
1122
1123		println!(
1124			"{} {}",
1125			colorize_info_message("Compressed blob in"),
1126			colorize_info_message(format!("{:?}", start.elapsed()).as_str())
1127		);
1128		Some(WasmBinary(compact_compressed_blob_path))
1129	} else {
1130		build_helper::warning!(
1131			"Writing uncompressed blob. Exceeded maximum size {}",
1132			CODE_BLOB_BOMB_LIMIT,
1133		);
1134		println!("{}", colorize_info_message("Skipping blob compression"));
1135		None
1136	}
1137}

1139/// Custom wrapper for a [`cargo_metadata::Package`] to store it in
1140/// a `HashSet`.

1144	identifier: String,
1145}

1150			package,
1151			identifier: format!("{}{}{:?}", package.name, package.version, package.source),
1152		}

1188/// Generate the `rerun-if-changed` instructions for cargo to make sure that the WASM binary is
1189/// rebuilt when needed.
1190fn generate_rerun_if_changed_instructions(
1191	cargo_manifest: &Path,
1192	project_folder: &Path,
1193	wasm_workspace: &Path,
1194	compressed_or_compact_wasm: Option<&WasmBinary>,
1195	bloaty_wasm: &WasmBinaryBloaty,
1196) {
1197	// Rerun `build.rs` if the `Cargo.lock` changes
1198	if let Some(cargo_lock) = find_cargo_lock(cargo_manifest) {
1199		rerun_if_changed(cargo_lock);
1200	}
1201
1202	let metadata = create_metadata_command(project_folder.join("Cargo.toml"))
1203		.exec()
1204		.expect("`cargo metadata` can not fail!");
1205
1206	let package = metadata
1207		.packages
1208		.iter()
1209		.find(|p| p.manifest_path == cargo_manifest)
1210		.expect("The crate package is contained in its own metadata; qed");
1211
1212	// Start with the dependencies of the crate we want to compile for wasm.
1213	let mut dependencies = package.dependencies.iter().collect::<Vec<_>>();
1214
1215	// Collect all packages by follow the dependencies of all packages we find.
1216	let mut packages = HashSet::new();
1217	packages.insert(DeduplicatePackage::from(package));
1218
1219	while let Some(dependency) = dependencies.pop() {
1220		// Ignore all dev dependencies
1221		if dependency.kind == DependencyKind::Development {
1222			continue
1223		}
1224
1225		let path_or_git_dep =
1226			dependency.source.as_ref().map(|s| s.starts_with("git+")).unwrap_or(true);
1227
1228		let package = metadata
1229			.packages
1230			.iter()
1231			.filter(|p| !p.manifest_path.starts_with(wasm_workspace))
1232			.find(|p| {
1233				// Check that the name matches and that the version matches or this is
1234				// a git or path dep. A git or path dependency can only occur once, so we don't
1235				// need to check the version.
1236				(path_or_git_dep || dependency.req.matches(&p.version)) && dependency.name == p.name
1237			});
1238
1239		if let Some(package) = package {
1240			if packages.insert(DeduplicatePackage::from(package)) {
1241				dependencies.extend(package.dependencies.iter());
1242			}
1243		}
1244	}
1245
1246	// Make sure that if any file/folder of a dependency change, we need to rerun the `build.rs`
1247	packages.iter().for_each(package_rerun_if_changed);
1248
1249	compressed_or_compact_wasm.map(|w| rerun_if_changed(w.wasm_binary_path()));
1250	rerun_if_changed(bloaty_wasm.bloaty_path());
1251
1252	// Register our env variables
1253	println!("cargo:rerun-if-env-changed={}", crate::SKIP_BUILD_ENV);
1254	println!("cargo:rerun-if-env-changed={}", crate::WASM_BUILD_TYPE_ENV);
1255	println!("cargo:rerun-if-env-changed={}", crate::WASM_BUILD_RUSTFLAGS_ENV);
1256	println!("cargo:rerun-if-env-changed={}", crate::WASM_TARGET_DIRECTORY);
1257	println!("cargo:rerun-if-env-changed={}", crate::WASM_BUILD_TOOLCHAIN);
1258	println!("cargo:rerun-if-env-changed={}", crate::WASM_BUILD_STD);
1259	println!("cargo:rerun-if-env-changed={}", crate::RUNTIME_TARGET);
1260	println!("cargo:rerun-if-env-changed={}", crate::WASM_BUILD_CARGO_ARGS);
1261}

1263/// Track files and paths related to the given package to rerun `build.rs` on any relevant change.
1264fn package_rerun_if_changed(package: &DeduplicatePackage) {
1265	let mut manifest_path = package.manifest_path.clone();
1266	if manifest_path.ends_with("Cargo.toml") {
1267		manifest_path.pop();
1268	}
1269
1270	WalkDir::new(&manifest_path)
1271		.into_iter()
1272		.filter_entry(|p| {
1273			// Ignore this entry if it is a directory that contains a `Cargo.toml` that is not the
1274			// `Cargo.toml` related to the current package. This is done to ignore sub-crates of a
1275			// crate. If such a sub-crate is a dependency, it will be processed independently
1276			// anyway.
1277			p.path() == manifest_path || !p.path().is_dir() || !p.path().join("Cargo.toml").exists()
1278		})
1279		.filter_map(|p| p.ok().map(|p| p.into_path()))
1280		.filter(|p| p.extension().map(|e| e == "rs" || e == "toml").unwrap_or_default())
1281		.for_each(rerun_if_changed);
1282}

1284/// Copy the blob binary to the target directory set in `WASM_TARGET_DIRECTORY` environment
1285/// variable. If the variable is not set, this is a no-op.
1286fn copy_blob_to_target_directory(cargo_manifest: &Path, blob_binary: &WasmBinary) {
1287	let target_dir = match env::var(crate::WASM_TARGET_DIRECTORY) {
1288		Ok(path) => PathBuf::from(path),
1289		Err(_) => return,
1290	};
1291
1292	if !target_dir.is_absolute() {
1293		// We use println! + exit instead of a panic in order to have a cleaner output.
1294		println!(
1295			"Environment variable `{}` with `{}` is not an absolute path!",
1296			crate::WASM_TARGET_DIRECTORY,
1297			target_dir.display(),
1298		);
1299		process::exit(1);
1300	}
1301
1302	fs::create_dir_all(&target_dir).expect("Creates `WASM_TARGET_DIRECTORY`.");
1303
1304	fs::copy(
1305		blob_binary.wasm_binary_path(),
1306		target_dir.join(format!("{}.wasm", get_blob_name(RuntimeTarget::Wasm, cargo_manifest))),
1307	)
1308	.expect("Copies blob binary to `WASM_TARGET_DIRECTORY`.");
1309}

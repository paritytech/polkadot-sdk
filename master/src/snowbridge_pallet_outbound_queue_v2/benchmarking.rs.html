25	/// Build `Upgrade` message with `MaxMessagePayloadSize`, in the worst-case.
26	fn build_message<T: Config>() -> (Message, OutboundMessage) {
27		let commands = vec![Command::Upgrade {
28			impl_address: H160::zero(),
29			impl_code_hash: H256::zero(),
30			initializer: Initializer {
31				params: core::iter::repeat_with(|| 1_u8)
32					.take(<T as Config>::MaxMessagePayloadSize::get() as usize)
33					.collect(),
34				maximum_required_gas: 200_000,
35			},
36		}];
37		let message = Message {
38			origin: Default::default(),
39			id: H256::default(),
40			fee: 0,
41			commands: BoundedVec::try_from(commands.clone()).unwrap(),
42		};
43		let wrapped_commands: Vec<OutboundCommandWrapper> = commands
44			.into_iter()
45			.map(|command| OutboundCommandWrapper {
46				kind: command.index(),
47				gas: T::GasMeter::maximum_dispatch_gas_used_at_most(&command),
48				payload: command.abi_encode(),
49			})
50			.collect();
51		let outbound_message = OutboundMessage {
52			origin: Default::default(),
53			nonce: 1,
54			topic: H256::default(),
55			commands: wrapped_commands.clone().try_into().unwrap(),
56		};
57		(message, outbound_message)
58	}

60	/// Initialize `MaxMessagesPerBlock` messages need to be committed, in the worst-case.
61	fn initialize_worst_case<T: Config>() {
62		for _ in 0..T::MaxMessagesPerBlock::get() {
63			initialize_with_one_message::<T>();
64		}
65	}
66
67	/// Initialize with a single message
68	fn initialize_with_one_message<T: Config>() {
69		let (message, outbound_message) = build_message::<T>();
70		let leaf = <T as Config>::Hashing::hash(&message.encode());
71		MessageLeaves::<T>::append(leaf);
72		Messages::<T>::append(outbound_message);
73	}
74
75	/// Benchmark for processing a message.
76	#[benchmark]
77	fn do_process_message() -> Result<(), BenchmarkError> {
78		let (enqueued_message, _) = build_message::<T>();
79		let origin = T::AggregateMessageOrigin::from([1; 32].into());
80		let message = enqueued_message.encode();
81
82		#[block]
83		{
84			let _ = OutboundQueue::<T>::do_process_message(origin, &message).unwrap();
85		}
86
87		assert_eq!(MessageLeaves::<T>::decode_len().unwrap(), 1);
88
89		Ok(())
90	}
91
92	/// Benchmark for producing final messages commitment, in the worst-case
93	#[benchmark]
94	fn commit() -> Result<(), BenchmarkError> {
95		initialize_worst_case::<T>();
96
97		#[block]
98		{
99			OutboundQueue::<T>::commit();
100		}
101
102		Ok(())
103	}

105	/// Benchmark for producing commitment for a single message, used to estimate the delivery

131	/// Benchmark the entire process flow in the worst-case. This can be used to determine
132	/// appropriate values for the configuration parameters `MaxMessagesPerBlock` and
133	/// `MaxMessagePayloadSize`
134	#[benchmark]
135	fn process() -> Result<(), BenchmarkError> {
136		initialize_worst_case::<T>();
137		let origin = T::AggregateMessageOrigin::from([1; 32].into());
138		let (enqueued_message, _) = build_message::<T>();
139		let message = enqueued_message.encode();
140
141		#[block]
142		{
143			OutboundQueue::<T>::on_initialize(1_u32.into());
144			for _ in 0..T::MaxMessagesPerBlock::get() {
145				OutboundQueue::<T>::do_process_message(origin.clone(), &message).unwrap();
146			}
147			OutboundQueue::<T>::commit();
148		}
149
150		Ok(())
151	}
152
153	#[benchmark]
154	fn submit_delivery_receipt() -> Result<(), BenchmarkError> {
155		let caller: T::AccountId = whitelisted_caller();
156
157		let message = make_submit_delivery_receipt_message();
158
159		T::Helper::initialize_storage(message.finalized_header, message.block_roots_root);
160
161		let receipt = DeliveryReceipt::try_from(&message.event.event_log).unwrap();
162
163		let order = PendingOrder {
164			nonce: receipt.nonce,
165			fee: 0,
166			block_number: frame_system::Pallet::<T>::current_block_number(),
167		};
168		<PendingOrders<T>>::insert(receipt.nonce, order);
169
170		#[block]
171		{
172			assert_ok!(OutboundQueue::<T>::submit_delivery_receipt(
173				RawOrigin::Signed(caller.clone()).into(),
174				Box::new(message.event),
175			));
176		}
177
178		Ok(())
179	}

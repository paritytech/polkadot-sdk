17use crate::AssetsInHolding;
18use core::result::Result;
19use xcm::latest::{prelude::*, Weight};
20
21/// Determine the weight of an XCM message.
22pub trait WeightBounds<RuntimeCall> {
23	/// Return the maximum amount of weight that an attempted execution of this message could
24	/// consume.
25	fn weight(
26		message: &mut Xcm<RuntimeCall>,
27		weight_limit: Weight,
28	) -> Result<Weight, InstructionError>;
29
30	/// Return the maximum amount of weight that an attempted execution of this instruction could
31	/// consume.
32	fn instr_weight(instruction: &mut Instruction<RuntimeCall>) -> Result<Weight, XcmError>;
33}
34
35/// Charge for weight in order to execute XCM.
36///
37/// A `WeightTrader` may also be put into a tuple, in which case the default behavior of
38/// `buy_weight` and `refund_weight` would be to attempt to call each tuple element's own
39/// implementation of these two functions, in the order of which they appear in the tuple,
40/// returning early when a successful result is returned.
41pub trait WeightTrader: Sized {
42	/// Create a new trader instance.
43	fn new() -> Self;
44
45	/// Purchase execution weight credit in return for up to a given `payment`. If less of the
46	/// payment is required then the surplus is returned. If the `payment` cannot be used to pay
47	/// for the `weight`, then an error is returned.
48	fn buy_weight(
49		&mut self,
50		weight: Weight,
51		payment: AssetsInHolding,
52		context: &XcmContext,
53	) -> Result<AssetsInHolding, XcmError>;
54
55	/// Attempt a refund of `weight` into some asset. The caller does not guarantee that the weight
56	/// was purchased using `buy_weight`.
57	///
58	/// Default implementation refunds nothing.
59	fn refund_weight(&mut self, _weight: Weight, _context: &XcmContext) -> Option<Asset> {
60		None
61	}
62}
63
64#[impl_trait_for_tuples::impl_for_tuples(30)]
65impl WeightTrader for Tuple {
66	fn new() -> Self {
67		for_tuples!( ( #( Tuple::new() ),* ) )
68	}
69
70	fn buy_weight(
71		&mut self,
72		weight: Weight,
73		payment: AssetsInHolding,
74		context: &XcmContext,
75	) -> Result<AssetsInHolding, XcmError> {
76		let mut too_expensive_error_found = false;
77		let mut last_error = None;
78		for_tuples!( #(
79			let weight_trader = core::any::type_name::<Tuple>();
80
81			match Tuple.buy_weight(weight, payment.clone(), context) {
82				Ok(assets) => {
83					tracing::trace!(
84						target: "xcm::buy_weight",
85						%weight_trader,
86						"Buy weight succeeded",
87					);
88
89					return Ok(assets)
90				},
91				Err(error) => {
92					if let XcmError::TooExpensive = error {
93						too_expensive_error_found = true;
94					}
95					last_error = Some(error);
96
97					tracing::trace!(
98						target: "xcm::buy_weight",
99						?error,
100						%weight_trader,
101						"Weight trader failed",
102					);
103				}
104			}
105		)* );
106
107		tracing::trace!(
108			target: "xcm::buy_weight",
109			"Buy weight failed",
110		);
111
112		// if we have multiple traders, and first one returns `TooExpensive` and others fail e.g.
113		// `AssetNotFound` then it is more accurate to return `TooExpensive` then `AssetNotFound`
114		Err(if too_expensive_error_found {
115			XcmError::TooExpensive
116		} else {
117			last_error.unwrap_or(XcmError::TooExpensive)
118		})
119	}
120
121	fn refund_weight(&mut self, weight: Weight, context: &XcmContext) -> Option<Asset> {
122		for_tuples!( #(
123			if let Some(asset) = Tuple.refund_weight(weight, context) {
124				return Some(asset);
125			}
126		)* );
127		None
128	}
129}

17use crate::traits::TransactAsset;
18use frame_support::traits::ContainsPair;
19use scale_info::TypeInfo;
20use sp_runtime::codec::{Decode, DecodeWithMemTracking, Encode};
21use xcm::prelude::*;
22
23/// Errors related to determining asset transfer support.
24#[derive(Copy, Clone, Encode, Decode, Eq, PartialEq, Debug, TypeInfo)]
25pub enum Error {
26	/// Reserve chain could not be determined for assets.
27	UnknownReserve,
28}
29
30/// Specify which type of asset transfer is required for a particular `(asset, dest)` combination.
31#[derive(Clone, Encode, Decode, DecodeWithMemTracking, PartialEq, Debug, TypeInfo)]
32pub enum TransferType {
33	/// should teleport `asset` to `dest`
34	Teleport,
35	/// should reserve-transfer `asset` to `dest`, using local chain as reserve
36	LocalReserve,
37	/// should reserve-transfer `asset` to `dest`, using `dest` as reserve
38	DestinationReserve,
39	/// should reserve-transfer `asset` to `dest`, using remote chain `Location` as reserve
40	RemoteReserve(VersionedLocation),
41}
42
43/// A trait for identifying asset transfer type based on `IsTeleporter` and `IsReserve`
44/// configurations.
45pub trait XcmAssetTransfers {
46	/// Combinations of (Asset, Location) pairs which we trust as reserves. Meaning
47	/// reserve-based-transfers are to be used for assets matching this filter.
48	type IsReserve: ContainsPair<Asset, Location>;
49
50	/// Combinations of (Asset, Location) pairs which we trust as teleporters. Meaning teleports are
51	/// to be used for assets matching this filter.
52	type IsTeleporter: ContainsPair<Asset, Location>;
53
54	/// How to withdraw and deposit an asset.
55	type AssetTransactor: TransactAsset;
56
57	/// Determine transfer type to be used for transferring `asset` from local chain to `dest`.
58	fn determine_for(asset: &Asset, dest: &Location) -> Result<TransferType, Error> {
59		if Self::IsTeleporter::contains(asset, dest) {
60			// we trust destination for teleporting asset
61			return Ok(TransferType::Teleport)
62		} else if Self::IsReserve::contains(asset, dest) {
63			// we trust destination as asset reserve location
64			return Ok(TransferType::DestinationReserve)
65		}
66
67		// try to determine reserve location based on asset id/location
68		let asset_location = asset.id.0.chain_location();
69		if asset_location == Location::here() ||
70			Self::IsTeleporter::contains(asset, &asset_location)
71		{
72			// if the asset is local, then it's a local reserve
73			// it's also a local reserve if the asset's location is not `here` but it's a location
74			// where it can be teleported to `here` => local reserve
75			Ok(TransferType::LocalReserve)
76		} else if Self::IsReserve::contains(asset, &asset_location) {
77			// remote location that is recognized as reserve location for asset
78			Ok(TransferType::RemoteReserve(asset_location.into()))
79		} else {
80			// remote location that is not configured either as teleporter or reserve => cannot
81			// determine asset reserve
82			Err(Error::UnknownReserve)
83		}
84	}
85}
86
87impl XcmAssetTransfers for () {
88	type IsReserve = ();
89	type IsTeleporter = ();
90	type AssetTransactor = ();
91	fn determine_for(_: &Asset, _: &Location) -> Result<TransferType, Error> {
92		return Err(Error::UnknownReserve);
93	}
94}

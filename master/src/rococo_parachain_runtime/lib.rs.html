145/// We assume that ~10% of the block weight is consumed by `on_initialize` handlers.
146/// This is used to limit the maximal weight of a single extrinsic.
147const AVERAGE_ON_INITIALIZE_RATIO: Perbill = Perbill::from_percent(10);
148/// We allow `Normal` extrinsics to fill up the block up to 75%, the rest can be used
149/// by  Operational  extrinsics.
150const NORMAL_DISPATCH_RATIO: Perbill = Perbill::from_percent(75);
151/// We allow for 2 seconds of compute with a 6 second average block time.
152const MAXIMUM_BLOCK_WEIGHT: Weight = Weight::from_parts(
153	WEIGHT_REF_TIME_PER_SECOND.saturating_mul(2),
154	cumulus_primitives_core::relay_chain::MAX_POV_SIZE as u64,
155);
156
157/// Maximum number of blocks simultaneously accepted by the Runtime, not yet included
158/// into the relay chain.
159const UNINCLUDED_SEGMENT_CAPACITY: u32 = 3;
160/// How many parachain blocks are processed by the relay chain per parent. Limits the
161/// number of blocks authored per slot.
162const BLOCK_PROCESSING_VELOCITY: u32 = 2;
163/// Relay chain slot duration, in milliseconds.
164const RELAY_CHAIN_SLOT_DURATION_MILLIS: u32 = 6000;

166parameter_types! {
167	pub const BlockHashCount: BlockNumber = 250;
168	pub const Version: RuntimeVersion = VERSION;
169	pub RuntimeBlockLength: BlockLength =
170		BlockLength::max_with_normal_ratio(5 * 1024 * 1024, NORMAL_DISPATCH_RATIO);
171	pub RuntimeBlockWeights: BlockWeights = BlockWeights::builder()
172		.base_block(BlockExecutionWeight::get())
173		.for_class(DispatchClass::all(), |weights| {
174			weights.base_extrinsic = ExtrinsicBaseWeight::get();
175		})
176		.for_class(DispatchClass::Normal, |weights| {
177			weights.max_total = Some(NORMAL_DISPATCH_RATIO * MAXIMUM_BLOCK_WEIGHT);
178		})
179		.for_class(DispatchClass::Operational, |weights| {
180			weights.max_total = Some(MAXIMUM_BLOCK_WEIGHT);
181			// Operational transactions have some extra reserved space, so that they
182			// are included even if block reached `MAXIMUM_BLOCK_WEIGHT`.
183			weights.reserved = Some(
184				MAXIMUM_BLOCK_WEIGHT - NORMAL_DISPATCH_RATIO * MAXIMUM_BLOCK_WEIGHT
185			);
186		})
187		.avg_block_initialization(AVERAGE_ON_INITIALIZE_RATIO)
188		.build_or_panic();
189	pub const SS58Prefix: u8 = 42;
190}

402/// This is the type we use to convert an (incoming) XCM origin into a local `Origin` instance,
403/// ready for dispatching a transaction with Xcm's `Transact`. There is an `OriginKind` which can
404/// biases the kind of local `Origin` it will become.
405pub type XcmOriginToTransactDispatchOrigin = (
406	// Sovereign account converter; this attempts to derive an `AccountId` from the origin location
407	// using `LocationToAccountId` and then turn that into the usual `Signed` origin. Useful for
408	// foreign chains who want to have a local sovereign account on this chain which they control.
409	SovereignSignedViaLocation<LocationToAccountId, RuntimeOrigin>,
410	// Native converter for Relay-chain (Parent) location; will convert to a `Relay` origin when
411	// recognised.
412	RelayChainAsNative<RelayChainOrigin, RuntimeOrigin>,
413	// Native converter for sibling Parachains; will convert to a `SiblingPara` origin when
414	// recognised.
415	SiblingParachainAsNative<cumulus_pallet_xcm::Origin, RuntimeOrigin>,
416	// Superuser converter for the Relay-chain (Parent) location. This will allow it to issue a
417	// transaction from the Root origin.
418	ParentAsSuperuser<RuntimeOrigin>,
419	// Native signed account converter; this just converts an `AccountId32` origin into a normal
420	// `RuntimeOrigin::Signed` origin of the same 32-byte value.
421	SignedAccountId32AsNative<RococoNetwork, RuntimeOrigin>,
422	// Xcm origins can be represented natively under the Xcm pallet's Xcm origin.
423	XcmPassthrough<RuntimeOrigin>,

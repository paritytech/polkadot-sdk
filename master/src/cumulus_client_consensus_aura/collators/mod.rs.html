46// This is an arbitrary value which is guaranteed to exceed the required depth for 500ms blocks
47// built with a relay parent offset of 1. It must be larger than the unincluded segment capacity.
48//
49// The formula we use to compute the capacity of the unincluded segment in the parachain runtime
50// is:
51// UNINCLUDED_SEGMENT_CAPACITY = (2 + RELAY_PARENT_OFFSET) * BLOCK_PROCESSING_VELOCITY + 1.
52//
53// Since we only search for parent blocks which have already been imported,
54// we can guarantee that all imported blocks respect the unincluded segment
55// rules specified by the parachain's runtime and thus will never be too deep. This is just an extra
56// sanity check.
57const PARENT_SEARCH_DEPTH: usize = 40;

59/// Check the `local_validation_code_hash` against the validation code hash in the relay chain
60/// state.
61///
62/// If the code hashes do not match, it prints a warning.
63async fn check_validation_code_or_log(
64	local_validation_code_hash: &ValidationCodeHash,
65	para_id: ParaId,
66	relay_client: &impl RelayChainInterface,
67	relay_parent: RelayHash,
68) {
69	let state_validation_code_hash = match relay_client
70		.validation_code_hash(relay_parent, para_id, OccupiedCoreAssumption::Included)
71		.await
72	{
73		Ok(hash) => hash,
74		Err(error) => {
75			tracing::debug!(
76				target: super::LOG_TARGET,
77				%error,
78				?relay_parent,
79				%para_id,
80				"Failed to fetch validation code hash",
81			);
82			return
83		},
84	};
85
86	match state_validation_code_hash {
87		Some(state) =>
88			if state != *local_validation_code_hash {
89				tracing::warn!(
90					target: super::LOG_TARGET,
91					%para_id,
92					?relay_parent,
93					?local_validation_code_hash,
94					relay_validation_code_hash = ?state,
95					"Parachain code doesn't match validation code stored in the relay chain state.",
96				);
97			},
98		None => {
99			tracing::warn!(
100				target: super::LOG_TARGET,
101				%para_id,
102				?relay_parent,
103				"Could not find validation code for parachain in the relay chain state.",
104			);
105		},
106	}
107}

109/// Fetch scheduling lookahead at given relay parent.
110async fn scheduling_lookahead(
111	relay_parent: RelayHash,
112	relay_client: &impl RelayChainInterface,
113) -> Option<u32> {
114	let runtime_api_version = relay_client
115		.version(relay_parent)
116		.await
117		.map_err(|e| {
118			tracing::error!(
119				target: super::LOG_TARGET,
120				error = ?e,
121				"Failed to fetch relay chain runtime version.",
122			)
123		})
124		.ok()?;
125
126	let parachain_host_runtime_api_version = runtime_api_version
127		.api_version(
128			&<dyn polkadot_primitives::runtime_api::ParachainHost<polkadot_primitives::Block>>::ID,
129		)
130		.unwrap_or_default();
131
132	if parachain_host_runtime_api_version <
133		RuntimeApiRequest::SCHEDULING_LOOKAHEAD_RUNTIME_REQUIREMENT
134	{
135		return None
136	}
137
138	match relay_client.scheduling_lookahead(relay_parent).await {
139		Ok(scheduling_lookahead) => Some(scheduling_lookahead),
140		Err(err) => {
141			tracing::error!(
142				target: crate::LOG_TARGET,
143				?err,
144				?relay_parent,
145				"Failed to fetch scheduling lookahead from relay chain",
146			);
147			None
148		},
149	}
150}

153async fn claim_queue_at(
154	relay_parent: RelayHash,
155	relay_client: &impl RelayChainInterface,
156) -> ClaimQueueSnapshot {
157	// Get `ClaimQueue` from runtime
158	match relay_client.claim_queue(relay_parent).await {
159		Ok(claim_queue) => claim_queue.into(),
160		Err(error) => {
161			tracing::error!(
162				target: crate::LOG_TARGET,
163				?error,
164				?relay_parent,
165				"Failed to query claim queue runtime API",
166			);
167			Default::default()
168		},
169	}
170}

172// Checks if we own the slot at the given block and whether there
173// is space in the unincluded segment.
174async fn can_build_upon<Block: BlockT, Client, P>(
175	para_slot: Slot,
176	relay_slot: Slot,
177	timestamp: Timestamp,
178	parent_hash: Block::Hash,
179	included_block: Block::Hash,
180	client: &Client,
181	keystore: &KeystorePtr,
182) -> Option<SlotClaim<P::Public>>
183where
184	Client: ProvideRuntimeApi<Block>,
185	Client::Api: AuraApi<Block, P::Public> + AuraUnincludedSegmentApi<Block> + ApiExt<Block>,
186	P: Pair,
187	P::Public: Codec,
188	P::Signature: Codec,
189{
190	let runtime_api = client.runtime_api();
191	let authorities = runtime_api.authorities(parent_hash).ok()?;
192	let author_pub = aura_internal::claim_slot::<P>(para_slot, &authorities, keystore).await?;
193
194	// This function is typically called when we want to build block N. At that point, the
195	// unincluded segment in the runtime is unaware of the hash of block N-1. If the unincluded
196	// segment in the runtime is full, but block N-1 is the included block, the unincluded segment
197	// should have length 0 and we can build. Since the hash is not available to the runtime
198	// however, we need this extra check here.
199	if parent_hash == included_block {
200		return Some(SlotClaim::unchecked::<P>(author_pub, para_slot, timestamp));
201	}
202
203	let api_version = runtime_api
204		.api_version::<dyn AuraUnincludedSegmentApi<Block>>(parent_hash)
205		.ok()
206		.flatten()?;
207
208	let slot = if api_version > 1 { relay_slot } else { para_slot };
209
210	runtime_api
211		.can_build_upon(parent_hash, included_block, slot)
212		.ok()?
213		.then(|| SlotClaim::unchecked::<P>(author_pub, para_slot, timestamp))
214}

216/// Use [`cumulus_client_consensus_common::find_potential_parents`] to find parachain blocks that
217/// we can build on. Once a list of potential parents is retrieved, return the last one of the
218/// longest chain.
219async fn find_parent<Block>(
220	relay_parent: RelayHash,
221	para_id: ParaId,
222	para_backend: &impl sc_client_api::Backend<Block>,
223	relay_client: &impl RelayChainInterface,
224) -> Option<(<Block as BlockT>::Header, consensus_common::PotentialParent<Block>)>
225where
226	Block: BlockT,
227{
228	let parent_search_params = ParentSearchParams {
229		relay_parent,
230		para_id,
231		ancestry_lookback: scheduling_lookahead(relay_parent, relay_client)
232			.await
233			.unwrap_or(DEFAULT_SCHEDULING_LOOKAHEAD)
234			.saturating_sub(1) as usize,
235		max_depth: PARENT_SEARCH_DEPTH,
236		ignore_alternative_branches: true,
237	};
238
239	let potential_parents = cumulus_client_consensus_common::find_potential_parents::<Block>(
240		parent_search_params,
241		para_backend,
242		relay_client,
243	)
244	.await;
245
246	let potential_parents = match potential_parents {
247		Err(e) => {
248			tracing::error!(
249				target: crate::LOG_TARGET,
250				?relay_parent,
251				err = ?e,
252				"Could not fetch potential parents to build upon"
253			);
254
255			return None
256		},
257		Ok(x) => x,
258	};
259
260	let included_block = potential_parents.iter().find(|x| x.depth == 0)?.header.clone();
261	potential_parents
262		.into_iter()
263		.max_by_key(|a| a.depth)
264		.map(|parent| (included_block, parent))
265}

287	async fn import_block<I: BlockImport<Block>>(
288		importer: &I,
289		block: Block,
290		origin: BlockOrigin,
291		import_as_best: bool,
292	) {
293		let (header, body) = block.deconstruct();
294
295		let mut block_import_params = BlockImportParams::new(origin, header);
296		block_import_params.fork_choice = Some(ForkChoiceStrategy::Custom(import_as_best));
297		block_import_params.body = Some(body);
298		importer.import_block(block_import_params).await.unwrap();
299	}

335	/// This tests a special scenario where the unincluded segment in the runtime
336	/// is full. We are calling `can_build_upon`, passing the last built block as the
337	/// included one. In the runtime we will not find the hash of the included block in the
338	/// unincluded segment. The `can_build_upon` runtime API would therefore return `false`, but
339	/// we are ensuring on the node side that we are are always able to build on the included block.
340	#[tokio::test]
341	async fn test_can_build_upon() {
342		let (client, keystore) = set_up_components();
343
344		let genesis_hash = client.chain_info().genesis_hash;
345		let mut last_hash = genesis_hash;
346
347		// Fill up the unincluded segment tracker in the runtime.
348		while can_build_upon::<_, _, sp_consensus_aura::sr25519::AuthorityPair>(
349			Slot::from(u64::MAX),
350			Slot::from(u64::MAX),
351			Timestamp::default(),
352			last_hash,
353			genesis_hash,
354			&*client,
355			&keystore,
356		)
357		.await
358		.is_some()
359		{
360			let block = build_and_import_block(&client, genesis_hash).await;
361			last_hash = block.header().hash();
362		}
363
364		// Blocks were built with the genesis hash set as included block.
365		// We call `can_build_upon` with the last built block as the included block.
366		let result = can_build_upon::<_, _, sp_consensus_aura::sr25519::AuthorityPair>(
367			Slot::from(u64::MAX),
368			Slot::from(u64::MAX),
369			Timestamp::default(),
370			last_hash,
371			last_hash,
372			&*client,
373			&keystore,
374		)
375		.await;
376		assert!(result.is_some());
377	}

380/// Holds a relay parent and its descendants.
381pub struct RelayParentData {
382	/// The relay parent block header
383	relay_parent: RelayHeader,
384	/// Ordered collection of descendant block headers, from oldest to newest
385	descendants: Vec<RelayHeader>,
386}
387
388impl RelayParentData {
389	/// Creates a new instance with the given relay parent and no descendants.
390	pub fn new(relay_parent: RelayHeader) -> Self {
391		Self { relay_parent, descendants: Default::default() }
392	}
393
394	/// Creates a new instance with the given relay parent and descendants.
395	pub fn new_with_descendants(relay_parent: RelayHeader, descendants: Vec<RelayHeader>) -> Self {
396		Self { relay_parent, descendants }
397	}
398
399	/// Returns a reference to the relay parent header.
400	pub fn relay_parent(&self) -> &RelayHeader {
401		&self.relay_parent
402	}
403
404	/// Returns the number of descendants.
405	#[cfg(test)]
406	pub fn descendants_len(&self) -> usize {
407		self.descendants.len()
408	}
409
410	/// Consumes the structure and returns a vector containing the relay parent followed by its
411	/// descendants in chronological order. The resulting list should be provided to the parachain
412	/// inherent data.
413	pub fn into_inherent_descendant_list(self) -> Vec<RelayHeader> {
414		let Self { relay_parent, mut descendants } = self;
415
416		if descendants.is_empty() {
417			return Default::default()
418		}
419
420		let mut result = vec![relay_parent];
421		result.append(&mut descendants);
422		result
423	}

66/// Error description of a message send failure.
67#[derive(Eq, PartialEq, Copy, Clone, RuntimeDebug, Encode, Decode)]
68pub enum MessageSendError {
69	/// The dispatch queue is full.
70	QueueFull,
71	/// There does not exist a channel for sending the message.
72	NoChannel,
73	/// The message is too big to ever fit in a channel.
74	TooBig,
75	/// Some other error.
76	Other,
77	/// There are too many channels open at once.
78	TooManyChannels,
79}
80
81impl From<MessageSendError> for &'static str {
82	fn from(e: MessageSendError) -> Self {
83		use MessageSendError::*;
84		match e {
85			QueueFull => "QueueFull",
86			NoChannel => "NoChannel",
87			TooBig => "TooBig",
88			Other => "Other",
89			TooManyChannels => "TooManyChannels",
90		}
91	}
92}

94/// The origin of an inbound message.
95#[derive(
96	Encode, Decode, DecodeWithMemTracking, MaxEncodedLen, Clone, Eq, PartialEq, TypeInfo, Debug,
97)]
98pub enum AggregateMessageOrigin {
99	/// The message came from the para-chain itself.
100	Here,
101	/// The message came from the relay-chain.
102	///
103	/// This is used by the DMP queue.
104	Parent,
105	/// The message came from a sibling para-chain.
106	///
107	/// This is used by the HRMP queue.
108	Sibling(ParaId),
109}
110
111impl From<AggregateMessageOrigin> for Location {
112	fn from(origin: AggregateMessageOrigin) -> Self {
113		match origin {
114			AggregateMessageOrigin::Here => Location::here(),
115			AggregateMessageOrigin::Parent => Location::parent(),
116			AggregateMessageOrigin::Sibling(id) => Location::new(1, Junction::Parachain(id.into())),
117		}
118	}
119}
120
121#[cfg(feature = "runtime-benchmarks")]
122impl From<u32> for AggregateMessageOrigin {
123	fn from(x: u32) -> Self {
124		match x {
125			0 => Self::Here,
126			1 => Self::Parent,
127			p => Self::Sibling(ParaId::from(p)),
128		}
129	}
130}

254/// Consensus header digests for Cumulus parachains.
255#[derive(Clone, Debug, Decode, Encode, PartialEq)]
256pub enum CumulusDigestItem {
257	/// A digest item indicating the relay-parent a parachain block was built against.
258	#[codec(index = 0)]
259	RelayParent(relay_chain::Hash),
260	/// A digest item providing information about the core selected on the relay chain for this
261	/// block.
262	#[codec(index = 1)]
263	CoreInfo(CoreInfo),
264}
265
266impl CumulusDigestItem {
267	/// Encode this as a Substrate [`DigestItem`].
268	pub fn to_digest_item(&self) -> DigestItem {
269		match self {
270			Self::RelayParent(_) => DigestItem::Consensus(CUMULUS_CONSENSUS_ID, self.encode()),
271			Self::CoreInfo(_) => DigestItem::PreRuntime(CUMULUS_CONSENSUS_ID, self.encode()),
272		}
273	}
274
275	/// Find [`CumulusDigestItem::CoreInfo`] in the given `digest`.
276	///
277	/// If there are multiple valid digests, this returns the value of the first one.
278	pub fn find_core_info(digest: &Digest) -> Option<CoreInfo> {
279		digest.convert_first(|d| match d {
280			DigestItem::PreRuntime(id, val) if id == &CUMULUS_CONSENSUS_ID => {
281				let Ok(CumulusDigestItem::CoreInfo(core_info)) =
282					CumulusDigestItem::decode_all(&mut &val[..])
283				else {
284					return None
285				};
286
287				Some(core_info)
288			},
289			_ => None,
290		})
291	}
292
293	/// Returns the found [`CoreInfo`] and iff [`Self::CoreInfo`] exists at max once in the given
294	/// `digest`.
295	pub fn core_info_exists_at_max_once(digest: &Digest) -> CoreInfoExistsAtMaxOnce {
296		let mut core_info = None;
297		if digest
298			.logs()
299			.iter()
300			.filter(|l| match l {
301				DigestItem::PreRuntime(CUMULUS_CONSENSUS_ID, d) => {
302					if let Ok(Self::CoreInfo(ci)) = Self::decode_all(&mut &d[..]) {
303						core_info = Some(ci);
304						true
305					} else {
306						false
307					}
308				},
309				_ => false,
310			})
311			.count() <= 1
312		{
313			core_info
314				.map(CoreInfoExistsAtMaxOnce::Once)
315				.unwrap_or(CoreInfoExistsAtMaxOnce::NotFound)
316		} else {
317			CoreInfoExistsAtMaxOnce::MoreThanOnce
318		}
319	}

321	/// Returns the [`RelayBlockIdentifier`] from the given `digest`.
322	///
323	/// The identifier corresponds to the relay parent used to build the parachain block.
324	pub fn find_relay_block_identifier(digest: &Digest) -> Option<RelayBlockIdentifier> {
325		digest.convert_first(|d| match d {
326			DigestItem::Consensus(id, val) if id == &CUMULUS_CONSENSUS_ID => {
327				let Ok(CumulusDigestItem::RelayParent(hash)) =
328					CumulusDigestItem::decode_all(&mut &val[..])
329				else {
330					return None
331				};
332
333				Some(RelayBlockIdentifier::ByHash(hash))
334			},
335			DigestItem::Consensus(id, val) if id == &rpsr_digest::RPSR_CONSENSUS_ID => {
336				let Ok((storage_root, block_number)) =
337					rpsr_digest::RpsrType::decode_all(&mut &val[..])
338				else {
339					return None
340				};
341
342				Some(RelayBlockIdentifier::ByStorageRoot {
343					storage_root,
344					block_number: block_number.into(),
345				})
346			},
347			_ => None,
348		})
349	}
350}

352///
353/// If there are multiple valid digests, this returns the value of the first one, although
354/// well-behaving runtimes should not produce headers with more than one.
355pub fn extract_relay_parent(digest: &Digest) -> Option<relay_chain::Hash> {
356	digest.convert_first(|d| match d {
357		DigestItem::Consensus(id, val) if id == &CUMULUS_CONSENSUS_ID =>
358			match CumulusDigestItem::decode(&mut &val[..]) {

362		_ => None,

366/// Utilities for handling the relay-parent storage root as a digest item.
367///
368/// This is not intended to be part of the public API, as it is a workaround for
369/// <https://github.com/paritytech/cumulus/issues/303> via
370/// <https://github.com/paritytech/polkadot/issues/7191>.
371///
372/// Runtimes using the parachain-system pallet are expected to produce this digest item,
373/// but will stop as soon as they are able to provide the relay-parent hash directly.
374///
375/// The relay-chain storage root is, in practice, a unique identifier of a block
376/// in the absence of equivocations (which are slashable). This assumes that the relay chain
377/// uses BABE or SASSAFRAS, because the slot and the author's VRF randomness are both included
378/// in the relay-chain storage root in both cases.
379///
380/// Therefore, the relay-parent storage root is a suitable identifier of unique relay chain
381/// blocks in low-value scenarios such as performance optimizations.
382#[doc(hidden)]
383pub mod rpsr_digest {
384	use super::{relay_chain, ConsensusEngineId, DecodeAll, Digest, DigestItem, Encode};
385	use codec::Compact;
386
387	/// The type used to store the relay-parent storage root and number.
388	pub type RpsrType = (relay_chain::Hash, Compact<relay_chain::BlockNumber>);
389
390	/// A consensus engine ID for relay-parent storage root digests.
391	pub const RPSR_CONSENSUS_ID: ConsensusEngineId = *b"RPSR";
392
393	/// Construct a digest item for relay-parent storage roots.
394	pub fn relay_parent_storage_root_item(
395		storage_root: relay_chain::Hash,
396		number: impl Into<Compact<relay_chain::BlockNumber>>,
397	) -> DigestItem {
398		DigestItem::Consensus(
399			RPSR_CONSENSUS_ID,
400			RpsrType::from((storage_root, number.into())).encode(),
401		)
402	}
403
404	/// Extract the relay-parent storage root and number from the provided header digest. Returns
405	/// `None` if none were found.
406	pub fn extract_relay_parent_storage_root(
407		digest: &Digest,
408	) -> Option<(relay_chain::Hash, relay_chain::BlockNumber)> {
409		digest.convert_first(|d| match d {
410			DigestItem::Consensus(id, val) if id == &RPSR_CONSENSUS_ID => {
411				let (h, n) = RpsrType::decode_all(&mut &val[..]).ok()?;
412
413				Some((h, n.0))
414			},
415			_ => None,
416		})
417	}
418}

470/// The schedule for the next relay chain slot.
471///
472/// Returns the maximum number of parachain blocks to produce and the block time per block to use.
473#[derive(Clone, Debug, codec::Decode, codec::Encode, PartialEq, TypeInfo)]
474pub struct NextSlotSchedule {
475	/// The maximum number of blocks to produce in the relay chain slot.
476	///
477	/// The node is free to produce less blocks.
478	pub number_of_blocks: u32,
479	/// The target block time in wall clock time for each block.
480	///
481	/// The maximum should be [`REF_TIME_PER_CORE_IN_SECS`] or otherwise blocks may fail to
482	/// validate on the relay chain.
483	pub block_time: Duration,
484}
485
486impl NextSlotSchedule {
487	/// Creates a schedule that produces one block, occupying an entire core.
488	pub fn one_block_using_one_core() -> Self {
489		Self { number_of_blocks: 1, block_time: Duration::from_secs(REF_TIME_PER_CORE_IN_SECS) }
490	}
491
492	/// A schedule that maps `x` blocks onto `y` cores.
493	pub fn x_blocks_using_y_cores(blocks: u32, cores: u32) -> Self {
494		let ref_time_per_core = Duration::from_secs(REF_TIME_PER_CORE_IN_SECS);
495
496		if blocks == 0 || cores == 0 {
497			return Self { number_of_blocks: 0, block_time: Duration::from_secs(0) }
498		}
499
500		// In wall clock time we can not go above `6s` (relay chain slot duration), so we need to
501		// cap there.
502		let block_time = (ref_time_per_core * cores).min(Duration::from_secs(6)) / blocks;
503		// One block can at max occupy one core.
504		let block_time = block_time.min(ref_time_per_core);
505
506		Self { block_time, number_of_blocks: blocks }
507	}
508}

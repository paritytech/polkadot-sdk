17//! Vote import logic.
18//!
19//! This module encapsulates the actual logic for importing new votes and provides easy access of
20//! the current state for votes for a particular candidate.
21//!
22//! In particular there is `CandidateVoteState` which tells what can be concluded for a particular
23//! set of votes. E.g. whether a dispute is ongoing, whether it is confirmed, concluded, ..
24//!
25//! Then there is `ImportResult` which reveals information about what changed once additional votes
26//! got imported on top of an existing `CandidateVoteState` and reveals "dynamic" information, like
27//! whether due to the import a dispute was raised/got confirmed, ...
28

44/// (Session) environment of a candidate.
45pub struct CandidateEnvironment<'a> {
46	/// The session the candidate appeared in.
47	session_index: SessionIndex,
48	/// Session for above index.
49	session: &'a SessionInfo,
50	/// Executor parameters for the session.
51	executor_params: &'a ExecutorParams,
52	/// Validator indices controlled by this node.
53	controlled_indices: HashSet<ValidatorIndex>,
54	/// Indices of on-chain disabled validators at the `relay_parent` combined
55	/// with the off-chain state.
56	disabled_indices: HashSet<ValidatorIndex>,
57}
58
59#[overseer::contextbounds(DisputeCoordinator, prefix = self::overseer)]
60impl<'a> CandidateEnvironment<'a> {
61	/// Create `CandidateEnvironment`.
62	///
63	/// Return: `None` in case session is outside of session window.
64	pub async fn new<Context>(
65		ctx: &mut Context,
66		runtime_info: &'a mut RuntimeInfo,
67		session_index: SessionIndex,
68		relay_parent: Hash,
69		disabled_offchain: impl IntoIterator<Item = ValidatorIndex>,
70		controlled_indices: &mut ControlledValidatorIndices,
71	) -> Option<CandidateEnvironment<'a>> {
72		let disabled_onchain = runtime_info
73			.get_disabled_validators(ctx.sender(), relay_parent)
74			.await
75			.unwrap_or_else(|err| {
76				gum::info!(target: LOG_TARGET, ?err, "Failed to get disabled validators");
77				Vec::new()
78			});
79
80		let (session, executor_params) = match runtime_info
81			.get_session_info_by_index(ctx.sender(), relay_parent, session_index)
82			.await
83		{
84			Ok(extended_session_info) =>
85				(&extended_session_info.session_info, &extended_session_info.executor_params),
86			Err(_) => return None,
87		};
88
89		let n_validators = session.validators.len();
90		let byzantine_threshold = polkadot_primitives::byzantine_threshold(n_validators);
91		// combine on-chain with off-chain disabled validators
92		// process disabled validators in the following order:
93		// - on-chain disabled validators
94		// - prioritized order of off-chain disabled validators
95		// deduplicate the list and take at most `byzantine_threshold` validators
96		let disabled_indices = {
97			let mut d: HashSet<ValidatorIndex> = HashSet::new();
98			for v in disabled_onchain.into_iter().chain(disabled_offchain.into_iter()) {
99				if d.len() == byzantine_threshold {
100					break
101				}
102				d.insert(v);
103			}
104			d
105		};
106
107		let controlled_indices = controlled_indices
108			.get(session_index, &session.validators)
109			.map_or(HashSet::new(), |index| HashSet::from([index]));
110
111		Some(Self { session_index, session, executor_params, controlled_indices, disabled_indices })
112	}
113

115	pub fn validators(&self) -> &IndexedVec<ValidatorIndex, ValidatorId> {
116		&self.session.validators
117	}

119	/// `SessionInfo` for the candidate's session.

145/// Whether or not we already issued some statement about a candidate.
146pub enum OwnVoteState {
147	/// Our votes, if any.
148	Voted(Vec<(ValidatorIndex, (DisputeStatement, ValidatorSignature))>),
149
150	/// We are not a parachain validator in the session.
151	///
152	/// Hence we cannot vote.
153	CannotVote,
154}
155
156impl OwnVoteState {
157	fn new(votes: &CandidateVotes, env: &CandidateEnvironment) -> Self {
158		let controlled_indices = env.controlled_indices();
159		if controlled_indices.is_empty() {
160			return Self::CannotVote
161		}
162
163		let our_valid_votes = controlled_indices
164			.iter()
165			.filter_map(|i| votes.valid.raw().get_key_value(i))
166			.map(|(index, (kind, sig))| {
167				(*index, (DisputeStatement::Valid(kind.clone()), sig.clone()))
168			});
169		let our_invalid_votes = controlled_indices
170			.iter()
171			.filter_map(|i| votes.invalid.get_key_value(i))
172			.map(|(index, (kind, sig))| (*index, (DisputeStatement::Invalid(*kind), sig.clone())));
173
174		Self::Voted(our_valid_votes.chain(our_invalid_votes).collect())
175	}
176
177	/// Is a vote from us missing but we are a validator able to vote?
178	fn vote_missing(&self) -> bool {
179		match self {
180			Self::Voted(votes) if votes.is_empty() => true,
181			Self::Voted(_) | Self::CannotVote => false,
182		}
183	}
184
185	/// Get own approval votes, if any.
186	///
187	/// Empty iterator means, no approval votes. `None` means, there will never be any (we cannot
188	/// vote).
189	fn approval_votes(
190		&self,
191	) -> Option<impl Iterator<Item = (ValidatorIndex, &ValidatorSignature)>> {
192		match self {
193			Self::Voted(votes) => Some(votes.iter().filter_map(|(index, (kind, sig))| {
194				if let DisputeStatement::Valid(ValidDisputeStatementKind::ApprovalChecking) = kind {
195					Some((*index, sig))
196				} else {
197					None
198				}
199			})),
200			Self::CannotVote => None,
201		}
202	}
203
204	/// Get our votes if there are any.
205	///
206	/// Empty iterator means, no votes. `None` means, there will never be any (we cannot
207	/// vote).
208	fn votes(&self) -> Option<&Vec<(ValidatorIndex, (DisputeStatement, ValidatorSignature))>> {
209		match self {
210			Self::Voted(votes) => Some(&votes),
211			Self::CannotVote => None,
212		}
213	}
214}

216/// Complete state of votes for a candidate.
217///
218/// All votes + information whether a dispute is ongoing, confirmed, concluded, whether we already
219/// voted, ...
220pub struct CandidateVoteState<Votes> {
221	/// Votes already existing for the candidate + receipt.
222	votes: Votes,
223
224	/// Information about own votes:
225	own_vote: OwnVoteState,
226
227	/// Current dispute status, if there is any.
228	dispute_status: Option<DisputeStatus>,
229
230	/// Are there `byzantine threshold + 1` invalid votes
231	byzantine_threshold_against: bool,
232}
233
234impl CandidateVoteState<CandidateVotes> {
235	/// Create an empty `CandidateVoteState`
236	///
237	/// in case there have not been any previous votes.
238	pub fn new_from_receipt(candidate_receipt: CandidateReceipt) -> Self {
239		let votes = CandidateVotes {
240			candidate_receipt,
241			valid: ValidCandidateVotes::new(),
242			invalid: BTreeMap::new(),
243		};
244		Self {
245			votes,
246			own_vote: OwnVoteState::CannotVote,
247			dispute_status: None,
248			byzantine_threshold_against: false,
249		}
250	}
251
252	/// Create a new `CandidateVoteState` from already existing votes.
253	pub fn new(votes: CandidateVotes, env: &CandidateEnvironment, now: Timestamp) -> Self {
254		let own_vote = OwnVoteState::new(&votes, env);
255
256		let n_validators = env.validators().len();
257
258		let supermajority_threshold = polkadot_primitives::supermajority_threshold(n_validators);
259
260		// We have a dispute, if we have votes on both sides, with at least one invalid vote
261		// from non-disabled validator or with votes on both sides and confirmed.
262		let has_non_disabled_invalid_votes =
263			votes.invalid.keys().any(|i| !env.disabled_indices().contains(i));
264		let byzantine_threshold = polkadot_primitives::byzantine_threshold(n_validators);
265		let votes_on_both_sides = !votes.valid.raw().is_empty() && !votes.invalid.is_empty();
266		let is_confirmed =
267			votes_on_both_sides && (votes.voted_indices().len() > byzantine_threshold);
268		let is_disputed =
269			is_confirmed || (has_non_disabled_invalid_votes && !votes.valid.raw().is_empty());
270
271		let (dispute_status, byzantine_threshold_against) = if is_disputed {
272			let mut status = DisputeStatus::active();
273			if is_confirmed {
274				status = status.confirm();
275			};
276			let concluded_for = votes.valid.raw().len() >= supermajority_threshold;
277			if concluded_for {
278				status = status.conclude_for(now);
279			};
280
281			let concluded_against = votes.invalid.len() >= supermajority_threshold;
282			if concluded_against {
283				status = status.conclude_against(now);
284			};
285			(Some(status), votes.invalid.len() > byzantine_threshold)
286		} else {
287			(None, false)
288		};
289
290		Self { votes, own_vote, dispute_status, byzantine_threshold_against }
291	}

293	/// Import fresh statements.
294	///
295	/// Result will be a new state plus information about things that changed due to the import.
296	pub fn import_statements(
297		self,
298		env: &CandidateEnvironment,
299		statements: Vec<(SignedDisputeStatement, ValidatorIndex)>,
300		now: Timestamp,
301	) -> ImportResult {
302		let (mut votes, old_state) = self.into_old_state();
303
304		let mut new_invalid_voters = Vec::new();
305		let mut imported_invalid_votes = 0;
306		let mut imported_valid_votes = 0;
307
308		let expected_candidate_hash = votes.candidate_receipt.hash();
309
310		for (statement, val_index) in statements {
311			if env
312				.validators()
313				.get(val_index)
314				.map_or(true, |v| v != statement.validator_public())
315			{
316				gum::error!(
317					target: LOG_TARGET,
318					?val_index,
319					session= ?env.session_index,
320					claimed_key = ?statement.validator_public(),
321					"Validator index doesn't match claimed key",
322				);
323
324				continue
325			}
326			if statement.candidate_hash() != &expected_candidate_hash {
327				gum::error!(
328					target: LOG_TARGET,
329					?val_index,
330					session= ?env.session_index,
331					given_candidate_hash = ?statement.candidate_hash(),
332					?expected_candidate_hash,
333					"Vote is for unexpected candidate!",
334				);
335				continue
336			}
337			if statement.session_index() != env.session_index() {
338				gum::error!(
339					target: LOG_TARGET,
340					?val_index,
341					session= ?env.session_index,
342					given_candidate_hash = ?statement.candidate_hash(),
343					?expected_candidate_hash,
344					"Vote is for unexpected session!",
345				);
346				continue
347			}
348
349			match statement.statement() {
350				DisputeStatement::Valid(valid_kind) => {
351					let fresh = votes.valid.insert_vote(
352						val_index,
353						valid_kind.clone(),
354						statement.into_validator_signature(),
355					);
356					if fresh {
357						imported_valid_votes += 1;
358					}
359				},
360				DisputeStatement::Invalid(invalid_kind) => {
361					let fresh = votes
362						.invalid
363						.insert(val_index, (*invalid_kind, statement.into_validator_signature()))
364						.is_none();
365					if fresh {
366						new_invalid_voters.push(val_index);
367						imported_invalid_votes += 1;
368					}
369				},
370			}
371		}
372
373		let new_state = Self::new(votes, env, now);
374
375		ImportResult {
376			old_state,
377			new_state,
378			imported_invalid_votes,
379			imported_valid_votes,
380			imported_approval_votes: 0,
381			new_invalid_voters,
382		}
383	}

398	/// Extract `CandidateVotes` for handling import of new statements.
399	fn into_old_state(self) -> (CandidateVotes, CandidateVoteState<()>) {
400		let CandidateVoteState { votes, own_vote, dispute_status, byzantine_threshold_against } =
401			self;
402		(
403			votes,
404			CandidateVoteState { votes: (), own_vote, dispute_status, byzantine_threshold_against },
405		)

409impl<V> CandidateVoteState<V> {
410	/// Whether or not we have an ongoing dispute.
411	pub fn is_disputed(&self) -> bool {
412		self.dispute_status.is_some()
413	}
414
415	/// Whether there is an ongoing confirmed dispute.
416	///
417	/// This checks whether there is a dispute ongoing and we have more than byzantine threshold
418	/// votes.
419	pub fn is_confirmed(&self) -> bool {
420		self.dispute_status.map_or(false, |s| s.is_confirmed_concluded())
421	}
422
423	/// Are we a validator in the session, but have not yet voted?
424	pub fn own_vote_missing(&self) -> bool {
425		self.own_vote.vote_missing()
426	}
427
428	/// Own approval votes if any:
429	pub fn own_approval_votes(
430		&self,
431	) -> Option<impl Iterator<Item = (ValidatorIndex, &ValidatorSignature)>> {
432		self.own_vote.approval_votes()
433	}
434
435	/// Get own votes if there are any.
436	pub fn own_votes(
437		&self,
438	) -> Option<&Vec<(ValidatorIndex, (DisputeStatement, ValidatorSignature))>> {
439		self.own_vote.votes()
440	}
441
442	/// Whether or not there is a dispute and it has already enough valid votes to conclude.
443	pub fn has_concluded_for(&self) -> bool {
444		self.dispute_status.map_or(false, |s| s.has_concluded_for())
445	}
446
447	/// Whether or not there is a dispute and it has already enough invalid votes to conclude.
448	pub fn has_concluded_against(&self) -> bool {
449		self.dispute_status.map_or(false, |s| s.has_concluded_against())
450	}
451
452	/// Get access to the dispute status, in case there is one.
453	pub fn dispute_status(&self) -> &Option<DisputeStatus> {
454		&self.dispute_status
455	}
456
457	/// Access to underlying votes.
458	pub fn votes(&self) -> &V {
459		&self.votes
460	}

463/// An ongoing statement/vote import.
464pub struct ImportResult {
465	/// The state we had before importing new statements.
466	old_state: CandidateVoteState<()>,
467	/// The new state after importing the new statements.
468	new_state: CandidateVoteState<CandidateVotes>,
469	/// New invalid voters as of this import.
470	new_invalid_voters: Vec<ValidatorIndex>,
471	/// Number of successfully imported valid votes.
472	imported_invalid_votes: u32,
473	/// Number of successfully imported invalid votes.
474	imported_valid_votes: u32,
475	/// Number of approval votes imported via `import_approval_votes()`.
476	///
477	/// And only those: If normal import included approval votes, those are not counted here.
478	///
479	/// In other words, without a call `import_approval_votes()` this will always be 0.
480	imported_approval_votes: u32,
481}
482
483impl ImportResult {
484	/// Whether or not anything has changed due to the import.
485	pub fn votes_changed(&self) -> bool {
486		self.imported_valid_votes != 0 || self.imported_invalid_votes != 0
487	}
488
489	/// The dispute state has changed in some way.
490	///
491	/// - freshly disputed
492	/// - freshly confirmed
493	/// - freshly concluded (valid or invalid)
494	pub fn dispute_state_changed(&self) -> bool {
495		self.is_freshly_disputed() || self.is_freshly_confirmed() || self.is_freshly_concluded()
496	}
497
498	/// State as it was before import.
499	pub fn old_state(&self) -> &CandidateVoteState<()> {
500		&self.old_state
501	}
502
503	/// State after import
504	pub fn new_state(&self) -> &CandidateVoteState<CandidateVotes> {
505		&self.new_state
506	}
507
508	/// New "invalid" voters encountered during import.
509	pub fn new_invalid_voters(&self) -> &Vec<ValidatorIndex> {
510		&self.new_invalid_voters
511	}
512
513	/// Number of imported valid votes.
514	pub fn imported_valid_votes(&self) -> u32 {
515		self.imported_valid_votes
516	}
517
518	/// Number of imported invalid votes.
519	pub fn imported_invalid_votes(&self) -> u32 {
520		self.imported_invalid_votes
521	}
522
523	/// Number of imported approval votes.
524	pub fn imported_approval_votes(&self) -> u32 {
525		self.imported_approval_votes
526	}
527
528	/// Whether we now have a dispute and did not prior to the import.
529	pub fn is_freshly_disputed(&self) -> bool {
530		!self.old_state().is_disputed() && self.new_state().is_disputed()
531	}
532
533	/// Whether we just surpassed the byzantine threshold.
534	pub fn is_freshly_confirmed(&self) -> bool {
535		!self.old_state().is_confirmed() && self.new_state().is_confirmed()
536	}
537
538	/// Whether or not any dispute just concluded valid due to the import.
539	pub fn is_freshly_concluded_for(&self) -> bool {
540		!self.old_state().has_concluded_for() && self.new_state().has_concluded_for()
541	}
542
543	/// Whether or not any dispute just concluded invalid due to the import.
544	pub fn is_freshly_concluded_against(&self) -> bool {
545		!self.old_state().has_concluded_against() && self.new_state().has_concluded_against()
546	}
547
548	/// Whether or not any dispute just concluded either invalid or valid due to the import.
549	pub fn is_freshly_concluded(&self) -> bool {
550		self.is_freshly_concluded_against() || self.is_freshly_concluded_for()
551	}
552
553	/// Whether or not the invalid vote count for the dispute went beyond the byzantine threshold
554	/// after the last import
555	pub fn has_fresh_byzantine_threshold_against(&self) -> bool {
556		!self.old_state().byzantine_threshold_against &&
557			self.new_state().byzantine_threshold_against
558	}
559
560	/// Modify this `ImportResult`s, by importing additional approval votes.
561	///
562	/// Both results and `new_state` will be changed as if those approval votes had been in the
563	/// original import.
564	pub fn import_approval_votes(
565		self,
566		env: &CandidateEnvironment,
567		approval_votes: HashMap<ValidatorIndex, (Vec<CandidateHash>, ValidatorSignature)>,
568		now: Timestamp,
569	) -> Self {
570		let Self {
571			old_state,
572			new_state,
573			new_invalid_voters,
574			mut imported_valid_votes,
575			imported_invalid_votes,
576			mut imported_approval_votes,
577		} = self;
578
579		let (mut votes, _) = new_state.into_old_state();
580
581		for (index, (candidate_hashes, sig)) in approval_votes.into_iter() {
582			debug_assert!(
583				{
584					let pub_key = &env.session_info().validators.get(index).expect("indices are validated by approval-voting subsystem; qed");
585					let session_index = env.session_index();
586					candidate_hashes.contains(&votes.candidate_receipt.hash()) && DisputeStatement::Valid(ValidDisputeStatementKind::ApprovalCheckingMultipleCandidates(candidate_hashes.clone()))
587						.check_signature(pub_key, *candidate_hashes.first().expect("Valid votes have at least one candidate; qed"), session_index, &sig)
588						.is_ok()
589				},
590				"Signature check for imported approval votes failed! This is a serious bug. Session: {:?}, candidate hash: {:?}, validator index: {:?}", env.session_index(), votes.candidate_receipt.hash(), index
591			);
592			if votes.valid.insert_vote(
593				index,
594				// There is a hidden dependency here between approval-voting and this subsystem.
595				// We should be able to start emitting
596				// ValidDisputeStatementKind::ApprovalCheckingMultipleCandidates only after:
597				// 1. Runtime have been upgraded to know about the new format.
598				// 2. All nodes have been upgraded to know about the new format.
599				// Once those two requirements have been met we should be able to increase
600				// max_approval_coalesce_count to values greater than 1.
601				if candidate_hashes.len() > 1 {
602					ValidDisputeStatementKind::ApprovalCheckingMultipleCandidates(candidate_hashes)
603				} else {
604					ValidDisputeStatementKind::ApprovalChecking
605				},
606				sig,
607			) {
608				imported_valid_votes += 1;
609				imported_approval_votes += 1;
610			}
611		}
612
613		let new_state = CandidateVoteState::new(votes, env, now);
614
615		Self {
616			old_state,
617			new_state,
618			new_invalid_voters,
619			imported_valid_votes,
620			imported_invalid_votes,
621			imported_approval_votes,
622		}
623	}
624
625	/// All done, give me those votes.
626	///
627	/// Returns: `None` in case nothing has changed (import was redundant).
628	pub fn into_updated_votes(self) -> Option<CandidateVotes> {
629		if self.votes_changed() {
630			let CandidateVoteState { votes, .. } = self.new_state;
631			Some(votes)
632		} else {
633			None
634		}
635	}

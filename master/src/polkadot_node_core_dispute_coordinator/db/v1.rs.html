43const RECENT_DISPUTES_KEY: &[u8; 15] = b"recent-disputes";
44const EARLIEST_SESSION_KEY: &[u8; 16] = b"earliest-session";
45const CANDIDATE_VOTES_SUBKEY: &[u8; 15] = b"candidate-votes";
46/// Until what session have votes been cleaned up already?
47const CLEANED_VOTES_WATERMARK_KEY: &[u8; 23] = b"cleaned-votes-watermark";
48
49/// Restrict number of cleanup operations.
50///
51/// On the first run we are starting at session 0 going up all the way to the current session -
52/// this should not be done at once, but rather in smaller batches so nodes won't get stalled by
53/// this.
54///
55/// 300 is with session duration of 1 hour and 30 parachains around <3_000_000 key purges in the
56/// worst case. Which is already quite a lot, at the same time we have around 21_000 sessions on
57/// Kusama. This means at 300 purged sessions per session, cleaning everything up will take
58/// around 3 days. Depending on how severe disk usage becomes, we might want to bump the batch
59/// size, at the cost of risking issues at session boundaries (performance).
60#[cfg(test)]
61const MAX_CLEAN_BATCH_SIZE: u32 = 10;
62#[cfg(not(test))]
63const MAX_CLEAN_BATCH_SIZE: u32 = 300;
64

67	config: ColumnConfiguration,

76	/// Cleanup old votes.
77	///
78	/// Should be called whenever a new earliest session gets written.
79	fn add_vote_cleanup_tx(
80		&mut self,
81		tx: &mut DBTransaction,
82		earliest_session: SessionIndex,
83	) -> FatalResult<()> {
84		// Cleanup old votes in db:
85		let watermark = load_cleaned_votes_watermark(&*self.inner, &self.config)?.unwrap_or(0);
86		let clean_until = if earliest_session.saturating_sub(watermark) > MAX_CLEAN_BATCH_SIZE {
87			watermark + MAX_CLEAN_BATCH_SIZE
88		} else {
89			earliest_session
90		};
91		gum::trace!(
92			target: LOG_TARGET,
93			?watermark,
94			?clean_until,
95			?earliest_session,
96			?MAX_CLEAN_BATCH_SIZE,
97			"WriteEarliestSession"
98		);
99
100		for index in watermark..clean_until {
101			gum::trace!(
102			target: LOG_TARGET,
103			?index,
104			encoded = ?candidate_votes_session_prefix(index),
105			"Cleaning votes for session index"
106			);
107			tx.delete_prefix(self.config.col_dispute_data, &candidate_votes_session_prefix(index));
108		}
109		// New watermark:
110		tx.put_vec(self.config.col_dispute_data, CLEANED_VOTES_WATERMARK_KEY, clean_until.encode());
111		Ok(())
112	}
113}

126	/// Load the candidate votes for the specific session-candidate pair, if any.
127	fn load_candidate_votes(
128		&self,
129		session: SessionIndex,
130		candidate_hash: &CandidateHash,

133	}
134
135	/// Atomically writes the list of operations, with later operations taking precedence over
136	/// prior.
137	///
138	/// This also takes care of purging old votes (of obsolete sessions).
139	fn write<I>(&mut self, ops: I) -> FatalResult<()>
140	where
141		I: IntoIterator<Item = BackendWriteOp>,
142	{
143		let mut tx = DBTransaction::new();
144		// Make sure the whole process is timed, including the actual transaction flush:
145		let mut cleanup_timer = None;
146		for op in ops {
147			match op {
148				BackendWriteOp::WriteEarliestSession(session) => {
149					cleanup_timer = match cleanup_timer.take() {
150						None => Some(self.metrics.time_vote_cleanup()),
151						Some(t) => Some(t),
152					};
153					self.add_vote_cleanup_tx(&mut tx, session)?;
154
155					// Actually write the earliest session.
156					tx.put_vec(
157						self.config.col_dispute_data,
158						EARLIEST_SESSION_KEY,
159						session.encode(),
160					);
161				},
162				BackendWriteOp::WriteRecentDisputes(recent_disputes) => {
163					tx.put_vec(
164						self.config.col_dispute_data,
165						RECENT_DISPUTES_KEY,
166						recent_disputes.encode(),
167					);
168				},
169				BackendWriteOp::WriteCandidateVotes(session, candidate_hash, votes) => {
170					gum::trace!(target: LOG_TARGET, ?session, "Writing candidate votes");
171					tx.put_vec(
172						self.config.col_dispute_data,
173						&candidate_votes_key(session, &candidate_hash),
174						votes.encode(),
175					);
176				},
177				BackendWriteOp::DeleteCandidateVotes(session, candidate_hash) => {
178					tx.delete(
179						self.config.col_dispute_data,
180						&candidate_votes_key(session, &candidate_hash),
181					);
182				},
183			}
184		}
185
186		self.inner.write(tx).map_err(FatalError::DbWriteFailed)
187	}
188}

190fn candidate_votes_key(session: SessionIndex, candidate_hash: &CandidateHash) -> [u8; 15 + 4 + 32] {
191	let mut buf = [0u8; 15 + 4 + 32];
192	buf[..15].copy_from_slice(CANDIDATE_VOTES_SUBKEY);
193
194	// big-endian encoding is used to ensure lexicographic ordering.
195	buf[15..][..4].copy_from_slice(&session.to_be_bytes());
196	candidate_hash.using_encoded(|s| buf[(15 + 4)..].copy_from_slice(s));
197
198	buf
199}
200
201fn candidate_votes_session_prefix(session: SessionIndex) -> [u8; 15 + 4] {
202	let mut buf = [0u8; 15 + 4];
203	buf[..15].copy_from_slice(CANDIDATE_VOTES_SUBKEY);
204
205	// big-endian encoding is used to ensure lexicographic ordering.
206	buf[15..][..4].copy_from_slice(&session.to_be_bytes());
207	buf
208}

217/// Tracked votes on candidates, for the purposes of dispute resolution.
218#[derive(Debug, Clone, Encode, Decode)]
219pub struct CandidateVotes {
220	/// The receipt of the candidate itself.
221	pub candidate_receipt: CandidateReceipt,
222	/// Votes of validity, sorted by validator index.
223	pub valid: Vec<(ValidDisputeStatementKind, ValidatorIndex, ValidatorSignature)>,
224	/// Votes of invalidity, sorted by validator index.
225	pub invalid: Vec<(InvalidDisputeStatementKind, ValidatorIndex, ValidatorSignature)>,
226}
227
228impl From<CandidateVotes> for polkadot_node_primitives::CandidateVotes {
229	fn from(db_votes: CandidateVotes) -> polkadot_node_primitives::CandidateVotes {
230		polkadot_node_primitives::CandidateVotes {
231			candidate_receipt: db_votes.candidate_receipt,
232			valid: db_votes.valid.into_iter().map(|(kind, i, sig)| (i, (kind, sig))).collect(),
233			invalid: db_votes.invalid.into_iter().map(|(kind, i, sig)| (i, (kind, sig))).collect(),
234		}
235	}
236}

276fn load_decode<D: Decode>(
277	db: &dyn Database,
278	col_dispute_data: u32,
279	key: &[u8],
280) -> Result<Option<D>> {
281	match db.get(col_dispute_data, key)? {
282		None => Ok(None),
283		Some(raw) => D::decode(&mut &raw[..]).map(Some).map_err(Into::into),
284	}
285}

316/// Maybe prune data in the DB based on the provided session index.
317///
318/// This is intended to be called on every block, and as such will be used to populate the DB on
319/// first launch. If the on-disk data does not need to be pruned, only a single storage read
320/// will be performed.
321///
322/// If one or more ancient sessions are pruned, all metadata on candidates within the ancient
323/// session will be deleted.
324pub(crate) fn note_earliest_session(
325	overlay_db: &mut OverlayedBackend<'_, impl Backend>,
326	new_earliest_session: SessionIndex,
327) -> FatalResult<()> {
328	match overlay_db.load_earliest_session()? {
329		None => {
330			// First launch - write new-earliest.
331			overlay_db.write_earliest_session(new_earliest_session);
332		},
333		Some(prev_earliest) if new_earliest_session > prev_earliest => {
334			// Prune all data in the outdated sessions.
335			overlay_db.write_earliest_session(new_earliest_session);
336
337			// Clear recent disputes metadata.
338			{
339				let mut recent_disputes = overlay_db.load_recent_disputes()?.unwrap_or_default();
340
341				let lower_bound = (new_earliest_session, CandidateHash(Hash::repeat_byte(0x00)));
342
343				let new_recent_disputes = recent_disputes.split_off(&lower_bound);
344				// Any remaining disputes are considered ancient and must be pruned.
345				let pruned_disputes = recent_disputes;
346
347				if pruned_disputes.len() != 0 {
348					overlay_db.write_recent_disputes(new_recent_disputes);
349					// Note: Deleting old candidate votes is handled in `write` based on the
350					// earliest session.
351				}
352			}
353		},
354		Some(_) => {
355			// nothing to do.
356		},
357	}
358
359	Ok(())
360}

362/// Until what session votes have been cleaned up already.
363///
364/// That is the db has already been purged of votes for sessions older than the returned
365/// `SessionIndex`.
366fn load_cleaned_votes_watermark(
367	db: &dyn Database,
368	config: &ColumnConfiguration,
369) -> FatalResult<Option<SessionIndex>> {
370	load_decode(db, config.col_dispute_data, CLEANED_VOTES_WATERMARK_KEY)
371		.map_err(|e| FatalError::DbReadFailed(e))
372}

620		let super_old_no_dispute = 1;
621		let very_old = 3;
622		let slightly_old = 4;
623		let very_recent = current_session - 1;
624
625		let blank_candidate_votes = || CandidateVotes {
626			candidate_receipt: dummy_candidate_receipt_v2(dummy_hash()),
627			valid: Vec::new(),
628			invalid: Vec::new(),
629		};
630
631		let mut overlay_db = OverlayedBackend::new(&backend);
632		overlay_db.write_earliest_session(prev_earliest_session);
633		overlay_db.write_recent_disputes(
634			vec![
635				((very_old, hash_a), DisputeStatus::Active),
636				((slightly_old, hash_b), DisputeStatus::Active),
637				((new_earliest_session, hash_c), DisputeStatus::Active),
638				((very_recent, hash_d), DisputeStatus::Active),
639			]
640			.into_iter()
641			.collect(),
642		);
643
644		overlay_db.write_candidate_votes(super_old_no_dispute, hash_a, blank_candidate_votes());
645		overlay_db.write_candidate_votes(very_old, hash_a, blank_candidate_votes());
646
647		overlay_db.write_candidate_votes(slightly_old, hash_b, blank_candidate_votes());
648
649		overlay_db.write_candidate_votes(new_earliest_session, hash_c, blank_candidate_votes());
650
651		overlay_db.write_candidate_votes(very_recent, hash_d, blank_candidate_votes());
652
653		let write_ops = overlay_db.into_write_ops();
654		backend.write(write_ops).unwrap();
655
656		let mut overlay_db = OverlayedBackend::new(&backend);
657		note_earliest_session(&mut overlay_db, new_earliest_session).unwrap();
658
659		assert_eq!(overlay_db.load_earliest_session().unwrap(), Some(new_earliest_session));
660
661		assert_eq!(
662			overlay_db.load_recent_disputes().unwrap().unwrap(),
663			vec![
664				((new_earliest_session, hash_c), DisputeStatus::Active),
665				((very_recent, hash_d), DisputeStatus::Active),
666			]
667			.into_iter()
668			.collect(),
669		);
670
671		// Votes are only cleaned up after actual write:
672		let write_ops = overlay_db.into_write_ops();
673		backend.write(write_ops).unwrap();
674
675		let overlay_db = OverlayedBackend::new(&backend);
676
677		assert!(overlay_db
678			.load_candidate_votes(super_old_no_dispute, &hash_a)
679			.unwrap()
680			.is_none());
681		assert!(overlay_db.load_candidate_votes(very_old, &hash_a).unwrap().is_none());
682		assert!(overlay_db.load_candidate_votes(slightly_old, &hash_b).unwrap().is_none());
683		assert!(overlay_db
684			.load_candidate_votes(new_earliest_session, &hash_c)
685			.unwrap()
686			.is_some());
687		assert!(overlay_db.load_candidate_votes(very_recent, &hash_d).unwrap().is_some());

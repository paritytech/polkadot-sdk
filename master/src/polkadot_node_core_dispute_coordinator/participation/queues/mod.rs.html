40/// How many potential garbage disputes we want to queue, before starting to drop requests.
41#[cfg(not(test))]
42const BEST_EFFORT_QUEUE_SIZE: usize = 100;
43#[cfg(test)]
44const BEST_EFFORT_QUEUE_SIZE: usize = 3;
45
46/// How many priority disputes can be queued.
47///
48/// Once the queue exceeds that size, we will start to drop the newest participation requests in
49/// the queue. Note that for each vote import the request will be re-added, if there is free
50/// capacity. This limit just serves as a safe guard, it is not expected to ever really be reached.
51///
52/// For 100 parachains, this would allow for every single candidate in 100 blocks on
53/// two forks to get disputed, which should be plenty to deal with any realistic attack.
54#[cfg(not(test))]
55const PRIORITY_QUEUE_SIZE: usize = 20_000;
56#[cfg(test)]
57const PRIORITY_QUEUE_SIZE: usize = 2;
58
59/// Queues for dispute participation.
60/// In both queues we have a strict ordering of candidates and participation will
61/// happen in that order. Refer to `CandidateComparator` for details on the ordering.
62pub struct Queues {
63	/// Set of best effort participation requests.
64	best_effort: BTreeMap<CandidateComparator, ParticipationRequest>,
65
66	/// Priority queue.
67	priority: BTreeMap<CandidateComparator, ParticipationRequest>,
68
69	/// Handle for recording queues data in metrics
70	metrics: Metrics,
71}

73/// A dispute participation request that can be queued.
74#[derive(Debug)]
75pub struct ParticipationRequest {
76	candidate_hash: CandidateHash,
77	candidate_receipt: CandidateReceipt,
78	session: SessionIndex,
79	executor_params: ExecutorParams,
80	request_timer: Option<prometheus::HistogramTimer>, // Sends metric data when request is dropped
81}
82
83/// Whether a `ParticipationRequest` should be put on best-effort or the priority queue.
84#[derive(Debug)]
85pub enum ParticipationPriority {
86	BestEffort,
87	Priority,
88}
89
90impl ParticipationPriority {
91	/// Create `ParticipationPriority` with either `Priority`
92	///
93	/// or `BestEffort`.
94	pub fn with_priority_if(is_priority: bool) -> Self {
95		if is_priority {
96			Self::Priority
97		} else {
98			Self::BestEffort
99		}
100	}
101
102	/// Whether or not this is a priority entry.
103	///
104	/// If false, it is best effort.
105	pub fn is_priority(&self) -> bool {
106		match self {
107			Self::Priority => true,
108			Self::BestEffort => false,
109		}
110	}
111}

113/// What can go wrong when queuing a request.
114#[derive(Debug, thiserror::Error)]
115pub enum QueueError {
116	#[error("Request could not be queued, because best effort queue was already full.")]
117	BestEffortFull,
118	#[error("Request could not be queued, because priority queue was already full.")]
119	PriorityFull,
120}
121
122impl ParticipationRequest {
123	/// Create a new `ParticipationRequest` to be queued.
124	pub fn new(
125		candidate_receipt: CandidateReceipt,
126		session: SessionIndex,
127		executor_params: ExecutorParams,
128		request_timer: Option<prometheus::HistogramTimer>,
129	) -> Self {
130		Self {
131			candidate_hash: candidate_receipt.hash(),
132			candidate_receipt,
133			session,
134			executor_params,
135			request_timer,
136		}
137	}
138
139	pub fn candidate_receipt(&'_ self) -> &'_ CandidateReceipt {
140		&self.candidate_receipt
141	}
142	pub fn candidate_hash(&'_ self) -> &'_ CandidateHash {
143		&self.candidate_hash
144	}
145	pub fn session(&self) -> SessionIndex {
146		self.session
147	}
148	pub fn executor_params(&self) -> ExecutorParams {
149		self.executor_params.clone()
150	}
151	pub fn discard_timer(&mut self) {
152		if let Some(timer) = self.request_timer.take() {
153			timer.stop_and_discard();
154		}
155	}
156	pub fn into_candidate_info(self) -> (CandidateHash, CandidateReceipt) {
157		let Self { candidate_hash, candidate_receipt, .. } = self;
158		(candidate_hash, candidate_receipt)
159	}
160}

183impl Queues {
184	/// Create new `Queues`.
185	pub fn new(metrics: Metrics) -> Self {
186		Self { best_effort: BTreeMap::new(), priority: BTreeMap::new(), metrics }
187	}
188
189	/// Will put message in queue, either priority or best effort depending on priority.
190	///
191	/// If the message was already previously present on best effort, it will be moved to priority
192	/// if it is considered priority now.
193	///
194	/// Returns error in case a queue was found full already.
195	pub async fn queue(
196		&mut self,
197		sender: &mut impl overseer::DisputeCoordinatorSenderTrait,
198		priority: ParticipationPriority,
199		req: ParticipationRequest,
200	) -> Result<()> {
201		let comparator = CandidateComparator::new(sender, &req.candidate_receipt).await?;
202
203		self.queue_with_comparator(comparator, priority, req)?;
204		Ok(())
205	}
206
207	/// Get the next best request for dispute participation if any.
208	/// First the priority queue is considered and then the best effort one.
209	pub fn dequeue(&mut self) -> Option<ParticipationRequest> {
210		if let Some(req) = self.pop_priority() {
211			self.metrics.report_priority_queue_size(self.priority.len() as u64);
212			return Some(req.1)
213		}
214		if let Some(req) = self.pop_best_effort() {
215			self.metrics.report_best_effort_queue_size(self.best_effort.len() as u64);
216			return Some(req.1)
217		}
218		None
219	}
220
221	/// Reprioritizes any participation requests pertaining to the
222	/// passed candidates from best effort to priority.
223	pub async fn prioritize_if_present(
224		&mut self,
225		sender: &mut impl overseer::DisputeCoordinatorSenderTrait,
226		receipt: &CandidateReceipt,
227	) -> Result<()> {
228		let comparator = CandidateComparator::new(sender, receipt).await?;
229		self.prioritize_with_comparator(comparator)?;
230		Ok(())
231	}
232
233	fn prioritize_with_comparator(
234		&mut self,
235		comparator: CandidateComparator,
236	) -> std::result::Result<(), QueueError> {
237		if self.priority.len() >= PRIORITY_QUEUE_SIZE {
238			return Err(QueueError::PriorityFull)
239		}
240		if let Some(request) = self.best_effort.remove(&comparator) {
241			self.priority.insert(comparator, request);
242			// Report changes to both queue sizes
243			self.metrics.report_priority_queue_size(self.priority.len() as u64);
244			self.metrics.report_best_effort_queue_size(self.best_effort.len() as u64);
245		}
246		Ok(())
247	}
248
249	/// Will put message in queue, either priority or best effort depending on priority.
250	///
251	/// If the message was already previously present on best effort, it will be moved to priority
252	/// if it is considered priority now.
253	///
254	/// Returns error in case a queue was found full already.
255	///
256	///  # Request timers
257	///
258	/// [`ParticipationRequest`]s contain request timers.
259	/// Where an old request would be replaced by a new one, we keep the old request.
260	/// This prevents request timers from resetting on each new request.
261	fn queue_with_comparator(
262		&mut self,
263		comparator: CandidateComparator,
264		priority: ParticipationPriority,
265		mut req: ParticipationRequest,
266	) -> std::result::Result<(), QueueError> {
267		if priority.is_priority() {
268			if self.priority.len() >= PRIORITY_QUEUE_SIZE {
269				return Err(QueueError::PriorityFull)
270			}
271			// Remove any best effort entry, using it to replace our new
272			// request.
273			if let Some(older_request) = self.best_effort.remove(&comparator) {
274				req.discard_timer();
275				req = older_request;
276			}
277			// Keeping old request if any.
278			match self.priority.entry(comparator) {
279				Entry::Occupied(_) => req.discard_timer(),
280				Entry::Vacant(vac) => {
281					gum::trace!(
282						target: LOG_TARGET,
283						candidate_hash = ?req.candidate_hash(),
284						"Added to priority participation queue"
285					);
286					vac.insert(req);
287				},
288			}
289			self.metrics.report_priority_queue_size(self.priority.len() as u64);
290			self.metrics.report_best_effort_queue_size(self.best_effort.len() as u64);
291		} else {
292			if self.priority.contains_key(&comparator) {
293				// The candidate is already in priority queue - don't
294				// add in in best effort too.
295				return Ok(())
296			}
297			if self.best_effort.len() >= BEST_EFFORT_QUEUE_SIZE {
298				return Err(QueueError::BestEffortFull)
299			}
300			// Keeping old request if any.
301			match self.best_effort.entry(comparator) {
302				Entry::Occupied(_) => req.discard_timer(),
303				Entry::Vacant(vac) => {
304					gum::trace!(
305						target: LOG_TARGET,
306						candidate_hash = ?req.candidate_hash(),
307						"Added to best effort participation queue"
308					);
309					vac.insert(req);
310				},
311			}
312			self.metrics.report_best_effort_queue_size(self.best_effort.len() as u64);
313		}
314		Ok(())
315	}
316
317	/// Get best from the best effort queue.
318	fn pop_best_effort(&mut self) -> Option<(CandidateComparator, ParticipationRequest)> {
319		return Self::pop_impl(&mut self.best_effort)
320	}
321
322	/// Get best priority queue entry.
323	fn pop_priority(&mut self) -> Option<(CandidateComparator, ParticipationRequest)> {
324		return Self::pop_impl(&mut self.priority)
325	}
326
327	// `pop_best_effort` and `pop_priority` do the same but on different `BTreeMap`s. This function
328	// has the extracted implementation
329	fn pop_impl(
330		target: &mut BTreeMap<CandidateComparator, ParticipationRequest>,
331	) -> Option<(CandidateComparator, ParticipationRequest)> {
332		// Once https://github.com/rust-lang/rust/issues/62924 is there, we can use a simple:
333		// target.pop_first().
334		if let Some((comparator, _)) = target.iter().next() {
335			let comparator = *comparator;
336			target
337				.remove(&comparator)
338				.map(|participation_request| (comparator, participation_request))
339		} else {
340			None
341		}
342	}
343}

345/// `Comparator` for ordering of disputes for candidates.
346///
347/// This `comparator` makes it possible to order disputes based on age and to ensure some fairness
348/// between chains in case of equally old disputes.
349///
350/// Objective ordering between nodes is important in case of lots disputes, so nodes will pull in
351/// the same direction and work on resolving the same disputes first. This ensures that we will
352/// conclude some disputes, even if there are lots of them. While any objective ordering would
353/// suffice for this goal, ordering by age ensures we are not only resolving disputes, but also
354/// resolve the oldest one first, which are also the most urgent and important ones to resolve.
355///
356/// Note: That by `oldest` we mean oldest in terms of relay chain block number, for any block
357/// number that has not yet been finalized. If a block has been finalized already it should be
358/// treated as low priority when it comes to disputes, as even in the case of a negative outcome,
359/// we are already too late. The ordering mechanism here serves to prevent this from happening in
360/// the first place.
361#[derive(Copy, Clone)]
362#[cfg_attr(test, derive(Debug))]
363struct CandidateComparator {
364	/// Block number of the relay parent. It's wrapped in an `Option<>` because there are cases
365	/// when it can't be obtained. For example when the node is lagging behind and new leaves are
366	/// received with a slight delay. Candidates with unknown relay parent are treated with the
367	/// lowest priority.
368	///
369	/// The order enforced by `CandidateComparator` is important because we want to participate in
370	/// the oldest disputes first.
371	///
372	/// Note: In theory it would make more sense to use the `BlockNumber` of the including
373	/// block, as inclusion time is the actual relevant event when it comes to ordering. The
374	/// problem is, that a candidate can get included multiple times on forks, so the `BlockNumber`
375	/// of the including block is not unique. We could theoretically work around that problem, by
376	/// just using the lowest `BlockNumber` of all available including blocks - the problem is,
377	/// that is not stable. If a new fork appears after the fact, we would start ordering the same
378	/// candidate differently, which would result in the same candidate getting queued twice.
379	relay_parent_block_number: Option<BlockNumber>,
380	/// By adding the `CandidateHash`, we can guarantee a unique ordering across candidates with
381	/// the same relay parent block number. Candidates without `relay_parent_block_number` are
382	/// ordered by the `candidate_hash` (and treated with the lowest priority, as already
383	/// mentioned).
384	candidate_hash: CandidateHash,
385}
386
387impl CandidateComparator {
388	/// Create a candidate comparator based on given (fake) values.
389	///
390	/// Useful for testing.
391	#[cfg(test)]
392	pub fn new_dummy(block_number: Option<BlockNumber>, candidate_hash: CandidateHash) -> Self {
393		Self { relay_parent_block_number: block_number, candidate_hash }
394	}
395
396	/// Create a candidate comparator for a given candidate.
397	///
398	/// Returns:
399	/// 	- `Ok(CandidateComparator{Some(relay_parent_block_number), candidate_hash})` when the
400	/// 	relay parent can be obtained. This is the happy case.
401	/// - `Ok(CandidateComparator{None, candidate_hash})` in case the candidate's relay parent
402	/// 	can't be obtained.
403	/// 	- `FatalError` in case the chain API call fails with an unexpected error.
404	pub async fn new(
405		sender: &mut impl overseer::DisputeCoordinatorSenderTrait,
406		candidate: &CandidateReceipt,
407	) -> FatalResult<Self> {
408		let candidate_hash = candidate.hash();
409		let n = get_block_number(sender, candidate.descriptor().relay_parent()).await?;
410
411		if n.is_none() {
412			gum::warn!(
413				target: LOG_TARGET,
414				candidate_hash = ?candidate_hash,
415				"Candidate's relay_parent could not be found via chain API - `CandidateComparator` \
416				with an empty relay parent block number will be provided!"
417			);
418		}
419
420		Ok(CandidateComparator { relay_parent_block_number: n, candidate_hash })
421	}
422}

424impl PartialEq for CandidateComparator {
425	fn eq(&self, other: &CandidateComparator) -> bool {
426		Ordering::Equal == self.cmp(other)
427	}
428}
429
430impl Eq for CandidateComparator {}
431
432impl PartialOrd for CandidateComparator {
433	fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
434		Some(self.cmp(other))
435	}
436}
437
438impl Ord for CandidateComparator {
439	fn cmp(&self, other: &Self) -> Ordering {
440		return match (self.relay_parent_block_number, other.relay_parent_block_number) {
441			(None, None) => {
442				// No relay parents for both -> compare hashes
443				self.candidate_hash.cmp(&other.candidate_hash)
444			},
445			(Some(self_relay_parent_block_num), Some(other_relay_parent_block_num)) => {
446				match self_relay_parent_block_num.cmp(&other_relay_parent_block_num) {
447					// if the relay parent is the same for both -> compare hashes
448					Ordering::Equal => self.candidate_hash.cmp(&other.candidate_hash),
449					// if not - return the result from comparing the relay parent block numbers
450					o => return o,
451				}
452			},
453			(Some(_), None) => {
454				// Candidates with known relay parents are always with priority
455				Ordering::Less
456			},
457			(None, Some(_)) => {
458				// Ditto
459				Ordering::Greater
460			},
461		}
462	}
463}

465async fn get_block_number(
466	sender: &mut impl overseer::DisputeCoordinatorSenderTrait,
467	relay_parent: Hash,
468) -> FatalResult<Option<BlockNumber>> {
469	let (tx, rx) = oneshot::channel();
470	sender.send_message(ChainApiMessage::BlockNumber(relay_parent, tx)).await;
471	rx.await
472		.map_err(|_| FatalError::ChainApiSenderDropped)?
473		.map_err(FatalError::ChainApiAncestors)
474}

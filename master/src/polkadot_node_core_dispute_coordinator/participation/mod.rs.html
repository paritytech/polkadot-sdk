54/// How many participation processes do we want to run in parallel the most.
55///
56/// This should be a relatively low value, while we might have a speedup once we fetched the data,
57/// due to multi-core architectures, but the fetching itself can not be improved by parallel
58/// requests. This means that higher numbers make it harder for a single dispute to resolve fast.
59#[cfg(not(test))]
60const MAX_PARALLEL_PARTICIPATIONS: usize = 3;
61#[cfg(test)]
62pub(crate) const MAX_PARALLEL_PARTICIPATIONS: usize = 1;
63
64/// Keep track of disputes we need to participate in.
65///
66/// - Prioritize and queue participations
67/// - Dequeue participation requests in order and launch participation worker.
68pub struct Participation {
69	/// Participations currently being processed.
70	running_participations: HashSet<CandidateHash>,
71	/// Priority and best effort queues.
72	queue: Queues,
73	/// Sender to be passed to worker tasks.
74	worker_sender: WorkerMessageSender,
75	/// Some recent block for retrieving validation code from chain.
76	recent_block: Option<(BlockNumber, Hash)>,
77	/// Metrics handle cloned from Initialized
78	metrics: Metrics,
79}
80
81/// Message from worker tasks.
82#[derive(Debug)]
83pub struct WorkerMessage(ParticipationStatement);
84
85/// Sender use by worker tasks.
86pub type WorkerMessageSender = mpsc::Sender<WorkerMessage>;
87
88/// Receiver to receive messages from worker tasks.
89pub type WorkerMessageReceiver = mpsc::Receiver<WorkerMessage>;
90
91/// Statement as result of the validation process.
92#[derive(Debug)]
93pub struct ParticipationStatement {
94	/// Relevant session.
95	pub session: SessionIndex,
96	/// The candidate the worker has been spawned for.
97	pub candidate_hash: CandidateHash,
98	/// Used receipt.
99	pub candidate_receipt: CandidateReceipt,
100	/// Actual result.
101	pub outcome: ParticipationOutcome,
102}
103
104/// Outcome of the validation process.
105#[derive(Copy, Clone, Debug)]
106pub enum ParticipationOutcome {
107	/// Candidate was found to be valid.
108	Valid,
109	/// Candidate was found to be invalid.
110	Invalid,
111	/// Candidate was found to be unavailable.
112	Unavailable,
113	/// Something went wrong (bug), details can be found in the logs.
114	Error,
115}
116
117impl ParticipationOutcome {
118	/// If validation was successful, get whether the candidate was valid or invalid.
119	pub fn validity(self) -> Option<bool> {
120		match self {
121			Self::Valid => Some(true),
122			Self::Invalid => Some(false),
123			Self::Unavailable | Self::Error => None,
124		}
125	}
126}
127
128impl WorkerMessage {
129	fn from_request(req: ParticipationRequest, outcome: ParticipationOutcome) -> Self {
130		let session = req.session();
131		let (candidate_hash, candidate_receipt) = req.into_candidate_info();
132		Self(ParticipationStatement { session, candidate_hash, candidate_receipt, outcome })
133	}
134}
135
136#[overseer::contextbounds(DisputeCoordinator, prefix = self::overseer)]
137impl Participation {
138	/// Get ready for managing dispute participation requests.
139	///
140	/// The passed in sender will be used by background workers to communicate back their results.
141	/// The calling context should make sure to call `Participation::on_worker_message()` for the
142	/// received messages.
143	pub fn new(sender: WorkerMessageSender, metrics: Metrics) -> Self {
144		Self {
145			running_participations: HashSet::new(),
146			queue: Queues::new(metrics.clone()),
147			worker_sender: sender,
148			recent_block: None,
149			metrics,
150		}
151	}
152
153	/// Queue a dispute for the node to participate in.
154	///
155	/// If capacity is available right now and we already got some relay chain head via
156	/// `on_active_leaves_update`, the participation will be launched right away.
157	///
158	/// Returns: false, if queues are already full.
159	pub async fn queue_participation<Context>(
160		&mut self,
161		ctx: &mut Context,
162		priority: ParticipationPriority,
163		mut req: ParticipationRequest,
164	) -> Result<()> {
165		// Participation already running - we can ignore that request, discarding its timer:
166		if self.running_participations.contains(req.candidate_hash()) {
167			req.discard_timer();
168			return Ok(())
169		}
170		// Available capacity - participate right away (if we already have a recent block):
171		if let Some((_, h)) = self.recent_block {
172			if self.running_participations.len() < MAX_PARALLEL_PARTICIPATIONS {
173				self.fork_participation(ctx, req, h)?;
174				return Ok(())
175			}
176		}
177		// Out of capacity/no recent block yet - queue:
178		self.queue.queue(ctx.sender(), priority, req).await
179	}
180
181	/// Message from a worker task was received - get the outcome.
182	///
183	/// Call this function to keep participations going and to receive `ParticipationStatement`s.
184	///
185	/// This message has to be called for each received worker message, in order to make sure
186	/// enough participation processes are running at any given time.
187	///
188	/// Returns: The received `ParticipationStatement` or a fatal error, in case
189	/// something went wrong when dequeuing more requests (tasks could not be spawned).
190	pub async fn get_participation_result<Context>(
191		&mut self,
192		ctx: &mut Context,
193		msg: WorkerMessage,
194	) -> FatalResult<ParticipationStatement> {
195		let WorkerMessage(statement) = msg;
196		self.running_participations.remove(&statement.candidate_hash);
197		let recent_block = self.recent_block.expect("We never ever reset recent_block to `None` and we already received a result, so it must have been set before. qed.");
198		self.dequeue_until_capacity(ctx, recent_block.1).await?;
199		Ok(statement)
200	}
201
202	/// Process active leaves update.
203	///
204	/// Make sure we to dequeue participations if that became possible and update most recent
205	/// block.
206	pub async fn process_active_leaves_update<Context>(
207		&mut self,
208		ctx: &mut Context,
209		update: &ActiveLeavesUpdate,
210	) -> FatalResult<()> {
211		if let Some(activated) = &update.activated {
212			match self.recent_block {
213				None => {
214					self.recent_block = Some((activated.number, activated.hash));
215					// Work got potentially unblocked:
216					self.dequeue_until_capacity(ctx, activated.hash).await?;
217				},
218				Some((number, _)) if activated.number > number => {
219					self.recent_block = Some((activated.number, activated.hash));
220				},
221				Some(_) => {},
222			}
223		}
224		Ok(())
225	}
226
227	/// Moving any request concerning the given candidates from best-effort to
228	/// priority, ignoring any candidates that don't have any queued participation requests.
229	pub async fn bump_to_priority_for_candidates<Context>(
230		&mut self,
231		ctx: &mut Context,
232		included_receipts: &Vec<CandidateReceipt>,
233	) -> Result<()> {
234		for receipt in included_receipts {
235			self.queue.prioritize_if_present(ctx.sender(), receipt).await?;
236		}
237		Ok(())
238	}
239
240	/// Dequeue until `MAX_PARALLEL_PARTICIPATIONS` is reached.
241	async fn dequeue_until_capacity<Context>(
242		&mut self,
243		ctx: &mut Context,
244		recent_head: Hash,
245	) -> FatalResult<()> {
246		while self.running_participations.len() < MAX_PARALLEL_PARTICIPATIONS {
247			if let Some(req) = self.queue.dequeue() {
248				self.fork_participation(ctx, req, recent_head)?;
249			} else {
250				break
251			}
252		}
253		Ok(())
254	}
255
256	/// Fork a participation task in the background.
257	fn fork_participation<Context>(
258		&mut self,
259		ctx: &mut Context,
260		req: ParticipationRequest,
261		recent_head: Hash,
262	) -> FatalResult<()> {
263		gum::trace!(
264			target: LOG_TARGET,
265			candidate_hash = ?req.candidate_hash(),
266			session = req.session(),
267			"Forking participation"
268		);
269
270		let participation_timer = self.metrics.time_participation();
271		if self.running_participations.insert(*req.candidate_hash()) {
272			let sender = ctx.sender().clone();
273			ctx.spawn(
274				"participation-worker",
275				participate(
276					self.worker_sender.clone(),
277					sender,
278					recent_head,
279					req,
280					participation_timer,
281				)
282				.boxed(),
283			)
284			.map_err(FatalError::SpawnFailed)?;
285		}
286		Ok(())
287	}
288}

290async fn participate(
291	mut result_sender: WorkerMessageSender,
292	mut sender: impl overseer::DisputeCoordinatorSenderTrait,
293	block_hash: Hash,
294	req: ParticipationRequest, // Sends metric data via request_timer field when dropped
295	_participation_timer: Option<prometheus::HistogramTimer>, // Sends metric data when dropped
296) {
297	#[cfg(test)]
298	// Hack for tests, so we get recovery messages not too early.
299	Delay::new(Duration::from_millis(100)).await;
300	// in order to validate a candidate we need to start by recovering the
301	// available data
302	let (recover_available_data_tx, recover_available_data_rx) = oneshot::channel();
303	sender
304		.send_message(AvailabilityRecoveryMessage::RecoverAvailableData(
305			req.candidate_receipt().clone(),
306			req.session(),
307			None,
308			None,
309			recover_available_data_tx,
310		))
311		.await;
312
313	let available_data = match recover_available_data_rx.await {
314		Err(oneshot::Canceled) => {
315			gum::warn!(
316				target: LOG_TARGET,
317				"`Oneshot` got cancelled when recovering available data {:?}",
318				req.candidate_hash(),
319			);
320			send_result(&mut result_sender, req, ParticipationOutcome::Error).await;
321			return
322		},
323		Ok(Ok(data)) => data,
324		Ok(Err(RecoveryError::Invalid)) => {
325			gum::debug!(
326				target: LOG_TARGET,
327				candidate_hash = ?req.candidate_hash(),
328				session = req.session(),
329				"Invalid availability data during participation"
330			);
331			// the available data was recovered but it is invalid, therefore we'll
332			// vote negatively for the candidate dispute
333			send_result(&mut result_sender, req, ParticipationOutcome::Invalid).await;
334			return
335		},
336		Ok(Err(RecoveryError::Unavailable)) | Ok(Err(RecoveryError::ChannelClosed)) => {
337			gum::debug!(
338				target: LOG_TARGET,
339				candidate_hash = ?req.candidate_hash(),
340				session = req.session(),
341				"Can't fetch availability data in participation"
342			);
343			send_result(&mut result_sender, req, ParticipationOutcome::Unavailable).await;
344			return
345		},
346	};
347
348	// we also need to fetch the validation code which we can reference by its
349	// hash as taken from the candidate descriptor
350	let validation_code = match get_validation_code_by_hash(
351		&mut sender,
352		block_hash,
353		req.candidate_receipt().descriptor.validation_code_hash(),
354	)
355	.await
356	{
357		Ok(Some(code)) => code,
358		Ok(None) => {
359			gum::warn!(
360				target: LOG_TARGET,
361				"Validation code unavailable for code hash {:?} in the state of block {:?}",
362				req.candidate_receipt().descriptor.validation_code_hash(),
363				block_hash,
364			);
365
366			send_result(&mut result_sender, req, ParticipationOutcome::Error).await;
367			return
368		},
369		Err(err) => {
370			gum::warn!(target: LOG_TARGET, ?err, "Error when fetching validation code.");
371			send_result(&mut result_sender, req, ParticipationOutcome::Error).await;
372			return
373		},
374	};
375
376	// Issue a request to validate the candidate with the provided exhaustive
377	// parameters
378	//
379	// We use the approval execution timeout because this is intended to
380	// be run outside of backing and therefore should be subject to the
381	// same level of leeway.
382	let (validation_tx, validation_rx) = oneshot::channel();
383	sender
384		.send_message(CandidateValidationMessage::ValidateFromExhaustive {
385			validation_data: available_data.validation_data,
386			validation_code,
387			candidate_receipt: req.candidate_receipt().clone(),
388			pov: available_data.pov,
389			executor_params: req.executor_params(),
390			exec_kind: PvfExecKind::Dispute,
391			response_sender: validation_tx,
392		})
393		.await;
394
395	// we cast votes (either positive or negative) depending on the outcome of
396	// the validation and if valid, whether the commitments hash matches
397	match validation_rx.await {
398		Err(oneshot::Canceled) => {
399			gum::warn!(
400				target: LOG_TARGET,
401				"`Oneshot` got cancelled when validating candidate {:?}",
402				req.candidate_hash(),
403			);
404			send_result(&mut result_sender, req, ParticipationOutcome::Error).await;
405			return
406		},
407		Ok(Err(err)) => {
408			gum::warn!(
409				target: LOG_TARGET,
410				"Candidate {:?} validation failed with: {:?}",
411				req.candidate_hash(),
412				err,
413			);
414
415			send_result(&mut result_sender, req, ParticipationOutcome::Error).await;
416		},
417
418		Ok(Ok(ValidationResult::Invalid(invalid))) => {
419			gum::warn!(
420				target: LOG_TARGET,
421				"Candidate {:?} considered invalid: {:?}",
422				req.candidate_hash(),
423				invalid,
424			);
425
426			send_result(&mut result_sender, req, ParticipationOutcome::Invalid).await;
427		},
428		Ok(Ok(ValidationResult::Valid(_, _))) => {
429			send_result(&mut result_sender, req, ParticipationOutcome::Valid).await;
430		},
431	}
432}

434/// Helper function for sending the result back and report any error.
435async fn send_result(
436	sender: &mut WorkerMessageSender,
437	req: ParticipationRequest,
438	outcome: ParticipationOutcome,
439) {
440	if let Err(err) = sender.feed(WorkerMessage::from_request(req, outcome)).await {
441		gum::error!(
442			target: LOG_TARGET,
443			?err,
444			"Sending back participation result failed. Dispute coordinator not working properly!"
445		);
446	}

17use polkadot_primitives::{BlockNumber, CandidateHash};
18use std::collections::{BTreeMap, HashMap, HashSet};
19
20/// Keeps `CandidateHash` in reference counted way.
21/// Each `insert` saves a value with `reference count == 1` or increases the reference
22/// count if the value already exists.
23/// Each `remove` decreases the reference count for the corresponding `CandidateHash`.
24/// If the reference count reaches 0 - the value is removed.
25struct RefCountedCandidates {
26	candidates: HashMap<CandidateHash, usize>,
27}
28
29impl RefCountedCandidates {
30	pub fn new() -> Self {
31		Self { candidates: HashMap::new() }
32	}
33	// If `CandidateHash` doesn't exist in the `HashMap` it is created and its reference
34	// count is set to 1.
35	// If `CandidateHash` already exists in the `HashMap` its reference count is increased.
36	pub fn insert(&mut self, candidate: CandidateHash) {
37		*self.candidates.entry(candidate).or_default() += 1;
38	}
39
40	// If a `CandidateHash` with reference count equals to 1 is about to be removed - the
41	// candidate is dropped from the container too.
42	// If a `CandidateHash` with reference count bigger than 1 is about to be removed - the
43	// reference count is decreased and the candidate remains in the container.
44	pub fn remove(&mut self, candidate: &CandidateHash) {
45		match self.candidates.get_mut(candidate) {
46			Some(v) if *v > 1 => *v -= 1,
47			Some(v) => {
48				assert!(*v == 1);
49				self.candidates.remove(candidate);
50			},
51			None => {},
52		}
53	}
54
55	pub fn contains(&self, candidate: &CandidateHash) -> bool {
56		self.candidates.contains_key(&candidate)
57	}
58}

60#[cfg(test)]
61mod ref_counted_candidates_tests {
62	use super::*;
63	use polkadot_primitives::{BlakeTwo256, HashT};
64
65	#[test]
66	fn element_is_removed_when_refcount_reaches_zero() {
67		let mut container = RefCountedCandidates::new();
68
69		let zero = CandidateHash(BlakeTwo256::hash(&vec![0]));
70		let one = CandidateHash(BlakeTwo256::hash(&vec![1]));
71		// add two separate candidates
72		container.insert(zero); // refcount == 1
73		container.insert(one);
74
75		// and increase the reference count for the first
76		container.insert(zero); // refcount == 2
77
78		assert!(container.contains(&zero));
79		assert!(container.contains(&one));
80
81		// remove once -> refcount == 1
82		container.remove(&zero);
83		assert!(container.contains(&zero));
84		assert!(container.contains(&one));
85
86		// remove once -> refcount == 0
87		container.remove(&zero);
88		assert!(!container.contains(&zero));
89		assert!(container.contains(&one));
90
91		// remove the other element
92		container.remove(&one);
93		assert!(!container.contains(&zero));
94		assert!(!container.contains(&one));
95	}
96}

98/// Keeps track of scraped candidates. Supports `insert`, `remove_up_to_height` and `contains`
99/// operations.
100pub struct ScrapedCandidates {
101	/// Main data structure which keeps the candidates we know about. `contains` does lookups only
102	/// here.
103	candidates: RefCountedCandidates,
104	/// Keeps track at which block number a candidate was inserted. Used in `remove_up_to_height`.
105	/// Without this tracking we won't be able to remove all candidates before block X.
106	candidates_by_block_number: BTreeMap<BlockNumber, HashSet<CandidateHash>>,
107}
108
109impl ScrapedCandidates {
110	pub fn new() -> Self {
111		Self {
112			candidates: RefCountedCandidates::new(),
113			candidates_by_block_number: BTreeMap::new(),
114		}
115	}
116
117	pub fn contains(&self, candidate_hash: &CandidateHash) -> bool {
118		self.candidates.contains(candidate_hash)
119	}
120
121	// Removes all candidates up to a given height. The candidates at the block height are NOT
122	// removed.
123	pub fn remove_up_to_height(&mut self, height: &BlockNumber) {
124		let not_stale = self.candidates_by_block_number.split_off(&height);
125		let stale = std::mem::take(&mut self.candidates_by_block_number);
126		self.candidates_by_block_number = not_stale;
127		for candidate in stale.values().flatten() {
128			self.candidates.remove(candidate);
129		}
130	}
131
132	pub fn insert(&mut self, block_number: BlockNumber, candidate_hash: CandidateHash) {
133		if self
134			.candidates_by_block_number
135			.entry(block_number)
136			.or_default()
137			.insert(candidate_hash)
138		{
139			self.candidates.insert(candidate_hash);
140		}
141	}
142
143	// Used only for tests to verify the pruning doesn't leak data.
144	#[cfg(test)]
145	pub fn candidates_by_block_number_is_empty(&self) -> bool {
146		self.candidates_by_block_number.is_empty()
147	}
148}

150#[cfg(test)]
151mod scraped_candidates_tests {
152	use super::*;
153	use polkadot_primitives::{BlakeTwo256, HashT};
154
155	#[test]
156	fn stale_candidates_are_removed() {
157		let mut candidates = ScrapedCandidates::new();
158		let target = CandidateHash(BlakeTwo256::hash(&vec![1, 2, 3]));
159		candidates.insert(1, target);
160		// Repeated inserts at same height should be fine:
161		candidates.insert(1, target);
162
163		assert!(candidates.contains(&target));
164
165		candidates.remove_up_to_height(&2);
166		assert!(!candidates.contains(&target));
167		assert!(candidates.candidates_by_block_number_is_empty());
168	}

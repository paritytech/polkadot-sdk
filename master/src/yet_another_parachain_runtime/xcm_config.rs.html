55/// Type for specifying how a `Location` can be converted into an `AccountId`. This is used
56/// when determining ownership of accounts for asset transacting and when attempting to use XCM
57/// `Transact` in order to determine the dispatch Origin.
58pub type LocationToAccountId = (
59	// The parent (Relay-chain) origin converts to the parent `AccountId`.
60	ParentIsPreset<AccountId>,
61	// Sibling parachain origins convert to AccountId via the `ParaId::into`.
62	SiblingParachainConvertsVia<Sibling, AccountId>,
63	// Straight up local `AccountId32` origins just alias directly to `AccountId`.
64	AccountId32Aliases<RelayNetwork, AccountId>,
65);

67/// Means for transacting assets on this chain.
68pub type LocalAssetTransactor = FungibleAdapter<
69	// Use this currency:
70	Balances,
71	// Use this currency when it is a fungible asset matching the given location or name:
72	IsConcrete<RelayLocation>,
73	// Do a simple punn to convert an AccountId32 Location into a native chain account ID:
74	LocationToAccountId,
75	// Our chain's account ID type (we can't get away without mentioning it explicitly):
76	AccountId,
77	// We don't track any teleports.
78	(),
79>;

81/// This is the type we use to convert an (incoming) XCM origin into a local `Origin` instance,
82/// ready for dispatching a transaction with Xcm's `Transact`. There is an `OriginKind` which can
83/// biases the kind of local `Origin` it will become.
84pub type XcmOriginToTransactDispatchOrigin = (
85	// Sovereign account converter; this attempts to derive an `AccountId` from the origin location
86	// using `LocationToAccountId` and then turn that into the usual `Signed` origin. Useful for
87	// foreign chains who want to have a local sovereign account on this chain which they control.
88	SovereignSignedViaLocation<LocationToAccountId, RuntimeOrigin>,
89	// Native converter for Relay-chain (Parent) location; will convert to a `Relay` origin when
90	// recognized.
91	RelayChainAsNative<RelayChainOrigin, RuntimeOrigin>,
92	// Native converter for sibling Parachains; will convert to a `SiblingPara` origin when
93	// recognized.
94	SiblingParachainAsNative<cumulus_pallet_xcm::Origin, RuntimeOrigin>,
95	// Native signed account converter; this just converts an `AccountId32` origin into a normal
96	// `RuntimeOrigin::Signed` origin of the same 32-byte value.
97	SignedAccountId32AsNative<RelayNetwork, RuntimeOrigin>,
98	// Xcm origins can be represented natively under the Xcm pallet's Xcm origin.
99	XcmPassthrough<RuntimeOrigin>,
100);

109pub struct ParentOrParentsExecutivePlurality;
110impl Contains<Location> for ParentOrParentsExecutivePlurality {
111	fn contains(location: &Location) -> bool {
112		matches!(location.unpack(), (1, []) | (1, [Plurality { id: BodyId::Executive, .. }]))
113	}
114}

116pub type Barrier = TrailingSetTopicAsId<
117	DenyThenTry<
118		DenyReserveTransferToRelayChain,
119		(
120			TakeWeightCredit,
121			WithComputedOrigin<
122				(
123					AllowTopLevelPaidExecutionFrom<Everything>,
124					AllowExplicitUnpaidExecutionFrom<ParentOrParentsExecutivePlurality>,
125					// ^^^ Parent and its exec plurality get free execution
126				),
127				UniversalLocation,
128				ConstU32<8>,
129			>,
130		),
131	>,
132>;

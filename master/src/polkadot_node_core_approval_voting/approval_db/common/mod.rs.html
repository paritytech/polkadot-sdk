37const STORED_BLOCKS_KEY: &[u8] = b"Approvals_StoredBlocks";

39/// A range from earliest..last block number stored within the DB.
40#[derive(Encode, Decode, Debug, Clone, PartialEq)]
41pub struct StoredBlockRange(pub BlockNumber, pub BlockNumber);
42/// The database config.

49/// `DbBackend` is a concrete implementation of the higher-level Backend trait
50pub struct DbBackend {
51	inner: Arc<dyn Database>,
52	config: Config,
53}
54
55impl DbBackend {
56	/// Create a new [`DbBackend`] with the supplied key-value store and
57	/// config.
58	pub fn new(db: Arc<dyn Database>, config: Config) -> Self {
59		DbBackend { inner: db, config }
60	}
61}

63/// Errors while accessing things from the DB.
64#[derive(Debug, derive_more::From, derive_more::Display)]
65pub enum Error {
66	Io(std::io::Error),
67	InvalidDecoding(codec::Error),
68	InternalError(SubsystemError),
69}
70
71impl std::error::Error for Error {}
72
73/// Result alias for DB errors.
74pub type Result<T> = std::result::Result<T, Error>;

76impl Backend for DbBackend {
77	fn load_block_entry(
78		&self,
79		block_hash: &Hash,
80	) -> SubsystemResult<Option<persisted_entries::BlockEntry>> {
81		load_block_entry(&*self.inner, &self.config, block_hash).map(|e| e.map(Into::into))
82	}
83
84	fn load_candidate_entry(
85		&self,

87	) -> SubsystemResult<Option<persisted_entries::CandidateEntry>> {
88		load_candidate_entry(&*self.inner, &self.config, candidate_hash).map(|e| e.map(Into::into))
89	}
90
91	fn load_blocks_at_height(&self, block_height: &BlockNumber) -> SubsystemResult<Vec<Hash>> {
92		load_blocks_at_height(&*self.inner, &self.config, block_height)
93	}
94
95	fn load_all_blocks(&self) -> SubsystemResult<Vec<Hash>> {
96		load_all_blocks(&*self.inner, &self.config)
97	}
98
99	fn load_stored_blocks(&self) -> SubsystemResult<Option<StoredBlockRange>> {
100		load_stored_blocks(&*self.inner, &self.config)
101	}
102
103	/// Atomically write the list of operations, with later operations taking precedence over prior.
104	fn write<I>(&mut self, ops: I) -> SubsystemResult<()>
105	where
106		I: IntoIterator<Item = BackendWriteOp>,
107	{
108		let mut tx = DBTransaction::new();
109		for op in ops {
110			match op {
111				BackendWriteOp::WriteStoredBlockRange(stored_block_range) => {
112					tx.put_vec(
113						self.config.col_approval_data,
114						&STORED_BLOCKS_KEY,
115						stored_block_range.encode(),
116					);
117				},
118				BackendWriteOp::DeleteStoredBlockRange => {
119					tx.delete(self.config.col_approval_data, &STORED_BLOCKS_KEY);
120				},
121				BackendWriteOp::WriteBlocksAtHeight(h, blocks) => {
122					tx.put_vec(
123						self.config.col_approval_data,
124						&blocks_at_height_key(h),
125						blocks.encode(),
126					);
127				},
128				BackendWriteOp::DeleteBlocksAtHeight(h) => {
129					tx.delete(self.config.col_approval_data, &blocks_at_height_key(h));
130				},
131				BackendWriteOp::WriteBlockEntry(block_entry) => {
132					let block_entry: BlockEntry = block_entry.into();
133					tx.put_vec(
134						self.config.col_approval_data,
135						&block_entry_key(&block_entry.block_hash),
136						block_entry.encode(),
137					);
138				},
139				BackendWriteOp::DeleteBlockEntry(hash) => {
140					tx.delete(self.config.col_approval_data, &block_entry_key(&hash));
141				},
142				BackendWriteOp::WriteCandidateEntry(candidate_entry) => {
143					let candidate_entry: CandidateEntry = candidate_entry.into();
144					tx.put_vec(
145						self.config.col_approval_data,
146						&candidate_entry_key(&candidate_entry.candidate.hash()),
147						candidate_entry.encode(),
148					);
149				},
150				BackendWriteOp::DeleteCandidateEntry(candidate_hash) => {
151					tx.delete(self.config.col_approval_data, &candidate_entry_key(&candidate_hash));
152				},
153			}
154		}
155
156		self.inner.write(tx).map_err(|e| e.into())
157	}
158}

196pub(crate) fn load_decode<D: Decode>(
197	store: &dyn Database,
198	col_approval_data: u32,
199	key: &[u8],
200) -> Result<Option<D>> {
201	match store.get(col_approval_data, key)? {
202		None => Ok(None),
203		Some(raw) => D::decode(&mut &raw[..]).map(Some).map_err(Into::into),
204	}
205}

207/// The key a given block entry is stored under.
208pub(crate) fn block_entry_key(block_hash: &Hash) -> [u8; 46] {
209	const BLOCK_ENTRY_PREFIX: [u8; 14] = *b"Approvals_blck";
210
211	let mut key = [0u8; 14 + 32];
212	key[0..14].copy_from_slice(&BLOCK_ENTRY_PREFIX);
213	key[14..][..32].copy_from_slice(block_hash.as_ref());
214
215	key
216}
217
218/// The key a given candidate entry is stored under.
219pub(crate) fn candidate_entry_key(candidate_hash: &CandidateHash) -> [u8; 46] {
220	const CANDIDATE_ENTRY_PREFIX: [u8; 14] = *b"Approvals_cand";
221
222	let mut key = [0u8; 14 + 32];
223	key[0..14].copy_from_slice(&CANDIDATE_ENTRY_PREFIX);
224	key[14..][..32].copy_from_slice(candidate_hash.0.as_ref());
225
226	key
227}
228
229/// The key a set of block hashes corresponding to a block number is stored under.
230pub(crate) fn blocks_at_height_key(block_number: BlockNumber) -> [u8; 16] {
231	const BLOCKS_AT_HEIGHT_PREFIX: [u8; 12] = *b"Approvals_at";
232
233	let mut key = [0u8; 12 + 4];
234	key[0..12].copy_from_slice(&BLOCKS_AT_HEIGHT_PREFIX);
235	block_number.using_encoded(|s| key[12..16].copy_from_slice(s));
236
237	key
238}

240/// Return all blocks which have entries in the DB, ascending, by height.
241pub fn load_all_blocks(store: &dyn Database, config: &Config) -> SubsystemResult<Vec<Hash>> {
242	let mut hashes = Vec::new();
243	if let Some(stored_blocks) = load_stored_blocks(store, config)? {
244		for height in stored_blocks.0..stored_blocks.1 {
245			let blocks = load_blocks_at_height(store, config, &height)?;
246			hashes.extend(blocks);
247		}
248	}
249
250	Ok(hashes)
251}
252
253/// Load the stored-blocks key from the state.
254pub fn load_stored_blocks(
255	store: &dyn Database,
256	config: &Config,
257) -> SubsystemResult<Option<StoredBlockRange>> {
258	load_decode(store, config.col_approval_data, STORED_BLOCKS_KEY)
259		.map_err(|e| SubsystemError::with_origin("approval-voting", e))
260}
261
262/// Load a blocks-at-height entry for a given block number.
263pub fn load_blocks_at_height(
264	store: &dyn Database,
265	config: &Config,
266	block_number: &BlockNumber,
267) -> SubsystemResult<Vec<Hash>> {
268	load_decode(store, config.col_approval_data, &blocks_at_height_key(*block_number))
269		.map(|x| x.unwrap_or_default())
270		.map_err(|e| SubsystemError::with_origin("approval-voting", e))
271}
272
273/// Load a block entry from the aux store.
274pub fn load_block_entry(
275	store: &dyn Database,
276	config: &Config,
277	block_hash: &Hash,
278) -> SubsystemResult<Option<BlockEntry>> {
279	load_decode(store, config.col_approval_data, &block_entry_key(block_hash))
280		.map(|u: Option<BlockEntry>| u.map(|v| v.into()))
281		.map_err(|e| SubsystemError::with_origin("approval-voting", e))
282}

284/// Load a candidate entry from the aux store in current version format.
285pub fn load_candidate_entry(
286	store: &dyn Database,
287	config: &Config,
288	candidate_hash: &CandidateHash,
289) -> SubsystemResult<Option<CandidateEntry>> {
290	load_decode(store, config.col_approval_data, &candidate_entry_key(candidate_hash))
291		.map(|u: Option<CandidateEntry>| u.map(|v| v.into()))
292		.map_err(|e| SubsystemError::with_origin("approval-voting", e))

32/// Migrates `BlockEntry`, `CandidateEntry`, `ApprovalEntry` and `OurApproval` to version 3.
33/// Returns on any error.
34/// Must only be used in parachains DB migration code - `polkadot-service` crate.
35pub fn v2_to_latest(db: Arc<dyn Database>, config: Config) -> Result<()> {
36	let mut backend = crate::DbBackend::new(db, config);
37	let all_blocks = backend
38		.load_all_blocks()
39		.map_err(|e| Error::InternalError(e))?
40		.iter()
41		.filter_map(|block_hash| {
42			backend
43				.load_block_entry_v2(block_hash)
44				.map_err(|e| Error::InternalError(e))
45				.ok()?
46		})
47		.collect::<Vec<_>>();
48
49	gum::info!(
50		target: crate::LOG_TARGET,
51		"Migrating candidate entries on top of {} blocks",
52		all_blocks.len()
53	);
54
55	let mut overlay = crate::OverlayedBackend::new(&backend);
56	let mut counter = 0;
57	// Get all candidate entries, approval entries and convert each of them.
58	for block in all_blocks {
59		for (candidate_index, (_core_index, candidate_hash)) in
60			block.candidates().iter().enumerate()
61		{
62			// Loading the candidate will also perform the conversion to the updated format and
63			// return that representation.
64			if let Some(candidate_entry) = backend
65				.load_candidate_entry_v2(&candidate_hash, candidate_index as CandidateIndex)
66				.map_err(|e| Error::InternalError(e))?
67			{
68				// Write the updated representation.
69				overlay.write_candidate_entry(candidate_entry);
70				counter += 1;
71			}
72		}
73		overlay.write_block_entry(block);
74	}
75
76	gum::info!(target: crate::LOG_TARGET, "Migrated {} entries", counter);
77
78	// Commit all changes to DB.
79	let write_ops = overlay.into_write_ops();
80	backend.write(write_ops).unwrap();
81
82	Ok(())
83}

85// Checks if the migration doesn't leave the DB in an unsane state.
86// This function is to be used in tests.
87pub fn v1_to_latest_sanity_check(
88	db: Arc<dyn Database>,
89	config: Config,
90	expected_candidates: HashSet<CandidateHash>,
91) -> Result<()> {
92	let backend = crate::DbBackend::new(db, config);
93
94	let all_blocks = backend
95		.load_all_blocks()
96		.unwrap()
97		.iter()
98		.map(|block_hash| backend.load_block_entry(block_hash).unwrap().unwrap())
99		.collect::<Vec<_>>();
100
101	let mut candidates = HashSet::new();
102
103	// Iterate all blocks and approval entries.
104	for block in all_blocks {
105		for (_core_index, candidate_hash) in block.candidates() {
106			// Loading the candidate will also perform the conversion to the updated format and
107			// return that representation.
108			if let Some(candidate_entry) = backend.load_candidate_entry(&candidate_hash).unwrap() {
109				candidates.insert(candidate_entry.candidate.hash());
110			}
111		}
112	}
113
114	assert_eq!(candidates, expected_candidates);
115
116	Ok(())
117}

119// Fills the db with dummy data in v2 scheme.
120pub fn v2_fill_test_data<F>(
121	db: Arc<dyn Database>,
122	config: Config,
123	dummy_candidate_create: F,
124) -> Result<HashSet<CandidateHash>>
125where
126	F: Fn(H256) -> CandidateReceipt<H256>,
127{
128	let mut backend = crate::DbBackend::new(db.clone(), config);
129	let mut overlay_db = crate::OverlayedBackend::new(&backend);
130	let mut expected_candidates = HashSet::new();
131
132	const RELAY_BLOCK_COUNT: u32 = 10;
133
134	let range = StoredBlockRange(1, 11);
135	overlay_db.write_stored_block_range(range.clone());
136
137	for relay_number in 1..=RELAY_BLOCK_COUNT {
138		let relay_hash = Hash::repeat_byte(relay_number as u8);
139		let assignment_core_index = CoreIndex(relay_number);
140		let candidate = dummy_candidate_create(relay_hash);
141		let candidate_hash = candidate.hash();
142
143		let at_height = vec![relay_hash];
144
145		let block_entry = make_block_entry_v2(
146			relay_hash,
147			Default::default(),
148			relay_number,
149			vec![(assignment_core_index, candidate_hash)],
150		);
151
152		let dummy_assignment = crate::approval_db::v2::OurAssignment {
153			cert: dummy_assignment_cert(AssignmentCertKind::RelayVRFModulo { sample: 0 }).into(),
154			tranche: 0,
155			validator_index: ValidatorIndex(0),
156			triggered: false,
157		};
158
159		let candidate_entry = crate::approval_db::v2::CandidateEntry {
160			candidate,
161			session: 123,
162			block_assignments: vec![(
163				relay_hash,
164				crate::approval_db::v2::ApprovalEntry {
165					tranches: Vec::new(),
166					backing_group: GroupIndex(1),
167					our_assignment: Some(dummy_assignment),
168					our_approval_sig: None,
169					approved: false,
170					assigned_validators: make_bitvec(1),
171				},
172			)]
173			.into_iter()
174			.collect(),
175			approvals: Default::default(),
176		};
177
178		overlay_db.write_blocks_at_height(relay_number, at_height.clone());
179		expected_candidates.insert(candidate_entry.candidate.hash());
180
181		db.write(write_candidate_entry_v2(candidate_entry, config)).unwrap();
182		db.write(write_block_entry_v2(block_entry, config)).unwrap();
183	}
184
185	let write_ops = overlay_db.into_write_ops();
186	backend.write(write_ops).unwrap();
187
188	Ok(expected_candidates)
189}

191fn make_block_entry_v2(
192	block_hash: Hash,
193	parent_hash: Hash,
194	block_number: BlockNumber,
195	candidates: Vec<(CoreIndex, CandidateHash)>,
196) -> crate::approval_db::v2::BlockEntry {
197	crate::approval_db::v2::BlockEntry {
198		block_hash,
199		parent_hash,
200		block_number,
201		session: 1,
202		slot: Slot::from(1),
203		relay_vrf_story: [0u8; 32],
204		approved_bitfield: make_bitvec(candidates.len()),
205		distributed_assignments: make_bitvec(candidates.len()),
206		candidates,
207		children: Vec::new(),
208	}
209}

211// Low level DB helper to write a candidate entry in v1 scheme.
212fn write_candidate_entry_v2(
213	candidate_entry: crate::approval_db::v2::CandidateEntry,
214	config: Config,
215) -> DBTransaction {
216	let mut tx = DBTransaction::new();
217	tx.put_vec(
218		config.col_approval_data,
219		&candidate_entry_key(&candidate_entry.candidate.hash()),
220		candidate_entry.encode(),
221	);
222	tx
223}

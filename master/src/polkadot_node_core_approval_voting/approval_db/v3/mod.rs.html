46/// Metadata about our approval signature
47#[derive(Encode, Decode, Debug, Clone, PartialEq)]
48pub struct OurApproval {
49	/// The signature for the candidates hashes pointed by indices.
50	pub signature: ValidatorSignature,
51	/// The indices of the candidates signed in this approval.
52	pub signed_candidates_indices: CandidateBitfield,
53}
54
55/// Metadata regarding approval of a particular candidate within the context of some
56/// particular block.
57#[derive(Encode, Decode, Debug, Clone, PartialEq)]
58pub struct ApprovalEntry {
59	pub tranches: Vec<TrancheEntry>,
60	pub backing_group: GroupIndex,
61	pub our_assignment: Option<OurAssignment>,
62	pub our_approval_sig: Option<OurApproval>,
63	// `n_validators` bits.
64	pub assigned_validators: Bitfield,
65	pub approved: bool,
66}
67
68/// Metadata regarding approval of a particular candidate.
69#[derive(Encode, Decode, Debug, Clone, PartialEq)]
70pub struct CandidateEntry {
71	pub candidate: CandidateReceipt,
72	pub session: SessionIndex,
73	// Assignments are based on blocks, so we need to track assignments separately
74	// based on the block we are looking at.
75	pub block_assignments: BTreeMap<Hash, ApprovalEntry>,
76	pub approvals: Bitfield,
77}
78
79/// Metadata regarding approval of a particular block, by way of approval of the
80/// candidates contained within it.
81#[derive(Encode, Decode, Debug, Clone, PartialEq)]
82pub struct BlockEntry {
83	pub block_hash: Hash,
84	pub block_number: BlockNumber,
85	pub parent_hash: Hash,
86	pub session: SessionIndex,
87	pub slot: Slot,
88	/// Random bytes derived from the VRF submitted within the block by the block
89	/// author as a credential and used as input to approval assignment criteria.
90	pub relay_vrf_story: [u8; 32],
91	// The candidates included as-of this block and the index of the core they are
92	// leaving. Sorted ascending by core index.
93	pub candidates: Vec<(CoreIndex, CandidateHash)>,
94	// A bitfield where the i'th bit corresponds to the i'th candidate in `candidates`.
95	// The i'th bit is `true` iff the candidate has been approved in the context of this
96	// block. The block can be considered approved if the bitfield has all bits set to `true`.
97	pub approved_bitfield: Bitfield,
98	pub children: Vec<Hash>,
99	// A list of candidates we have checked, but didn't not sign and
100	// advertise the vote yet.
101	pub candidates_pending_signature: BTreeMap<CandidateIndex, CandidateSigningContext>,
102	// Assignments we already distributed. A 1 bit means the candidate index for which
103	// we already have sent out an assignment. We need this to avoid distributing
104	// multiple core assignments more than once.
105	pub distributed_assignments: Bitfield,
106}
107
108#[derive(Encode, Decode, Debug, Clone, PartialEq)]
109/// Context needed for creating an approval signature for a given candidate.
110pub struct CandidateSigningContext {
111	/// The candidate hash, to be included in the signature.
112	pub candidate_hash: CandidateHash,
113	/// The latest tick we have to create and send the approval.
114	pub sign_no_later_than_tick: Tick,
115}

117/// Load a candidate entry from the aux store in v2 format.
118pub fn load_candidate_entry_v2(
119	store: &dyn Database,
120	config: &Config,
121	candidate_hash: &CandidateHash,
122) -> SubsystemResult<Option<super::v2::CandidateEntry>> {
123	load_decode(store, config.col_approval_data, &candidate_entry_key(candidate_hash))
124		.map(|u: Option<super::v2::CandidateEntry>| u.map(|v| v.into()))
125		.map_err(|e| SubsystemError::with_origin("approval-voting", e))
126}
127
128/// Load a block entry from the aux store in v2 format.
129pub fn load_block_entry_v2(
130	store: &dyn Database,
131	config: &Config,
132	block_hash: &Hash,
133) -> SubsystemResult<Option<super::v2::BlockEntry>> {
134	load_decode(store, config.col_approval_data, &block_entry_key(block_hash))
135		.map(|u: Option<super::v2::BlockEntry>| u.map(|v| v.into()))
136		.map_err(|e| SubsystemError::with_origin("approval-voting", e))
137}

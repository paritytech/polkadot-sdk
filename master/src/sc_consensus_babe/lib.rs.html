19//! # BABE (Blind Assignment for Blockchain Extension)
20//!
21//! BABE is a slot-based block production mechanism which uses a VRF PRNG to
22//! randomly perform the slot allocation. On every slot, all the authorities
23//! generate a new random number with the VRF function and if it is lower than a
24//! given threshold (which is proportional to their weight/stake) they have a
25//! right to produce a block. The proof of the VRF function execution will be
26//! used by other peer to validate the legitimacy of the slot claim.
27//!
28//! The engine is also responsible for collecting entropy on-chain which will be
29//! used to seed the given VRF PRNG. An epoch is a contiguous number of slots
30//! under which we will be using the same authority set. During an epoch all VRF
31//! outputs produced as a result of block production will be collected on an
32//! on-chain randomness pool. Epoch changes are announced one epoch in advance,
33//! i.e. when ending epoch N, we announce the parameters (randomness,
34//! authorities, etc.) for epoch N+2.
35//!
36//! Since the slot assignment is randomized, it is possible that a slot is
37//! assigned to multiple validators in which case we will have a temporary fork,
38//! or that a slot is assigned to no validator in which case no block is
39//! produced. Which means that block times are not deterministic.
40//!
41//! The protocol has a parameter `c` [0, 1] for which `1 - c` is the probability
42//! of a slot being empty. The choice of this parameter affects the security of
43//! the protocol relating to maximum tolerable network delays.
44//!
45//! In addition to the VRF-based slot assignment described above, which we will
46//! call primary slots, the engine also supports a deterministic secondary slot
47//! assignment. Primary slots take precedence over secondary slots, when
48//! authoring the node starts by trying to claim a primary slot and falls back
49//! to a secondary slot claim attempt. The secondary slot assignment is done
50//! by picking the authority at index:
51//!
52//! `blake2_256(epoch_randomness ++ slot_number) % authorities_len`.
53//!
54//! The secondary slots supports either a `SecondaryPlain` or `SecondaryVRF`
55//! variant. Comparing with `SecondaryPlain` variant, the `SecondaryVRF` variant
56//! generates an additional VRF output. The output is not included in beacon
57//! randomness, but can be consumed by parachains.
58//!
59//! The fork choice rule is weight-based, where weight equals the number of
60//! primary blocks in the chain. We will pick the heaviest chain (more primary
61//! blocks) and will go with the longest one in case of a tie.
62//!
63//! An in-depth description and analysis of the protocol can be found here:
64//! <https://research.web3.foundation/Polkadot/protocols/block-production/Babe>

213impl Epoch {
214	/// Create the genesis epoch (epoch #0).
215	///
216	/// This is defined to start at the slot of the first block, so that has to be provided.
217	pub fn genesis(genesis_config: &BabeConfiguration, slot: Slot) -> Epoch {
218		sp_consensus_babe::Epoch {
219			epoch_index: 0,
220			start_slot: slot,
221			duration: genesis_config.epoch_length,
222			authorities: genesis_config.authorities.clone(),
223			randomness: genesis_config.randomness,
224			config: BabeEpochConfiguration {
225				c: genesis_config.c,
226				allowed_slots: genesis_config.allowed_slots,
227			},
228		}
229		.into()
230	}
231
232	/// Clone and tweak epoch information to refer to the specified slot.
233	///
234	/// All the information which depends on the slot value is recomputed and assigned
235	/// to the returned epoch instance.
236	///
237	/// The `slot` must be greater than or equal the original epoch start slot,
238	/// if is less this operation is equivalent to a simple clone.
239	pub fn clone_for_slot(&self, slot: Slot) -> Epoch {
240		let mut epoch = self.clone();
241
242		let skipped_epochs = *slot.saturating_sub(self.start_slot) / self.duration;
243
244		let epoch_index = epoch.epoch_index.checked_add(skipped_epochs).expect(
245			"epoch number is u64; it should be strictly smaller than number of slots; \
246				slots relate in some way to wall clock time; \
247				if u64 is not enough we should crash for safety; qed.",
248		);
249
250		let start_slot = skipped_epochs
251			.checked_mul(epoch.duration)
252			.and_then(|skipped_slots| epoch.start_slot.checked_add(skipped_slots))
253			.expect(
254				"slot number is u64; it should relate in some way to wall clock time; \
255				 if u64 is not enough we should crash for safety; qed.",
256			);
257
258		epoch.epoch_index = epoch_index;
259		epoch.start_slot = Slot::from(start_slot);
260
261		epoch
262	}

380/// Read configuration from the runtime state at current best block.
381pub fn configuration<B: BlockT, C>(client: &C) -> ClientResult<BabeConfiguration>
382where
383	C: AuxStore + ProvideRuntimeApi<B> + UsageProvider<B>,
384	C::Api: BabeApi<B>,
385{
386	let at_hash = if client.usage_info().chain.finalized_state.is_some() {
387		client.usage_info().chain.best_hash
388	} else {
389		debug!(target: LOG_TARGET, "No finalized state is available. Reading config from genesis");
390		client.usage_info().chain.genesis_hash
391	};
392
393	let runtime_api = client.runtime_api();
394	let version = runtime_api.api_version::<dyn BabeApi<B>>(at_hash)?;
395
396	let config = match version {
397		Some(1) => {
398			#[allow(deprecated)]
399			{
400				runtime_api.configuration_before_version_2(at_hash)?.into()
401			}
402		},
403		Some(2) => runtime_api.configuration(at_hash)?,
404		_ =>
405			return Err(sp_blockchain::Error::VersionInvalid(
406				"Unsupported or invalid BabeApi version".to_string(),
407			)),
408	};
409	Ok(config)
410}

535// Remove obsolete block's weight data by leveraging finality notifications.
536// This includes data for all finalized blocks (excluding the most recent one)
537// and all stale branches.
538fn aux_storage_cleanup<C: HeaderMetadata<Block> + HeaderBackend<Block>, Block: BlockT>(
539	client: &C,
540	notification: &FinalityNotification<Block>,
541) -> AuxDataOperations {
542	let mut hashes = HashSet::new();
543
544	let first = notification.tree_route.first().unwrap_or(&notification.hash);
545	match client.header_metadata(*first) {
546		Ok(meta) => {
547			hashes.insert(meta.parent);
548		},
549		Err(err) => {
550			warn!(target: LOG_TARGET, "Failed to lookup metadata for block `{:?}`: {}", first, err,)
551		},
552	}
553
554	// Cleans data for finalized block's ancestors
555	hashes.extend(
556		notification
557			.tree_route
558			.iter()
559			// Ensure we don't prune latest finalized block.
560			// This should not happen, but better be safe than sorry!
561			.filter(|h| **h != notification.hash),
562	);
563
564	hashes.extend(notification.stale_blocks.iter().map(|b| b.hash));
565
566	hashes
567		.into_iter()
568		.map(|val| (aux_schema::block_weight_key(val), None))
569		.collect()
570}

572async fn answer_requests<B: BlockT, C>(
573	mut request_rx: Receiver<BabeRequest<B>>,
574	config: BabeConfiguration,
575	client: Arc<C>,
576	epoch_changes: SharedEpochChanges<B, Epoch>,
577) where
578	C: HeaderBackend<B> + HeaderMetadata<B, Error = ClientError>,
579{
580	while let Some(request) = request_rx.next().await {
581		match request {
582			BabeRequest::EpochData(response) => {
583				let _ = response.send(epoch_changes.shared_data().clone());
584			},
585			BabeRequest::EpochDataForChildOf(parent_hash, parent_number, slot, response) => {
586				let lookup = || {
587					let epoch_changes = epoch_changes.shared_data();
588					epoch_changes
589						.epoch_data_for_child_of(
590							descendent_query(&*client),
591							&parent_hash,
592							parent_number,
593							slot,
594							|slot| Epoch::genesis(&config, slot),
595						)
596						.map_err(|e| Error::<B>::ForkTree(Box::new(e)))?
597						.ok_or(Error::<B>::FetchEpoch(parent_hash))
598				};
599
600				let _ = response.send(lookup());
601			},
602		}
603	}
604}

892/// Extract the BABE pre digest from the given header. Pre-runtime digests are
893/// mandatory, the function will return `Err` if none is found.
894pub fn find_pre_digest<B: BlockT>(header: &B::Header) -> Result<PreDigest, Error<B>> {
895	// genesis block doesn't contain a pre digest so let's generate a
896	// dummy one to not break any invariants in the rest of the code
897	if header.number().is_zero() {
898		return Ok(PreDigest::SecondaryPlain(SecondaryPlainPreDigest {
899			slot: 0.into(),
900			authority_index: 0,
901		}))
902	}
903
904	let mut pre_digest: Option<_> = None;
905	for log in header.digest().logs() {
906		trace!(target: LOG_TARGET, "Checking log {:?}, looking for pre runtime digest", log);
907		match (log.as_babe_pre_digest(), pre_digest.is_some()) {
908			(Some(_), true) => return Err(babe_err(Error::MultiplePreRuntimeDigests)),
909			(None, _) => trace!(target: LOG_TARGET, "Ignoring digest not meant for us"),
910			(s, false) => pre_digest = s,
911		}
912	}
913	pre_digest.ok_or_else(|| babe_err(Error::NoPreRuntimeDigest))
914}

916/// Extract the BABE epoch change digest from the given header, if it exists.
917pub fn find_next_epoch_digest<B: BlockT>(
918	header: &B::Header,
919) -> Result<Option<NextEpochDescriptor>, Error<B>> {
920	let mut epoch_digest: Option<_> = None;
921	for log in header.digest().logs() {
922		trace!(target: LOG_TARGET, "Checking log {:?}, looking for epoch change digest.", log);
923		let log = log.try_to::<ConsensusLog>(OpaqueDigestItemId::Consensus(&BABE_ENGINE_ID));
924		match (log, epoch_digest.is_some()) {
925			(Some(ConsensusLog::NextEpochData(_)), true) =>
926				return Err(babe_err(Error::MultipleEpochChangeDigests)),
927			(Some(ConsensusLog::NextEpochData(epoch)), false) => epoch_digest = Some(epoch),
928			_ => trace!(target: LOG_TARGET, "Ignoring digest not meant for us"),
929		}
930	}
931
932	Ok(epoch_digest)
933}

983impl<Block, Client> Verifier<Block> for BabeVerifier<Block, Client>
984where
985	Block: BlockT,
986	Client: HeaderMetadata<Block, Error = sp_blockchain::Error>
987		+ HeaderBackend<Block>
988		+ ProvideRuntimeApi<Block>
989		+ Send
990		+ Sync
991		+ AuxStore,
992	Client::Api: BlockBuilderApi<Block> + BabeApi<Block>,
993{
994	async fn verify(
995		&self,
996		mut block: BlockImportParams<Block>,
997	) -> Result<BlockImportParams<Block>, String> {
998		trace!(
999			target: LOG_TARGET,
1000			"Verifying origin: {:?} header: {:?} justification(s): {:?} body: {:?}",
1001			block.origin,
1002			block.header,
1003			block.justifications,
1004			block.body,
1005		);
1006
1007		let hash = block.header.hash();
1008		let parent_hash = *block.header.parent_hash();
1009
1010		let number = block.header.number();
1011
1012		if is_state_sync_or_gap_sync_import(&*self.client, &block) {
1013			return Ok(block)
1014		}
1015
1016		debug!(
1017			target: LOG_TARGET,
1018			"We have {:?} logs in this header",
1019			block.header.digest().logs().len()
1020		);
1021
1022		let slot_now = Slot::from_timestamp(Timestamp::current(), self.slot_duration);
1023
1024		let pre_digest = find_pre_digest::<Block>(&block.header)?;
1025		let (check_header, epoch_descriptor) = {
1026			let (epoch_descriptor, viable_epoch) = query_epoch_changes(
1027				&self.epoch_changes,
1028				self.client.as_ref(),
1029				&self.config,
1030				*number,
1031				pre_digest.slot(),
1032				parent_hash,
1033			)?;
1034
1035			// We add one to the current slot to allow for some small drift.
1036			// FIXME #1019 in the future, alter this queue to allow deferring of headers
1037			let v_params = verification::VerificationParams {
1038				header: block.header.clone(),
1039				pre_digest: Some(pre_digest),
1040				slot_now: slot_now + 1,
1041				epoch: viable_epoch.as_ref(),
1042			};
1043
1044			(verification::check_header::<Block>(v_params)?, epoch_descriptor)
1045		};
1046
1047		match check_header {
1048			CheckedHeader::Checked(pre_header, verified_info) => {
1049				trace!(target: LOG_TARGET, "Checked {:?}; importing.", pre_header);
1050				telemetry!(
1051					self.telemetry;
1052					CONSENSUS_TRACE;
1053					"babe.checked_and_importing";
1054					"pre_header" => ?pre_header,
1055				);
1056
1057				block.header = pre_header;
1058				block.post_digests.push(verified_info.seal);
1059				block.insert_intermediate(
1060					INTERMEDIATE_KEY,
1061					BabeIntermediate::<Block> { epoch_descriptor },
1062				);
1063				block.post_hash = Some(hash);
1064
1065				Ok(block)
1066			},
1067			CheckedHeader::Deferred(a, b) => {
1068				debug!(target: LOG_TARGET, "Checking {:?} failed; {:?}, {:?}.", hash, a, b);
1069				telemetry!(
1070					self.telemetry;
1071					CONSENSUS_DEBUG;
1072					"babe.header_too_far_in_future";
1073					"hash" => ?hash, "a" => ?a, "b" => ?b
1074				);
1075				Err(Error::<Block>::TooFarInFuture(hash).into())
1076			},
1077		}
1078	}
1079}

1083/// 2. When importing whole state we don't calculate epoch descriptor, but rather read it from the

1092	info.block_gap.map_or(false, |gap| gap.start <= number && number <= gap.end) ||
1093		block.with_state()
1094}

1163	Inner::Error: Into<ConsensusError>,
1164	Client: HeaderBackend<Block>
1165		+ HeaderMetadata<Block, Error = sp_blockchain::Error>
1166		+ AuxStore
1167		+ ProvideRuntimeApi<Block>
1168		+ Send
1169		+ Sync,
1170	Client::Api: BlockBuilderApi<Block> + BabeApi<Block> + ApiExt<Block>,
1171	CIDP: CreateInherentDataProviders<Block, ()>,
1172	CIDP::InherentDataProviders: InherentDataProviderExt + Send,
1173	SC: sp_consensus::SelectChain<Block> + 'static,
1174{
1175	/// Import whole state after warp sync.
1176	// This function makes multiple transactions to the DB. If one of them fails we may
1177	// end up in an inconsistent state and have to resync.
1178	async fn import_state(
1179		&self,
1180		mut block: BlockImportParams<Block>,
1181	) -> Result<ImportResult, ConsensusError> {
1182		let hash = block.post_hash();
1183		let parent_hash = *block.header.parent_hash();
1184		let number = *block.header.number();
1185
1186		block.fork_choice = Some(ForkChoiceStrategy::Custom(true));
1187		// Reset block weight.
1188		aux_schema::write_block_weight(hash, 0, |values| {
1189			block
1190				.auxiliary
1191				.extend(values.iter().map(|(k, v)| (k.to_vec(), Some(v.to_vec()))))
1192		});
1193
1194		// First make the client import the state.
1195		let import_result = self.inner.import_block(block).await;
1196		let aux = match import_result {
1197			Ok(ImportResult::Imported(aux)) => aux,
1198			Ok(r) =>
1199				return Err(ConsensusError::ClientImport(format!(
1200					"Unexpected import result: {:?}",
1201					r
1202				))),
1203			Err(r) => return Err(r.into()),
1204		};
1205
1206		// Read epoch info from the imported state.
1207		let current_epoch = self.client.runtime_api().current_epoch(hash).map_err(|e| {
1208			ConsensusError::ClientImport(babe_err::<Block>(Error::RuntimeApi(e)).into())
1209		})?;
1210		let next_epoch = self.client.runtime_api().next_epoch(hash).map_err(|e| {
1211			ConsensusError::ClientImport(babe_err::<Block>(Error::RuntimeApi(e)).into())
1212		})?;
1213
1214		let mut epoch_changes = self.epoch_changes.shared_data_locked();
1215		epoch_changes.reset(parent_hash, hash, number, current_epoch.into(), next_epoch.into());

1227		block: &mut BlockImportParams<Block>,
1228	) -> Result<(), ConsensusError> {
1229		if is_state_sync_or_gap_sync_import(&*self.client, block) {
1230			return Ok(())
1231		}

1233		let parent_hash = *block.header.parent_hash();

1257				number,

1265				.authorities

1285	async fn check_inherents(
1286		&self,
1287		block: &mut BlockImportParams<Block>,
1288		at_hash: Block::Hash,
1289		slot: Slot,
1290		create_inherent_data_providers: CIDP::InherentDataProviders,
1291	) -> Result<(), ConsensusError> {
1292		if block.state_action.skip_execution_checks() {
1293			return Ok(())
1294		}

1297			let new_block = Block::new(block.header.clone(), inner_body);
1298			// if the body is passed through and the block was executed,
1299			// we need to use the runtime to check that the internally-set
1300			// timestamp in the inherents actually matches the slot set in the seal.
1301			let mut inherent_data = create_inherent_data_providers
1302				.create_inherent_data()
1303				.await
1304				.map_err(|e| ConsensusError::Other(Box::new(e)))?;

1308

1310				self.client.clone(),
1311				at_hash,

1314				inherent_data,

1320						Error::<Block>::CreateInherents(e),
1321					CheckInherentsError::Client(e) => Error::RuntimeApi(e),
1322					CheckInherentsError::CheckInherents(e) => Error::CheckInherents(e),
1323					CheckInherentsError::CheckInherentsUnknownError(id) =>
1324						Error::CheckInherentsUnhandled(id),
1325				}))
1326			})?;

1334	async fn check_and_report_equivocation(
1335		&self,
1336		slot_now: Slot,
1337		slot: Slot,
1338		header: &Block::Header,
1339		author: &AuthorityId,
1340		origin: &BlockOrigin,
1341	) -> Result<(), Error<Block>> {
1342		// don't report any equivocations during initial sync
1343		// as they are most likely stale.
1344		if *origin == BlockOrigin::NetworkInitialSync {
1345			return Ok(())
1346		}
1347
1348		// check if authorship of this header is an equivocation and return a proof if so.
1349		let Some(equivocation_proof) =
1350			check_equivocation(&*self.client, slot_now, slot, header, author)
1351				.map_err(Error::Client)?
1352		else {
1353			return Ok(())
1354		};
1355
1356		info!(
1357			"Slot author {:?} is equivocating at slot {} with headers {:?} and {:?}",
1358			author,
1359			slot,
1360			equivocation_proof.first_header.hash(),
1361			equivocation_proof.second_header.hash(),
1362		);
1363
1364		// get the best block on which we will build and send the equivocation report.
1365		let best_hash = self
1366			.select_chain
1367			.best_chain()
1368			.await
1369			.map(|h| h.hash())
1370			.map_err(|e| Error::Client(e.into()))?;
1371
1372		// generate a key ownership proof. we start by trying to generate the
1373		// key ownership proof at the parent of the equivocating header, this
1374		// will make sure that proof generation is successful since it happens
1375		// during the on-going session (i.e. session keys are available in the
1376		// state to be able to generate the proof). this might fail if the
1377		// equivocation happens on the first block of the session, in which case
1378		// its parent would be on the previous session. if generation on the
1379		// parent header fails we try with best block as well.
1380		let generate_key_owner_proof = |at_hash: Block::Hash| {
1381			self.client
1382				.runtime_api()
1383				.generate_key_ownership_proof(at_hash, slot, equivocation_proof.offender.clone())
1384				.map_err(Error::RuntimeApi)
1385		};
1386
1387		let parent_hash = *header.parent_hash();
1388		let key_owner_proof = match generate_key_owner_proof(parent_hash)? {
1389			Some(proof) => proof,
1390			None => match generate_key_owner_proof(best_hash)? {
1391				Some(proof) => proof,
1392				None => {
1393					debug!(
1394						target: LOG_TARGET,
1395						"Equivocation offender is not part of the authority set."
1396					);
1397					return Ok(())
1398				},
1399			},
1400		};
1401
1402		// submit equivocation report at best block.
1403		let mut runtime_api = self.client.runtime_api();
1404
1405		// Register the offchain tx pool to be able to use it from the runtime.
1406		runtime_api
1407			.register_extension(self.offchain_tx_pool_factory.offchain_transaction_pool(best_hash));
1408
1409		runtime_api
1410			.submit_report_equivocation_unsigned_extrinsic(
1411				best_hash,
1412				equivocation_proof,
1413				key_owner_proof,
1414			)
1415			.map_err(Error::RuntimeApi)?;
1416
1417		info!(target: LOG_TARGET, "Submitted equivocation report for author {:?}", author);
1418
1419		Ok(())
1420	}

1443	async fn import_block(
1444		&self,
1445		mut block: BlockImportParams<Block>,
1446	) -> Result<ImportResult, Self::Error> {
1447		let hash = block.post_hash();
1448		let number = *block.header.number();
1449		let info = self.client.info();
1450
1451		self.check_inherents_and_equivocations(&mut block).await?;
1452
1453		let block_status = self
1454			.client
1455			.status(hash)
1456			.map_err(|e| ConsensusError::ClientImport(e.to_string()))?;
1457
1458		// Skip babe logic if block already in chain or importing blocks during initial sync,
1459		// otherwise the check for epoch changes will error because trying to re-import an
1460		// epoch change or because of missing epoch data in the tree, respectively.
1461		if info.block_gap.map_or(false, |gap| gap.start <= number && number <= gap.end) ||
1462			block_status == BlockStatus::InChain
1463		{
1464			// When re-importing existing block strip away intermediates.
1465			// In case of initial sync intermediates should not be present...
1466			let _ = block.remove_intermediate::<BabeIntermediate<Block>>(INTERMEDIATE_KEY);
1467			block.fork_choice = Some(ForkChoiceStrategy::Custom(false));
1468			return self.inner.import_block(block).await.map_err(Into::into)
1469		}
1470
1471		if block.with_state() {
1472			return self.import_state(block).await
1473		}
1474
1475		let pre_digest = find_pre_digest::<Block>(&block.header).expect(
1476			"valid babe headers must contain a predigest; header has been already verified; qed",
1477		);
1478		let slot = pre_digest.slot();
1479
1480		let parent_hash = *block.header.parent_hash();
1481		let parent_header = self
1482			.client
1483			.header(parent_hash)
1484			.map_err(|e| ConsensusError::ChainLookup(e.to_string()))?
1485			.ok_or_else(|| {
1486				ConsensusError::ChainLookup(
1487					babe_err(Error::<Block>::ParentUnavailable(parent_hash, hash)).into(),
1488				)
1489			})?;
1490
1491		let parent_slot = find_pre_digest::<Block>(&parent_header).map(|d| d.slot()).expect(
1492			"parent is non-genesis; valid BABE headers contain a pre-digest; header has already \
1493			 been verified; qed",
1494		);
1495
1496		// make sure that slot number is strictly increasing
1497		if slot <= parent_slot {
1498			return Err(ConsensusError::ClientImport(
1499				babe_err(Error::<Block>::SlotMustIncrease(parent_slot, slot)).into(),
1500			))
1501		}
1502
1503		// if there's a pending epoch we'll save the previous epoch changes here
1504		// this way we can revert it if there's any error
1505		let mut old_epoch_changes = None;
1506
1507		// Use an extra scope to make the compiler happy, because otherwise it complains about the
1508		// mutex, even if we dropped it...
1509		let mut epoch_changes = {
1510			let mut epoch_changes = self.epoch_changes.shared_data_locked();
1511
1512			// check if there's any epoch change expected to happen at this slot.
1513			// `epoch` is the epoch to verify the block under, and `first_in_epoch` is true
1514			// if this is the first block in its chain for that epoch.
1515			//
1516			// also provides the total weight of the chain, including the imported block.
1517			let (epoch_descriptor, first_in_epoch, parent_weight) = {
1518				let parent_weight = if *parent_header.number() == Zero::zero() {
1519					0
1520				} else {
1521					aux_schema::load_block_weight(&*self.client, parent_hash)
1522						.map_err(|e| ConsensusError::ClientImport(e.to_string()))?
1523						.ok_or_else(|| {
1524							ConsensusError::ClientImport(
1525								babe_err(Error::<Block>::ParentBlockNoAssociatedWeight(hash))
1526									.into(),
1527							)
1528						})?
1529				};
1530
1531				let intermediate =
1532					block.remove_intermediate::<BabeIntermediate<Block>>(INTERMEDIATE_KEY)?;
1533
1534				let epoch_descriptor = intermediate.epoch_descriptor;
1535				let first_in_epoch = parent_slot < epoch_descriptor.start_slot();
1536				(epoch_descriptor, first_in_epoch, parent_weight)
1537			};
1538
1539			let total_weight = parent_weight + pre_digest.added_weight();
1540
1541			// search for this all the time so we can reject unexpected announcements.
1542			let next_epoch_digest = find_next_epoch_digest::<Block>(&block.header)
1543				.map_err(|e| ConsensusError::ClientImport(e.to_string()))?;
1544			let next_config_digest = find_next_config_digest::<Block>(&block.header)
1545				.map_err(|e| ConsensusError::ClientImport(e.to_string()))?;
1546
1547			match (first_in_epoch, next_epoch_digest.is_some(), next_config_digest.is_some()) {
1548				(true, true, _) => {},
1549				(false, false, false) => {},
1550				(false, false, true) =>
1551					return Err(ConsensusError::ClientImport(
1552						babe_err(Error::<Block>::UnexpectedConfigChange).into(),
1553					)),
1554				(true, false, _) =>
1555					return Err(ConsensusError::ClientImport(
1556						babe_err(Error::<Block>::ExpectedEpochChange(hash, slot)).into(),
1557					)),
1558				(false, true, _) =>
1559					return Err(ConsensusError::ClientImport(
1560						babe_err(Error::<Block>::UnexpectedEpochChange).into(),
1561					)),
1562			}
1563
1564			if let Some(next_epoch_descriptor) = next_epoch_digest {
1565				old_epoch_changes = Some((*epoch_changes).clone());
1566
1567				let mut viable_epoch = epoch_changes
1568					.viable_epoch(&epoch_descriptor, |slot| Epoch::genesis(&self.config, slot))
1569					.ok_or_else(|| {
1570						ConsensusError::ClientImport(Error::<Block>::FetchEpoch(parent_hash).into())
1571					})?
1572					.into_cloned();
1573
1574				let epoch_config = next_config_digest
1575					.map(Into::into)
1576					.unwrap_or_else(|| viable_epoch.as_ref().config.clone());
1577
1578				// restrict info logging during initial sync to avoid spam
1579				let log_level = if block.origin == BlockOrigin::NetworkInitialSync {
1580					log::Level::Debug
1581				} else {
1582					log::Level::Info
1583				};
1584
1585				if viable_epoch.as_ref().end_slot() <= slot {
1586					// Some epochs must have been skipped as our current slot fits outside the
1587					// current epoch. We will figure out which epoch it belongs to and we will
1588					// re-use the same data for that epoch.
1589					// Notice that we are only updating a local copy of the `Epoch`, this
1590					// makes it so that when we insert the next epoch into `EpochChanges` below
1591					// (after incrementing it), it will use the correct epoch index and start slot.
1592					// We do not update the original epoch that will be re-used because there might
1593					// be other forks (that we haven't imported) where the epoch isn't skipped, and
1594					// to import those forks we want to keep the original epoch data. Not updating
1595					// the original epoch works because when we search the tree for which epoch to
1596					// use for a given slot, we will search in-depth with the predicate
1597					// `epoch.start_slot <= slot` which will still match correctly without updating
1598					// `start_slot` to the correct value as below.
1599					let epoch = viable_epoch.as_mut();
1600					let prev_index = epoch.epoch_index;
1601					*epoch = epoch.clone_for_slot(slot);
1602
1603					warn!(
1604						target: LOG_TARGET,
1605						"๐ถ Epoch(s) skipped: from {} to {}", prev_index, epoch.epoch_index,
1606					);
1607				}
1608
1609				log!(
1610					target: LOG_TARGET,
1611					log_level,
1612					"๐ถ New epoch {} launching at block {} (block slot {} >= start slot {}).",
1613					viable_epoch.as_ref().epoch_index,
1614					hash,
1615					slot,
1616					viable_epoch.as_ref().start_slot,
1617				);
1618
1619				let next_epoch = viable_epoch.increment((next_epoch_descriptor, epoch_config));
1620
1621				log!(
1622					target: LOG_TARGET,
1623					log_level,
1624					"๐ถ Next epoch starts at slot {}",
1625					next_epoch.as_ref().start_slot,
1626				);
1627
1628				// prune the tree of epochs not part of the finalized chain or
1629				// that are not live anymore, and then track the given epoch change
1630				// in the tree.
1631				// NOTE: it is important that these operations are done in this
1632				// order, otherwise if pruning after import the `is_descendent_of`
1633				// used by pruning may not know about the block that is being
1634				// imported.
1635				let prune_and_import = || {
1636					prune_finalized(self.client.clone(), &mut epoch_changes)?;
1637
1638					epoch_changes
1639						.import(
1640							descendent_query(&*self.client),
1641							hash,
1642							number,
1643							*block.header.parent_hash(),
1644							next_epoch,
1645						)
1646						.map_err(|e| {
1647							ConsensusError::ClientImport(format!(
1648								"Error importing epoch changes: {}",
1649								e
1650							))
1651						})?;
1652					Ok(())
1653				};
1654
1655				if let Err(e) = prune_and_import() {
1656					debug!(target: LOG_TARGET, "Failed to launch next epoch: {}", e);
1657					*epoch_changes =
1658						old_epoch_changes.expect("set `Some` above and not taken; qed");
1659					return Err(e)
1660				}
1661
1662				crate::aux_schema::write_epoch_changes::<Block, _, _>(&*epoch_changes, |insert| {
1663					block
1664						.auxiliary
1665						.extend(insert.iter().map(|(k, v)| (k.to_vec(), Some(v.to_vec()))))
1666				});
1667			}
1668
1669			aux_schema::write_block_weight(hash, total_weight, |values| {
1670				block
1671					.auxiliary
1672					.extend(values.iter().map(|(k, v)| (k.to_vec(), Some(v.to_vec()))))
1673			});
1674
1675			// The fork choice rule is that we pick the heaviest chain (i.e.
1676			// more primary blocks), if there's a tie we go with the longest
1677			// chain.
1678			block.fork_choice = {
1679				let (last_best, last_best_number) = (info.best_hash, info.best_number);
1680
1681				let last_best_weight = if &last_best == block.header.parent_hash() {
1682					// the parent=genesis case is already covered for loading parent weight,
1683					// so we don't need to cover again here.
1684					parent_weight
1685				} else {
1686					aux_schema::load_block_weight(&*self.client, last_best)
1687						.map_err(|e| ConsensusError::ChainLookup(e.to_string()))?
1688						.ok_or_else(|| {
1689							ConsensusError::ChainLookup(
1690								"No block weight for parent header.".to_string(),
1691							)
1692						})?
1693				};
1694
1695				Some(ForkChoiceStrategy::Custom(if total_weight > last_best_weight {
1696					true
1697				} else if total_weight == last_best_weight {
1698					number > last_best_number
1699				} else {
1700					false
1701				}))
1702			};
1703
1704			// Release the mutex, but it stays locked
1705			epoch_changes.release_mutex()
1706		};
1707
1708		let import_result = self.inner.import_block(block).await;
1709
1710		// revert to the original epoch changes in case there's an error
1711		// importing the block
1712		if import_result.is_err() {
1713			if let Some(old_epoch_changes) = old_epoch_changes {
1714				*epoch_changes.upgrade() = old_epoch_changes;
1715			}
1716		}
1717
1718		import_result.map_err(Into::into)
1719	}

1781		+ HeaderMetadata<Block, Error = sp_blockchain::Error>
1782		+ PreCommitActions<Block>
1783		+ 'static,
1784{
1785	let epoch_changes = aux_schema::load_epoch_changes::<Block, _>(&*client, &config)?;
1786	let link = BabeLink { epoch_changes: epoch_changes.clone(), config: config.clone() };
1787
1788	// NOTE: this isn't entirely necessary, but since we didn't use to prune the
1789	// epoch tree it is useful as a migration, so that nodes prune long trees on
1790	// startup rather than waiting until importing the next epoch change block.
1791	prune_finalized(client.clone(), &mut epoch_changes.shared_data())?;
1792
1793	let client_weak = Arc::downgrade(&client);
1794	let on_finality = move |summary: &FinalityNotification<Block>| {
1795		if let Some(client) = client_weak.upgrade() {
1796			aux_storage_cleanup(client.as_ref(), summary)
1797		} else {
1798			Default::default()
1799		}
1800	};
1801	client.register_finality_action(Box::new(on_finality));
1802
1803	let import = BabeBlockImport::new(
1804		client,
1805		epoch_changes,
1806		wrapped_block_import,
1807		config,
1808		create_inherent_data_providers,
1809		select_chain,
1810		offchain_tx_pool_factory,
1811	);
1812
1813	Ok((import, link))
1814}

1837///
1838/// This method returns the import queue, some data that needs to be passed to the block authoring
1839/// logic (`BabeLink`), and a future that must be run to
1840/// completion and is responsible for listening to finality notifications and
1841/// pruning the epoch changes tree.
1842///
1843/// The block import object provided must be the `BabeBlockImport` or a wrapper
1844/// of it, otherwise crucial import logic will be omitted.
1845pub fn import_queue<Block: BlockT, Client, BI, Spawn>(
1846	ImportQueueParams {
1847		link: babe_link,
1848		block_import,
1849		justification_import,
1850		client,
1851		slot_duration,
1852		spawner,
1853		registry,
1854		telemetry,
1855	}: ImportQueueParams<'_, Block, BI, Client, Spawn>,
1856) -> ClientResult<(DefaultImportQueue<Block>, BabeWorkerHandle<Block>)>
1857where
1858	BI: BlockImport<Block, Error = ConsensusError> + Send + Sync + 'static,
1859	Client: ProvideRuntimeApi<Block>
1860		+ HeaderBackend<Block>
1861		+ HeaderMetadata<Block, Error = sp_blockchain::Error>
1862		+ AuxStore
1863		+ Send
1864		+ Sync
1865		+ 'static,
1866	Client::Api: BlockBuilderApi<Block> + BabeApi<Block> + ApiExt<Block>,
1867	Spawn: SpawnEssentialNamed,
1868{
1869	const HANDLE_BUFFER_SIZE: usize = 1024;
1870
1871	let verifier = BabeVerifier {
1872		slot_duration,
1873		config: babe_link.config.clone(),
1874		epoch_changes: babe_link.epoch_changes.clone(),
1875		telemetry,
1876		client: client.clone(),
1877	};
1878
1879	let (worker_tx, worker_rx) = channel(HANDLE_BUFFER_SIZE);
1880
1881	let answer_requests =
1882		answer_requests(worker_rx, babe_link.config, client, babe_link.epoch_changes);
1883
1884	spawner.spawn_essential("babe-worker", Some("babe"), answer_requests.boxed());

1886	Ok((
1887		BasicQueue::new(verifier, Box::new(block_import), justification_import, spawner, registry),
1888		BabeWorkerHandle(worker_tx),
1889	))
1890}

1892/// Reverts protocol aux data to at most the last finalized block.
1893/// In particular, epoch-changes and block weights announced after the revert
1894/// point are removed.
1895pub fn revert<Block, Client, Backend>(
1896	client: Arc<Client>,
1897	backend: Arc<Backend>,
1898	blocks: NumberFor<Block>,
1899) -> ClientResult<()>
1900where
1901	Block: BlockT,
1902	Client: AuxStore
1903		+ HeaderMetadata<Block, Error = sp_blockchain::Error>
1904		+ HeaderBackend<Block>
1905		+ ProvideRuntimeApi<Block>
1906		+ UsageProvider<Block>,
1907	Client::Api: BabeApi<Block>,
1908	Backend: BackendT<Block>,
1909{
1910	let best_number = client.info().best_number;
1911	let finalized = client.info().finalized_number;
1912
1913	let revertible = blocks.min(best_number - finalized);
1914	if revertible == Zero::zero() {
1915		return Ok(())
1916	}
1917
1918	let revert_up_to_number = best_number - revertible;
1919	let revert_up_to_hash = client.hash(revert_up_to_number)?.ok_or(ClientError::Backend(
1920		format!("Unexpected hash lookup failure for block number: {}", revert_up_to_number),
1921	))?;
1922
1923	// Revert epoch changes tree.
1924
1925	// This config is only used on-genesis.
1926	let config = configuration(&*client)?;
1927	let epoch_changes = aux_schema::load_epoch_changes::<Block, Client>(&*client, &config)?;
1928	let mut epoch_changes = epoch_changes.shared_data();
1929
1930	if revert_up_to_number == Zero::zero() {
1931		// Special case, no epoch changes data were present on genesis.
1932		*epoch_changes = EpochChangesFor::<Block, Epoch>::default();
1933	} else {
1934		epoch_changes.revert(descendent_query(&*client), revert_up_to_hash, revert_up_to_number);
1935	}
1936
1937	// Remove block weights added after the revert point.
1938
1939	let mut weight_keys = HashSet::with_capacity(revertible.saturated_into());
1940
1941	let leaves = backend.blockchain().leaves()?.into_iter().filter(|&leaf| {
1942		sp_blockchain::tree_route(&*client, revert_up_to_hash, leaf)
1943			.map(|route| route.retracted().is_empty())
1944			.unwrap_or_default()
1945	});
1946
1947	for leaf in leaves {
1948		let mut hash = leaf;
1949		loop {
1950			let meta = client.header_metadata(hash)?;
1951			if meta.number <= revert_up_to_number ||
1952				!weight_keys.insert(aux_schema::block_weight_key(hash))
1953			{
1954				// We've reached the revert point or an already processed branch, stop here.
1955				break
1956			}
1957			hash = meta.parent;
1958		}
1959	}
1960
1961	let weight_keys: Vec<_> = weight_keys.iter().map(|val| val.as_slice()).collect();
1962
1963	// Write epoch changes and remove weights in one shot.
1964	aux_schema::write_epoch_changes::<Block, _, _>(&epoch_changes, |values| {
1965		client.insert_aux(values, weight_keys.iter())
1966	})
1967}

1969fn query_epoch_changes<Block, Client>(
1970	epoch_changes: &SharedEpochChanges<Block, Epoch>,
1971	client: &Client,
1972	config: &BabeConfiguration,
1973	block_number: NumberFor<Block>,
1974	slot: Slot,
1975	parent_hash: Block::Hash,
1976) -> Result<
1977	(ViableEpochDescriptor<Block::Hash, NumberFor<Block>, Epoch>, ViableEpoch<Epoch>),
1978	Error<Block>,
1979>
1980where
1981	Block: BlockT,
1982	Client: HeaderBackend<Block> + HeaderMetadata<Block, Error = sp_blockchain::Error>,
1983{
1984	let epoch_changes = epoch_changes.shared_data();
1985	let epoch_descriptor = epoch_changes
1986		.epoch_descriptor_for_child_of(
1987			descendent_query(client),
1988			&parent_hash,
1989			block_number - 1u32.into(),
1990			slot,
1991		)
1992		.map_err(|e| Error::<Block>::ForkTree(Box::new(e)))?
1993		.ok_or(Error::<Block>::FetchEpoch(parent_hash))?;
1994	let viable_epoch = epoch_changes
1995		.viable_epoch(&epoch_descriptor, |slot| Epoch::genesis(&config, slot))
1996		.ok_or(Error::<Block>::FetchEpoch(parent_hash))?;
1997	Ok((epoch_descriptor, viable_epoch.into_cloned()))
1998}

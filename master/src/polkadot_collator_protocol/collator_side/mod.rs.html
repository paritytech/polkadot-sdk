94/// Ensure that collator updates its connection requests to validators

110/// Future that when resolved indicates that we should update reserved peer-set

113/// `Pending` variant never finishes and should be used when there're no peers

117/// A future that returns a candidate hash along with validator discovery
118/// keys once a timeout hit.
119///
120/// If a validator doesn't manage to fetch a collation within this timeout
121/// we should reset its interest in this advertisement in a buffer. For example,
122/// when the PoV was already requested from another peer.
123struct ResetInterestTimeout {
124	fut: futures_timer::Delay,
125	candidate_hash: CandidateHash,
126	peer_id: PeerId,
127}

132		Self { fut: futures_timer::Delay::new(delay), candidate_hash, peer_id }

148enum ShouldAdvertiseTo {
149	Yes,
150	NotAuthority,
151	AlreadyAdvertised,
152}

154/// Info about validators we are currently connected to.
155///
156/// It keeps track to which validators we advertised our collation.
157#[derive(Debug, Default)]
158struct ValidatorGroup {
159	/// Validators discovery ids. Lazily initialized when first
160	/// distributing a collation.
161	validators: Vec<AuthorityDiscoveryId>,
162
163	/// Bits indicating which validators have already seen the announcement
164	/// per candidate.
165	advertised_to: HashMap<CandidateHash, BitVec>,
166}
167
168impl ValidatorGroup {
169	/// Returns `true` if we should advertise our collation to the given peer.
170	fn should_advertise_to(
171		&self,
172		candidate_hash: &CandidateHash,
173		peer_ids: &HashMap<PeerId, HashSet<AuthorityDiscoveryId>>,
174		peer: &PeerId,
175	) -> ShouldAdvertiseTo {
176		let authority_ids = match peer_ids.get(peer) {
177			Some(authority_ids) => authority_ids,
178			None => return ShouldAdvertiseTo::NotAuthority,
179		};
180
181		for id in authority_ids {
182			// One peer id may correspond to different discovery ids across sessions,
183			// having a non-empty intersection is sufficient to assume that this peer
184			// belongs to this particular validator group.
185			let validator_index = match self.validators.iter().position(|v| v == id) {
186				Some(idx) => idx,
187				None => continue,
188			};
189
190			// Either the candidate is unseen by this validator group
191			// or the corresponding bit is not set.
192			if self
193				.advertised_to
194				.get(candidate_hash)
195				.map_or(true, |advertised| !advertised[validator_index])
196			{
197				return ShouldAdvertiseTo::Yes
198			} else {
199				return ShouldAdvertiseTo::AlreadyAdvertised
200			}
201		}
202
203		ShouldAdvertiseTo::NotAuthority
204	}
205
206	/// Should be called after we advertised our collation to the given `peer` to keep track of it.
207	fn advertised_to_peer(
208		&mut self,
209		candidate_hash: &CandidateHash,
210		peer_ids: &HashMap<PeerId, HashSet<AuthorityDiscoveryId>>,
211		peer: &PeerId,
212	) {
213		if let Some(authority_ids) = peer_ids.get(peer) {
214			for id in authority_ids {
215				let validator_index = match self.validators.iter().position(|v| v == id) {
216					Some(idx) => idx,
217					None => continue,
218				};
219				self.advertised_to
220					.entry(*candidate_hash)
221					.or_insert_with(|| bitvec![0; self.validators.len()])
222					.set(validator_index, true);
223			}
224		}
225	}
226}

269struct PerRelayParent {
270	/// Per core index validators group responsible for backing candidates built
271	/// on top of this relay parent.
272	validator_group: HashMap<CoreIndex, ValidatorGroup>,
273	/// Distributed collations.
274	collations: HashMap<CandidateHash, CollationData>,
275	/// Number of assignments per core
276	assignments: HashMap<CoreIndex, usize>,
277	/// The relay parent block number
278	block_number: Option<BlockNumber>,
279	/// The session index of this relay parent.
280	session_index: SessionIndex,
281}
282
283impl PerRelayParent {
284	#[overseer::contextbounds(CollatorProtocol, prefix = self::overseer)]
285	async fn new<Context>(
286		ctx: &mut Context,
287		runtime: &mut RuntimeInfo,
288		para_id: ParaId,
289		claim_queue: ClaimQueueSnapshot,
290		block_number: Option<BlockNumber>,
291		block_hash: Hash,
292		session_index: SessionIndex,
293	) -> Result<Self> {
294		let assignments =
295			claim_queue.iter_all_claims().fold(HashMap::new(), |mut acc, (core, claims)| {
296				let n_claims = claims.iter().filter(|para| para == &&para_id).count();
297				if n_claims > 0 {
298					acc.insert(*core, n_claims);
299				}
300				acc
301			});
302
303		let mut validator_groups = HashMap::default();
304
305		for (core, _) in &assignments {
306			let GroupValidators { validators } =
307				determine_our_validators(ctx, runtime, *core, block_hash).await?;
308			let mut group = ValidatorGroup::default();
309			group.validators = validators;
310			validator_groups.insert(*core, group);
311		}
312
313		Ok(Self {
314			validator_group: validator_groups,
315			collations: HashMap::new(),
316			assignments,
317			block_number,
318			session_index,
319		})
320	}
321}

420/// Distribute a collation.
421///
422/// Figure out the core our para is assigned to and the relevant validators.
423/// Issue a connection request to these validators.
424/// If the para is not scheduled or next up on any core, at the relay-parent,
425/// or the relay-parent isn't in the implicit ancestry, we ignore the message
426/// as it must be invalid in that case - although this indicates a logic error
427/// elsewhere in the node.
428#[overseer::contextbounds(CollatorProtocol, prefix = self::overseer)]
429async fn distribute_collation<Context>(
430	ctx: &mut Context,
431	state: &mut State,
432	id: ParaId,
433	receipt: CandidateReceipt,
434	pov: PoV,
435	parent_head_data: HeadData,
436	result_sender: Option<oneshot::Sender<CollationSecondedSignal>>,
437	core_index: CoreIndex,
438) -> Result<()> {
439	let candidate_relay_parent = receipt.descriptor.relay_parent();
440	let candidate_hash = receipt.hash();
441
442	// We should already be connected to the validators, but if we aren't, we will try to connect to
443	// them now.
444	update_validator_connections(
445		ctx,
446		&state.peer_ids,
447		&state.implicit_view,
448		&state.per_relay_parent,
449		id,
450		true,
451	)
452	.await;
453
454	let per_relay_parent = match state.per_relay_parent.get_mut(&candidate_relay_parent) {
455		Some(per_relay_parent) => per_relay_parent,
456		None => {
457			gum::debug!(
458				target: LOG_TARGET,
459				para_id = %id,
460				candidate_relay_parent = %candidate_relay_parent,
461				candidate_hash = ?candidate_hash,
462				"Candidate relay parent is out of our view",
463			);
464			return Ok(())
465		},
466	};
467
468	let Some(collations_limit) = per_relay_parent.assignments.get(&core_index) else {
469		gum::warn!(
470			target: LOG_TARGET,
471			para_id = %id,
472			relay_parent = ?candidate_relay_parent,
473			cores = ?per_relay_parent.assignments.keys(),
474			?core_index,
475			"Attempting to distribute collation for a core we are not assigned to ",
476		);
477
478		return Ok(())
479	};
480
481	let current_collations_count = per_relay_parent
482		.collations
483		.values()
484		.filter(|c| c.core_index() == &core_index)
485		.count();
486	if current_collations_count >= *collations_limit {
487		gum::debug!(
488			target: LOG_TARGET,
489			?candidate_relay_parent,
490			"The limit of {} collations per relay parent for core {} is already reached",
491			collations_limit,
492			core_index.0,
493		);
494		return Ok(())
495	}
496
497	// We have already seen collation for this relay parent.
498	if per_relay_parent.collations.contains_key(&candidate_hash) {
499		gum::debug!(
500			target: LOG_TARGET,
501			?candidate_relay_parent,
502			?candidate_hash,
503			"Already seen this candidate",
504		);
505		return Ok(())
506	}
507
508	let elastic_scaling = per_relay_parent.assignments.len() > 1;
509	if elastic_scaling {
510		gum::debug!(
511			target: LOG_TARGET,
512			para_id = %id,
513			cores = ?per_relay_parent.assignments.keys(),
514			"{} is assigned to {} cores at {}", id, per_relay_parent.assignments.len(), candidate_relay_parent,
515		);
516	}
517
518	let validators = per_relay_parent
519		.validator_group
520		.get(&core_index)
521		.map(|v| v.validators.clone())
522		.unwrap_or_default();
523
524	if validators.is_empty() {
525		gum::warn!(
526			target: LOG_TARGET,
527			core = ?core_index,
528			"there are no validators assigned to core",
529		);
530
531		return Ok(())
532	}
533
534	gum::debug!(
535		target: LOG_TARGET,
536		para_id = %id,
537		candidate_relay_parent = %candidate_relay_parent,
538		?candidate_hash,
539		pov_hash = ?pov.hash(),
540		?core_index,
541		current_validators = ?validators,
542		"Accepted collation, connecting to validators."
543	);
544
545	// Insert validator group for the `core_index` at relay parent.
546	per_relay_parent.validator_group.entry(core_index).or_insert_with(|| {
547		let mut group = ValidatorGroup::default();
548		group.validators = validators;
549		group
550	});
551
552	if let Some(result_sender) = result_sender {
553		state.collation_result_senders.insert(candidate_hash, result_sender);
554	}
555
556	let para_head = receipt.descriptor.para_head();
557	let pov_hash = pov.hash();
558	per_relay_parent.collations.insert(
559		candidate_hash,
560		CollationData {
561			collation: Collation {
562				receipt,
563				pov,
564				parent_head_data,
565				status: CollationStatus::Created,
566			},
567			core_index,
568			session_index: per_relay_parent.session_index,
569			stats: per_relay_parent.block_number.map(|n| {
570				CollationStats::new(
571					para_head,
572					n,
573					candidate_relay_parent,
574					&state.metrics,
575					*candidate_hash,
576					pov_hash,
577				)
578			}),
579		},
580	);
581
582	// The leaf should be present in the allowed ancestry of some leaf.
583	//
584	// It's collation-producer responsibility to verify that there exists
585	// a hypothetical membership in a fragment chain for the candidate.
586	let interested = state
587		.peer_data
588		.iter()
589		.filter(|(_, PeerData { view: v, .. })| {
590			v.iter().any(|block_hash| {
591				state.implicit_view.as_ref().map(|implicit_view| {
592					implicit_view
593						.known_allowed_relay_parents_under(block_hash, Some(id))
594						.unwrap_or_default()
595						.contains(&candidate_relay_parent)
596				}) == Some(true)
597			})
598		})
599		.map(|(id, _)| id);
600
601	// Make sure already connected peers get collations:
602	for peer_id in interested {
603		advertise_collation(
604			ctx,
605			candidate_relay_parent,
606			per_relay_parent,
607			peer_id,
608			&state.peer_ids,
609			&mut state.advertisement_timeouts,
610			&state.metrics,
611		)
612		.await;
613	}
614
615	Ok(())
616}

618/// Validators of a particular group index.
619#[derive(Debug)]
620struct GroupValidators {
621	/// The validators of above group (their discovery keys).
622	validators: Vec<AuthorityDiscoveryId>,
623}

625/// Figure out current group of validators assigned to the para being collated on.
626///
627/// Returns [`ValidatorId`]'s of current group as determined based on the `relay_parent`.
628#[overseer::contextbounds(CollatorProtocol, prefix = self::overseer)]
629async fn determine_our_validators<Context>(
630	ctx: &mut Context,
631	runtime: &mut RuntimeInfo,
632	core_index: CoreIndex,
633	relay_parent: Hash,
634) -> Result<GroupValidators> {
635	let session_index = runtime.get_session_index_for_child(ctx.sender(), relay_parent).await?;
636	let info = &runtime
637		.get_session_info_by_index(ctx.sender(), relay_parent, session_index)
638		.await?
639		.session_info;
640	gum::debug!(target: LOG_TARGET, ?session_index, "Received session info");
641	let groups = &info.validator_groups;
642	let num_cores = groups.len();
643	let rotation_info = get_group_rotation_info(ctx.sender(), relay_parent).await?;
644
645	let current_group_index = rotation_info.group_for_core(core_index, num_cores);
646	let current_validators =
647		groups.get(current_group_index).map(|v| v.as_slice()).unwrap_or_default();
648
649	let validators = &info.discovery_keys;
650
651	let current_validators =
652		current_validators.iter().map(|i| validators[i.0 as usize].clone()).collect();
653
654	let current_validators = GroupValidators { validators: current_validators };
655
656	Ok(current_validators)
657}

682/// Checks whether there are any core assignments for our para on any active relay chain leaves.
683fn has_assigned_cores(
684	implicit_view: &Option<ImplicitView>,
685	per_relay_parent: &HashMap<Hash, PerRelayParent>,
686) -> bool {
687	let Some(implicit_view) = implicit_view else { return false };
688
689	for leaf in implicit_view.leaves() {
690		if let Some(relay_parent) = per_relay_parent.get(leaf) {
691			if !relay_parent.assignments.is_empty() {
692				return true;
693			}
694		}
695	}
696
697	false
698}

700/// Get a list of backing validators at all allowed relay parents.
701/// If `pending_collation` is true, we will only return the validators
702/// that have a collation pending.
703fn list_of_backing_validators_in_view(
704	implicit_view: &Option<ImplicitView>,
705	per_relay_parent: &HashMap<Hash, PerRelayParent>,
706	para_id: ParaId,
707	pending_collation: bool,
708) -> Vec<AuthorityDiscoveryId> {
709	let mut backing_validators = HashSet::new();
710	let Some(implicit_view) = implicit_view else { return vec![] };
711
712	for leaf in implicit_view.leaves() {
713		let allowed_ancestry = implicit_view
714			.known_allowed_relay_parents_under(leaf, Some(para_id))
715			.unwrap_or_default();
716
717		for allowed_relay_parent in allowed_ancestry {
718			let Some(relay_parent) = per_relay_parent.get(allowed_relay_parent) else { continue };
719
720			if pending_collation {
721				// Check if there is any collation for this relay parent.
722				for collation_data in relay_parent.collations.values() {
723					let core_index = collation_data.core_index();
724					if let Some(group) = relay_parent.validator_group.get(core_index) {
725						backing_validators.extend(group.validators.iter().cloned());
726					}
727				}
728			} else {
729				for group in relay_parent.validator_group.values() {
730					backing_validators.extend(group.validators.iter().cloned());
731				}
732			}
733		}
734	}
735
736	backing_validators.into_iter().collect()
737}

739/// Connect or disconnect to/from all backers at all viable relay parents.
740#[overseer::contextbounds(CollatorProtocol, prefix = self::overseer)]
741async fn update_validator_connections<Context>(
742	ctx: &mut Context,
743	peer_ids: &HashMap<PeerId, HashSet<AuthorityDiscoveryId>>,
744	implicit_view: &Option<ImplicitView>,
745	per_relay_parent: &HashMap<Hash, PerRelayParent>,
746	para_id: ParaId,
747	connect: bool,
748) {
749	gum::trace!(target: LOG_TARGET, ?para_id, ?connect, "update_validator_connections");
750
751	// Ignore address resolution failure, will reissue a new request on new collation.
752	let (failed, _) = oneshot::channel();
753
754	let msg = if connect {
755		let cores_assigned = has_assigned_cores(implicit_view, per_relay_parent);
756		// If no cores are assigned to the para, we still need to send a ConnectToValidators request
757		// to the network bridge passing an empty list of validator ids. Otherwise, it will keep
758		// connecting to the last requested validators until a new request is issued.
759		let validator_ids = if cores_assigned {
760			list_of_backing_validators_in_view(implicit_view, per_relay_parent, para_id, false)
761		} else {
762			Vec::new()
763		};
764
765		gum::trace!(
766			target: LOG_TARGET,
767			?cores_assigned,
768			"Sending connection request to validators: {:?}",
769			validator_ids,
770		);
771		NetworkBridgeTxMessage::ConnectToValidators {
772			validator_ids,
773			peer_set: PeerSet::Collation,
774			failed,
775		}
776	} else {
777		if peer_ids.is_empty() {
778			return
779		}
780
781		let validator_ids =
782			list_of_backing_validators_in_view(implicit_view, per_relay_parent, para_id, true);
783
784		gum::trace!(
785			target: LOG_TARGET,
786			"Keeping connections to validators with pending collations: {:?}",
787			validator_ids,
788		);
789
790		// Disconnect from all validators with no pending collations.
791		NetworkBridgeTxMessage::ConnectToValidators {
792			validator_ids,
793			peer_set: PeerSet::Collation,
794			failed,
795		}
796	};
797
798	ctx.send_message(msg).await;
799}

801/// Advertise collation to the given `peer`.
802///
803/// This will only advertise a collation if there exists at least one for the given
804/// `relay_parent` and the given `peer` is set as validator for our para at the given
805/// `relay_parent`.
806///
807/// We also make sure not to advertise the same collation multiple times to the same validator.
808#[overseer::contextbounds(CollatorProtocol, prefix = self::overseer)]
809async fn advertise_collation<Context>(
810	ctx: &mut Context,
811	relay_parent: Hash,
812	per_relay_parent: &mut PerRelayParent,
813	peer: &PeerId,
814	peer_ids: &HashMap<PeerId, HashSet<AuthorityDiscoveryId>>,
815	advertisement_timeouts: &mut FuturesUnordered<ResetInterestTimeout>,
816	metrics: &Metrics,
817) {
818	for (candidate_hash, collation_and_core) in per_relay_parent.collations.iter_mut() {
819		let core_index = *collation_and_core.core_index();
820		let collation = collation_and_core.collation_mut();
821
822		let Some(validator_group) = per_relay_parent.validator_group.get_mut(&core_index) else {
823			gum::debug!(
824				target: LOG_TARGET,
825				?relay_parent,
826				?core_index,
827				"Skipping advertising to validator, validator group for core not found",
828			);
829			return
830		};
831
832		let should_advertise = validator_group.should_advertise_to(candidate_hash, peer_ids, &peer);
833		match should_advertise {
834			ShouldAdvertiseTo::Yes => {},
835			ShouldAdvertiseTo::NotAuthority | ShouldAdvertiseTo::AlreadyAdvertised => {
836				gum::trace!(
837					target: LOG_TARGET,
838					?relay_parent,
839					?candidate_hash,
840					peer_id = %peer,
841					reason = ?should_advertise,
842					"Not advertising collation"
843				);
844				continue
845			},
846		}
847
848		gum::debug!(
849			target: LOG_TARGET,
850			?relay_parent,
851			?candidate_hash,
852			peer_id = %peer,
853			"Advertising collation.",
854		);
855
856		collation.status.advance_to_advertised();
857
858		ctx.send_message(NetworkBridgeTxMessage::SendCollationMessage(
859			vec![*peer],
860			CollationProtocols::V2(protocol_v2::CollationProtocol::CollatorProtocol(
861				protocol_v2::CollatorProtocolMessage::AdvertiseCollation {
862					relay_parent,
863					candidate_hash: *candidate_hash,
864					parent_head_data_hash: collation.parent_head_data.hash(),
865				},
866			)),
867		))
868		.await;
869
870		validator_group.advertised_to_peer(candidate_hash, &peer_ids, peer);
871
872		advertisement_timeouts.push(ResetInterestTimeout::new(
873			*candidate_hash,
874			*peer,
875			RESET_INTEREST_TIMEOUT,
876		));
877
878		metrics.on_advertisement_made();
879	}
880}

882/// The main incoming message dispatching switch.
883#[overseer::contextbounds(CollatorProtocol, prefix = self::overseer)]
884async fn process_msg<Context>(
885	ctx: &mut Context,
886	runtime: &mut RuntimeInfo,
887	state: &mut State,
888	msg: CollatorProtocolMessage,
889) -> Result<()> {
890	use CollatorProtocolMessage::*;
891
892	match msg {
893		ConnectToBackingGroups => {
894			gum::debug!(
895				target: LOG_TARGET,
896				"Received PreConnectToBackingGroups message."
897			);
898			state.connect_to_backers = true;
899
900			if let Some(para_id) = state.collating_on {
901				update_validator_connections(
902					ctx,
903					&state.peer_ids,
904					&state.implicit_view,
905					&state.per_relay_parent,
906					para_id,
907					state.connect_to_backers,
908				)
909				.await;
910			}
911		},
912		DisconnectFromBackingGroups => {
913			gum::debug!(
914				target: LOG_TARGET,
915				"Received DisconnectFromBackingGroups message."
916			);
917			state.connect_to_backers = false;
918
919			if let Some(para_id) = state.collating_on {
920				update_validator_connections(
921					ctx,
922					&state.peer_ids,
923					&state.implicit_view,
924					&state.per_relay_parent,
925					para_id,
926					state.connect_to_backers,
927				)
928				.await;
929			}
930		},
931		CollateOn(id) => {
932			state.collating_on = Some(id);
933			state.implicit_view = Some(ImplicitView::new(Some(id)));
934		},
935		DistributeCollation {
936			candidate_receipt,
937			parent_head_data_hash: _,
938			pov,
939			parent_head_data,
940			result_sender,
941			core_index,
942		} => {
943			match state.collating_on {
944				Some(id) if candidate_receipt.descriptor.para_id() != id => {
945					// If the ParaId of a collation requested to be distributed does not match
946					// the one we expect, we ignore the message.
947					gum::warn!(
948						target: LOG_TARGET,
949						para_id = %candidate_receipt.descriptor.para_id(),
950						collating_on = %id,
951						"DistributeCollation for unexpected para_id",
952					);
953				},
954				Some(id) => {
955					let _ = state.metrics.time_collation_distribution("distribute");
956					distribute_collation(
957						ctx,
958						state,
959						id,
960						candidate_receipt,
961						pov,
962						parent_head_data,
963						result_sender,
964						core_index,
965					)
966					.await?;
967				},
968				None => {
969					gum::warn!(
970						target: LOG_TARGET,
971						para_id = %candidate_receipt.descriptor.para_id(),
972						"DistributeCollation message while not collating on any",
973					);
974				},
975			}
976		},
977		NetworkBridgeUpdate(event) => {
978			// We should count only this shoulder in the histogram, as other shoulders are just
979			// introducing noise
980			let _ = state.metrics.time_process_msg();
981
982			if let Err(e) = handle_network_msg(ctx, runtime, state, event).await {
983				gum::warn!(
984					target: LOG_TARGET,
985					err = ?e,
986					"Failed to handle incoming network message",
987				);
988			}
989		},
990		msg @ (Invalid(..) | Seconded(..)) => {
991			gum::warn!(
992				target: LOG_TARGET,
993				"{:?} message is not expected on the collator side of the protocol",
994				msg,
995			);
996		},
997	}
998
999	Ok(())
1000}

1002/// Issue a response to a previously requested collation.
1003async fn send_collation(
1004	state: &mut State,
1005	request: VersionedCollationRequest,
1006	receipt: CandidateReceipt,
1007	pov: PoV,
1008	parent_head_data: HeadData,
1009) {
1010	let (tx, rx) = oneshot::channel();
1011
1012	let relay_parent = request.relay_parent();
1013	let peer_id = request.peer_id();
1014	let candidate_hash = receipt.hash();
1015
1016	let result = Ok(request_v2::CollationFetchingResponse::CollationWithParentHeadData {
1017		receipt,
1018		pov,
1019		parent_head_data,
1020	});
1021
1022	let response =
1023		OutgoingResponse { result, reputation_changes: Vec::new(), sent_feedback: Some(tx) };
1024
1025	if let Err(_) = request.send_outgoing_response(response) {
1026		gum::warn!(target: LOG_TARGET, "Sending collation response failed");
1027	}
1028
1029	state.active_collation_fetches.push(
1030		async move {
1031			let r = rx.timeout(MAX_UNSHARED_UPLOAD_TIME).await;
1032			let timed_out = r.is_none();
1033
1034			CollationSendResult { relay_parent, candidate_hash, peer_id, timed_out }
1035		}
1036		.boxed(),
1037	);
1038
1039	state.metrics.on_collation_sent();
1040}

1042/// A networking messages switch.
1043#[overseer::contextbounds(CollatorProtocol, prefix = self::overseer)]
1044async fn handle_incoming_peer_message<Context>(
1045	ctx: &mut Context,
1046	runtime: &mut RuntimeInfo,
1047	state: &mut State,
1048	origin: PeerId,
1049	msg: CollationProtocols<
1050		protocol_v1::CollatorProtocolMessage,
1051		protocol_v2::CollatorProtocolMessage,
1052	>,
1053) -> Result<()> {
1054	use protocol_v1::CollatorProtocolMessage as V1;
1055	use protocol_v2::CollatorProtocolMessage as V2;
1056
1057	match msg {
1058		CollationProtocols::V1(V1::Declare(..)) | CollationProtocols::V2(V2::Declare(..)) => {
1059			gum::trace!(
1060				target: LOG_TARGET,
1061				?origin,
1062				"Declare message is not expected on the collator side of the protocol",
1063			);
1064
1065			// If we are declared to, this is another collator, and we should disconnect.
1066			ctx.send_message(NetworkBridgeTxMessage::DisconnectPeers(
1067				vec![origin],
1068				PeerSet::Collation,
1069			))
1070			.await;
1071		},
1072		CollationProtocols::V1(V1::AdvertiseCollation(_)) |
1073		CollationProtocols::V2(V2::AdvertiseCollation { .. }) => {
1074			gum::trace!(
1075				target: LOG_TARGET,
1076				?origin,
1077				"AdvertiseCollation message is not expected on the collator side of the protocol",
1078			);
1079
1080			modify_reputation(&mut state.reputation, ctx.sender(), origin, COST_UNEXPECTED_MESSAGE)
1081				.await;
1082
1083			// If we are advertised to, this is another collator, and we should disconnect.
1084			ctx.send_message(NetworkBridgeTxMessage::DisconnectPeers(
1085				vec![origin],
1086				PeerSet::Collation,
1087			))
1088			.await;
1089		},
1090		CollationProtocols::V1(V1::CollationSeconded(relay_parent, statement)) => {
1091			// Impossible, we no longer accept connections on v1.
1092			gum::warn!(
1093				target: LOG_TARGET,
1094				?statement,
1095				?origin,
1096				?relay_parent,
1097				"Collation seconded message received on unsupported protocol version 1",
1098			);
1099		},
1100		CollationProtocols::V2(V2::CollationSeconded(relay_parent, statement)) => {
1101			if !matches!(statement.unchecked_payload(), Statement::Seconded(_)) {
1102				gum::warn!(
1103					target: LOG_TARGET,
1104					?statement,
1105					?origin,
1106					"Collation seconded message received with none-seconded statement.",
1107				);
1108			} else {
1109				let statement = runtime
1110					.check_signature(ctx.sender(), relay_parent, statement)
1111					.await?
1112					.map_err(Error::InvalidStatementSignature)?;
1113
1114				let removed =
1115					state.collation_result_senders.remove(&statement.payload().candidate_hash());
1116
1117				if let Some(sender) = removed {
1118					gum::trace!(
1119						target: LOG_TARGET,
1120						?statement,
1121						?origin,
1122						"received a valid `CollationSeconded`, forwarding result to collator",
1123					);
1124					let _ = sender.send(CollationSecondedSignal { statement, relay_parent });
1125				} else {
1126					// Checking whether the `CollationSeconded` statement is unexpected
1127					let relay_parent = match state.per_relay_parent.get(&relay_parent) {
1128						Some(per_relay_parent) => per_relay_parent,
1129						None => {
1130							gum::debug!(
1131								target: LOG_TARGET,
1132								candidate_relay_parent = %relay_parent,
1133								candidate_hash = ?&statement.payload().candidate_hash(),
1134								"Seconded statement relay parent is out of our view",
1135							);
1136							return Ok(())
1137						},
1138					};
1139					match relay_parent.collations.get(&statement.payload().candidate_hash()) {
1140						Some(_) => {
1141							// We've seen this collation before, so a seconded statement is expected
1142							gum::trace!(
1143								target: LOG_TARGET,
1144								?statement,
1145								?origin,
1146								"received a valid `CollationSeconded`",
1147							);
1148						},
1149						None => {
1150							gum::debug!(
1151								target: LOG_TARGET,
1152								candidate_hash = ?&statement.payload().candidate_hash(),
1153								?origin,
1154								"received an unexpected `CollationSeconded`: unknown statement",
1155							);
1156						},
1157					}
1158				}
1159			}
1160		},
1161	}
1162
1163	Ok(())
1164}

1166/// Process an incoming network request for a collation.

1196			};

1198				if let Some(collation_with_core) = collation_with_core {
1199					let collation = collation_with_core.collation_mut();
1200					collation.status.advance_to_requested();

1219			let candidate_hash = receipt.hash();

1231				)
1232				.await;
1233				return Ok(())
1234			}

1241				let _ = state.metrics.time_collation_distribution("send");

1244			}
1245		},
1246		Some(our_para_id) => {
1247			gum::warn!(
1248				target: LOG_TARGET,
1249				for_para_id = %para_id,
1250				our_para_id = %our_para_id,
1251				"received a `CollationFetchingRequest` for unexpected para_id",
1252			);
1253		},
1254		None => {
1255			gum::warn!(
1256				target: LOG_TARGET,
1257				for_para_id = %para_id,
1258				"received a `RequestCollation` while not collating on any para",
1259			);
1260		},
1261	}
1262	Ok(())

1265/// Peer's view has changed. Send advertisements for new relay parents
1266/// if there're any.
1267#[overseer::contextbounds(CollatorProtocol, prefix = self::overseer)]
1268async fn handle_peer_view_change<Context>(
1269	ctx: &mut Context,
1270	state: &mut State,
1271	peer_id: PeerId,

1273) {

1276	};
1277
1278	let added: Vec<Hash> = view.difference(&*current).cloned().collect();
1279
1280	*current = view;
1281
1282	for added in added.into_iter() {
1283		let block_hashes = match state.per_relay_parent.contains_key(&added) {
1284			true => state

1288					implicit_view.known_allowed_relay_parents_under(&added, state.collating_on)
1289				})
1290				.unwrap_or_default(),
1291			false => {
1292				gum::trace!(
1293					target: LOG_TARGET,
1294					?peer_id,
1295					new_leaf = ?added,
1296					"New leaf in peer's view is unknown",
1297				);
1298
1299				unknown_heads.insert(added, ());
1300
1301				continue
1302			},
1303		};
1304
1305		for block_hash in block_hashes {
1306			let Some(per_relay_parent) = state.per_relay_parent.get_mut(block_hash) else {
1307				continue
1308			};
1309
1310			advertise_collation(
1311				ctx,
1312				*block_hash,
1313				per_relay_parent,
1314				&peer_id,
1315				&state.peer_ids,
1316				&mut state.advertisement_timeouts,
1317				&state.metrics,
1318			)
1319			.await;
1320		}
1321	}
1322}

1324/// Bridge messages switch.
1325#[overseer::contextbounds(CollatorProtocol, prefix = self::overseer)]
1326async fn handle_network_msg<Context>(
1327	ctx: &mut Context,
1328	runtime: &mut RuntimeInfo,
1329	state: &mut State,
1330	bridge_message: NetworkBridgeEvent<net_protocol::CollatorProtocolMessage>,
1331) -> Result<()> {
1332	use NetworkBridgeEvent::*;
1333
1334	match bridge_message {
1335		PeerConnected(peer_id, observed_role, protocol_version, maybe_authority) => {
1336			// If it is possible that a disconnected validator would attempt a reconnect
1337			// it should be handled here.
1338			gum::trace!(target: LOG_TARGET, ?peer_id, ?observed_role, ?maybe_authority, "Peer connected");
1339
1340			let version: CollationVersion = match protocol_version.try_into() {
1341				Ok(version) => version,
1342				Err(err) => {
1343					// Network bridge is expected to handle this.
1344					gum::error!(
1345						target: LOG_TARGET,
1346						?peer_id,
1347						?observed_role,
1348						?err,
1349						"Unsupported protocol version"
1350					);
1351					return Ok(())
1352				},
1353			};
1354			if version == CollationVersion::V1 {
1355				gum::warn!(
1356					target: LOG_TARGET,
1357					?peer_id,
1358					?observed_role,
1359					"Unsupported protocol version v1"
1360				);
1361
1362				// V1 no longer supported, we should disconnect.
1363				ctx.send_message(NetworkBridgeTxMessage::DisconnectPeers(
1364					vec![peer_id],
1365					PeerSet::Collation,
1366				))
1367				.await;
1368				return Ok(())
1369			}
1370
1371			state.peer_data.entry(peer_id).or_insert_with(|| PeerData {
1372				view: View::default(),
1373				// Unlikely that the collator is falling 10 blocks behind and if so, it probably is
1374				// not able to keep up any way.
1375				unknown_heads: LruMap::new(ByLength::new(10)),
1376			});
1377
1378			if let Some(authority_ids) = maybe_authority {
1379				gum::trace!(
1380					target: LOG_TARGET,
1381					?authority_ids,
1382					?peer_id,
1383					"Connected to requested validator"
1384				);
1385				state.peer_ids.insert(peer_id, authority_ids);
1386
1387				declare(ctx, state, &peer_id).await;
1388			}
1389		},
1390		PeerViewChange(peer_id, view) => {
1391			gum::trace!(target: LOG_TARGET, ?peer_id, ?view, "Peer view change");
1392			handle_peer_view_change(ctx, state, peer_id, view).await;
1393		},
1394		PeerDisconnected(peer_id) => {
1395			gum::trace!(target: LOG_TARGET, ?peer_id, "Peer disconnected");
1396			state.peer_data.remove(&peer_id);
1397			state.peer_ids.remove(&peer_id);
1398		},
1399		OurViewChange(view) => {
1400			gum::trace!(target: LOG_TARGET, ?view, "Own view change");
1401			handle_our_view_change(ctx, runtime, state, view).await?;
1402			// Connect only if we are collating on a para.
1403			if let Some(para_id) = state.collating_on {
1404				update_validator_connections(
1405					ctx,
1406					&state.peer_ids,
1407					&state.implicit_view,
1408					&state.per_relay_parent,
1409					para_id,
1410					state.connect_to_backers,
1411				)
1412				.await;
1413			}
1414		},
1415		PeerMessage(remote, msg) => {
1416			handle_incoming_peer_message(ctx, runtime, state, remote, msg).await?;
1417		},
1418		UpdatedAuthorityIds(peer_id, authority_ids) => {
1419			gum::trace!(target: LOG_TARGET, ?peer_id, ?authority_ids, "Updated authority ids");
1420			if state.peer_data.contains_key(&peer_id) {
1421				if state.peer_ids.insert(peer_id, authority_ids).is_none() {
1422					declare(ctx, state, &peer_id).await;
1423				}
1424			}
1425		},
1426		NewGossipTopology { .. } => {
1427			// impossible!
1428		},
1429	}
1430
1431	Ok(())
1432}

1434/// Update collation tracker with the backed and included candidates.
1435#[overseer::contextbounds(CollatorProtocol, prefix = crate::overseer)]
1436async fn process_block_events<Context>(
1437	ctx: &mut Context,
1438	collation_tracker: &mut CollationTracker,
1439	leaf: Hash,
1440	maybe_block_number: Option<BlockNumber>,
1441	para_id: ParaId,
1442) {
1443	if let Ok(events) = get_candidate_events(ctx.sender(), leaf).await {
1444		let Some(block_number) = maybe_block_number else {
1445			// This should not happen. If it does this log message explains why
1446			// metrics and logs are missing for the candidates under this block.
1447			gum::debug!(
1448				target: crate::LOG_TARGET_STATS,
1449				relay_block = ?leaf,
1450				?para_id,
1451				"Failed to get relay chain block number",
1452			);
1453			return
1454		};
1455
1456		for ev in events {
1457			match ev {
1458				CandidateEvent::CandidateIncluded(receipt, _, _, _) => {
1459					if receipt.descriptor.para_id() != para_id {
1460						continue
1461					}
1462					collation_tracker.collation_included(block_number, leaf, receipt);
1463				},
1464				CandidateEvent::CandidateBacked(receipt, _, _, _) => {
1465					if receipt.descriptor.para_id() != para_id {
1466						continue
1467					}
1468
1469					let Some(block_number) = maybe_block_number else { continue };
1470					collation_tracker.collation_backed(block_number, leaf, receipt);
1471				},
1472				_ => {
1473					// do not care about other events
1474				},
1475			}
1476		}
1477	}
1478}

1480/// Handles our view changes.
1481#[overseer::contextbounds(CollatorProtocol, prefix = crate::overseer)]
1482async fn handle_our_view_change<Context>(
1483	ctx: &mut Context,
1484	runtime: &mut RuntimeInfo,
1485	state: &mut State,
1486	view: OurView,
1487) -> Result<()> {
1488	let Some(implicit_view) = &mut state.implicit_view else { return Ok(()) };
1489	let Some(para_id) = state.collating_on else { return Ok(()) };
1490
1491	let removed: Vec<_> = implicit_view.leaves().filter(|h| !view.contains(h)).copied().collect();
1492	let added: Vec<_> = view.iter().filter(|h| !implicit_view.contains_leaf(h)).collect();
1493
1494	for leaf in added {
1495		let session_index = runtime.get_session_index_for_child(ctx.sender(), *leaf).await?;
1496
1497		let claim_queue = fetch_claim_queue(ctx.sender(), *leaf).await?;
1498
1499		implicit_view
1500			.activate_leaf(ctx.sender(), *leaf)
1501			.await
1502			.map_err(Error::ImplicitViewFetchError)?;
1503
1504		let block_number = implicit_view.block_number(leaf);
1505
1506		state.per_relay_parent.insert(
1507			*leaf,
1508			PerRelayParent::new(
1509				ctx,
1510				runtime,
1511				para_id,
1512				claim_queue,
1513				block_number,
1514				*leaf,
1515				session_index,
1516			)
1517			.await?,
1518		);
1519
1520		process_block_events(ctx, &mut state.collation_tracker, *leaf, block_number, para_id).await;
1521		let allowed_ancestry = implicit_view
1522			.known_allowed_relay_parents_under(leaf, state.collating_on)
1523			.unwrap_or_default();
1524
1525		// Get the peers that already reported us this head, but we didn't know it at this
1526		// point.
1527		let peers = state
1528			.peer_data
1529			.iter_mut()
1530			.filter_map(|(id, data)| data.unknown_heads.remove(leaf).map(|_| id))
1531			.collect::<Vec<_>>();
1532
1533		for block_hash in allowed_ancestry {
1534			let block_number = implicit_view.block_number(block_hash);
1535
1536			let per_relay_parent = match state.per_relay_parent.entry(*block_hash) {
1537				Entry::Vacant(entry) => {
1538					let claim_queue = match fetch_claim_queue(ctx.sender(), *block_hash).await {
1539						Ok(cq) => cq,
1540						Err(error) => {
1541							gum::debug!(
1542								target: LOG_TARGET,
1543								?block_hash,
1544								?error,
1545								"Failed to fetch claim queue while iterating allowed ancestry",
1546							);
1547							continue
1548						},
1549					};
1550					let session_index =
1551						match runtime.get_session_index_for_child(ctx.sender(), *leaf).await {
1552							Ok(si) => si,
1553							Err(error) => {
1554								gum::debug!(
1555									target: LOG_TARGET,
1556									?block_hash,
1557									?error,
1558									"Failed to fetch session index while iterating allowed ancestry",
1559								);
1560								continue
1561							},
1562						};
1563
1564					entry.insert(
1565						PerRelayParent::new(
1566							ctx,
1567							runtime,
1568							para_id,
1569							claim_queue,
1570							block_number,
1571							*block_hash,
1572							session_index,
1573						)
1574						.await?,
1575					)
1576				},
1577				Entry::Occupied(entry) => entry.into_mut(),
1578			};
1579
1580			// Announce relevant collations to these peers.
1581			for peer_id in &peers {
1582				advertise_collation(
1583					ctx,
1584					*block_hash,
1585					per_relay_parent,
1586					&peer_id,
1587					&state.peer_ids,
1588					&mut state.advertisement_timeouts,
1589					&state.metrics,
1590				)
1591				.await;
1592			}
1593		}
1594	}
1595
1596	let highest_session_index = state.per_relay_parent.values().map(|pr| pr.session_index).max();
1597
1598	for leaf in removed {
1599		// If the leaf is deactivated it still may stay in the view as a part
1600		// of implicit ancestry. Only update the state after the hash is actually
1601		// pruned from the block info storage.
1602		let maybe_block_number = implicit_view.block_number(&leaf);
1603		let pruned = implicit_view.deactivate_leaf(leaf);
1604
1605		for removed in &pruned {
1606			gum::debug!(
1607				target: LOG_TARGET,
1608				relay_parent = ?removed,
1609				"Removing relay parent because our view changed.",
1610			);
1611
1612			if let Some(block_number) = maybe_block_number {
1613				let expired_collations = state.collation_tracker.drain_expired(block_number);
1614				process_expired_collations(expired_collations, para_id, &state.metrics);
1615			}
1616
1617			// Get all the collations built on top of the removed leaf.
1618			let collations = state
1619				.per_relay_parent
1620				.remove(removed)
1621				.map(|per_relay_parent| per_relay_parent.collations)
1622				.unwrap_or_default();
1623
1624			for collation_with_core in collations.into_values() {
1625				let collation = collation_with_core.collation();
1626				let candidate_hash = collation.receipt.hash();
1627
1628				state.collation_result_senders.remove(&candidate_hash);
1629
1630				process_out_of_view_collation(
1631					&mut state.collation_tracker,
1632					collation_with_core,
1633					highest_session_index,
1634				);
1635			}
1636
1637			state.waiting_collation_fetches.remove(removed);
1638		}
1639	}
1640	Ok(())
1641}

1643fn process_out_of_view_collation(
1644	collation_tracker: &mut CollationTracker,
1645	mut collation_with_core: CollationData,
1646	highest_session_index: Option<SessionIndex>,
1647) {
1648	let is_same_session =
1649		highest_session_index.map_or(true, |hs| hs == collation_with_core.session_index);
1650	let collation = collation_with_core.collation_mut();
1651	let candidate_hash = collation.receipt.hash();
1652
1653	match collation.status {
1654		CollationStatus::Created =>
1655			if is_same_session {
1656				gum::warn!(
1657					target: LOG_TARGET,
1658					?candidate_hash,
1659					pov_hash = ?collation.pov.hash(),
1660					"Collation wasn't advertised to any validator.",
1661				)
1662			} else {
1663				gum::debug!(
1664					target: LOG_TARGET,
1665					?candidate_hash,
1666					pov_hash = ?collation.pov.hash(),
1667					"Collation wasn't advertised because it was built on a relay chain block that is now part of an old session.",
1668				)
1669			},
1670		CollationStatus::Advertised => gum::debug!(
1671			target: LOG_TARGET,
1672			?candidate_hash,
1673			pov_hash = ?collation.pov.hash(),
1674			"Collation was advertised but not requested by any validator.",
1675		),
1676		CollationStatus::Requested => {
1677			gum::debug!(
1678				target: LOG_TARGET,
1679				?candidate_hash,
1680				pov_hash = ?collation.pov.hash(),
1681				"Collation was requested.",
1682			);
1683		},
1684	}
1685
1686	let collation_status = collation.status.clone();
1687	let Some(mut stats) = collation_with_core.take_stats() else { return };
1688
1689	// If the collation stats are still available, it means it was never
1690	// succesfully fetched, even if a fetch request was received, but not succeed.
1691	//
1692	// Will expire in it's current state at the next block import.
1693	stats.set_pre_backing_status(collation_status);
1694	collation_tracker.track(stats);
1695}

1697/// Process collations that were expired
1698///
1699/// Collations no more tracked after this call.
1700fn process_expired_collations(
1701	expired_collations: Vec<CollationStats>,
1702	para_id: ParaId,
1703	metrics: &Metrics,
1704) {
1705	for expired_collation in expired_collations {
1706		let collation_state = expired_collation.expiry_state();
1707		let age = expired_collation.expired().unwrap_or_default();
1708		let candidate_hash = expired_collation.candidate_hash();
1709		let pov_hash = expired_collation.pov_hash();
1710		let relay_parent = expired_collation.relay_parent();
1711		gum::debug!(
1712			target: crate::LOG_TARGET_STATS,
1713			?age,
1714			?collation_state,
1715			?relay_parent,
1716			?para_id,
1717			head = ?expired_collation.head(),
1718			?candidate_hash,
1719			?pov_hash,
1720			"Collation expired",
1721		);
1722
1723		// Report metrics for collations that were expired after being backed.
1724		// The only way this can happen is if they time out availability.
1725		if let Some(latency) = expired_collation.backed() {
1726			metrics.on_collation_backed(latency as f64);
1727		}
1728		metrics.on_collation_expired(age as f64, collation_state);
1729	}
1730}

1732/// Process collations that may have been finalized on the relay chain.
1733///
1734/// Collations are no longer tracked after this call.
1735///
1736/// ## Limitations
1737///
1738/// We only track the inclusion block number, not the inclusion block hash.
1739/// This means a collation included in a fork that was later dropped could still be counted as
1740/// finalized. This is acceptable for metrics because such cases are rare.
1741async fn process_possibly_finalized_collations(
1742	collations: Vec<CollationStats>,
1743	last_finalized: (Hash, BlockNumber),
1744	sender: &mut impl overseer::SubsystemSender<ChainApiMessage>,
1745	metrics: &Metrics,
1746) {
1747	if collations.is_empty() {
1748		return
1749	}
1750
1751	let (last_hash, last_number) = last_finalized;
1752	let mut blocks_to_request = collations
1753		.iter()
1754		.map(|stats| stats.relay_parent().1)
1755		.filter(|n| *n != last_number) // No need to request `last_hash` for `last_number`
1756		.collect::<Vec<_>>();
1757	blocks_to_request.sort_unstable();
1758	blocks_to_request.dedup();
1759
1760	gum::debug!(target: LOG_TARGET_STATS, ?blocks_to_request, "Collations possibly finalized on blocks");
1761
1762	let mut finalized = vec![(last_hash, last_number)];
1763
1764	for chunk in blocks_to_request.as_slice().chunks(MAX_PARALLEL_CHAIN_API_REQUESTS) {
1765		let futures = chunk.iter().map(|&bn| {
1766				let mut sender = sender.clone();
1767				async move {
1768					let (tx, rx) = oneshot::channel();
1769					sender.send_message(ChainApiMessage::FinalizedBlockHash(bn, tx)).await;
1770					match rx.await {
1771						Ok(Ok(Some(bh))) => Some((bh, bn)),
1772						_ => {
1773							gum::warn!(target: LOG_TARGET_STATS, block_number = ?bn, "Can't request hash for the finalized block from ChainApi");
1774							None
1775						},
1776					}
1777				}
1778			});
1779		finalized.extend(futures::future::join_all(futures).await.into_iter().flatten());
1780	}
1781
1782	for collation in collations {
1783		if !finalized.contains(&collation.relay_parent()) {
1784			// Omit collations built on forks: they're dropped but it's expected
1785			gum::debug!(
1786				target: crate::LOG_TARGET_STATS,
1787				relay_parent = ?collation.relay_parent(),
1788				head = ?collation.head(),
1789				"Collation is built on a fork, skipping",
1790			);
1791			continue;
1792		}
1793
1794		// Report metrics for finalized collations
1795		if let Some(latency) = collation.backed() {
1796			metrics.on_collation_backed(latency as f64);
1797		}
1798		if let Some(latency) = collation.included() {
1799			metrics.on_collation_included(latency as f64);
1800		}
1801		gum::debug!(
1802			target: crate::LOG_TARGET_STATS,
1803			relay_parent = ?collation.relay_parent(),
1804			head = ?collation.head(),
1805			"Collation finalized, stop tracking",
1806		);
1807	}
1808}

1831#[overseer::contextbounds(CollatorProtocol, prefix = crate::overseer)]
1832async fn run_inner<Context>(
1833	mut ctx: Context,
1834	local_peer_id: PeerId,
1835	collator_pair: CollatorPair,
1836	mut req_v2_receiver: IncomingRequestReceiver<request_v2::CollationFetchingRequest>,
1837	metrics: Metrics,
1838	reputation: ReputationAggregator,
1839	reputation_interval: Duration,
1840) -> std::result::Result<(), FatalError> {
1841	use OverseerSignal::*;
1842
1843	let new_reputation_delay = || futures_timer::Delay::new(reputation_interval).fuse();
1844	let mut reputation_delay = new_reputation_delay();
1845
1846	let mut state = State::new(local_peer_id, collator_pair, metrics.clone(), reputation);
1847	let mut runtime = RuntimeInfo::new(None);
1848
1849	loop {
1850		let reputation_changes = || vec![COST_INVALID_REQUEST];
1851		let recv_req_v2 = req_v2_receiver.recv(reputation_changes).fuse();
1852		pin_mut!(recv_req_v2);
1853
1854		let mut reconnect_timeout = &mut state.reconnect_timeout;
1855		select! {
1856			_ = reputation_delay => {
1857				state.reputation.send(ctx.sender()).await;
1858				reputation_delay = new_reputation_delay();
1859			},
1860			msg = ctx.recv().fuse() => match msg.map_err(FatalError::SubsystemReceive)? {
1861				FromOrchestra::Communication { msg } => {
1862					log_error(
1863						process_msg(&mut ctx, &mut runtime, &mut state, msg).await,
1864						"Failed to process message"
1865					)?;
1866				},
1867				FromOrchestra::Signal(ActiveLeaves(update)) => {
1868					if update.activated.is_some() {
1869						*reconnect_timeout = futures_timer::Delay::new(RECONNECT_AFTER_LEAF_TIMEOUT).fuse();
1870					}
1871				}
1872				FromOrchestra::Signal(BlockFinalized(hash, number)) => {
1873					let possibly_finalized = state.collation_tracker.drain_finalized(number);
1874					process_possibly_finalized_collations(possibly_finalized, (hash, number), ctx.sender(), &metrics).await;
1875				}
1876				FromOrchestra::Signal(Conclude) => return Ok(()),
1877			},
1878			CollationSendResult { relay_parent, candidate_hash, peer_id, timed_out } =
1879				state.active_collation_fetches.select_next_some() => {
1880
1881				let next = if let Some(waiting) = state.waiting_collation_fetches.get_mut(&relay_parent) {
1882					if timed_out {
1883						gum::debug!(
1884							target: LOG_TARGET_STATS,
1885							?relay_parent,
1886							?peer_id,
1887							?candidate_hash,
1888							"Sending collation to validator timed out, carrying on with next validator."
1889						);
1890						// We try to throttle requests per relay parent to give validators
1891						// more bandwidth, but if the collation is not received within the
1892						// timeout, we simply start processing next request.
1893						// The request it still alive, it should be kept in a waiting queue.
1894					} else {
1895						waiting.waiting_peers.remove(&(peer_id, candidate_hash));
1896
1897						// Update collation status to fetched.
1898						if let Some(per_relay_parent) =  state.per_relay_parent.get_mut(&relay_parent) {
1899							if let Some(collation_with_core) = per_relay_parent.collations.get_mut(&candidate_hash) {
1900								let maybe_stats = collation_with_core.take_stats();
1901								let our_para_id = collation_with_core.collation().receipt.descriptor.para_id();
1902
1903								if let Some(mut stats) = maybe_stats {
1904									// Update the timestamp when collation has been sent (from subsysytem perspective)
1905									stats.set_fetched_at(std::time::Instant::now());
1906									gum::debug!(
1907										target: LOG_TARGET_STATS,
1908										para_head = ?stats.head(),
1909										%our_para_id,
1910										"Collation fetch latency is {}ms",
1911										stats.fetch_latency().unwrap_or_default().as_millis(),
1912									);
1913
1914									// Update the pre-backing status. Should be requested at this point.
1915									stats.set_pre_backing_status(collation_with_core.collation().status.clone());
1916									debug_assert_eq!(collation_with_core.collation().status, CollationStatus::Requested);
1917
1918									// Observe fetch latency metric.
1919									stats.take_fetch_latency_metric();
1920									stats.set_backed_latency_metric(metrics.time_collation_backing_latency());
1921
1922									// Next step is to measure backing latency.
1923									state.collation_tracker.track(stats);
1924								}
1925							}
1926						}
1927					}
1928
1929					if let Some(next) = waiting.req_queue.pop_front() {
1930						next
1931					} else {
1932						waiting.collation_fetch_active = false;
1933						continue
1934					}
1935				} else {
1936					// No waiting collation fetches means we already removed the relay parent from our view.
1937					continue
1938				};
1939
1940				let next_collation_with_core = {
1941					let per_relay_parent = match state.per_relay_parent.get(&relay_parent) {
1942						Some(per_relay_parent) => per_relay_parent,
1943						None => continue,
1944					};
1945
1946					per_relay_parent.collations.get(&next.candidate_hash())
1947				};
1948
1949				if let Some(collation_with_core) = next_collation_with_core {
1950					let collation = collation_with_core.collation();
1951					let receipt = collation.receipt.clone();
1952					let pov = collation.pov.clone();
1953					let parent_head_data = collation.parent_head_data.clone();
1954
1955					send_collation(&mut state, next, receipt, pov, parent_head_data).await;
1956				}
1957			},
1958			(candidate_hash, peer_id) = state.advertisement_timeouts.select_next_some() => {
1959				gum::debug!(
1960					target: LOG_TARGET,
1961					?candidate_hash,
1962					?peer_id,
1963					"Advertising to peer timed out"
1964				);
1965			}
1966			_ = reconnect_timeout => {
1967				// Connect only if we are collating on a para.
1968				if let Some(para_id) = state.collating_on {
1969					update_validator_connections(
1970						&mut ctx,
1971						&state.peer_ids,
1972						&state.implicit_view,
1973						&state.per_relay_parent,
1974						para_id,
1975						state.connect_to_backers,
1976					)
1977					.await;
1978				}
1979
1980				gum::trace!(
1981					target: LOG_TARGET,
1982					timeout = ?RECONNECT_AFTER_LEAF_TIMEOUT,
1983					"Peer-set updated due to a timeout"
1984				);
1985			},
1986			in_req = recv_req_v2 => {
1987				let request = in_req.map(VersionedCollationRequest::from);
1988
1989				log_error(
1990					handle_incoming_request(&mut ctx, &mut state, request).await,
1991					"Handling incoming collation fetch request V2"
1992				)?;
1993			}

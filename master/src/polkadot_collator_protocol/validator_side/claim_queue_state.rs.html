17//! `ClaimQueueState` tracks the state of the claim queue over a set of relay blocks. Refer to
18//! [`ClaimQueueState`] for more details.
19
20use std::collections::VecDeque;
21
22use polkadot_primitives::{Hash, Id as ParaId};
23
24/// Represents a single claim from the claim queue, mapped to the relay chain block where it could
25/// be backed on-chain.
26#[derive(Debug, PartialEq)]
27struct ClaimInfo {
28	// Hash of the relay chain block. Can be `None` if it is still not known (a future block).
29	hash: Option<Hash>,
30	/// Represents the `ParaId` scheduled for the block. Can be `None` if nothing is scheduled.
31	claim: Option<ParaId>,
32	/// The length of the claim queue at the block. It is used to determine the 'block window'
33	/// where a claim can be made.
34	claim_queue_len: usize,
35	/// A flag that indicates if the slot is claimed or not.
36	claimed: bool,
37}
38
39/// Tracks the state of the claim queue over a set of relay blocks.
40///
41/// Generally the claim queue represents the `ParaId` that should be scheduled at the current block
42/// (the first element of the claim queue) and N other `ParaId`s which are supposed to be scheduled
43/// on the next relay blocks. In other words the claim queue is a rolling window giving a hint what
44/// should be built/fetched/accepted (depending on the context) at each block.
45///
46/// Since the claim queue peeks into the future blocks there is a relation between the claim queue
47/// state between the current block and the future blocks.
48/// Let's see an example with 2 co-scheduled parachains:
49/// - relay parent 1; Claim queue: [A, B, A]
50/// - relay parent 2; Claim queue: [B, A, B]
51/// - relay parent 3; Claim queue: [A, B, A]
52/// - and so on
53///
54/// Note that at rp1 the second element in the claim queue is equal to the first one in rp2. Also
55/// the third element of the claim queue at rp1 is equal to the second one in rp2 and the first one
56/// in rp3.
57///
58/// So if we want to claim the third slot at rp 1 we are also claiming the second at rp2 and first
59/// at rp3. To track this in a simple way we can project the claim queue onto the relay blocks like
60/// this:
61///               [A]   [B]   [A] -> this is the claim queue at rp3
62///         [B]   [A]   [B]       -> this is the claim queue at rp2
63///   [A]   [B]   [A]	          -> this is the claim queue at rp1
64/// [RP 1][RP 2][RP 3][RP X][RP Y] -> relay blocks, RP x and RP Y are future blocks
65///
66/// Note that the claims at each column are the same so we can simplify this by just projecting a
67/// single claim over a block:
68///   [A]   [B]   [A]   [B]   [A]  -> claims effectively are the same
69/// [RP 1][RP 2][RP 3][RP X][RP Y] -> relay blocks, RP x and RP Y are future blocks
70///
71/// Basically this is how `ClaimQueueState` works. It keeps track of claims at each block by mapping
72/// claims to relay blocks.
73///
74/// How making a claim works?
75/// At each relay block we keep track how long is the claim queue. This is a 'window' where we can
76/// make a claim. So adding a claim just looks for a free spot at this window and claims it.
77///
78/// Note on adding a new leaf.
79/// When a new leaf is added we check if the first element in its claim queue matches with the
80/// projection on the first element in 'future blocks'. If yes - the new relay block inherits this
81/// claim. If not - this means that the claim queue changed for some reason so the claim can't be
82/// inherited. This should not happen under normal circumstances. But if it happens it means that we
83/// have got one claim which won't be satisfied in the worst case scenario.
84pub(crate) struct ClaimQueueState {
85	block_state: VecDeque<ClaimInfo>,
86	future_blocks: VecDeque<ClaimInfo>,
87}
88
89impl ClaimQueueState {
90	pub(crate) fn new() -> Self {
91		Self { block_state: VecDeque::new(), future_blocks: VecDeque::new() }
92	}
93
94	// Appends a new leaf
95	pub(crate) fn add_leaf(&mut self, hash: &Hash, claim_queue: &Vec<ParaId>) {
96		if self.block_state.iter().any(|s| s.hash == Some(*hash)) {
97			return
98		}
99
100		// First check if our view for the future blocks is consistent with the one in the claim
101		// queue of the new block. If not - the claim queue has changed for some reason and we need
102		// to readjust our view.
103		for (idx, expected_claim) in claim_queue.iter().enumerate() {
104			match self.future_blocks.get_mut(idx) {
105				Some(future_block) =>
106					if future_block.claim.as_ref() != Some(expected_claim) {
107						// There is an inconsistency. Update our view with the one from the claim
108						// queue. `claimed` can't be true anymore since the `ParaId` has changed.
109						future_block.claimed = false;
110						future_block.claim = Some(*expected_claim);
111					},
112				None => {
113					self.future_blocks.push_back(ClaimInfo {
114						hash: None,
115						claim: Some(*expected_claim),
116						// For future blocks we don't know the size of the claim queue.
117						// `claim_queue_len` could be an option but there is not much benefit from
118						// the extra boilerplate code to handle it. We set it to one since we
119						// usually know about one claim at each future block but this value is not
120						// used anywhere in the code.
121						claim_queue_len: 1,
122						claimed: false,
123					});
124				},
125			}
126		}
127
128		// Now pop the first future block and add it as a leaf
129		let claim_info = if let Some(new_leaf) = self.future_blocks.pop_front() {
130			ClaimInfo {
131				hash: Some(*hash),
132				claim: claim_queue.first().copied(),
133				claim_queue_len: claim_queue.len(),
134				claimed: new_leaf.claimed,
135			}
136		} else {
137			// maybe the claim queue was empty but we still need to add a leaf
138			ClaimInfo {
139				hash: Some(*hash),
140				claim: claim_queue.first().copied(),
141				claim_queue_len: claim_queue.len(),
142				claimed: false,
143			}
144		};
145
146		// `future_blocks` can't be longer than the length of the claim queue at the last block - 1.
147		// For example this can happen if at relay block N we have got a claim queue of a length 4
148		// and it's shrunk to 2.
149		self.future_blocks.truncate(claim_queue.len().saturating_sub(1));
150
151		self.block_state.push_back(claim_info);
152	}
153
154	fn get_window<'a>(
155		&'a mut self,
156		relay_parent: &'a Hash,
157	) -> impl Iterator<Item = &'a mut ClaimInfo> + 'a {
158		let mut window = self
159			.block_state
160			.iter_mut()
161			.skip_while(|b| b.hash != Some(*relay_parent))
162			.peekable();
163		let cq_len = window.peek().map_or(0, |b| b.claim_queue_len);
164		window.chain(self.future_blocks.iter_mut()).take(cq_len)
165	}
166
167	pub(crate) fn claim_at(&mut self, relay_parent: &Hash, para_id: &ParaId) -> bool {
168		self.find_a_claim(relay_parent, para_id, true)
169	}
170
171	pub(crate) fn can_claim_at(&mut self, relay_parent: &Hash, para_id: &ParaId) -> bool {
172		self.find_a_claim(relay_parent, para_id, false)
173	}
174
175	// Returns `true` if there is a claim within `relay_parent`'s view of the claim queue for
176	// `para_id`. If `claim_it` is set to `true` the slot is claimed. Otherwise the function just
177	// reports the availability of the slot.
178	fn find_a_claim(&mut self, relay_parent: &Hash, para_id: &ParaId, claim_it: bool) -> bool {
179		let window = self.get_window(relay_parent);
180
181		for w in window {
182			if !w.claimed && w.claim == Some(*para_id) {
183				w.claimed = claim_it;
184				return true
185			}
186		}
187
188		false
189	}
190
191	pub(crate) fn unclaimed_at(&mut self, relay_parent: &Hash) -> Vec<ParaId> {
192		let window = self.get_window(relay_parent);
193
194		window.filter(|b| !b.claimed).filter_map(|b| b.claim).collect()
195	}
196}

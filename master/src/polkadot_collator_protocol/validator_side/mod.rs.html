166impl PeerData {
167	/// Update the view, clearing all advertisements that are no longer in the
168	/// current view.
169	fn update_view(
170		&mut self,
171		implicit_view: &ImplicitView,
172		active_leaves: &HashSet<Hash>,
173		new_view: View,
174	) {
175		let old_view = std::mem::replace(&mut self.view, new_view);
176		if let PeerState::Collating(ref mut peer_state) = self.state {
177			for removed in old_view.difference(&self.view) {
178				// Remove relay parent advertisements if it went out of our (implicit) view.
179				let keep = is_relay_parent_in_implicit_view(
180					removed,
181					implicit_view,
182					active_leaves,
183					peer_state.para_id,
184				);
185
186				if !keep {
187					peer_state.advertisements.remove(&removed);
188				}
189			}
190		}
191	}
192
193	/// Prune old advertisements relative to our view.
194	fn prune_old_advertisements(
195		&mut self,
196		implicit_view: &ImplicitView,
197		active_leaves: &HashSet<Hash>,
198	) {
199		if let PeerState::Collating(ref mut peer_state) = self.state {
200			peer_state.advertisements.retain(|hash, _| {
201				// Either
202				// - Relay parent is an active leaf
203				// - It belongs to allowed ancestry under some leaf
204				// Discard otherwise.
205				is_relay_parent_in_implicit_view(
206					hash,
207					implicit_view,
208					active_leaves,
209					peer_state.para_id,
210				)
211			});
212		}
213	}
214
215	/// Performs sanity check for an advertisement and notes it as advertised.
216	fn insert_advertisement(
217		&mut self,
218		on_relay_parent: Hash,
219		candidate_hash: Option<CandidateHash>,
220		implicit_view: &ImplicitView,
221		active_leaves: &HashSet<Hash>,
222		per_relay_parent: &PerRelayParent,
223	) -> std::result::Result<(CollatorId, ParaId), InsertAdvertisementError> {
224		match self.state {
225			PeerState::Connected(_) => Err(InsertAdvertisementError::UndeclaredCollator),
226			PeerState::Collating(ref mut state) => {
227				if !is_relay_parent_in_implicit_view(
228					&on_relay_parent,
229					implicit_view,
230					active_leaves,
231					state.para_id,
232				) {
233					return Err(InsertAdvertisementError::OutOfOurView)
234				}
235
236				if let Some(candidate_hash) = candidate_hash {
237					if state
238						.advertisements
239						.get(&on_relay_parent)
240						.map_or(false, |candidates| candidates.contains(&candidate_hash))
241					{
242						return Err(InsertAdvertisementError::Duplicate)
243					}
244
245					let candidates = state.advertisements.entry(on_relay_parent).or_default();
246
247					// Current assignments is equal to the length of the claim queue. No honest
248					// collator should send that many advertisements.
249					if candidates.len() > per_relay_parent.assignment.current.len() {
250						return Err(InsertAdvertisementError::PeerLimitReached)
251					}
252
253					candidates.insert(candidate_hash);
254				} else {
255					if self.version != CollationVersion::V1 {
256						gum::error!(
257							target: LOG_TARGET,
258							"Programming error, `candidate_hash` can not be `None` \
259							 for non `V1` networking.",
260						);
261					}
262
263					if state.advertisements.contains_key(&on_relay_parent) {
264						return Err(InsertAdvertisementError::Duplicate)
265					}
266
267					state
268						.advertisements
269						.insert(on_relay_parent, HashSet::from_iter(candidate_hash));
270				};
271
272				state.last_active = Instant::now();
273				Ok((state.collator_id.clone(), state.para_id))
274			},
275		}
276	}
277
278	/// Whether a peer is collating.
279	fn is_collating(&self) -> bool {
280		match self.state {
281			PeerState::Connected(_) => false,
282			PeerState::Collating(_) => true,
283		}
284	}
285
286	/// Note that a peer is now collating with the given collator and para id.
287	///
288	/// This will overwrite any previous call to `set_collating` and should only be called
289	/// if `is_collating` is false.
290	fn set_collating(&mut self, collator_id: CollatorId, para_id: ParaId) {
291		self.state = PeerState::Collating(CollatingPeerState {
292			collator_id,
293			para_id,
294			advertisements: HashMap::new(),
295			last_active: Instant::now(),
296		});
297	}
298
299	fn collator_id(&self) -> Option<&CollatorId> {
300		match self.state {
301			PeerState::Connected(_) => None,
302			PeerState::Collating(ref state) => Some(&state.collator_id),
303		}
304	}
305
306	fn collating_para(&self) -> Option<ParaId> {
307		match self.state {
308			PeerState::Connected(_) => None,
309			PeerState::Collating(ref state) => Some(state.para_id),
310		}
311	}
312
313	/// Whether the peer has advertised the given collation.
314	fn has_advertised(
315		&self,
316		relay_parent: &Hash,
317		maybe_candidate_hash: Option<CandidateHash>,
318	) -> bool {
319		let collating_state = match self.state {
320			PeerState::Connected(_) => return false,
321			PeerState::Collating(ref state) => state,
322		};
323
324		if let Some(ref candidate_hash) = maybe_candidate_hash {
325			collating_state
326				.advertisements
327				.get(relay_parent)
328				.map_or(false, |candidates| candidates.contains(candidate_hash))
329		} else {
330			collating_state.advertisements.contains_key(relay_parent)
331		}
332	}
333
334	/// Whether the peer is now inactive according to the current instant and the eviction policy.
335	fn is_inactive(&self, policy: &crate::CollatorEvictionPolicy) -> bool {
336		match self.state {
337			PeerState::Connected(connected_at) => connected_at.elapsed() >= policy.undeclared,
338			PeerState::Collating(ref state) =>
339				state.last_active.elapsed() >= policy.inactive_collator,
340		}
341	}

350/// Represents the hold off state of a relay parent.
351enum RelayParentHoldOffState {
352	/// No Advertisements from non-invulnerable collators were received so far. Nothing was held
353	/// off.
354	NotStarted,
355	/// One or more advertisements from non-invulnerable collators are being held off. Any new
356	/// advertisements received in this state are added to the queue.
357	HoldingOff(VecDeque<HeldOffAdvertisement>),
358	/// One or more advertisements were held off. We are not holding off any new advertisements.
359	Done,
360}
361
362impl RelayParentHoldOffState {
363	/// If called in `HoldingOff` state - returns all held off advertisements (there should be at
364	/// least one). If the function is called in a bad state - an error is returned.
365	fn on_hold_off_complete(
366		&mut self,
367	) -> std::result::Result<VecDeque<HeldOffAdvertisement>, HoldOffError> {
368		match std::mem::replace(self, RelayParentHoldOffState::Done) {
369			Self::NotStarted => {
370				*self = Self::NotStarted;
371				Err(HoldOffError::InvalidStateNotStarted)
372			},
373			Self::HoldingOff(held_off) => {
374				*self = RelayParentHoldOffState::Done;
375				if held_off.is_empty() {
376					// the queue should never be empty
377					Err(HoldOffError::QueueEmpty)
378				} else {
379					Ok(held_off)
380				}
381			},
382			Self::Done => {
383				*self = Self::Done;
384				Err(HoldOffError::InvalidStateDone)
385			},
386		}
387	}
388
389	/// Holds off the advertisement, if nothing was held off for this relay parent or there are
390	/// other advertisements being held off. Return false if the hold off was complete for the
391	/// relay parent.
392	fn hold_off_if_necessary(&mut self, advertisement: HeldOffAdvertisement) -> bool {
393		match self {
394			Self::NotStarted => {
395				*self = Self::HoldingOff(vec![advertisement].into());
396				true
397			},
398			Self::HoldingOff(advertisements) => {
399				advertisements.push_back(advertisement);
400				true
401			},
402			Self::Done => false,
403		}
404	}
405}

499impl State {
500	// Returns the number of seconded and pending collations for a specific `ParaId`. Pending
501	// collations are:
502	// 1. Collations being fetched from a collator.
503	// 2. Collations waiting for validation from backing subsystem.
504	// 3. Collations blocked from seconding due to parent not being known by backing subsystem.
505	fn seconded_and_pending_for_para(&self, relay_parent: &Hash, para_id: &ParaId) -> usize {
506		let seconded = self
507			.per_relay_parent
508			.get(relay_parent)
509			.map_or(0, |per_relay_parent| per_relay_parent.collations.seconded_for_para(para_id));
510
511		let pending_fetch = self.per_relay_parent.get(relay_parent).map_or(0, |rp_state| {
512			match rp_state.collations.status {
513				CollationStatus::Fetching(pending_para_id) if pending_para_id == *para_id => 1,
514				_ => 0,
515			}
516		});
517
518		let waiting_for_validation = self
519			.fetched_candidates
520			.keys()
521			.filter(|fc| fc.relay_parent == *relay_parent && fc.para_id == *para_id)
522			.count();
523
524		let blocked_from_seconding =
525			self.blocked_from_seconding.values().fold(0, |acc, blocked_collations| {
526				acc + blocked_collations
527					.iter()
528					.filter(|pc| {
529						pc.candidate_receipt.descriptor.para_id() == *para_id &&
530							pc.candidate_receipt.descriptor.relay_parent() == *relay_parent
531					})
532					.count()
533			});
534
535		gum::trace!(
536			target: LOG_TARGET,
537			?relay_parent,
538			?para_id,
539			seconded,
540			pending_fetch,
541			waiting_for_validation,
542			blocked_from_seconding,
543			"Seconded and pending collations for para",
544		);
545
546		seconded + pending_fetch + waiting_for_validation + blocked_from_seconding
547	}
548
549	/// Returns the number of collations pending to be fetched for a `ParaId`
550	fn in_waiting_queue_for_para(&self, relay_parent: &Hash, para_id: &ParaId) -> usize {
551		self.per_relay_parent
552			.get(relay_parent)
553			.map_or(0, |rp_state| rp_state.collations.queued_for_para(para_id))
554	}
555}

557fn is_relay_parent_in_implicit_view(
558	relay_parent: &Hash,
559	implicit_view: &ImplicitView,
560	active_leaves: &HashSet<Hash>,
561	para_id: ParaId,
562) -> bool {
563	active_leaves.iter().any(|hash| {
564		implicit_view
565			.known_allowed_relay_parents_under(hash, Some(para_id))
566			.unwrap_or_default()
567			.contains(relay_parent)
568	})

571async fn construct_per_relay_parent<Sender>(
572	sender: &mut Sender,
573	current_assignments: &mut HashMap<ParaId, usize>,
574	keystore: &KeystorePtr,
575	relay_parent: Hash,
576	v2_receipts: bool,
577	session_index: SessionIndex,
578) -> Result<Option<PerRelayParent>>
579where
580	Sender: CollatorProtocolSenderTrait,
581{
582	let validators = polkadot_node_subsystem_util::request_validators(relay_parent, sender)
583		.await
584		.await
585		.map_err(Error::CancelledActiveValidators)??;
586
587	let (groups, rotation_info) =
588		polkadot_node_subsystem_util::request_validator_groups(relay_parent, sender)
589			.await
590			.await
591			.map_err(Error::CancelledValidatorGroups)??;
592
593	let core_now = if let Some(group) =
594		polkadot_node_subsystem_util::signing_key_and_index(&validators, keystore).and_then(
595			|(_, index)| polkadot_node_subsystem_util::find_validator_group(&groups, index),
596		) {
597		rotation_info.core_for_group(group, groups.len())
598	} else {
599		gum::trace!(target: LOG_TARGET, ?relay_parent, "Not a validator");
600		return Ok(None)
601	};
602
603	let mut claim_queue = request_claim_queue(relay_parent, sender)
604		.await
605		.await
606		.map_err(Error::CancelledClaimQueue)??;
607
608	let assigned_paras = claim_queue.remove(&core_now).unwrap_or_else(|| VecDeque::new());
609
610	for para_id in assigned_paras.iter() {
611		let entry = current_assignments.entry(*para_id).or_default();
612		*entry += 1;
613		if *entry == 1 {
614			gum::debug!(
615				target: LOG_TARGET,
616				?relay_parent,
617				para_id = ?para_id,
618				"Assigned to a parachain",
619			);
620		}
621	}
622
623	let assignment = GroupAssignments { current: assigned_paras.into_iter().collect() };
624	let collations = Collations::new(&assignment.current);
625
626	Ok(Some(PerRelayParent {
627		assignment,
628		collations,
629		v2_receipts,
630		session_index,
631		current_core: core_now,

674/// Another subsystem has requested to fetch collations on a particular leaf for some para.
675async fn fetch_collation(
676	sender: &mut impl overseer::CollatorProtocolSenderTrait,
677	state: &mut State,
678	pc: PendingCollation,
679	id: CollatorId,
680) -> std::result::Result<(), FetchError> {
681	let PendingCollation { relay_parent, peer_id, prospective_candidate, .. } = pc;
682	let candidate_hash = prospective_candidate.as_ref().map(ProspectiveCandidate::candidate_hash);
683
684	let peer_data = state.peer_data.get(&peer_id).ok_or(FetchError::UnknownPeer)?;
685
686	if peer_data.has_advertised(&relay_parent, candidate_hash) {
687		request_collation(sender, state, pc, id.clone(), peer_data.version).await?;
688		let timeout = |collator_id, candidate_hash, relay_parent| async move {
689			Delay::new(MAX_UNSHARED_DOWNLOAD_TIME).await;
690			(collator_id, candidate_hash, relay_parent)
691		};
692		state
693			.collation_fetch_timeouts
694			.push(timeout(id.clone(), candidate_hash, relay_parent).boxed());
695
696		Ok(())
697	} else {
698		Err(FetchError::NotAdvertised)
699	}
700}

750/// A peer's view has changed. A number of things should be done:
751///  - Ongoing collation requests have to be canceled.
752///  - Advertisements by this peer that are no longer relevant have to be removed.
753fn handle_peer_view_change(state: &mut State, peer_id: PeerId, view: View) {
754	let peer_data = match state.peer_data.get_mut(&peer_id) {
755		Some(peer_data) => peer_data,
756		None => return,
757	};
758
759	peer_data.update_view(&state.implicit_view, &state.active_leaves, view);
760	state.collation_requests_cancel_handles.retain(|pc, handle| {

762		if !keep {

764		}
765		keep
766	});

769/// Request a collation from the network.
770/// This function will
771///  - Check for duplicate requests.
772///  - Check if the requested collation is in our view.
773/// And as such invocations of this function may rely on that.
774async fn request_collation(
775	sender: &mut impl overseer::CollatorProtocolSenderTrait,
776	state: &mut State,
777	pending_collation: PendingCollation,
778	collator_id: CollatorId,
779	peer_protocol_version: CollationVersion,
780) -> std::result::Result<(), FetchError> {
781	if state.collation_requests_cancel_handles.contains_key(&pending_collation) {
782		return Err(FetchError::AlreadyRequested)
783	}
784
785	let PendingCollation { relay_parent, para_id, peer_id, prospective_candidate, .. } =
786		pending_collation;
787	let per_relay_parent = state
788		.per_relay_parent
789		.get_mut(&relay_parent)
790		.ok_or(FetchError::RelayParentOutOfView)?;
791
792	let (requests, response_recv) = match (peer_protocol_version, prospective_candidate) {
793		(CollationVersion::V1, _) => {
794			let (req, response_recv) = OutgoingRequest::new(
795				Recipient::Peer(peer_id),
796				request_v1::CollationFetchingRequest { relay_parent, para_id },
797			);
798			let requests = Requests::CollationFetchingV1(req);
799			(requests, response_recv.boxed())
800		},
801		(CollationVersion::V2, Some(ProspectiveCandidate { candidate_hash, .. })) => {
802			let (req, response_recv) = OutgoingRequest::new(
803				Recipient::Peer(peer_id),
804				request_v2::CollationFetchingRequest { relay_parent, para_id, candidate_hash },
805			);
806			let requests = Requests::CollationFetchingV2(req);
807			(requests, response_recv.boxed())
808		},
809		_ => return Err(FetchError::ProtocolMismatch),
810	};
811
812	let cancellation_token = CancellationToken::new();
813	let collation_request = CollationFetchRequest {
814		pending_collation,
815		collator_id: collator_id.clone(),
816		collator_protocol_version: peer_protocol_version,
817		from_collator: response_recv,
818		cancellation_token: cancellation_token.clone(),
819		_lifetime_timer: state.metrics.time_collation_request_duration(),
820	};
821
822	state.collation_requests.push(collation_request);
823	state
824		.collation_requests_cancel_handles
825		.insert(pending_collation, cancellation_token);
826
827	gum::debug!(
828		target: LOG_TARGET,
829		peer_id = %peer_id,
830		%para_id,
831		?relay_parent,
832		"Requesting collation",
833	);
834
835	let maybe_candidate_hash =
836		prospective_candidate.as_ref().map(ProspectiveCandidate::candidate_hash);
837	per_relay_parent.collations.status = CollationStatus::Fetching(para_id);
838	per_relay_parent
839		.collations
840		.fetching_from
841		.replace((collator_id, maybe_candidate_hash));
842
843	sender
844		.send_message(NetworkBridgeTxMessage::SendRequests(
845			vec![requests],
846			IfDisconnected::ImmediateError,
847		))
848		.await;
849	Ok(())
850}

852/// Networking message has been received.
853#[overseer::contextbounds(CollatorProtocol, prefix = overseer)]
854async fn process_incoming_peer_message<Context>(
855	ctx: &mut Context,
856	state: &mut State,
857	origin: PeerId,
858	msg: CollationProtocols<
859		protocol_v1::CollatorProtocolMessage,
860		protocol_v2::CollatorProtocolMessage,
861	>,
862) {
863	use protocol_v1::CollatorProtocolMessage as V1;
864	use protocol_v2::CollatorProtocolMessage as V2;
865	use sp_runtime::traits::AppVerify;
866
867	match msg {
868		CollationProtocols::V1(V1::Declare(collator_id, para_id, signature)) |
869		CollationProtocols::V2(V2::Declare(collator_id, para_id, signature)) => {
870			if collator_peer_id(&state.peer_data, &collator_id).is_some() {
871				modify_reputation(
872					&mut state.reputation,
873					ctx.sender(),
874					origin,
875					COST_UNEXPECTED_MESSAGE,
876				)
877				.await;
878				return
879			}
880
881			let peer_data = match state.peer_data.get_mut(&origin) {
882				Some(p) => p,
883				None => {
884					gum::debug!(
885						target: LOG_TARGET,
886						peer_id = ?origin,
887						?para_id,
888						"Unknown peer",
889					);
890					modify_reputation(
891						&mut state.reputation,
892						ctx.sender(),
893						origin,
894						COST_UNEXPECTED_MESSAGE,
895					)
896					.await;
897					return
898				},
899			};
900
901			if peer_data.is_collating() {
902				gum::debug!(
903					target: LOG_TARGET,
904					peer_id = ?origin,
905					?para_id,
906					"Peer is already in the collating state",
907				);
908				modify_reputation(
909					&mut state.reputation,
910					ctx.sender(),
911					origin,
912					COST_UNEXPECTED_MESSAGE,
913				)
914				.await;
915				return
916			}
917
918			if !signature.verify(&*protocol_v1::declare_signature_payload(&origin), &collator_id) {
919				gum::debug!(
920					target: LOG_TARGET,
921					peer_id = ?origin,
922					?para_id,
923					"Signature verification failure",
924				);
925				modify_reputation(
926					&mut state.reputation,
927					ctx.sender(),
928					origin,
929					COST_INVALID_SIGNATURE,
930				)
931				.await;
932				return
933			}
934
935			if state.current_assignments.contains_key(&para_id) {
936				gum::debug!(
937					target: LOG_TARGET,
938					peer_id = ?origin,
939					?collator_id,
940					?para_id,
941					"Declared as collator for current para",
942				);
943
944				peer_data.set_collating(collator_id, para_id);
945			} else {
946				gum::debug!(
947					target: LOG_TARGET,
948					peer_id = ?origin,
949					?collator_id,
950					?para_id,
951					"Declared as collator for unneeded para. Current assignments: {:?}",
952					&state.current_assignments
953				);
954
955				modify_reputation(
956					&mut state.reputation,
957					ctx.sender(),
958					origin,
959					COST_UNNEEDED_COLLATOR,
960				)
961				.await;
962				gum::trace!(target: LOG_TARGET, "Disconnecting unneeded collator");
963				disconnect_peer(ctx.sender(), origin).await;
964			}
965		},
966		CollationProtocols::V1(V1::AdvertiseCollation(relay_parent)) => {
967			if let Err(err) =
968				handle_advertisement(ctx.sender(), state, relay_parent, origin, None).await
969			{
970				gum::debug!(
971					target: LOG_TARGET,
972					peer_id = ?origin,
973					?relay_parent,
974					error = ?err,
975					"Rejected v1 advertisement",
976				);
977
978				if let Some(rep) = err.reputation_changes() {
979					modify_reputation(&mut state.reputation, ctx.sender(), origin, rep).await;
980				}
981			}
982		},
983		CollationProtocols::V2(V2::AdvertiseCollation {
984			relay_parent,
985			candidate_hash,
986			parent_head_data_hash,
987		}) => {
988			if let Err(err) = handle_advertisement(
989				ctx.sender(),
990				state,
991				relay_parent,
992				origin,
993				Some((candidate_hash, parent_head_data_hash)),
994			)
995			.await
996			{
997				gum::debug!(
998					target: LOG_TARGET,
999					peer_id = ?origin,
1000					?relay_parent,
1001					?candidate_hash,
1002					error = ?err,
1003					"Rejected v2 advertisement",
1004				);
1005
1006				if let Some(rep) = err.reputation_changes() {
1007					modify_reputation(&mut state.reputation, ctx.sender(), origin, rep).await;
1008				}
1009			}
1010		},
1011		CollationProtocols::V1(V1::CollationSeconded(..)) |
1012		CollationProtocols::V2(V2::CollationSeconded(..)) => {
1013			gum::warn!(
1014				target: LOG_TARGET,
1015				peer_id = ?origin,
1016				"Unexpected `CollationSeconded` message, decreasing reputation",
1017			);
1018
1019			modify_reputation(&mut state.reputation, ctx.sender(), origin, COST_UNEXPECTED_MESSAGE)
1020				.await;
1021		},
1022	}
1023}

1025// Holds off an AssetHub collation, if it's not from an invulnerable AssetHub collator. Returns
1026// `true` if the collation was held off and `false` otherwise.
1027fn hold_off_asset_hub_collation_if_needed(
1028	state: &mut State,
1029	peer_id: PeerId,
1030	collator_id: &CollatorId,
1031	relay_parent: Hash,
1032	prospective_candidate: Option<(CandidateHash, Hash)>,
1033) -> bool {
1034	// If we don't know the peer we should reject the advertisement but to avoid verbosity and
1035	// copy-pasted logic we'll just return `false` and let the caller handle it.
1036	let maybe_para_id = state.peer_data.get(&peer_id).and_then(|pd| pd.collating_para());
1037	let hold_off_duration = state.hold_off_duration;
1038	let collator_id = collator_id.clone();
1039	let peer_is_invulnerable = state.ah_invulnerables.contains(&peer_id);
1040	let invulnerables_set_is_empty = state.ah_invulnerables.is_empty();
1041
1042	if maybe_para_id != Some(ASSET_HUB_PARA_ID) ||
1043		peer_is_invulnerable ||
1044		invulnerables_set_is_empty
1045	{
1046		gum::trace!(
1047			target: LOG_TARGET,
1048			?maybe_para_id,
1049			peer_is_invulnerable,
1050			invulnerables_set_is_empty,
1051			?peer_id,
1052			?relay_parent,
1053			?prospective_candidate,
1054			"Collation not held off",
1055		);
1056
1057		return false
1058	}
1059
1060	let Some(rp_state) = state.per_relay_parent.get_mut(&relay_parent) else {
1061		// this should never happen
1062		gum::warn!(
1063			target: LOG_TARGET,
1064			?peer_id,
1065			?relay_parent,
1066			"Trying to hold off AssetHub collation, but the relay parent is not known",
1067		);
1068		return false
1069	};
1070
1071	if rp_state.ah_held_off_advertisements.hold_off_if_necessary(HeldOffAdvertisement {
1072		relay_parent,
1073		peer_id,
1074		collator_id,
1075		prospective_candidate,
1076	}) {
1077		state.ah_held_off_rp_timers.push(Box::pin(async move {
1078			Delay::new(hold_off_duration).await;
1079			relay_parent
1080		}));
1081
1082		gum::debug!(
1083			target: LOG_TARGET,
1084			?peer_id,
1085			?relay_parent,
1086			?prospective_candidate,
1087			"AssetHub collation held off, not from invulnerable collator",
1088		);
1089
1090		return true;
1091	} else {
1092		gum::debug!(
1093			target: LOG_TARGET,
1094			?peer_id,
1095			?relay_parent,
1096			?prospective_candidate,
1097			"AssetHub collation from non-invulnerable collator not held off - already done for this relay parent",
1098		);
1099
1100		return false
1101	}
1102}

1107	RelayParentUnknown,
1108	/// Peer is not present in the subsystem state.

1135	}

1224fn ensure_seconding_limit_is_respected(
1225	relay_parent: &Hash,
1226	para_id: ParaId,
1227	state: &State,
1228) -> std::result::Result<(), AdvertisementError> {
1229	let paths = state.implicit_view.paths_via_relay_parent(relay_parent);
1230
1231	gum::trace!(
1232		target: LOG_TARGET,
1233		?relay_parent,
1234		?para_id,
1235		?paths,
1236		"Checking seconding limit",
1237	);
1238
1239	let mut has_claim_at_some_path = false;
1240	for path in paths {
1241		let mut cq_state = ClaimQueueState::new();
1242		for ancestor in &path {
1243			let seconded_and_pending = state.seconded_and_pending_for_para(&ancestor, &para_id) +
1244				state.in_waiting_queue_for_para(relay_parent, &para_id);
1245			cq_state.add_leaf(
1246				&ancestor,
1247				&state
1248					.per_relay_parent
1249					.get(ancestor)
1250					.ok_or(AdvertisementError::RelayParentUnknown)?
1251					.assignment
1252					.current,
1253			);
1254			for _ in 0..seconded_and_pending {
1255				cq_state.claim_at(ancestor, &para_id);
1256			}
1257		}
1258
1259		if cq_state.can_claim_at(relay_parent, &para_id) {
1260			has_claim_at_some_path = true;
1261			break
1262		}
1263	}
1264
1265	// If there is a place in the claim queue for the candidate at at least one path we will accept
1266	// it.
1267	if has_claim_at_some_path {
1268		gum::trace!(
1269			target: LOG_TARGET,
1270			?relay_parent,
1271			?para_id,
1272			"Seconding limit respected",
1273		);
1274		Ok(())
1275	} else {
1276		gum::trace!(
1277			target: LOG_TARGET,
1278			?relay_parent,
1279			?para_id,
1280			"Seconding limit not respected",
1281		);
1282		Err(AdvertisementError::SecondedLimitReached)
1283	}
1284}

1286async fn handle_advertisement<Sender>(
1287	sender: &mut Sender,
1288	state: &mut State,
1289	relay_parent: Hash,
1290	peer_id: PeerId,
1291	prospective_candidate: Option<(CandidateHash, Hash)>,
1292) -> std::result::Result<(), AdvertisementError>
1293where
1294	Sender: CollatorProtocolSenderTrait,
1295{
1296	let peer_data = state.peer_data.get_mut(&peer_id).ok_or(AdvertisementError::UnknownPeer)?;
1297
1298	if peer_data.version == CollationVersion::V1 && !state.active_leaves.contains(&relay_parent) {
1299		return Err(AdvertisementError::ProtocolMisuse)
1300	}
1301
1302	let per_relay_parent = state
1303		.per_relay_parent
1304		.get(&relay_parent)
1305		.ok_or(AdvertisementError::RelayParentUnknown)?;
1306
1307	let assignment = &per_relay_parent.assignment;
1308
1309	let collator_para_id =
1310		peer_data.collating_para().ok_or(AdvertisementError::UndeclaredCollator)?;
1311
1312	// Check if this is assigned to us.
1313	if !assignment.current.contains(&collator_para_id) {
1314		return Err(AdvertisementError::InvalidAssignment)
1315	}
1316
1317	// Always insert advertisements that pass all the checks for spam protection.
1318	let candidate_hash = prospective_candidate.map(|(hash, ..)| hash);
1319	let (collator_id, para_id) = peer_data
1320		.insert_advertisement(
1321			relay_parent,
1322			candidate_hash,
1323			&state.implicit_view,
1324			&state.active_leaves,
1325			&per_relay_parent,
1326		)
1327		.map_err(AdvertisementError::Invalid)?;
1328
1329	if hold_off_asset_hub_collation_if_needed(
1330		state,
1331		peer_id,
1332		&collator_id,
1333		relay_parent,
1334		prospective_candidate,
1335	) {
1336		return Ok(())
1337	}
1338
1339	process_advertisement(
1340		sender,
1341		state,
1342		relay_parent,
1343		para_id,
1344		peer_id,
1345		collator_id,
1346		prospective_candidate,
1347	)
1348	.await
1349}

1351async fn process_advertisement<Sender>(
1352	sender: &mut Sender,
1353	state: &mut State,
1354	relay_parent: Hash,
1355	para_id: ParaId,
1356	peer_id: PeerId,
1357	collator_id: CollatorId,
1358	prospective_candidate: Option<(CandidateHash, Hash)>,
1359) -> std::result::Result<(), AdvertisementError>
1360where
1361	Sender: CollatorProtocolSenderTrait,
1362{
1363	ensure_seconding_limit_is_respected(&relay_parent, para_id, state)?;
1364
1365	if let Some((candidate_hash, parent_head_data_hash)) = prospective_candidate {
1366		// Check if backing subsystem allows to second this candidate.
1367		//
1368		// This is also only important when async backing or elastic scaling is enabled.
1369		let can_second =
1370			can_second(sender, para_id, relay_parent, candidate_hash, parent_head_data_hash).await;
1371
1372		if !can_second {
1373			return Err(AdvertisementError::BlockedByBacking)
1374		}
1375	}
1376
1377	let result = enqueue_collation(
1378		sender,
1379		state,
1380		relay_parent,
1381		para_id,
1382		peer_id,
1383		collator_id,
1384		prospective_candidate,
1385	)
1386	.await;
1387
1388	if let Err(fetch_error) = result {
1389		gum::debug!(
1390			target: LOG_TARGET,
1391			relay_parent = ?relay_parent,
1392			para_id = ?para_id,
1393			peer_id = ?peer_id,
1394			error = %fetch_error,
1395			"Failed to request advertised collation",
1396		);
1397	}
1398
1399	Ok(())
1400}

1402/// Enqueue collation for fetching. The advertisement is expected to be validated and the seconding
1403/// limit checked.
1404async fn enqueue_collation<Sender>(
1405	sender: &mut Sender,
1406	state: &mut State,
1407	relay_parent: Hash,
1408	para_id: ParaId,
1409	peer_id: PeerId,
1410	collator_id: CollatorId,
1411	prospective_candidate: Option<(CandidateHash, Hash)>,
1412) -> std::result::Result<(), FetchError>
1413where
1414	Sender: CollatorProtocolSenderTrait,
1415{
1416	gum::debug!(
1417		target: LOG_TARGET,
1418		peer_id = ?peer_id,
1419		%para_id,
1420		?relay_parent,
1421		"Received advertise collation",
1422	);
1423	let per_relay_parent = match state.per_relay_parent.get_mut(&relay_parent) {
1424		Some(rp_state) => rp_state,
1425		None => {
1426			// Race happened, not an error.
1427			gum::trace!(
1428				target: LOG_TARGET,
1429				peer_id = ?peer_id,
1430				%para_id,
1431				?relay_parent,
1432				?prospective_candidate,
1433				"Candidate relay parent went out of view for valid advertisement",
1434			);
1435			return Ok(())
1436		},
1437	};
1438	let prospective_candidate =
1439		prospective_candidate.map(|(candidate_hash, parent_head_data_hash)| ProspectiveCandidate {
1440			candidate_hash,
1441			parent_head_data_hash,
1442		});
1443
1444	let collations = &mut per_relay_parent.collations;
1445	let pending_collation =
1446		PendingCollation::new(relay_parent, para_id, &peer_id, prospective_candidate);
1447
1448	match collations.status {
1449		CollationStatus::Fetching(_) | CollationStatus::WaitingOnValidation => {
1450			gum::trace!(
1451				target: LOG_TARGET,
1452				peer_id = ?peer_id,
1453				%para_id,
1454				?relay_parent,
1455				"Added collation to the pending list"
1456			);
1457			collations.add_to_waiting_queue((pending_collation, collator_id));
1458		},
1459		CollationStatus::Waiting => {
1460			// We were waiting for a collation to be advertised to us (we were idle) so we can fetch
1461			// the new collation immediately
1462			fetch_collation(sender, state, pending_collation, collator_id).await?;
1463		},
1464	}
1465
1466	Ok(())
1467}

1469/// Our view has changed.
1470async fn handle_our_view_change<Sender>(
1471	sender: &mut Sender,
1472	state: &mut State,
1473	keystore: &KeystorePtr,
1474	view: OurView,
1475) -> Result<()>
1476where
1477	Sender: CollatorProtocolSenderTrait,
1478{
1479	let current_leaves = state.active_leaves.clone();
1480
1481	let removed = current_leaves.iter().filter(|h| !view.contains(h));
1482	let added = view.iter().filter(|h| !current_leaves.contains(h));
1483
1484	for leaf in added {
1485		let session_index = request_session_index_for_child(*leaf, sender)
1486			.await
1487			.await
1488			.map_err(Error::CancelledSessionIndex)??;
1489
1490		let v2_receipts = request_node_features(*leaf, session_index, sender)
1491			.await
1492			.await
1493			.map_err(Error::CancelledNodeFeatures)??
1494			.get(node_features::FeatureIndex::CandidateReceiptV2 as usize)
1495			.map(|b| *b)
1496			.unwrap_or(false);
1497
1498		let Some(per_relay_parent) = construct_per_relay_parent(
1499			sender,
1500			&mut state.current_assignments,
1501			keystore,
1502			*leaf,
1503			v2_receipts,
1504			session_index,
1505		)
1506		.await?
1507		else {
1508			continue
1509		};
1510
1511		state.active_leaves.insert(*leaf);
1512		state.per_relay_parent.insert(*leaf, per_relay_parent);
1513
1514		state
1515			.implicit_view
1516			.activate_leaf(sender, *leaf)
1517			.await
1518			.map_err(Error::ImplicitViewFetchError)?;
1519
1520		// Order is always descending.
1521		let allowed_ancestry = state
1522			.implicit_view
1523			.known_allowed_relay_parents_under(leaf, None)
1524			.unwrap_or_default();
1525		for block_hash in allowed_ancestry {
1526			if let Entry::Vacant(entry) = state.per_relay_parent.entry(*block_hash) {
1527				// Safe to use the same v2 receipts config for the allowed relay parents as well
1528				// as the same session index since they must be in the same session.
1529				if let Some(per_relay_parent) = construct_per_relay_parent(
1530					sender,
1531					&mut state.current_assignments,
1532					keystore,
1533					*block_hash,
1534					v2_receipts,
1535					session_index,
1536				)
1537				.await?
1538				{
1539					entry.insert(per_relay_parent);
1540				}
1541			}
1542		}
1543	}
1544
1545	for removed in removed {
1546		gum::trace!(
1547			target: LOG_TARGET,
1548			?view,
1549			?removed,
1550			"handle_our_view_change - removed",
1551		);
1552
1553		state.active_leaves.remove(removed);
1554		// If the leaf is deactivated it still may stay in the view as a part
1555		// of implicit ancestry. Only update the state after the hash is actually
1556		// pruned from the block info storage.
1557		let pruned = state.implicit_view.deactivate_leaf(*removed);
1558
1559		for removed in pruned {
1560			if let Some(per_relay_parent) = state.per_relay_parent.remove(&removed) {
1561				remove_outgoing(&mut state.current_assignments, per_relay_parent);
1562			}
1563
1564			state.collation_requests_cancel_handles.retain(|pc, handle| {
1565				let keep = pc.relay_parent != removed;
1566				if !keep {
1567					handle.cancel();
1568				}
1569				keep
1570			});
1571			state.fetched_candidates.retain(|k, _| k.relay_parent != removed);
1572		}
1573	}
1574
1575	// Remove blocked seconding requests that left the view.
1576	state.blocked_from_seconding.retain(|_, collations| {
1577		collations.retain(|collation| {
1578			state
1579				.per_relay_parent
1580				.contains_key(&collation.candidate_receipt.descriptor.relay_parent())
1581		});
1582
1583		!collations.is_empty()
1584	});
1585
1586	for (peer_id, peer_data) in state.peer_data.iter_mut() {
1587		peer_data.prune_old_advertisements(&state.implicit_view, &state.active_leaves);
1588
1589		// Disconnect peers who are not relevant to our current or next para.
1590		//
1591		// If the peer hasn't declared yet, they will be disconnected if they do not
1592		// declare.
1593		if let Some(para_id) = peer_data.collating_para() {
1594			if !state.current_assignments.contains_key(&para_id) {
1595				gum::trace!(
1596					target: LOG_TARGET,
1597					?peer_id,
1598					?para_id,
1599					"Disconnecting peer on view change (not current parachain id)"
1600				);
1601				disconnect_peer(sender, *peer_id).await;
1602			}
1603		}
1604	}
1605
1606	Ok(())
1607}

1609/// Bridge event switch.
1610#[overseer::contextbounds(CollatorProtocol, prefix = self::overseer)]
1611async fn handle_network_msg<Context>(
1612	ctx: &mut Context,
1613	state: &mut State,
1614	keystore: &KeystorePtr,
1615	bridge_message: NetworkBridgeEvent<net_protocol::CollatorProtocolMessage>,
1616) -> Result<()> {
1617	use NetworkBridgeEvent::*;
1618
1619	match bridge_message {
1620		PeerConnected(peer_id, observed_role, protocol_version, _) => {
1621			let version = match protocol_version.try_into() {
1622				Ok(version) => version,
1623				Err(err) => {
1624					// Network bridge is expected to handle this.
1625					gum::error!(
1626						target: LOG_TARGET,
1627						?peer_id,
1628						?observed_role,
1629						?err,
1630						"Unsupported protocol version"
1631					);
1632					return Ok(())
1633				},
1634			};
1635
1636			// Since the AssetHub system parachain is very important for the operation of the
1637			// relay chain we need to ensure that we always accept connections from the invulnerable
1638			// AssetHub collators.
1639			// However there is a connection limit in `polkadot-node-network-protocol` of the number
1640			// of peers which can connect to an authority node. Once this limit is reached all new
1641			// connections are dropped.
1642			// This means we need to be extra careful when accepting connections from new collators.
1643			// If we are at the point where there are not enough slots for
1644			// invulnerable AssetHub collators we need to drop new connections.
1645			if !can_accept_new_collator_connection(
1646				&peer_id,
1647				&state.peer_data,
1648				&state.ah_invulnerables,
1649			) {
1650				gum::debug!(
1651					target: LOG_TARGET,
1652					?peer_id,
1653					?observed_role,
1654					"Peer connection refused, no slots for invulnerable AssetHub collators",
1655				);
1656				disconnect_peer(ctx.sender(), peer_id).await;
1657				return Ok(())
1658			}
1659
1660			state.peer_data.entry(peer_id).or_insert_with(|| PeerData {
1661				view: View::default(),
1662				state: PeerState::Connected(Instant::now()),
1663				version,
1664			});
1665			state.metrics.note_collator_peer_count(state.peer_data.len());
1666		},
1667		PeerDisconnected(peer_id) => {
1668			state.peer_data.remove(&peer_id);
1669			state.metrics.note_collator_peer_count(state.peer_data.len());
1670		},
1671		NewGossipTopology { .. } => {
1672			// impossible!
1673		},
1674		PeerViewChange(peer_id, view) => {
1675			handle_peer_view_change(state, peer_id, view);
1676		},
1677		OurViewChange(view) => {
1678			handle_our_view_change(ctx.sender(), state, keystore, view).await?;
1679		},
1680		PeerMessage(remote, msg) => {
1681			process_incoming_peer_message(ctx, state, remote, msg).await;
1682		},
1683		UpdatedAuthorityIds { .. } => {
1684			// The validator side doesn't deal with `AuthorityDiscoveryId`s.
1685		},
1686	}
1687
1688	Ok(())
1689}

1691/// The main message receiver switch.
1692#[overseer::contextbounds(CollatorProtocol, prefix = self::overseer)]
1693async fn process_msg<Context>(
1694	ctx: &mut Context,
1695	keystore: &KeystorePtr,
1696	msg: CollatorProtocolMessage,
1697	state: &mut State,
1698) {
1699	use CollatorProtocolMessage::*;
1700
1701	let _timer = state.metrics.time_process_msg();
1702
1703	match msg {
1704		CollateOn(id) => {
1705			gum::warn!(
1706				target: LOG_TARGET,
1707				para_id = %id,
1708				"CollateOn message is not expected on the validator side of the protocol",
1709			);
1710		},
1711		DistributeCollation { .. } => {
1712			gum::warn!(
1713				target: LOG_TARGET,
1714				"DistributeCollation message is not expected on the validator side of the protocol",
1715			);
1716		},
1717		NetworkBridgeUpdate(event) => {
1718			if let Err(e) = handle_network_msg(ctx, state, keystore, event).await {
1719				gum::warn!(
1720					target: LOG_TARGET,
1721					err = ?e,
1722					"Failed to handle incoming network message",
1723				);
1724			}
1725		},
1726		Seconded(parent, stmt) => {
1727			let receipt = match stmt.payload() {
1728				Statement::Seconded(receipt) => receipt,
1729				Statement::Valid(_) => {
1730					gum::warn!(
1731						target: LOG_TARGET,
1732						?stmt,
1733						relay_parent = %parent,
1734						"Seconded message received with a `Valid` statement",
1735					);
1736					return
1737				},
1738			};
1739			let output_head_data = receipt.commitments.head_data.clone();
1740			let output_head_data_hash = receipt.descriptor.para_head();
1741			let fetched_collation = FetchedCollation::from(&receipt.to_plain());
1742			if let Some(CollationEvent { collator_id, pending_collation, .. }) =
1743				state.fetched_candidates.remove(&fetched_collation)
1744			{
1745				let PendingCollation {
1746					relay_parent, peer_id, prospective_candidate, para_id, ..
1747				} = pending_collation;
1748				note_good_collation(
1749					&mut state.reputation,
1750					ctx.sender(),
1751					&state.peer_data,
1752					collator_id.clone(),
1753				)
1754				.await;
1755				if let Some(peer_data) = state.peer_data.get(&peer_id) {
1756					notify_collation_seconded(
1757						ctx.sender(),
1758						peer_id,
1759						peer_data.version,
1760						relay_parent,
1761						stmt,
1762					)
1763					.await;
1764				}
1765
1766				if let Some(rp_state) = state.per_relay_parent.get_mut(&parent) {
1767					rp_state.collations.note_seconded(para_id);
1768				}
1769
1770				// See if we've unblocked other collations for seconding.
1771				second_unblocked_collations(
1772					ctx,
1773					state,
1774					fetched_collation.para_id,
1775					output_head_data,
1776					output_head_data_hash,
1777				)
1778				.await;
1779
1780				// If async backing is enabled, make an attempt to fetch next collation.
1781				let maybe_candidate_hash =
1782					prospective_candidate.as_ref().map(ProspectiveCandidate::candidate_hash);
1783				dequeue_next_collation_and_fetch(
1784					ctx,
1785					state,
1786					parent,
1787					(collator_id, maybe_candidate_hash),
1788				)
1789				.await;
1790			} else {
1791				gum::debug!(
1792					target: LOG_TARGET,
1793					relay_parent = ?parent,
1794					"Collation has been seconded, but the relay parent is deactivated",
1795				);
1796			}
1797		},
1798		Invalid(parent, candidate_receipt) => {
1799			// Remove collations which were blocked from seconding and had this candidate as parent.
1800			state.blocked_from_seconding.remove(&BlockedCollationId {
1801				para_id: candidate_receipt.descriptor.para_id(),
1802				parent_head_data_hash: candidate_receipt.descriptor.para_head(),
1803			});
1804
1805			let fetched_collation = FetchedCollation::from(&candidate_receipt);
1806			let candidate_hash = fetched_collation.candidate_hash;
1807			let id = match state.fetched_candidates.entry(fetched_collation) {
1808				Entry::Occupied(entry)
1809					if entry.get().pending_collation.commitments_hash ==
1810						Some(candidate_receipt.commitments_hash) =>
1811					entry.remove().collator_id,
1812				Entry::Occupied(_) => {
1813					gum::error!(
1814						target: LOG_TARGET,
1815						relay_parent = ?parent,
1816						candidate = ?candidate_receipt.hash(),
1817						"Reported invalid candidate for unknown `pending_candidate`!",
1818					);
1819					return
1820				},
1821				Entry::Vacant(_) => return,
1822			};
1823
1824			report_collator(&mut state.reputation, ctx.sender(), &state.peer_data, id.clone())
1825				.await;
1826
1827			dequeue_next_collation_and_fetch(ctx, state, parent, (id, Some(candidate_hash))).await;
1828		},
1829	}
1830}

1855#[overseer::contextbounds(CollatorProtocol, prefix = self::overseer)]
1856async fn run_inner<Context>(
1857	mut ctx: Context,
1858	keystore: KeystorePtr,
1859	eviction_policy: crate::CollatorEvictionPolicy,
1860	metrics: Metrics,
1861	reputation: ReputationAggregator,
1862	reputation_interval: Duration,
1863	ah_invulnerables: HashSet<PeerId>,
1864	hold_off_duration: Duration,
1865) -> std::result::Result<(), SubsystemError> {
1866	let new_reputation_delay = || futures_timer::Delay::new(reputation_interval).fuse();
1867	let mut reputation_delay = new_reputation_delay();
1868
1869	let mut state =
1870		State { metrics, reputation, ah_invulnerables, hold_off_duration, ..Default::default() };
1871
1872	let next_inactivity_stream = tick_stream(ACTIVITY_POLL);
1873	futures::pin_mut!(next_inactivity_stream);
1874
1875	let mut network_error_freq = gum::Freq::new();
1876	let mut canceled_freq = gum::Freq::new();
1877
1878	loop {
1879		select! {
1880			_ = reputation_delay => {
1881				state.reputation.send(ctx.sender()).await;
1882				reputation_delay = new_reputation_delay();
1883			},
1884			res = ctx.recv().fuse() => {
1885				match res {
1886					Ok(FromOrchestra::Communication { msg }) => {
1887						gum::trace!(target: LOG_TARGET, msg = ?msg, "received a message");
1888						process_msg(
1889							&mut ctx,
1890							&keystore,
1891							msg,
1892							&mut state,
1893						).await;
1894					}
1895					Ok(FromOrchestra::Signal(OverseerSignal::Conclude)) | Err(_) => break,
1896					Ok(FromOrchestra::Signal(_)) => continue,
1897				}
1898			},
1899			_ = next_inactivity_stream.next() => {
1900				disconnect_inactive_peers(ctx.sender(), &eviction_policy, &state.peer_data).await;
1901			},
1902			resp = state.collation_requests.select_next_some() => {
1903				let relay_parent = resp.0.pending_collation.relay_parent;
1904				let res = match handle_collation_fetch_response(
1905					&mut state,
1906					resp,
1907					&mut network_error_freq,
1908					&mut canceled_freq,
1909				).await {
1910					Err(Some((peer_id, rep))) => {
1911						modify_reputation(&mut state.reputation, ctx.sender(), peer_id, rep).await;
1912						// Reset the status for the relay parent
1913						state.per_relay_parent.get_mut(&relay_parent).map(|rp| {
1914							rp.collations.status.back_to_waiting();
1915						});
1916						continue
1917					},
1918					Err(None) => {
1919						// Reset the status for the relay parent
1920						state.per_relay_parent.get_mut(&relay_parent).map(|rp| {
1921							rp.collations.status.back_to_waiting();
1922						});
1923						continue
1924					},
1925					Ok(res) => res
1926				};
1927
1928				let CollationEvent {collator_id, pending_collation, .. } = res.collation_event.clone();
1929
1930				match kick_off_seconding(&mut ctx, &mut state, res).await {
1931					Err(err) => {
1932						gum::warn!(
1933							target: LOG_TARGET,
1934							relay_parent = ?pending_collation.relay_parent,
1935							para_id = ?pending_collation.para_id,
1936							peer_id = ?pending_collation.peer_id,
1937							error = %err,
1938							"Seconding aborted due to an error",
1939						);
1940
1941						if err.is_malicious() {
1942							// Report malicious peer.
1943							modify_reputation(&mut state.reputation, ctx.sender(), pending_collation.peer_id, COST_REPORT_BAD).await;
1944						}
1945						let maybe_candidate_hash =
1946						pending_collation.prospective_candidate.as_ref().map(ProspectiveCandidate::candidate_hash);
1947						dequeue_next_collation_and_fetch(
1948							&mut ctx,
1949							&mut state,
1950							pending_collation.relay_parent,
1951							(collator_id, maybe_candidate_hash),
1952						)
1953						.await;
1954					},
1955					Ok(false) => {
1956						// No hard error occurred, but we can try fetching another collation.
1957						let maybe_candidate_hash =
1958						pending_collation.prospective_candidate.as_ref().map(ProspectiveCandidate::candidate_hash);
1959						dequeue_next_collation_and_fetch(
1960							&mut ctx,
1961							&mut state,
1962							pending_collation.relay_parent,
1963							(collator_id, maybe_candidate_hash),
1964						)
1965						.await;
1966					}
1967					Ok(true) => {}
1968				}
1969			},
1970			res = state.collation_fetch_timeouts.select_next_some() => {
1971				let (collator_id, maybe_candidate_hash, relay_parent) = res;
1972				gum::debug!(
1973					target: LOG_TARGET,
1974					?relay_parent,
1975					?collator_id,
1976					"Timeout hit - already seconded?"
1977				);
1978				dequeue_next_collation_and_fetch(
1979					&mut ctx,
1980					&mut state,
1981					relay_parent,
1982					(collator_id, maybe_candidate_hash),
1983				)
1984				.await;
1985			},
1986			rp = state.ah_held_off_rp_timers.select_next_some() => {
1987				let Some(held_off_advertisements) = state.per_relay_parent.get_mut(&rp).map(|rp_state| rp_state.ah_held_off_advertisements.on_hold_off_complete()) else {
1988					gum::debug!(
1989						target: LOG_TARGET,
1990						?rp,
1991						"Received AssetHub held off advertisements timeout for unknown relay parent",
1992					);
1993					continue
1994				};
1995
1996				let held_off_advertisements = match held_off_advertisements {
1997					Ok(held_off) => held_off,
1998					Err(err) => {
1999						gum::warn!(
2000							target: LOG_TARGET,
2001							?rp,
2002							?err,
2003							"Failed to get held off advertisements for relay parent",
2004						);
2005						continue;
2006					}
2007				};
2008
2009				for held_off_advertisement in held_off_advertisements {
2010					let HeldOffAdvertisement{relay_parent, peer_id, collator_id, prospective_candidate} = held_off_advertisement;
2011					gum::debug!(
2012						target: LOG_TARGET,
2013						?relay_parent,
2014						?peer_id,
2015						?prospective_candidate,
2016						"Processing held off advertisement for AssetHub",
2017					);
2018
2019					if let Err(err) = process_advertisement(
2020						ctx.sender(),
2021						&mut state,
2022						relay_parent,
2023						ASSET_HUB_PARA_ID,
2024						peer_id,
2025						collator_id,
2026						prospective_candidate,
2027					)
2028					.await {
2029						gum::debug!(
2030								target: LOG_TARGET,
2031								?err,
2032								?relay_parent,
2033								?peer_id,
2034								?prospective_candidate,
2035								"Failed to handle held off advertisement",
2036							);
2037						if let Some(rep) = err.reputation_changes() {
2038							modify_reputation(&mut state.reputation, ctx.sender(), peer_id, rep).await;
2039						}
2040					}
2041				}
2042			},
2043		}
2044	}
2045
2046	Ok(())
2047}

2129/// Handle a fetched collation result.
2130/// Returns whether or not seconding has begun.
2131#[overseer::contextbounds(CollatorProtocol, prefix = self::overseer)]
2132async fn kick_off_seconding<Context>(
2133	ctx: &mut Context,
2134	state: &mut State,
2135	PendingCollationFetch { mut collation_event, candidate_receipt, pov, maybe_parent_head_data }: PendingCollationFetch,
2136) -> std::result::Result<bool, SecondingError> {
2137	let pending_collation = collation_event.pending_collation;
2138	let relay_parent = pending_collation.relay_parent;
2139
2140	let per_relay_parent = match state.per_relay_parent.get_mut(&relay_parent) {
2141		Some(state) => state,
2142		None => {
2143			// Relay parent went out of view, not an error.
2144			gum::trace!(
2145				target: LOG_TARGET,
2146				relay_parent = ?relay_parent,
2147				"Fetched collation for a parent out of view",
2148			);
2149			return Ok(false)
2150		},
2151	};
2152
2153	// Sanity check of the candidate receipt version.
2154	descriptor_version_sanity_check(candidate_receipt.descriptor(), per_relay_parent)?;
2155
2156	let collations = &mut per_relay_parent.collations;
2157
2158	let fetched_collation = FetchedCollation::from(&candidate_receipt);
2159	if let Entry::Vacant(entry) = state.fetched_candidates.entry(fetched_collation) {
2160		collation_event.pending_collation.commitments_hash =
2161			Some(candidate_receipt.commitments_hash);
2162
2163		let (maybe_pvd, maybe_parent_head, maybe_parent_head_hash) = match (
2164			collation_event.collator_protocol_version,
2165			collation_event.pending_collation.prospective_candidate,
2166		) {
2167			(CollationVersion::V2, Some(ProspectiveCandidate { parent_head_data_hash, .. })) => {
2168				let pvd = request_prospective_validation_data(
2169					ctx.sender(),
2170					relay_parent,
2171					parent_head_data_hash,
2172					pending_collation.para_id,
2173					maybe_parent_head_data.clone(),
2174				)
2175				.await?;
2176
2177				(pvd, maybe_parent_head_data, Some(parent_head_data_hash))
2178			},
2179			(CollationVersion::V1, _) => {
2180				let pvd = request_persisted_validation_data(
2181					ctx.sender(),
2182					candidate_receipt.descriptor().relay_parent(),
2183					candidate_receipt.descriptor().para_id(),
2184				)
2185				.await?;
2186				(
2187					Some(pvd.ok_or(SecondingError::PersistedValidationDataNotFound)?),
2188					maybe_parent_head_data,
2189					None,
2190				)
2191			},
2192			_ => {
2193				// `handle_advertisement` checks for protocol mismatch.
2194				return Ok(false)
2195			},
2196		};
2197
2198		let pvd = match (maybe_pvd, maybe_parent_head.clone(), maybe_parent_head_hash) {
2199			(Some(pvd), _, _) => pvd,
2200			(None, None, Some(parent_head_data_hash)) => {
2201				// In this case, the collator did not supply the head data and neither could
2202				// prospective-parachains. We add this to the blocked_from_seconding collection
2203				// until we second its parent.
2204				let blocked_collation = PendingCollationFetch {
2205					collation_event,
2206					candidate_receipt,
2207					pov,
2208					maybe_parent_head_data: None,
2209				};
2210				gum::debug!(
2211					target: LOG_TARGET,
2212					candidate_hash = ?blocked_collation.candidate_receipt.hash(),
2213					relay_parent = ?blocked_collation.candidate_receipt.descriptor.relay_parent(),
2214					"Collation having parent head data hash {} is blocked from seconding. Waiting on its parent to be validated.",
2215					parent_head_data_hash
2216				);
2217				state
2218					.blocked_from_seconding
2219					.entry(BlockedCollationId {
2220						para_id: blocked_collation.candidate_receipt.descriptor.para_id(),
2221						parent_head_data_hash,
2222					})
2223					.or_insert_with(Vec::new)
2224					.push(blocked_collation);
2225
2226				return Ok(false)
2227			},
2228			(None, _, _) => {
2229				// Even though we already have the parent head data, the pvd fetching failed. We
2230				// don't need to wait for seconding another collation outputting this head data.
2231				return Err(SecondingError::PersistedValidationDataNotFound)
2232			},
2233		};
2234
2235		fetched_collation_sanity_check(
2236			&collation_event.pending_collation,
2237			&candidate_receipt,
2238			&pvd,
2239			maybe_parent_head.and_then(|head| maybe_parent_head_hash.map(|hash| (head, hash))),
2240		)?;
2241
2242		ctx.send_message(CandidateBackingMessage::Second(
2243			relay_parent,
2244			candidate_receipt,
2245			pvd,
2246			pov,
2247		))
2248		.await;
2249		// There's always a single collation being fetched at any moment of time.
2250		// In case of a failure, we reset the status back to waiting.
2251		collations.status = CollationStatus::WaitingOnValidation;
2252
2253		entry.insert(collation_event);
2254		Ok(true)
2255	} else {
2256		Err(SecondingError::Duplicate)
2257	}
2258}

2276/// Handle a collation fetch response.
2277async fn handle_collation_fetch_response(
2278	state: &mut State,
2279	response: <CollationFetchRequest as Future>::Output,
2280	network_error_freq: &mut gum::Freq,
2281	canceled_freq: &mut gum::Freq,
2282) -> std::result::Result<PendingCollationFetch, Option<(PeerId, Rep)>> {
2283	let (CollationEvent { collator_id, collator_protocol_version, pending_collation }, response) =
2284		response;
2285	// Remove the cancellation handle, as the future already completed.
2286	state.collation_requests_cancel_handles.remove(&pending_collation);
2287
2288	let response = match response {
2289		Err(CollationFetchError::Cancelled) => {
2290			gum::debug!(
2291				target: LOG_TARGET,
2292				hash = ?pending_collation.relay_parent,
2293				para_id = ?pending_collation.para_id,
2294				peer_id = ?pending_collation.peer_id,
2295				"Request was cancelled from the validator side"
2296			);
2297			return Err(None)
2298		},
2299		Err(CollationFetchError::Request(req_error)) => Err(req_error),
2300		Ok(resp) => Ok(resp),
2301	};
2302
2303	let _timer = state.metrics.time_handle_collation_request_result();
2304
2305	let mut metrics_result = Err(());
2306

2308		Err(RequestError::InvalidResponse(err)) => {
2309			gum::warn!(
2310				target: LOG_TARGET,
2311				hash = ?pending_collation.relay_parent,
2312				para_id = ?pending_collation.para_id,
2313				peer_id = ?pending_collation.peer_id,
2314				err = ?err,
2315				"Collator provided response that could not be decoded"
2316			);
2317			Err(Some((pending_collation.peer_id, COST_CORRUPTED_MESSAGE)))
2318		},

2320			gum::debug!(
2321				target: LOG_TARGET,
2322				hash = ?pending_collation.relay_parent,
2323				para_id = ?pending_collation.para_id,
2324				peer_id = ?pending_collation.peer_id,
2325				"Request timed out"
2326			);

2330		},

2350				freq: canceled_freq,
2351				max_rate: gum::Times::PerHour(100),

2353				hash = ?pending_collation.relay_parent,
2354				para_id = ?pending_collation.para_id,
2355				peer_id = ?pending_collation.peer_id,
2356				err = ?err,
2357				"Canceled should be handled by `is_timed_out` above - this is a bug!"
2358			);
2359			Err(None)
2360		},
2361		Ok(
2362			request_v1::CollationFetchingResponse::Collation(receipt, _) |

2365			request_v2::CollationFetchingResponse::CollationWithParentHeadData { receipt, .. },

2370				got_para_id = ?receipt.descriptor().para_id(),
2371				peer_id = ?pending_collation.peer_id,
2372				"Got wrong para ID for requested collation."
2373			);
2374
2375			Err(Some((pending_collation.peer_id, COST_WRONG_PARA)))
2376		},
2377		Ok(request_v1::CollationFetchingResponse::Collation(candidate_receipt, pov)) => {
2378			gum::debug!(
2379				target: LOG_TARGET,
2380				para_id = %pending_collation.para_id,
2381				hash = ?pending_collation.relay_parent,
2382				candidate_hash = ?candidate_receipt.hash(),
2383				"Received collation",
2384			);
2385
2386			metrics_result = Ok(());
2387			Ok(PendingCollationFetch {
2388				collation_event: CollationEvent {
2389					collator_id,
2390					pending_collation,
2391					collator_protocol_version,
2392				},
2393				candidate_receipt,
2394				pov,
2395				maybe_parent_head_data: None,
2396			})
2397		},
2398		Ok(request_v2::CollationFetchingResponse::CollationWithParentHeadData {
2399			receipt,
2400			pov,
2401			parent_head_data,
2402		}) => {
2403			gum::debug!(
2404				target: LOG_TARGET,
2405				para_id = %pending_collation.para_id,
2406				hash = ?pending_collation.relay_parent,
2407				candidate_hash = ?receipt.hash(),
2408				"Received collation (v3)",
2409			);
2410
2411			metrics_result = Ok(());
2412			Ok(PendingCollationFetch {
2413				collation_event: CollationEvent {
2414					collator_id,
2415					pending_collation,
2416					collator_protocol_version,
2417				},
2418				candidate_receipt: receipt,
2419				pov,
2420				maybe_parent_head_data: Some(parent_head_data),
2421			})
2422		},
2423	};
2424	state.metrics.on_request(metrics_result);
2425	result
2426}

2428// Returns the claim queue without fetched or pending advertisement. The resulting `Vec` keeps the
2429// order in the claim queue so the earlier an element is located in the `Vec` the higher its
2430// priority is.
2431fn unfulfilled_claim_queue_entries(relay_parent: &Hash, state: &State) -> Result<Vec<ParaId>> {
2432	let relay_parent_state = state
2433		.per_relay_parent
2434		.get(relay_parent)
2435		.ok_or(Error::RelayParentStateNotFound)?;
2436	let scheduled_paras = relay_parent_state.assignment.current.iter().collect::<HashSet<_>>();
2437	let paths = state.implicit_view.paths_via_relay_parent(relay_parent);
2438
2439	let mut claim_queue_states = Vec::new();
2440	for path in paths {
2441		let mut cq_state = ClaimQueueState::new();
2442		for ancestor in &path {
2443			cq_state.add_leaf(
2444				&ancestor,
2445				&state
2446					.per_relay_parent
2447					.get(&ancestor)
2448					.ok_or(Error::RelayParentStateNotFound)?
2449					.assignment
2450					.current,
2451			);
2452
2453			for para_id in &scheduled_paras {
2454				let seconded_and_pending = state.seconded_and_pending_for_para(&ancestor, &para_id);
2455				for _ in 0..seconded_and_pending {
2456					cq_state.claim_at(&ancestor, &para_id);
2457				}
2458			}
2459		}
2460		claim_queue_states.push(cq_state);
2461	}
2462
2463	// From the claim queue state for each leaf we have to return a combined single one. Go for a
2464	// simple solution and return the longest one. In theory we always prefer the earliest entries
2465	// in the claim queue so there is a good chance that the longest path is the one with
2466	// unsatisfied entries in the beginning. This is not guaranteed as we might have fetched 2nd or
2467	// 3rd spot from the claim queue but it should be good enough.
2468	let unfulfilled_entries = claim_queue_states
2469		.iter_mut()
2470		.map(|cq| cq.unclaimed_at(relay_parent))
2471		.max_by(|a, b| a.len().cmp(&b.len()))
2472		.unwrap_or_default();
2473
2474	Ok(unfulfilled_entries)
2475}

2477/// Returns the next collation to fetch from the `waiting_queue` and reset the status back to
2478/// `Waiting`.
2479fn get_next_collation_to_fetch(
2480	finished_one: &(CollatorId, Option<CandidateHash>),
2481	relay_parent: Hash,
2482	state: &mut State,
2483) -> Option<(PendingCollation, CollatorId)> {
2484	let unfulfilled_entries = match unfulfilled_claim_queue_entries(&relay_parent, &state) {
2485		Ok(entries) => entries,
2486		Err(err) => {
2487			gum::error!(
2488				target: LOG_TARGET,
2489				?relay_parent,
2490				?err,
2491				"Failed to get unfulfilled claim queue entries"
2492			);
2493			return None
2494		},
2495	};
2496	let rp_state = match state.per_relay_parent.get_mut(&relay_parent) {
2497		Some(rp_state) => rp_state,
2498		None => {
2499			gum::error!(
2500				target: LOG_TARGET,
2501				?relay_parent,
2502				"Failed to get relay parent state"
2503			);
2504			return None
2505		},
2506	};
2507
2508	// If finished one does not match waiting_collation, then we already dequeued another fetch
2509	// to replace it.
2510	if let Some((collator_id, maybe_candidate_hash)) = rp_state.collations.fetching_from.as_ref() {
2511		// If a candidate hash was saved previously, `finished_one` must include this too.
2512		if collator_id != &finished_one.0 &&
2513			maybe_candidate_hash.map_or(true, |hash| Some(&hash) != finished_one.1.as_ref())
2514		{
2515			gum::trace!(
2516				target: LOG_TARGET,
2517				waiting_collation = ?rp_state.collations.fetching_from,
2518				?finished_one,
2519				"Not proceeding to the next collation - has already been done."
2520			);
2521			return None
2522		}
2523	}
2524	rp_state.collations.status.back_to_waiting();
2525	rp_state.collations.pick_a_collation_to_fetch(unfulfilled_entries)
2526}

2528// Sanity check the candidate descriptor version.
2529fn descriptor_version_sanity_check(
2530	descriptor: &CandidateDescriptorV2,
2531	per_relay_parent: &PerRelayParent,
2532) -> std::result::Result<(), SecondingError> {
2533	match descriptor.version() {
2534		CandidateDescriptorVersion::V1 => Ok(()),
2535		CandidateDescriptorVersion::V2 if per_relay_parent.v2_receipts => {
2536			if let Some(core_index) = descriptor.core_index() {
2537				if core_index != per_relay_parent.current_core {
2538					return Err(SecondingError::InvalidCoreIndex(
2539						core_index.0,
2540						per_relay_parent.current_core.0,
2541					))
2542				}
2543			}
2544
2545			if let Some(session_index) = descriptor.session_index() {
2546				if session_index != per_relay_parent.session_index {
2547					return Err(SecondingError::InvalidSessionIndex(
2548						session_index,
2549						per_relay_parent.session_index,
2550					))
2551				}
2552			}
2553
2554			Ok(())
2555		},
2556		descriptor_version => Err(SecondingError::InvalidReceiptVersion(descriptor_version)),
2557	}
2558}

2560// Checks that there are enough free connection slots for AssetHub invulnerable collators.
2561fn can_accept_new_collator_connection(
2562	new_peer_id: &PeerId,
2563	currently_connected: &HashMap<PeerId, PeerData>,
2564	ah_invulnerables: &HashSet<PeerId>,
2565) -> bool {
2566	if ah_invulnerables.contains(new_peer_id) || ah_invulnerables.is_empty() {
2567		// If the new peer is AH invulnerable, we always accept it.
2568		return true;
2569	}
2570
2571	// We want to keep some space for new connections. If we use the whole limit we could end up in
2572	// the case when a collator keeps on making a connection requests and with some luck (or
2573	// something else I don't anticipate right now) it keeps on connecting before the other peers
2574	// and thus blocking AssetHub invulnerables from connecting.
2575	let max_accepted_connections = MAX_AUTHORITY_INCOMING_STREAMS as usize - 10;
2576	let ah_invulnerables_connected = currently_connected
2577		.keys()
2578		.filter(|peer_id| ah_invulnerables.contains(peer_id))
2579		.count();
2580
2581	// Reserve space for any unconnected AH invulnerable which might want to connect to us.
2582	let connection_limit =
2583		max_accepted_connections - ah_invulnerables.len() + ah_invulnerables_connected;
2584
2585	gum::trace!(
2586		target: LOG_TARGET,
2587		?new_peer_id,
2588		?ah_invulnerables,
2589		currently_connected=currently_connected.len(),
2590		?connection_limit,
2591		"Checking if we can accept a new collator connection",
2592	);
2593
2594	currently_connected.len() < connection_limit
2595}

166impl PeerData {
167	/// Update the view, clearing all advertisements that are no longer in the
168	/// current view.
169	fn update_view(
170		&mut self,
171		implicit_view: &ImplicitView,
172		active_leaves: &HashSet<Hash>,
173		new_view: View,
174	) {
175		let old_view = std::mem::replace(&mut self.view, new_view);
176		if let PeerState::Collating(ref mut peer_state) = self.state {
177			for removed in old_view.difference(&self.view) {
178				// Remove relay parent advertisements if it went out of our (implicit) view.
179				let keep = is_relay_parent_in_implicit_view(
180					removed,
181					implicit_view,
182					active_leaves,
183					peer_state.para_id,
184				);
185
186				if !keep {
187					peer_state.advertisements.remove(&removed);
188				}
189			}
190		}
191	}
192
193	/// Prune old advertisements relative to our view.
194	fn prune_old_advertisements(
195		&mut self,
196		implicit_view: &ImplicitView,
197		active_leaves: &HashSet<Hash>,
198	) {
199		if let PeerState::Collating(ref mut peer_state) = self.state {
200			peer_state.advertisements.retain(|hash, _| {
201				// Either
202				// - Relay parent is an active leaf
203				// - It belongs to allowed ancestry under some leaf
204				// Discard otherwise.
205				is_relay_parent_in_implicit_view(
206					hash,
207					implicit_view,
208					active_leaves,
209					peer_state.para_id,
210				)
211			});
212		}
213	}
214
215	/// Performs sanity check for an advertisement and notes it as advertised.
216	fn insert_advertisement(
217		&mut self,
218		on_relay_parent: Hash,
219		candidate_hash: Option<CandidateHash>,
220		implicit_view: &ImplicitView,
221		active_leaves: &HashSet<Hash>,
222		per_relay_parent: &PerRelayParent,
223	) -> std::result::Result<(CollatorId, ParaId), InsertAdvertisementError> {
224		match self.state {
225			PeerState::Connected(_) => Err(InsertAdvertisementError::UndeclaredCollator),
226			PeerState::Collating(ref mut state) => {
227				if !is_relay_parent_in_implicit_view(
228					&on_relay_parent,
229					implicit_view,
230					active_leaves,
231					state.para_id,
232				) {
233					return Err(InsertAdvertisementError::OutOfOurView)
234				}
235
236				if let Some(candidate_hash) = candidate_hash {
237					if state
238						.advertisements
239						.get(&on_relay_parent)
240						.map_or(false, |candidates| candidates.contains(&candidate_hash))
241					{
242						return Err(InsertAdvertisementError::Duplicate)
243					}
244
245					let candidates = state.advertisements.entry(on_relay_parent).or_default();
246
247					// Current assignments is equal to the length of the claim queue. No honest
248					// collator should send that many advertisements.
249					if candidates.len() > per_relay_parent.assignment.current.len() {
250						return Err(InsertAdvertisementError::PeerLimitReached)
251					}
252
253					candidates.insert(candidate_hash);
254				} else {
255					if self.version != CollationVersion::V1 {
256						gum::error!(
257							target: LOG_TARGET,
258							"Programming error, `candidate_hash` can not be `None` \
259							 for non `V1` networking.",
260						);
261					}
262
263					if state.advertisements.contains_key(&on_relay_parent) {
264						return Err(InsertAdvertisementError::Duplicate)
265					}
266
267					state
268						.advertisements
269						.insert(on_relay_parent, HashSet::from_iter(candidate_hash));
270				};
271
272				state.last_active = Instant::now();
273				Ok((state.collator_id.clone(), state.para_id))
274			},
275		}
276	}
277
278	/// Whether a peer is collating.
279	fn is_collating(&self) -> bool {
280		match self.state {
281			PeerState::Connected(_) => false,
282			PeerState::Collating(_) => true,
283		}
284	}
285
286	/// Note that a peer is now collating with the given collator and para id.
287	///
288	/// This will overwrite any previous call to `set_collating` and should only be called
289	/// if `is_collating` is false.
290	fn set_collating(&mut self, collator_id: CollatorId, para_id: ParaId) {
291		self.state = PeerState::Collating(CollatingPeerState {
292			collator_id,
293			para_id,
294			advertisements: HashMap::new(),
295			last_active: Instant::now(),
296		});
297	}
298
299	fn collator_id(&self) -> Option<&CollatorId> {
300		match self.state {
301			PeerState::Connected(_) => None,
302			PeerState::Collating(ref state) => Some(&state.collator_id),
303		}
304	}
305
306	fn collating_para(&self) -> Option<ParaId> {
307		match self.state {
308			PeerState::Connected(_) => None,
309			PeerState::Collating(ref state) => Some(state.para_id),
310		}
311	}
312
313	/// Whether the peer has advertised the given collation.
314	fn has_advertised(
315		&self,
316		relay_parent: &Hash,
317		maybe_candidate_hash: Option<CandidateHash>,
318	) -> bool {
319		let collating_state = match self.state {
320			PeerState::Connected(_) => return false,
321			PeerState::Collating(ref state) => state,
322		};
323
324		if let Some(ref candidate_hash) = maybe_candidate_hash {
325			collating_state
326				.advertisements
327				.get(relay_parent)
328				.map_or(false, |candidates| candidates.contains(candidate_hash))
329		} else {
330			collating_state.advertisements.contains_key(relay_parent)
331		}
332	}
333
334	/// Whether the peer is now inactive according to the current instant and the eviction policy.
335	fn is_inactive(&self, policy: &crate::CollatorEvictionPolicy) -> bool {
336		match self.state {
337			PeerState::Connected(connected_at) => connected_at.elapsed() >= policy.undeclared,
338			PeerState::Collating(ref state) =>
339				state.last_active.elapsed() >= policy.inactive_collator,
340		}
341	}

350/// Represents the result from a hold off operation.
351enum HoldOffOperationOutcome {
352	/// The advertisement was held off and this is the first hold off for the relay parent.
353	FirstHoldOff,
354	/// The advertisement was held off and this is a subsequent hold off for this relay parent.
355	SubsequentHoldOff,
356	/// The advertisement was not held off. The hold off period for this relay parent has already
357	/// expired.
358	NoHoldOff,
359}
360
361/// Represents the hold off state of a relay parent.
362enum RelayParentHoldOffState {
363	/// No Advertisements from non-invulnerable collators were received so far. Nothing was held
364	/// off.
365	NotStarted,
366	/// One or more advertisements from non-invulnerable collators are being held off. Any new
367	/// advertisements received in this state are added to the queue.
368	HoldingOff(VecDeque<HeldOffAdvertisement>),
369	/// One or more advertisements were held off. We are not holding off any new advertisements.
370	Done,
371}
372
373impl RelayParentHoldOffState {
374	/// If called in `HoldingOff` state - returns all held off advertisements (there should be at
375	/// least one). If the function is called in a bad state - an error is returned.
376	fn on_hold_off_complete(
377		&mut self,
378	) -> std::result::Result<VecDeque<HeldOffAdvertisement>, HoldOffError> {
379		match std::mem::replace(self, RelayParentHoldOffState::Done) {
380			Self::NotStarted => {
381				*self = Self::NotStarted;
382				Err(HoldOffError::InvalidStateNotStarted)
383			},
384			Self::HoldingOff(held_off) => {
385				*self = RelayParentHoldOffState::Done;
386				if held_off.is_empty() {
387					// the queue should never be empty
388					Err(HoldOffError::QueueEmpty)
389				} else {
390					Ok(held_off)
391				}
392			},
393			Self::Done => {
394				*self = Self::Done;
395				Err(HoldOffError::InvalidStateDone)
396			},
397		}
398	}
399
400	/// Holds off the advertisement, if nothing was held off for this relay parent or there are
401	/// other advertisements being held off. Return false if the hold off was complete for the
402	/// relay parent.
403	fn hold_off_if_necessary(
404		&mut self,
405		advertisement: HeldOffAdvertisement,
406	) -> HoldOffOperationOutcome {
407		match self {
408			Self::NotStarted => {
409				*self = Self::HoldingOff(vec![advertisement].into());
410				HoldOffOperationOutcome::FirstHoldOff
411			},
412			Self::HoldingOff(advertisements) => {
413				advertisements.push_back(advertisement);
414				HoldOffOperationOutcome::SubsequentHoldOff
415			},
416			Self::Done => HoldOffOperationOutcome::NoHoldOff,
417		}
418	}
419}

513impl State {
514	// Returns the number of seconded and pending collations for a specific `ParaId`. Pending
515	// collations are:
516	// 1. Collations being fetched from a collator.
517	// 2. Collations waiting for validation from backing subsystem.
518	// 3. Collations blocked from seconding due to parent not being known by backing subsystem.
519	fn seconded_and_pending_for_para(&self, relay_parent: &Hash, para_id: &ParaId) -> usize {
520		let seconded = self
521			.per_relay_parent
522			.get(relay_parent)
523			.map_or(0, |per_relay_parent| per_relay_parent.collations.seconded_for_para(para_id));
524
525		let pending_fetch = self.per_relay_parent.get(relay_parent).map_or(0, |rp_state| {
526			match rp_state.collations.status {
527				CollationStatus::Fetching(pending_para_id) if pending_para_id == *para_id => 1,
528				_ => 0,
529			}
530		});
531
532		let waiting_for_validation = self
533			.fetched_candidates
534			.keys()
535			.filter(|fc| fc.relay_parent == *relay_parent && fc.para_id == *para_id)
536			.count();
537
538		let blocked_from_seconding =
539			self.blocked_from_seconding.values().fold(0, |acc, blocked_collations| {
540				acc + blocked_collations
541					.iter()
542					.filter(|pc| {
543						pc.candidate_receipt.descriptor.para_id() == *para_id &&
544							pc.candidate_receipt.descriptor.relay_parent() == *relay_parent
545					})
546					.count()
547			});
548
549		gum::trace!(
550			target: LOG_TARGET,
551			?relay_parent,
552			?para_id,
553			seconded,
554			pending_fetch,
555			waiting_for_validation,
556			blocked_from_seconding,
557			"Seconded and pending collations for para",
558		);
559
560		seconded + pending_fetch + waiting_for_validation + blocked_from_seconding
561	}
562
563	/// Returns the number of collations pending to be fetched for a `ParaId`
564	fn in_waiting_queue_for_para(&self, relay_parent: &Hash, para_id: &ParaId) -> usize {
565		self.per_relay_parent
566			.get(relay_parent)
567			.map_or(0, |rp_state| rp_state.collations.queued_for_para(para_id))
568	}
569}

571fn is_relay_parent_in_implicit_view(
572	relay_parent: &Hash,
573	implicit_view: &ImplicitView,

578		implicit_view
579			.known_allowed_relay_parents_under(hash, Some(para_id))

581			.contains(relay_parent)
582	})

585async fn construct_per_relay_parent<Sender>(
586	sender: &mut Sender,
587	current_assignments: &mut HashMap<ParaId, usize>,
588	keystore: &KeystorePtr,
589	relay_parent: Hash,
590	v2_receipts: bool,
591	session_index: SessionIndex,
592) -> Result<Option<PerRelayParent>>
593where
594	Sender: CollatorProtocolSenderTrait,
595{
596	let validators = polkadot_node_subsystem_util::request_validators(relay_parent, sender)
597		.await
598		.await
599		.map_err(Error::CancelledActiveValidators)??;
600
601	let (groups, rotation_info) =
602		polkadot_node_subsystem_util::request_validator_groups(relay_parent, sender)
603			.await
604			.await
605			.map_err(Error::CancelledValidatorGroups)??;
606
607	let core_now = if let Some(group) =
608		polkadot_node_subsystem_util::signing_key_and_index(&validators, keystore).and_then(
609			|(_, index)| polkadot_node_subsystem_util::find_validator_group(&groups, index),
610		) {
611		rotation_info.core_for_group(group, groups.len())
612	} else {
613		gum::trace!(target: LOG_TARGET, ?relay_parent, "Not a validator");
614		return Ok(None)
615	};
616
617	let mut claim_queue = request_claim_queue(relay_parent, sender)
618		.await
619		.await
620		.map_err(Error::CancelledClaimQueue)??;
621
622	let assigned_paras = claim_queue.remove(&core_now).unwrap_or_else(|| VecDeque::new());
623
624	for para_id in assigned_paras.iter() {
625		let entry = current_assignments.entry(*para_id).or_default();
626		*entry += 1;
627		if *entry == 1 {
628			gum::debug!(
629				target: LOG_TARGET,
630				?relay_parent,
631				para_id = ?para_id,
632				"Assigned to a parachain",
633			);
634		}
635	}
636
637	let assignment = GroupAssignments { current: assigned_paras.into_iter().collect() };
638	let collations = Collations::new(&assignment.current);
639
640	Ok(Some(PerRelayParent {
641		assignment,
642		collations,
643		v2_receipts,
644		session_index,
645		current_core: core_now,
646		ah_held_off_advertisements: RelayParentHoldOffState::NotStarted,
647	}))
648}

653) {

667		}
668	}

688/// Another subsystem has requested to fetch collations on a particular leaf for some para.
689async fn fetch_collation(
690	sender: &mut impl overseer::CollatorProtocolSenderTrait,
691	state: &mut State,
692	pc: PendingCollation,
693	id: CollatorId,
694) -> std::result::Result<(), FetchError> {
695	let PendingCollation { relay_parent, peer_id, prospective_candidate, .. } = pc;
696	let candidate_hash = prospective_candidate.as_ref().map(ProspectiveCandidate::candidate_hash);
697
698	let peer_data = state.peer_data.get(&peer_id).ok_or(FetchError::UnknownPeer)?;
699
700	if peer_data.has_advertised(&relay_parent, candidate_hash) {
701		request_collation(sender, state, pc, id.clone(), peer_data.version).await?;
702		let timeout = |collator_id, candidate_hash, relay_parent| async move {
703			Delay::new(MAX_UNSHARED_DOWNLOAD_TIME).await;
704			(collator_id, candidate_hash, relay_parent)
705		};
706		state
707			.collation_fetch_timeouts
708			.push(timeout(id.clone(), candidate_hash, relay_parent).boxed());
709
710		Ok(())
711	} else {
712		Err(FetchError::NotAdvertised)
713	}
714}

764/// A peer's view has changed. A number of things should be done:
765///  - Ongoing collation requests have to be canceled.
766///  - Advertisements by this peer that are no longer relevant have to be removed.
767fn handle_peer_view_change(state: &mut State, peer_id: PeerId, view: View) {
768	let peer_data = match state.peer_data.get_mut(&peer_id) {
769		Some(peer_data) => peer_data,
770		None => return,
771	};
772
773	peer_data.update_view(&state.implicit_view, &state.active_leaves, view);
774	state.collation_requests_cancel_handles.retain(|pc, handle| {
775		let keep = pc.peer_id != peer_id || peer_data.has_advertised(&pc.relay_parent, None);
776		if !keep {
777			handle.cancel();
778		}
779		keep
780	});
781}

783/// Request a collation from the network.
784/// This function will
785///  - Check for duplicate requests.
786///  - Check if the requested collation is in our view.
787/// And as such invocations of this function may rely on that.
788async fn request_collation(
789	sender: &mut impl overseer::CollatorProtocolSenderTrait,
790	state: &mut State,
791	pending_collation: PendingCollation,
792	collator_id: CollatorId,
793	peer_protocol_version: CollationVersion,
794) -> std::result::Result<(), FetchError> {
795	if state.collation_requests_cancel_handles.contains_key(&pending_collation) {
796		return Err(FetchError::AlreadyRequested)
797	}
798
799	let PendingCollation { relay_parent, para_id, peer_id, prospective_candidate, .. } =
800		pending_collation;
801	let per_relay_parent = state
802		.per_relay_parent
803		.get_mut(&relay_parent)
804		.ok_or(FetchError::RelayParentOutOfView)?;
805
806	let (requests, response_recv) = match (peer_protocol_version, prospective_candidate) {
807		(CollationVersion::V1, _) => {
808			let (req, response_recv) = OutgoingRequest::new(
809				Recipient::Peer(peer_id),
810				request_v1::CollationFetchingRequest { relay_parent, para_id },
811			);
812			let requests = Requests::CollationFetchingV1(req);
813			(requests, response_recv.boxed())
814		},
815		(CollationVersion::V2, Some(ProspectiveCandidate { candidate_hash, .. })) => {
816			let (req, response_recv) = OutgoingRequest::new(
817				Recipient::Peer(peer_id),
818				request_v2::CollationFetchingRequest { relay_parent, para_id, candidate_hash },
819			);
820			let requests = Requests::CollationFetchingV2(req);
821			(requests, response_recv.boxed())
822		},
823		_ => return Err(FetchError::ProtocolMismatch),
824	};
825
826	let cancellation_token = CancellationToken::new();
827	let collation_request = CollationFetchRequest {
828		pending_collation,
829		collator_id: collator_id.clone(),
830		collator_protocol_version: peer_protocol_version,
831		from_collator: response_recv,
832		cancellation_token: cancellation_token.clone(),
833		_lifetime_timer: state.metrics.time_collation_request_duration(),
834	};
835
836	state.collation_requests.push(collation_request);
837	state
838		.collation_requests_cancel_handles
839		.insert(pending_collation, cancellation_token);
840
841	gum::debug!(
842		target: LOG_TARGET,
843		peer_id = %peer_id,
844		%para_id,
845		?relay_parent,
846		"Requesting collation",
847	);
848
849	let maybe_candidate_hash =
850		prospective_candidate.as_ref().map(ProspectiveCandidate::candidate_hash);
851	per_relay_parent.collations.status = CollationStatus::Fetching(para_id);
852	per_relay_parent
853		.collations
854		.fetching_from
855		.replace((collator_id, maybe_candidate_hash));
856
857	sender
858		.send_message(NetworkBridgeTxMessage::SendRequests(
859			vec![requests],
860			IfDisconnected::ImmediateError,
861		))
862		.await;
863	Ok(())
864}

866/// Networking message has been received.
867#[overseer::contextbounds(CollatorProtocol, prefix = overseer)]
868async fn process_incoming_peer_message<Context>(
869	ctx: &mut Context,
870	state: &mut State,
871	origin: PeerId,
872	msg: CollationProtocols<
873		protocol_v1::CollatorProtocolMessage,
874		protocol_v2::CollatorProtocolMessage,
875	>,
876) {
877	use protocol_v1::CollatorProtocolMessage as V1;
878	use protocol_v2::CollatorProtocolMessage as V2;
879	use sp_runtime::traits::AppVerify;
880
881	match msg {
882		CollationProtocols::V1(V1::Declare(collator_id, para_id, signature)) |
883		CollationProtocols::V2(V2::Declare(collator_id, para_id, signature)) => {
884			if collator_peer_id(&state.peer_data, &collator_id).is_some() {
885				modify_reputation(
886					&mut state.reputation,
887					ctx.sender(),
888					origin,
889					COST_UNEXPECTED_MESSAGE,
890				)
891				.await;
892				return
893			}
894
895			let peer_data = match state.peer_data.get_mut(&origin) {
896				Some(p) => p,
897				None => {
898					gum::debug!(
899						target: LOG_TARGET,
900						peer_id = ?origin,
901						?para_id,
902						"Unknown peer",
903					);
904					modify_reputation(
905						&mut state.reputation,
906						ctx.sender(),
907						origin,
908						COST_UNEXPECTED_MESSAGE,
909					)
910					.await;
911					return
912				},
913			};
914
915			if peer_data.is_collating() {
916				gum::debug!(
917					target: LOG_TARGET,
918					peer_id = ?origin,
919					?para_id,
920					"Peer is already in the collating state",
921				);
922				modify_reputation(
923					&mut state.reputation,
924					ctx.sender(),
925					origin,
926					COST_UNEXPECTED_MESSAGE,
927				)
928				.await;
929				return
930			}
931
932			if !signature.verify(&*protocol_v1::declare_signature_payload(&origin), &collator_id) {
933				gum::debug!(
934					target: LOG_TARGET,
935					peer_id = ?origin,
936					?para_id,
937					"Signature verification failure",
938				);
939				modify_reputation(
940					&mut state.reputation,
941					ctx.sender(),
942					origin,
943					COST_INVALID_SIGNATURE,
944				)
945				.await;
946				return
947			}
948
949			if state.current_assignments.contains_key(&para_id) {
950				gum::debug!(
951					target: LOG_TARGET,
952					peer_id = ?origin,
953					?collator_id,
954					?para_id,
955					"Declared as collator for current para",
956				);
957
958				peer_data.set_collating(collator_id, para_id);
959			} else {
960				gum::debug!(
961					target: LOG_TARGET,
962					peer_id = ?origin,
963					?collator_id,
964					?para_id,
965					"Declared as collator for unneeded para. Current assignments: {:?}",
966					&state.current_assignments
967				);
968
969				modify_reputation(
970					&mut state.reputation,
971					ctx.sender(),
972					origin,
973					COST_UNNEEDED_COLLATOR,
974				)
975				.await;
976				gum::trace!(target: LOG_TARGET, "Disconnecting unneeded collator");
977				disconnect_peer(ctx.sender(), origin).await;
978			}
979		},
980		CollationProtocols::V1(V1::AdvertiseCollation(relay_parent)) => {
981			if let Err(err) =
982				handle_advertisement(ctx.sender(), state, relay_parent, origin, None).await
983			{
984				gum::debug!(
985					target: LOG_TARGET,
986					peer_id = ?origin,
987					?relay_parent,
988					error = ?err,
989					"Rejected v1 advertisement",
990				);
991
992				if let Some(rep) = err.reputation_changes() {
993					modify_reputation(&mut state.reputation, ctx.sender(), origin, rep).await;
994				}
995			}
996		},
997		CollationProtocols::V2(V2::AdvertiseCollation {
998			relay_parent,
999			candidate_hash,
1000			parent_head_data_hash,
1001		}) => {
1002			if let Err(err) = handle_advertisement(
1003				ctx.sender(),
1004				state,
1005				relay_parent,
1006				origin,
1007				Some((candidate_hash, parent_head_data_hash)),
1008			)
1009			.await
1010			{
1011				gum::debug!(
1012					target: LOG_TARGET,
1013					peer_id = ?origin,
1014					?relay_parent,
1015					?candidate_hash,
1016					error = ?err,
1017					"Rejected v2 advertisement",
1018				);
1019
1020				if let Some(rep) = err.reputation_changes() {
1021					modify_reputation(&mut state.reputation, ctx.sender(), origin, rep).await;
1022				}
1023			}
1024		},
1025		CollationProtocols::V1(V1::CollationSeconded(..)) |
1026		CollationProtocols::V2(V2::CollationSeconded(..)) => {
1027			gum::warn!(
1028				target: LOG_TARGET,
1029				peer_id = ?origin,
1030				"Unexpected `CollationSeconded` message, decreasing reputation",
1031			);
1032
1033			modify_reputation(&mut state.reputation, ctx.sender(), origin, COST_UNEXPECTED_MESSAGE)
1034				.await;
1035		},
1036	}
1037}

1039// Holds off an AssetHub collation, if it's not from an invulnerable AssetHub collator. Returns
1040// `true` if the collation was held off and `false` otherwise.
1041fn hold_off_asset_hub_collation_if_needed(
1042	state: &mut State,
1043	peer_id: PeerId,
1044	collator_id: &CollatorId,
1045	relay_parent: Hash,
1046	prospective_candidate: Option<(CandidateHash, Hash)>,
1047) -> bool {
1048	// If we don't know the peer we should reject the advertisement but to avoid verbosity and
1049	// copy-pasted logic we'll just return `false` and let the caller handle it.
1050	let maybe_para_id = state.peer_data.get(&peer_id).and_then(|pd| pd.collating_para());
1051	let hold_off_duration = state.hold_off_duration;
1052	let collator_id = collator_id.clone();
1053	let peer_is_invulnerable = state.ah_invulnerables.contains(&peer_id);
1054	let invulnerables_set_is_empty = state.ah_invulnerables.is_empty();
1055
1056	if maybe_para_id != Some(ASSET_HUB_PARA_ID) ||
1057		peer_is_invulnerable ||
1058		invulnerables_set_is_empty
1059	{
1060		gum::trace!(
1061			target: LOG_TARGET,
1062			?maybe_para_id,
1063			peer_is_invulnerable,
1064			invulnerables_set_is_empty,
1065			?peer_id,
1066			?relay_parent,
1067			?prospective_candidate,
1068			"Collation not held off",
1069		);
1070
1071		return false
1072	}
1073
1074	let Some(rp_state) = state.per_relay_parent.get_mut(&relay_parent) else {
1075		// this should never happen
1076		gum::warn!(
1077			target: LOG_TARGET,
1078			?peer_id,
1079			?relay_parent,
1080			"Trying to hold off AssetHub collation, but the relay parent is not known",
1081		);
1082		return false
1083	};
1084
1085	let hold_off_outcome =
1086		rp_state.ah_held_off_advertisements.hold_off_if_necessary(HeldOffAdvertisement {
1087			relay_parent,
1088			peer_id,
1089			collator_id,
1090			prospective_candidate,
1091		});
1092
1093	match hold_off_outcome {
1094		HoldOffOperationOutcome::FirstHoldOff => {
1095			state.ah_held_off_rp_timers.push(Box::pin(async move {
1096				Delay::new(hold_off_duration).await;
1097				relay_parent
1098			}));
1099
1100			gum::debug!(
1101				target: LOG_TARGET,
1102				?peer_id,
1103				?relay_parent,
1104				?prospective_candidate,
1105				"AssetHub collation held off, not from invulnerable collator. First hold off.",
1106			);
1107
1108			true
1109		},
1110		HoldOffOperationOutcome::SubsequentHoldOff => {
1111			gum::debug!(
1112				target: LOG_TARGET,
1113				?peer_id,
1114				?relay_parent,
1115				?prospective_candidate,
1116				"AssetHub collation held off, not from invulnerable collator. Subsequent hold off.",
1117			);
1118			true
1119		},
1120		HoldOffOperationOutcome::NoHoldOff => {
1121			gum::debug!(
1122				target: LOG_TARGET,
1123				?peer_id,
1124				?relay_parent,
1125				?prospective_candidate,
1126				"AssetHub collation from non-invulnerable collator not held off - already done for this relay parent",
1127			);
1128			false
1129		},
1130	}
1131}

1133#[derive(Debug)]
1134enum AdvertisementError {
1135	/// Relay parent is unknown.
1136	RelayParentUnknown,
1137	/// Peer is not present in the subsystem state.
1138	UnknownPeer,
1139	/// Peer has not declared its para id.
1140	UndeclaredCollator,
1141	/// We're assigned to a different para at the given relay parent.
1142	InvalidAssignment,
1143	/// Para reached a limit of seconded candidates for this relay parent.
1144	SecondedLimitReached,
1145	/// Collator trying to advertise a collation using V1 protocol for an async backing relay
1146	/// parent.
1147	ProtocolMisuse,
1148	/// Advertisement is invalid.
1149	#[allow(dead_code)]
1150	Invalid(InsertAdvertisementError),
1151	/// Seconding not allowed by backing subsystem
1152	BlockedByBacking,
1153}
1154
1155impl AdvertisementError {
1156	fn reputation_changes(&self) -> Option<Rep> {
1157		use AdvertisementError::*;
1158		match self {
1159			InvalidAssignment => Some(COST_WRONG_PARA),
1160			ProtocolMisuse => Some(COST_PROTOCOL_MISUSE),
1161			RelayParentUnknown | UndeclaredCollator | Invalid(_) => Some(COST_UNEXPECTED_MESSAGE),
1162			UnknownPeer | SecondedLimitReached | BlockedByBacking => None,
1163		}
1164	}
1165}

1266	);
1267
1268	let mut has_claim_at_some_path = false;
1269	for path in paths {
1270		let mut cq_state = ClaimQueueState::new();
1271		for ancestor in &path {
1272			let seconded_and_pending = state.seconded_and_pending_for_para(&ancestor, &para_id) +
1273				state.in_waiting_queue_for_para(relay_parent, &para_id);
1274			cq_state.add_leaf(
1275				&ancestor,
1276				&state
1277					.per_relay_parent
1278					.get(ancestor)
1279					.ok_or(AdvertisementError::RelayParentUnknown)?
1280					.assignment
1281					.current,
1282			);
1283			for _ in 0..seconded_and_pending {
1284				cq_state.claim_at(ancestor, &para_id);
1285			}
1286		}
1287
1288		if cq_state.can_claim_at(relay_parent, &para_id) {
1289			has_claim_at_some_path = true;
1290			break
1291		}
1292	}
1293
1294	// If there is a place in the claim queue for the candidate at at least one path we will accept
1295	// it.
1296	if has_claim_at_some_path {
1297		gum::trace!(
1298			target: LOG_TARGET,
1299			?relay_parent,
1300			?para_id,
1301			"Seconding limit respected",
1302		);
1303		Ok(())
1304	} else {
1305		gum::trace!(
1306			target: LOG_TARGET,
1307			?relay_parent,
1308			?para_id,
1309			"Seconding limit not respected",
1310		);
1311		Err(AdvertisementError::SecondedLimitReached)
1312	}
1313}

1315async fn handle_advertisement<Sender>(
1316	sender: &mut Sender,
1317	state: &mut State,
1318	relay_parent: Hash,
1319	peer_id: PeerId,
1320	prospective_candidate: Option<(CandidateHash, Hash)>,
1321) -> std::result::Result<(), AdvertisementError>
1322where
1323	Sender: CollatorProtocolSenderTrait,
1324{
1325	let peer_data = state.peer_data.get_mut(&peer_id).ok_or(AdvertisementError::UnknownPeer)?;
1326
1327	if peer_data.version == CollationVersion::V1 && !state.active_leaves.contains(&relay_parent) {
1328		return Err(AdvertisementError::ProtocolMisuse)
1329	}
1330
1331	let per_relay_parent = state
1332		.per_relay_parent
1333		.get(&relay_parent)
1334		.ok_or(AdvertisementError::RelayParentUnknown)?;
1335
1336	let assignment = &per_relay_parent.assignment;
1337
1338	let collator_para_id =
1339		peer_data.collating_para().ok_or(AdvertisementError::UndeclaredCollator)?;
1340
1341	// Check if this is assigned to us.
1342	if !assignment.current.contains(&collator_para_id) {
1343		return Err(AdvertisementError::InvalidAssignment)
1344	}
1345
1346	// Always insert advertisements that pass all the checks for spam protection.
1347	let candidate_hash = prospective_candidate.map(|(hash, ..)| hash);
1348	let (collator_id, para_id) = peer_data
1349		.insert_advertisement(
1350			relay_parent,
1351			candidate_hash,
1352			&state.implicit_view,
1353			&state.active_leaves,
1354			&per_relay_parent,
1355		)
1356		.map_err(AdvertisementError::Invalid)?;
1357
1358	if hold_off_asset_hub_collation_if_needed(
1359		state,
1360		peer_id,
1361		&collator_id,
1362		relay_parent,
1363		prospective_candidate,
1364	) {
1365		return Ok(())
1366	}
1367
1368	process_advertisement(
1369		sender,
1370		state,
1371		relay_parent,
1372		para_id,
1373		peer_id,
1374		collator_id,
1375		prospective_candidate,
1376	)
1377	.await
1378}

1380async fn process_advertisement<Sender>(
1381	sender: &mut Sender,
1382	state: &mut State,
1383	relay_parent: Hash,
1384	para_id: ParaId,
1385	peer_id: PeerId,
1386	collator_id: CollatorId,
1387	prospective_candidate: Option<(CandidateHash, Hash)>,
1388) -> std::result::Result<(), AdvertisementError>
1389where
1390	Sender: CollatorProtocolSenderTrait,
1391{
1392	ensure_seconding_limit_is_respected(&relay_parent, para_id, state)?;
1393
1394	if let Some((candidate_hash, parent_head_data_hash)) = prospective_candidate {
1395		// Check if backing subsystem allows to second this candidate.
1396		//
1397		// This is also only important when async backing or elastic scaling is enabled.
1398		let can_second =
1399			can_second(sender, para_id, relay_parent, candidate_hash, parent_head_data_hash).await;
1400
1401		if !can_second {
1402			return Err(AdvertisementError::BlockedByBacking)
1403		}
1404	}
1405
1406	let result = enqueue_collation(
1407		sender,
1408		state,
1409		relay_parent,
1410		para_id,
1411		peer_id,
1412		collator_id,
1413		prospective_candidate,
1414	)
1415	.await;
1416
1417	if let Err(fetch_error) = result {
1418		gum::debug!(
1419			target: LOG_TARGET,
1420			relay_parent = ?relay_parent,
1421			para_id = ?para_id,
1422			peer_id = ?peer_id,
1423			error = %fetch_error,
1424			"Failed to request advertised collation",
1425		);
1426	}
1427
1428	Ok(())
1429}

1431/// Enqueue collation for fetching. The advertisement is expected to be validated and the seconding
1432/// limit checked.
1433async fn enqueue_collation<Sender>(
1434	sender: &mut Sender,
1435	state: &mut State,
1436	relay_parent: Hash,
1437	para_id: ParaId,
1438	peer_id: PeerId,
1439	collator_id: CollatorId,
1440	prospective_candidate: Option<(CandidateHash, Hash)>,
1441) -> std::result::Result<(), FetchError>
1442where
1443	Sender: CollatorProtocolSenderTrait,
1444{
1445	gum::debug!(
1446		target: LOG_TARGET,
1447		peer_id = ?peer_id,
1448		%para_id,
1449		?relay_parent,
1450		"Received advertise collation",
1451	);
1452	let per_relay_parent = match state.per_relay_parent.get_mut(&relay_parent) {
1453		Some(rp_state) => rp_state,
1454		None => {
1455			// Race happened, not an error.
1456			gum::trace!(
1457				target: LOG_TARGET,
1458				peer_id = ?peer_id,
1459				%para_id,
1460				?relay_parent,
1461				?prospective_candidate,
1462				"Candidate relay parent went out of view for valid advertisement",
1463			);
1464			return Ok(())
1465		},
1466	};
1467	let prospective_candidate =
1468		prospective_candidate.map(|(candidate_hash, parent_head_data_hash)| ProspectiveCandidate {
1469			candidate_hash,
1470			parent_head_data_hash,
1471		});
1472
1473	let collations = &mut per_relay_parent.collations;
1474	let pending_collation =
1475		PendingCollation::new(relay_parent, para_id, &peer_id, prospective_candidate);
1476
1477	match collations.status {
1478		CollationStatus::Fetching(_) | CollationStatus::WaitingOnValidation => {
1479			gum::trace!(
1480				target: LOG_TARGET,
1481				peer_id = ?peer_id,
1482				%para_id,
1483				?relay_parent,
1484				"Added collation to the pending list"
1485			);
1486			collations.add_to_waiting_queue((pending_collation, collator_id));
1487		},
1488		CollationStatus::Waiting => {
1489			// We were waiting for a collation to be advertised to us (we were idle) so we can fetch
1490			// the new collation immediately
1491			fetch_collation(sender, state, pending_collation, collator_id).await?;
1492		},
1493	}
1494
1495	Ok(())
1496}

1498/// Our view has changed.
1499async fn handle_our_view_change<Sender>(
1500	sender: &mut Sender,
1501	state: &mut State,
1502	keystore: &KeystorePtr,
1503	view: OurView,
1504) -> Result<()>
1505where
1506	Sender: CollatorProtocolSenderTrait,
1507{
1508	let current_leaves = state.active_leaves.clone();
1509
1510	let removed = current_leaves.iter().filter(|h| !view.contains(h));
1511	let added = view.iter().filter(|h| !current_leaves.contains(h));
1512
1513	for leaf in added {
1514		let session_index = request_session_index_for_child(*leaf, sender)
1515			.await
1516			.await
1517			.map_err(Error::CancelledSessionIndex)??;
1518
1519		let v2_receipts = request_node_features(*leaf, session_index, sender)
1520			.await
1521			.await
1522			.map_err(Error::CancelledNodeFeatures)??
1523			.get(node_features::FeatureIndex::CandidateReceiptV2 as usize)
1524			.map(|b| *b)
1525			.unwrap_or(false);
1526
1527		let Some(per_relay_parent) = construct_per_relay_parent(
1528			sender,
1529			&mut state.current_assignments,
1530			keystore,
1531			*leaf,
1532			v2_receipts,
1533			session_index,
1534		)
1535		.await?
1536		else {
1537			continue
1538		};
1539
1540		state.active_leaves.insert(*leaf);
1541		state.per_relay_parent.insert(*leaf, per_relay_parent);
1542
1543		state
1544			.implicit_view
1545			.activate_leaf(sender, *leaf)
1546			.await
1547			.map_err(Error::ImplicitViewFetchError)?;
1548
1549		// Order is always descending.
1550		let allowed_ancestry = state
1551			.implicit_view
1552			.known_allowed_relay_parents_under(leaf, None)
1553			.unwrap_or_default();
1554		for block_hash in allowed_ancestry {
1555			if let Entry::Vacant(entry) = state.per_relay_parent.entry(*block_hash) {
1556				// Safe to use the same v2 receipts config for the allowed relay parents as well
1557				// as the same session index since they must be in the same session.
1558				if let Some(per_relay_parent) = construct_per_relay_parent(
1559					sender,
1560					&mut state.current_assignments,
1561					keystore,
1562					*block_hash,
1563					v2_receipts,
1564					session_index,
1565				)
1566				.await?
1567				{
1568					entry.insert(per_relay_parent);
1569				}
1570			}
1571		}
1572	}
1573
1574	for removed in removed {
1575		gum::trace!(
1576			target: LOG_TARGET,
1577			?view,
1578			?removed,
1579			"handle_our_view_change - removed",
1580		);
1581
1582		state.active_leaves.remove(removed);
1583		// If the leaf is deactivated it still may stay in the view as a part
1584		// of implicit ancestry. Only update the state after the hash is actually
1585		// pruned from the block info storage.
1586		let pruned = state.implicit_view.deactivate_leaf(*removed);
1587
1588		for removed in pruned {
1589			if let Some(per_relay_parent) = state.per_relay_parent.remove(&removed) {
1590				remove_outgoing(&mut state.current_assignments, per_relay_parent);
1591			}
1592
1593			state.collation_requests_cancel_handles.retain(|pc, handle| {
1594				let keep = pc.relay_parent != removed;
1595				if !keep {
1596					handle.cancel();
1597				}
1598				keep
1599			});
1600			state.fetched_candidates.retain(|k, _| k.relay_parent != removed);
1601		}
1602	}
1603
1604	// Remove blocked seconding requests that left the view.
1605	state.blocked_from_seconding.retain(|_, collations| {
1606		collations.retain(|collation| {
1607			state
1608				.per_relay_parent
1609				.contains_key(&collation.candidate_receipt.descriptor.relay_parent())
1610		});
1611
1612		!collations.is_empty()
1613	});
1614
1615	for (peer_id, peer_data) in state.peer_data.iter_mut() {
1616		peer_data.prune_old_advertisements(&state.implicit_view, &state.active_leaves);
1617
1618		// Disconnect peers who are not relevant to our current or next para.
1619		//
1620		// If the peer hasn't declared yet, they will be disconnected if they do not
1621		// declare.
1622		if let Some(para_id) = peer_data.collating_para() {
1623			if !state.current_assignments.contains_key(&para_id) {
1624				gum::trace!(
1625					target: LOG_TARGET,
1626					?peer_id,
1627					?para_id,
1628					"Disconnecting peer on view change (not current parachain id)"
1629				);
1630				disconnect_peer(sender, *peer_id).await;
1631			}
1632		}
1633	}
1634
1635	Ok(())
1636}

1638/// Bridge event switch.
1639#[overseer::contextbounds(CollatorProtocol, prefix = self::overseer)]
1640async fn handle_network_msg<Context>(
1641	ctx: &mut Context,
1642	state: &mut State,
1643	keystore: &KeystorePtr,
1644	bridge_message: NetworkBridgeEvent<net_protocol::CollatorProtocolMessage>,
1645) -> Result<()> {
1646	use NetworkBridgeEvent::*;
1647
1648	match bridge_message {
1649		PeerConnected(peer_id, observed_role, protocol_version, _) => {
1650			let version = match protocol_version.try_into() {
1651				Ok(version) => version,
1652				Err(err) => {
1653					// Network bridge is expected to handle this.
1654					gum::error!(
1655						target: LOG_TARGET,
1656						?peer_id,
1657						?observed_role,
1658						?err,
1659						"Unsupported protocol version"
1660					);
1661					return Ok(())
1662				},
1663			};
1664
1665			// Since the AssetHub system parachain is very important for the operation of the
1666			// relay chain we need to ensure that we always accept connections from the invulnerable
1667			// AssetHub collators.
1668			// However there is a connection limit in `polkadot-node-network-protocol` of the number
1669			// of peers which can connect to an authority node. Once this limit is reached all new
1670			// connections are dropped.
1671			// This means we need to be extra careful when accepting connections from new collators.
1672			// If we are at the point where there are not enough slots for
1673			// invulnerable AssetHub collators we need to drop new connections.
1674			if !can_accept_new_collator_connection(
1675				&peer_id,
1676				&state.peer_data,
1677				&state.ah_invulnerables,
1678			) {
1679				gum::debug!(
1680					target: LOG_TARGET,
1681					?peer_id,
1682					?observed_role,
1683					"Peer connection refused, no slots for invulnerable AssetHub collators",
1684				);
1685				disconnect_peer(ctx.sender(), peer_id).await;
1686				return Ok(())
1687			}
1688
1689			state.peer_data.entry(peer_id).or_insert_with(|| PeerData {
1690				view: View::default(),
1691				state: PeerState::Connected(Instant::now()),
1692				version,
1693			});
1694			state.metrics.note_collator_peer_count(state.peer_data.len());
1695		},
1696		PeerDisconnected(peer_id) => {
1697			state.peer_data.remove(&peer_id);
1698			state.metrics.note_collator_peer_count(state.peer_data.len());
1699		},
1700		NewGossipTopology { .. } => {
1701			// impossible!
1702		},
1703		PeerViewChange(peer_id, view) => {
1704			handle_peer_view_change(state, peer_id, view);
1705		},
1706		OurViewChange(view) => {
1707			handle_our_view_change(ctx.sender(), state, keystore, view).await?;
1708		},
1709		PeerMessage(remote, msg) => {
1710			process_incoming_peer_message(ctx, state, remote, msg).await;
1711		},
1712		UpdatedAuthorityIds { .. } => {
1713			// The validator side doesn't deal with `AuthorityDiscoveryId`s.
1714		},
1715	}
1716
1717	Ok(())
1718}

1720/// The main message receiver switch.
1721#[overseer::contextbounds(CollatorProtocol, prefix = self::overseer)]
1722async fn process_msg<Context>(
1723	ctx: &mut Context,
1724	keystore: &KeystorePtr,
1725	msg: CollatorProtocolMessage,
1726	state: &mut State,
1727) {
1728	use CollatorProtocolMessage::*;
1729
1730	let _timer = state.metrics.time_process_msg();
1731
1732	match msg {
1733		ConnectToBackingGroups => {
1734			gum::warn!(
1735				target: LOG_TARGET,
1736				"PreConnectToBackingGroups message is not expected on the validator side of the protocol",
1737			);
1738		},
1739		DisconnectFromBackingGroups => {
1740			gum::warn!(
1741				target: LOG_TARGET,
1742				"DisconnectFromBackingGroups message is not expected on the validator side of the protocol",
1743			);
1744		},
1745		CollateOn(id) => {
1746			gum::warn!(
1747				target: LOG_TARGET,
1748				para_id = %id,
1749				"CollateOn message is not expected on the validator side of the protocol",
1750			);
1751		},
1752		DistributeCollation { .. } => {
1753			gum::warn!(
1754				target: LOG_TARGET,
1755				"DistributeCollation message is not expected on the validator side of the protocol",
1756			);
1757		},
1758		NetworkBridgeUpdate(event) => {
1759			if let Err(e) = handle_network_msg(ctx, state, keystore, event).await {
1760				gum::warn!(
1761					target: LOG_TARGET,
1762					err = ?e,
1763					"Failed to handle incoming network message",
1764				);
1765			}
1766		},
1767		Seconded(parent, stmt) => {
1768			let receipt = match stmt.payload() {
1769				Statement::Seconded(receipt) => receipt,
1770				Statement::Valid(_) => {
1771					gum::warn!(
1772						target: LOG_TARGET,
1773						?stmt,
1774						relay_parent = %parent,
1775						"Seconded message received with a `Valid` statement",
1776					);
1777					return
1778				},
1779			};
1780			let output_head_data = receipt.commitments.head_data.clone();
1781			let output_head_data_hash = receipt.descriptor.para_head();
1782			let fetched_collation = FetchedCollation::from(&receipt.to_plain());
1783			if let Some(CollationEvent { collator_id, pending_collation, .. }) =
1784				state.fetched_candidates.remove(&fetched_collation)
1785			{
1786				let PendingCollation {
1787					relay_parent, peer_id, prospective_candidate, para_id, ..
1788				} = pending_collation;
1789				note_good_collation(
1790					&mut state.reputation,
1791					ctx.sender(),
1792					&state.peer_data,
1793					collator_id.clone(),
1794				)
1795				.await;
1796				if let Some(peer_data) = state.peer_data.get(&peer_id) {
1797					notify_collation_seconded(
1798						ctx.sender(),
1799						peer_id,
1800						peer_data.version,
1801						relay_parent,
1802						stmt,
1803					)
1804					.await;
1805				}
1806
1807				if let Some(rp_state) = state.per_relay_parent.get_mut(&parent) {
1808					rp_state.collations.note_seconded(para_id);
1809				}
1810
1811				// See if we've unblocked other collations for seconding.
1812				second_unblocked_collations(
1813					ctx,
1814					state,
1815					fetched_collation.para_id,
1816					output_head_data,
1817					output_head_data_hash,
1818				)
1819				.await;
1820
1821				// If async backing is enabled, make an attempt to fetch next collation.
1822				let maybe_candidate_hash =
1823					prospective_candidate.as_ref().map(ProspectiveCandidate::candidate_hash);
1824				dequeue_next_collation_and_fetch(
1825					ctx,
1826					state,
1827					parent,
1828					(collator_id, maybe_candidate_hash),
1829				)
1830				.await;
1831			} else {
1832				gum::debug!(
1833					target: LOG_TARGET,
1834					relay_parent = ?parent,
1835					"Collation has been seconded, but the relay parent is deactivated",
1836				);
1837			}
1838		},
1839		Invalid(parent, candidate_receipt) => {
1840			// Remove collations which were blocked from seconding and had this candidate as parent.
1841			state.blocked_from_seconding.remove(&BlockedCollationId {
1842				para_id: candidate_receipt.descriptor.para_id(),
1843				parent_head_data_hash: candidate_receipt.descriptor.para_head(),
1844			});
1845
1846			let fetched_collation = FetchedCollation::from(&candidate_receipt);
1847			let candidate_hash = fetched_collation.candidate_hash;
1848			let id = match state.fetched_candidates.entry(fetched_collation) {
1849				Entry::Occupied(entry)
1850					if entry.get().pending_collation.commitments_hash ==
1851						Some(candidate_receipt.commitments_hash) =>
1852					entry.remove().collator_id,
1853				Entry::Occupied(_) => {
1854					gum::error!(
1855						target: LOG_TARGET,
1856						relay_parent = ?parent,
1857						candidate = ?candidate_receipt.hash(),
1858						"Reported invalid candidate for unknown `pending_candidate`!",
1859					);
1860					return
1861				},
1862				Entry::Vacant(_) => return,
1863			};
1864
1865			report_collator(&mut state.reputation, ctx.sender(), &state.peer_data, id.clone())
1866				.await;
1867
1868			dequeue_next_collation_and_fetch(ctx, state, parent, (id, Some(candidate_hash))).await;
1869		},
1870	}
1871}

1898	mut ctx: Context,
1899	keystore: KeystorePtr,
1900	eviction_policy: crate::CollatorEvictionPolicy,
1901	metrics: Metrics,
1902	reputation: ReputationAggregator,
1903	reputation_interval: Duration,
1904	ah_invulnerables: HashSet<PeerId>,
1905	hold_off_duration: Duration,
1906) -> std::result::Result<(), SubsystemError> {
1907	let new_reputation_delay = || futures_timer::Delay::new(reputation_interval).fuse();
1908	let mut reputation_delay = new_reputation_delay();
1909
1910	let mut state =
1911		State { metrics, reputation, ah_invulnerables, hold_off_duration, ..Default::default() };
1912
1913	let next_inactivity_stream = tick_stream(ACTIVITY_POLL);
1914	futures::pin_mut!(next_inactivity_stream);
1915
1916	let mut network_error_freq = gum::Freq::new();
1917	let mut canceled_freq = gum::Freq::new();
1918
1919	loop {
1920		select! {
1921			_ = reputation_delay => {
1922				state.reputation.send(ctx.sender()).await;
1923				reputation_delay = new_reputation_delay();
1924			},
1925			res = ctx.recv().fuse() => {
1926				match res {
1927					Ok(FromOrchestra::Communication { msg }) => {
1928						gum::trace!(target: LOG_TARGET, msg = ?msg, "received a message");
1929						process_msg(
1930							&mut ctx,
1931							&keystore,
1932							msg,
1933							&mut state,
1934						).await;
1935					}
1936					Ok(FromOrchestra::Signal(OverseerSignal::Conclude)) | Err(_) => break,
1937					Ok(FromOrchestra::Signal(_)) => continue,
1938				}
1939			},
1940			_ = next_inactivity_stream.next() => {
1941				disconnect_inactive_peers(ctx.sender(), &eviction_policy, &state.peer_data).await;
1942			},
1943			resp = state.collation_requests.select_next_some() => {
1944				let relay_parent = resp.0.pending_collation.relay_parent;
1945				let res = match handle_collation_fetch_response(
1946					&mut state,
1947					resp,
1948					&mut network_error_freq,
1949					&mut canceled_freq,
1950				).await {
1951					Err(Some((peer_id, rep))) => {
1952						modify_reputation(&mut state.reputation, ctx.sender(), peer_id, rep).await;
1953						// Reset the status for the relay parent
1954						state.per_relay_parent.get_mut(&relay_parent).map(|rp| {
1955							rp.collations.status.back_to_waiting();
1956						});
1957						continue
1958					},
1959					Err(None) => {
1960						// Reset the status for the relay parent
1961						state.per_relay_parent.get_mut(&relay_parent).map(|rp| {
1962							rp.collations.status.back_to_waiting();
1963						});
1964						continue
1965					},
1966					Ok(res) => res
1967				};
1968
1969				let CollationEvent {collator_id, pending_collation, .. } = res.collation_event.clone();
1970
1971				match kick_off_seconding(&mut ctx, &mut state, res).await {
1972					Err(err) => {
1973						gum::warn!(
1974							target: LOG_TARGET,
1975							relay_parent = ?pending_collation.relay_parent,
1976							para_id = ?pending_collation.para_id,
1977							peer_id = ?pending_collation.peer_id,
1978							error = %err,
1979							"Seconding aborted due to an error",
1980						);
1981
1982						if err.is_malicious() {
1983							// Report malicious peer.
1984							modify_reputation(&mut state.reputation, ctx.sender(), pending_collation.peer_id, COST_REPORT_BAD).await;
1985						}
1986						let maybe_candidate_hash =
1987						pending_collation.prospective_candidate.as_ref().map(ProspectiveCandidate::candidate_hash);
1988						dequeue_next_collation_and_fetch(
1989							&mut ctx,
1990							&mut state,
1991							pending_collation.relay_parent,
1992							(collator_id, maybe_candidate_hash),
1993						)
1994						.await;
1995					},
1996					Ok(false) => {
1997						// No hard error occurred, but we can try fetching another collation.
1998						let maybe_candidate_hash =
1999						pending_collation.prospective_candidate.as_ref().map(ProspectiveCandidate::candidate_hash);
2000						dequeue_next_collation_and_fetch(
2001							&mut ctx,
2002							&mut state,
2003							pending_collation.relay_parent,
2004							(collator_id, maybe_candidate_hash),
2005						)
2006						.await;
2007					}
2008					Ok(true) => {}
2009				}
2010			},
2011			res = state.collation_fetch_timeouts.select_next_some() => {
2012				let (collator_id, maybe_candidate_hash, relay_parent) = res;
2013				gum::debug!(
2014					target: LOG_TARGET,
2015					?relay_parent,
2016					?collator_id,
2017					"Timeout hit - already seconded?"
2018				);
2019				dequeue_next_collation_and_fetch(
2020					&mut ctx,
2021					&mut state,
2022					relay_parent,
2023					(collator_id, maybe_candidate_hash),
2024				)
2025				.await;
2026			},
2027			rp = state.ah_held_off_rp_timers.select_next_some() => {
2028				let Some(held_off_advertisements) = state.per_relay_parent.get_mut(&rp).map(|rp_state| rp_state.ah_held_off_advertisements.on_hold_off_complete()) else {
2029					gum::debug!(
2030						target: LOG_TARGET,
2031						?rp,
2032						"Received AssetHub held off advertisements timeout for unknown relay parent",
2033					);
2034					continue
2035				};
2036
2037				let held_off_advertisements = match held_off_advertisements {
2038					Ok(held_off) => held_off,
2039					Err(err) => {
2040						gum::warn!(
2041							target: LOG_TARGET,
2042							?rp,
2043							?err,
2044							"Failed to get held off advertisements for relay parent",
2045						);
2046						continue;
2047					}
2048				};
2049
2050				for held_off_advertisement in held_off_advertisements {
2051					let HeldOffAdvertisement{relay_parent, peer_id, collator_id, prospective_candidate} = held_off_advertisement;
2052					gum::debug!(
2053						target: LOG_TARGET,
2054						?relay_parent,
2055						?peer_id,
2056						?prospective_candidate,
2057						"Processing held off advertisement for AssetHub",
2058					);
2059
2060					if let Err(err) = process_advertisement(
2061						ctx.sender(),
2062						&mut state,
2063						relay_parent,
2064						ASSET_HUB_PARA_ID,
2065						peer_id,
2066						collator_id,
2067						prospective_candidate,
2068					)
2069					.await {
2070						gum::debug!(
2071								target: LOG_TARGET,
2072								?err,
2073								?relay_parent,
2074								?peer_id,
2075								?prospective_candidate,
2076								"Failed to handle held off advertisement",
2077							);
2078						if let Some(rep) = err.reputation_changes() {
2079							modify_reputation(&mut state.reputation, ctx.sender(), peer_id, rep).await;
2080						}
2081					}
2082				}
2083			},
2084		}
2085	}
2086
2087	Ok(())
2088}

2170/// Handle a fetched collation result.
2171/// Returns whether or not seconding has begun.
2172#[overseer::contextbounds(CollatorProtocol, prefix = self::overseer)]
2173async fn kick_off_seconding<Context>(
2174	ctx: &mut Context,
2175	state: &mut State,
2176	PendingCollationFetch { mut collation_event, candidate_receipt, pov, maybe_parent_head_data }: PendingCollationFetch,
2177) -> std::result::Result<bool, SecondingError> {
2178	let pending_collation = collation_event.pending_collation;
2179	let relay_parent = pending_collation.relay_parent;
2180
2181	let per_relay_parent = match state.per_relay_parent.get_mut(&relay_parent) {
2182		Some(state) => state,
2183		None => {
2184			// Relay parent went out of view, not an error.
2185			gum::trace!(
2186				target: LOG_TARGET,
2187				relay_parent = ?relay_parent,
2188				"Fetched collation for a parent out of view",
2189			);
2190			return Ok(false)
2191		},
2192	};
2193
2194	// Sanity check of the candidate receipt version.
2195	descriptor_version_sanity_check(candidate_receipt.descriptor(), per_relay_parent)?;
2196
2197	let collations = &mut per_relay_parent.collations;
2198
2199	let fetched_collation = FetchedCollation::from(&candidate_receipt);
2200	if let Entry::Vacant(entry) = state.fetched_candidates.entry(fetched_collation) {
2201		collation_event.pending_collation.commitments_hash =
2202			Some(candidate_receipt.commitments_hash);
2203
2204		let (maybe_pvd, maybe_parent_head, maybe_parent_head_hash) = match (
2205			collation_event.collator_protocol_version,
2206			collation_event.pending_collation.prospective_candidate,
2207		) {
2208			(CollationVersion::V2, Some(ProspectiveCandidate { parent_head_data_hash, .. })) => {
2209				let pvd = request_prospective_validation_data(
2210					ctx.sender(),
2211					relay_parent,
2212					parent_head_data_hash,
2213					pending_collation.para_id,
2214					maybe_parent_head_data.clone(),
2215				)
2216				.await?;
2217
2218				(pvd, maybe_parent_head_data, Some(parent_head_data_hash))
2219			},
2220			(CollationVersion::V1, _) => {
2221				let pvd = request_persisted_validation_data(
2222					ctx.sender(),
2223					candidate_receipt.descriptor().relay_parent(),
2224					candidate_receipt.descriptor().para_id(),
2225				)
2226				.await?;
2227				(
2228					Some(pvd.ok_or(SecondingError::PersistedValidationDataNotFound)?),
2229					maybe_parent_head_data,
2230					None,
2231				)
2232			},
2233			_ => {
2234				// `handle_advertisement` checks for protocol mismatch.
2235				return Ok(false)
2236			},
2237		};
2238
2239		let pvd = match (maybe_pvd, maybe_parent_head.clone(), maybe_parent_head_hash) {
2240			(Some(pvd), _, _) => pvd,
2241			(None, None, Some(parent_head_data_hash)) => {
2242				// In this case, the collator did not supply the head data and neither could
2243				// prospective-parachains. We add this to the blocked_from_seconding collection
2244				// until we second its parent.
2245				let blocked_collation = PendingCollationFetch {
2246					collation_event,
2247					candidate_receipt,
2248					pov,
2249					maybe_parent_head_data: None,
2250				};
2251				gum::debug!(
2252					target: LOG_TARGET,
2253					candidate_hash = ?blocked_collation.candidate_receipt.hash(),
2254					relay_parent = ?blocked_collation.candidate_receipt.descriptor.relay_parent(),
2255					"Collation having parent head data hash {} is blocked from seconding. Waiting on its parent to be validated.",
2256					parent_head_data_hash
2257				);
2258				state
2259					.blocked_from_seconding
2260					.entry(BlockedCollationId {
2261						para_id: blocked_collation.candidate_receipt.descriptor.para_id(),
2262						parent_head_data_hash,
2263					})
2264					.or_insert_with(Vec::new)
2265					.push(blocked_collation);
2266
2267				return Ok(false)
2268			},
2269			(None, _, _) => {
2270				// Even though we already have the parent head data, the pvd fetching failed. We
2271				// don't need to wait for seconding another collation outputting this head data.
2272				return Err(SecondingError::PersistedValidationDataNotFound)
2273			},
2274		};
2275
2276		fetched_collation_sanity_check(
2277			&collation_event.pending_collation,
2278			&candidate_receipt,
2279			&pvd,
2280			maybe_parent_head.and_then(|head| maybe_parent_head_hash.map(|hash| (head, hash))),
2281		)?;
2282
2283		ctx.send_message(CandidateBackingMessage::Second(
2284			relay_parent,
2285			candidate_receipt,
2286			pvd,
2287			pov,
2288		))
2289		.await;
2290		// There's always a single collation being fetched at any moment of time.
2291		// In case of a failure, we reset the status back to waiting.
2292		collations.status = CollationStatus::WaitingOnValidation;
2293
2294		entry.insert(collation_event);
2295		Ok(true)
2296	} else {
2297		Err(SecondingError::Duplicate)
2298	}
2299}

2317/// Handle a collation fetch response.
2318async fn handle_collation_fetch_response(
2319	state: &mut State,
2320	response: <CollationFetchRequest as Future>::Output,
2321	network_error_freq: &mut gum::Freq,
2322	canceled_freq: &mut gum::Freq,
2323) -> std::result::Result<PendingCollationFetch, Option<(PeerId, Rep)>> {
2324	let (CollationEvent { collator_id, collator_protocol_version, pending_collation }, response) =
2325		response;
2326	// Remove the cancellation handle, as the future already completed.
2327	state.collation_requests_cancel_handles.remove(&pending_collation);
2328
2329	let response = match response {
2330		Err(CollationFetchError::Cancelled) => {
2331			gum::debug!(
2332				target: LOG_TARGET,
2333				hash = ?pending_collation.relay_parent,
2334				para_id = ?pending_collation.para_id,
2335				peer_id = ?pending_collation.peer_id,
2336				"Request was cancelled from the validator side"
2337			);
2338			return Err(None)
2339		},
2340		Err(CollationFetchError::Request(req_error)) => Err(req_error),
2341		Ok(resp) => Ok(resp),
2342	};
2343
2344	let _timer = state.metrics.time_handle_collation_request_result();
2345
2346	let mut metrics_result = Err(());
2347
2348	let result = match response {
2349		Err(RequestError::InvalidResponse(err)) => {
2350			gum::warn!(
2351				target: LOG_TARGET,
2352				hash = ?pending_collation.relay_parent,
2353				para_id = ?pending_collation.para_id,
2354				peer_id = ?pending_collation.peer_id,
2355				err = ?err,
2356				"Collator provided response that could not be decoded"
2357			);
2358			Err(Some((pending_collation.peer_id, COST_CORRUPTED_MESSAGE)))
2359		},
2360		Err(err) if err.is_timed_out() => {
2361			gum::debug!(
2362				target: LOG_TARGET,
2363				hash = ?pending_collation.relay_parent,
2364				para_id = ?pending_collation.para_id,
2365				peer_id = ?pending_collation.peer_id,
2366				"Request timed out"
2367			);
2368			// For now we don't want to change reputation on timeout, to mitigate issues like
2369			// this: https://github.com/paritytech/polkadot/issues/4617
2370			Err(None)
2371		},
2372		Err(RequestError::NetworkError(err)) => {
2373			gum::warn_if_frequent!(
2374				freq: network_error_freq,
2375				max_rate: gum::Times::PerHour(100),
2376				target: LOG_TARGET,
2377				hash = ?pending_collation.relay_parent,
2378				para_id = ?pending_collation.para_id,
2379				peer_id = ?pending_collation.peer_id,
2380				err = ?err,
2381				"Fetching collation failed due to network error"
2382			);
2383			// A minor decrease in reputation for any network failure seems
2384			// sensible. In theory this could be exploited, by DoSing this node,
2385			// which would result in reduced reputation for proper nodes, but the
2386			// same can happen for penalties on timeouts, which we also have.
2387			Err(Some((pending_collation.peer_id, COST_NETWORK_ERROR)))
2388		},
2389		Err(RequestError::Canceled(err)) => {
2390			gum::warn_if_frequent!(
2391				freq: canceled_freq,
2392				max_rate: gum::Times::PerHour(100),
2393				target: LOG_TARGET,
2394				hash = ?pending_collation.relay_parent,
2395				para_id = ?pending_collation.para_id,
2396				peer_id = ?pending_collation.peer_id,
2397				err = ?err,
2398				"Canceled should be handled by `is_timed_out` above - this is a bug!"
2399			);
2400			Err(None)
2401		},
2402		Ok(
2403			request_v1::CollationFetchingResponse::Collation(receipt, _) |
2404			request_v2::CollationFetchingResponse::Collation(receipt, _) |
2405			request_v1::CollationFetchingResponse::CollationWithParentHeadData { receipt, .. } |
2406			request_v2::CollationFetchingResponse::CollationWithParentHeadData { receipt, .. },
2407		) if receipt.descriptor().para_id() != pending_collation.para_id => {
2408			gum::debug!(
2409				target: LOG_TARGET,
2410				expected_para_id = ?pending_collation.para_id,
2411				got_para_id = ?receipt.descriptor().para_id(),
2412				peer_id = ?pending_collation.peer_id,
2413				"Got wrong para ID for requested collation."
2414			);
2415
2416			Err(Some((pending_collation.peer_id, COST_WRONG_PARA)))
2417		},
2418		Ok(request_v1::CollationFetchingResponse::Collation(candidate_receipt, pov)) => {
2419			gum::debug!(
2420				target: LOG_TARGET,
2421				para_id = %pending_collation.para_id,
2422				hash = ?pending_collation.relay_parent,
2423				candidate_hash = ?candidate_receipt.hash(),
2424				"Received collation",
2425			);
2426
2427			metrics_result = Ok(());
2428			Ok(PendingCollationFetch {
2429				collation_event: CollationEvent {
2430					collator_id,
2431					pending_collation,
2432					collator_protocol_version,
2433				},
2434				candidate_receipt,
2435				pov,
2436				maybe_parent_head_data: None,
2437			})
2438		},
2439		Ok(request_v2::CollationFetchingResponse::CollationWithParentHeadData {
2440			receipt,
2441			pov,
2442			parent_head_data,
2443		}) => {
2444			gum::debug!(
2445				target: LOG_TARGET,
2446				para_id = %pending_collation.para_id,
2447				hash = ?pending_collation.relay_parent,
2448				candidate_hash = ?receipt.hash(),
2449				"Received collation (v3)",
2450			);
2451
2452			metrics_result = Ok(());
2453			Ok(PendingCollationFetch {
2454				collation_event: CollationEvent {
2455					collator_id,
2456					pending_collation,
2457					collator_protocol_version,
2458				},
2459				candidate_receipt: receipt,
2460				pov,
2461				maybe_parent_head_data: Some(parent_head_data),
2462			})
2463		},
2464	};
2465	state.metrics.on_request(metrics_result);
2466	result
2467}

2476		.ok_or(Error::RelayParentStateNotFound)?;
2477	let scheduled_paras = relay_parent_state.assignment.current.iter().collect::<HashSet<_>>();
2478	let paths = state.implicit_view.paths_via_relay_parent(relay_parent);
2479
2480	let mut claim_queue_states = Vec::new();
2481	for path in paths {
2482		let mut cq_state = ClaimQueueState::new();
2483		for ancestor in &path {
2484			cq_state.add_leaf(
2485				&ancestor,
2486				&state
2487					.per_relay_parent
2488					.get(&ancestor)
2489					.ok_or(Error::RelayParentStateNotFound)?
2490					.assignment
2491					.current,
2492			);
2493
2494			for para_id in &scheduled_paras {
2495				let seconded_and_pending = state.seconded_and_pending_for_para(&ancestor, &para_id);
2496				for _ in 0..seconded_and_pending {
2497					cq_state.claim_at(&ancestor, &para_id);
2498				}
2499			}
2500		}
2501		claim_queue_states.push(cq_state);
2502	}
2503
2504	// From the claim queue state for each leaf we have to return a combined single one. Go for a
2505	// simple solution and return the longest one. In theory we always prefer the earliest entries
2506	// in the claim queue so there is a good chance that the longest path is the one with
2507	// unsatisfied entries in the beginning. This is not guaranteed as we might have fetched 2nd or
2508	// 3rd spot from the claim queue but it should be good enough.
2509	let unfulfilled_entries = claim_queue_states
2510		.iter_mut()
2511		.map(|cq| cq.unclaimed_at(relay_parent))
2512		.max_by(|a, b| a.len().cmp(&b.len()))
2513		.unwrap_or_default();
2514
2515	Ok(unfulfilled_entries)
2516}

2518/// Returns the next collation to fetch from the `waiting_queue` and reset the status back to
2519/// `Waiting`.
2520fn get_next_collation_to_fetch(
2521	finished_one: &(CollatorId, Option<CandidateHash>),
2522	relay_parent: Hash,
2523	state: &mut State,
2524) -> Option<(PendingCollation, CollatorId)> {
2525	let unfulfilled_entries = match unfulfilled_claim_queue_entries(&relay_parent, &state) {
2526		Ok(entries) => entries,
2527		Err(err) => {
2528			gum::error!(
2529				target: LOG_TARGET,
2530				?relay_parent,
2531				?err,
2532				"Failed to get unfulfilled claim queue entries"
2533			);
2534			return None
2535		},
2536	};
2537	let rp_state = match state.per_relay_parent.get_mut(&relay_parent) {
2538		Some(rp_state) => rp_state,
2539		None => {
2540			gum::error!(
2541				target: LOG_TARGET,
2542				?relay_parent,
2543				"Failed to get relay parent state"
2544			);
2545			return None
2546		},
2547	};
2548
2549	// If finished one does not match waiting_collation, then we already dequeued another fetch
2550	// to replace it.
2551	if let Some((collator_id, maybe_candidate_hash)) = rp_state.collations.fetching_from.as_ref() {
2552		// If a candidate hash was saved previously, `finished_one` must include this too.
2553		if collator_id != &finished_one.0 &&
2554			maybe_candidate_hash.map_or(true, |hash| Some(&hash) != finished_one.1.as_ref())
2555		{
2556			gum::trace!(
2557				target: LOG_TARGET,
2558				waiting_collation = ?rp_state.collations.fetching_from,
2559				?finished_one,
2560				"Not proceeding to the next collation - has already been done."
2561			);
2562			return None
2563		}
2564	}
2565	rp_state.collations.status.back_to_waiting();
2566	rp_state.collations.pick_a_collation_to_fetch(unfulfilled_entries)
2567}

2569// Sanity check the candidate descriptor version.
2570fn descriptor_version_sanity_check(
2571	descriptor: &CandidateDescriptorV2,
2572	per_relay_parent: &PerRelayParent,
2573) -> std::result::Result<(), SecondingError> {
2574	match descriptor.version() {
2575		CandidateDescriptorVersion::V1 => Ok(()),
2576		CandidateDescriptorVersion::V2 if per_relay_parent.v2_receipts => {
2577			if let Some(core_index) = descriptor.core_index() {
2578				if core_index != per_relay_parent.current_core {
2579					return Err(SecondingError::InvalidCoreIndex(
2580						core_index.0,
2581						per_relay_parent.current_core.0,
2582					))
2583				}
2584			}
2585
2586			if let Some(session_index) = descriptor.session_index() {
2587				if session_index != per_relay_parent.session_index {
2588					return Err(SecondingError::InvalidSessionIndex(
2589						session_index,
2590						per_relay_parent.session_index,
2591					))
2592				}
2593			}
2594
2595			Ok(())
2596		},
2597		descriptor_version => Err(SecondingError::InvalidReceiptVersion(descriptor_version)),
2598	}
2599}

2601// Checks that there are enough free connection slots for AssetHub invulnerable collators.
2602fn can_accept_new_collator_connection(
2603	new_peer_id: &PeerId,
2604	currently_connected: &HashMap<PeerId, PeerData>,
2605	ah_invulnerables: &HashSet<PeerId>,
2606) -> bool {
2607	if ah_invulnerables.contains(new_peer_id) || ah_invulnerables.is_empty() {
2608		// If the new peer is AH invulnerable, we always accept it.
2609		return true;
2610	}
2611
2612	// We want to keep some space for new connections. If we use the whole limit we could end up in
2613	// the case when a collator keeps on making a connection requests and with some luck (or
2614	// something else I don't anticipate right now) it keeps on connecting before the other peers
2615	// and thus blocking AssetHub invulnerables from connecting.
2616	let max_accepted_connections = MAX_AUTHORITY_INCOMING_STREAMS as usize - 10;
2617	let ah_invulnerables_connected = currently_connected
2618		.keys()
2619		.filter(|peer_id| ah_invulnerables.contains(peer_id))
2620		.count();
2621
2622	// Reserve space for any unconnected AH invulnerable which might want to connect to us.
2623	let connection_limit =
2624		max_accepted_connections - ah_invulnerables.len() + ah_invulnerables_connected;
2625
2626	gum::trace!(
2627		target: LOG_TARGET,
2628		?new_peer_id,
2629		?ah_invulnerables,
2630		currently_connected=currently_connected.len(),
2631		?connection_limit,
2632		"Checking if we can accept a new collator connection",
2633	);
2634
2635	currently_connected.len() < connection_limit
2636}

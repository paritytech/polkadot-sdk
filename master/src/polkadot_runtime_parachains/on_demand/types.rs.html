33/// Meta data for full queue.
34///
35/// This includes elements with affinity and free entries.
36///
37/// The actual queue is implemented via multiple priority queues. One for each core, for entries
38/// which currently have a core affinity and one free queue, with entries without any affinity yet.
39///
40/// The design aims to have most queue accessess be O(1) or O(log(N)). Absolute worst case is O(N).
41/// Importantly this includes all accessess that happen in a single block. Even with 50 cores, the
42/// total complexity of all operations in the block should maintain above complexities. In
43/// particular O(N) stays O(N), it should never be O(N*cores).
44///
45/// More concrete rundown on complexity:
46///
47///  - insert: O(1) for placing an order, O(log(N)) for push backs.
48///  - pop_assignment_for_core: O(log(N)), O(N) worst case: Can only happen for one core, next core
49///  is already less work.
50///  - report_processed & push back: If affinity dropped to 0, then O(N) in the worst case. Again
51///  this divides per core.
52///
53///  Reads still exist, also improved slightly, but worst case we fetch all entries.
54#[derive(Encode, Decode, TypeInfo)]
55pub struct QueueStatusType {
56	/// Last calculated traffic value.
57	pub traffic: FixedU128,
58	/// The next index to use.
59	pub next_index: QueueIndex,
60	/// Smallest index still in use.
61	///
62	/// In case of a completely empty queue (free + affinity queues), `next_index - smallest_index
63	/// == 0`.
64	pub smallest_index: QueueIndex,
65	/// Indices that have been freed already.
66	///
67	/// But have a hole to `smallest_index`, so we can not yet bump `smallest_index`. This binary
68	/// heap is roughly bounded in the number of on demand cores:
69	///
70	/// For a single core, elements will always be processed in order. With each core added, a
71	/// level of out of order execution is added.
72	pub freed_indices: BinaryHeap<ReverseQueueIndex>,
73}
74
75impl Default for QueueStatusType {
76	fn default() -> QueueStatusType {
77		QueueStatusType {
78			traffic: FixedU128::default(),
79			next_index: QueueIndex(0),
80			smallest_index: QueueIndex(0),
81			freed_indices: BinaryHeap::new(),
82		}
83	}
84}
85
86impl QueueStatusType {
87	/// How many orders are queued in total?
88	///
89	/// This includes entries which have core affinity.
90	pub fn size(&self) -> u32 {
91		self.next_index
92			.0
93			.overflowing_sub(self.smallest_index.0)
94			.0
95			.saturating_sub(self.freed_indices.len() as u32)
96	}
97
98	/// Get current next index
99	///
100	/// to use for an element newly pushed to the back of the queue.
101	pub fn push_back(&mut self) -> QueueIndex {
102		let QueueIndex(next_index) = self.next_index;
103		self.next_index = QueueIndex(next_index.overflowing_add(1).0);
104		QueueIndex(next_index)
105	}
106
107	/// Push something to the front of the queue
108	pub fn push_front(&mut self) -> QueueIndex {
109		self.smallest_index = QueueIndex(self.smallest_index.0.overflowing_sub(1).0);
110		self.smallest_index
111	}
112
113	/// The given index is no longer part of the queue.
114	///
115	/// This updates `smallest_index` if need be.
116	pub fn consume_index(&mut self, removed_index: QueueIndex) {
117		if removed_index != self.smallest_index {
118			self.freed_indices.push(removed_index.reverse());
119			return
120		}
121		let mut index = self.smallest_index.0.overflowing_add(1).0;
122		// Even more to advance?
123		while self.freed_indices.peek() == Some(&ReverseQueueIndex(index)) {
124			index = index.overflowing_add(1).0;
125			self.freed_indices.pop();
126		}
127		self.smallest_index = QueueIndex(index);
128	}
129}

131/// Type used for priority indices.
132//  NOTE: The `Ord` implementation for this type is unsound in the general case.
133//        Do not use it for anything but it's intended purpose.
134#[derive(Encode, Decode, TypeInfo, Debug, PartialEq, Clone, Eq, Copy)]
135pub struct QueueIndex(pub u32);
136
137/// QueueIndex with reverse ordering.
138///
139/// Same as `Reverse(QueueIndex)`, but with all the needed traits implemented.
140#[derive(Encode, Decode, TypeInfo, Debug, PartialEq, Clone, Eq, Copy)]
141pub struct ReverseQueueIndex(pub u32);
142
143impl QueueIndex {
144	fn reverse(self) -> ReverseQueueIndex {
145		ReverseQueueIndex(self.0)
146	}
147}
148
149impl Ord for QueueIndex {
150	fn cmp(&self, other: &Self) -> Ordering {
151		let diff = self.0.overflowing_sub(other.0).0;
152		if diff == 0 {
153			Ordering::Equal
154		} else if diff <= ON_DEMAND_MAX_QUEUE_MAX_SIZE {
155			Ordering::Greater
156		} else {
157			Ordering::Less
158		}
159	}
160}
161
162impl PartialOrd for QueueIndex {
163	fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
164		Some(self.cmp(other))
165	}
166}
167
168impl Ord for ReverseQueueIndex {
169	fn cmp(&self, other: &Self) -> Ordering {
170		QueueIndex(other.0).cmp(&QueueIndex(self.0))
171	}
172}
173impl PartialOrd for ReverseQueueIndex {
174	fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
175		Some(self.cmp(&other))
176	}
177}

179/// Internal representation of an order after it has been enqueued already.
180///
181/// This data structure is provided for a min BinaryHeap (Ord compares in reverse order with regards
182/// to its elements)
183#[derive(Encode, Decode, TypeInfo, Debug, PartialEq, Clone, Eq)]
184pub struct EnqueuedOrder {
185	pub para_id: ParaId,
186	pub idx: QueueIndex,
187}
188
189impl EnqueuedOrder {
190	pub fn new(idx: QueueIndex, para_id: ParaId) -> Self {
191		Self { idx, para_id }
192	}
193}
194
195impl PartialOrd for EnqueuedOrder {
196	fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
197		match other.idx.partial_cmp(&self.idx) {
198			Some(Ordering::Equal) => other.para_id.partial_cmp(&self.para_id),
199			o => o,
200		}
201	}
202}
203
204impl Ord for EnqueuedOrder {
205	fn cmp(&self, other: &Self) -> Ordering {
206		match other.idx.cmp(&self.idx) {
207			Ordering::Equal => other.para_id.cmp(&self.para_id),
208			o => o,
209		}
210	}
211}

3use crate::{PublicKey, Signature};
4use codec::{Decode, DecodeWithMemTracking, Encode};
5use frame_support::{ensure, PalletError};
6pub use milagro_bls::{
7	AggregatePublicKey, AggregateSignature, PublicKey as PublicKeyPrepared,
8	Signature as SignaturePrepared,
9};
10use scale_info::TypeInfo;
11use sp_core::H256;
12use sp_std::prelude::*;
13use Debug;
14
15#[derive(
16	Copy, Clone, Encode, Decode, DecodeWithMemTracking, Eq, PartialEq, TypeInfo, Debug, PalletError,
17)]
18pub enum BlsError {
19	InvalidSignature,
20	InvalidPublicKey,
21	InvalidAggregatePublicKeys,
22	SignatureVerificationFailed,
23}
24
25/// fast_aggregate_verify optimized with aggregate key subtracting absent ones.
26pub fn fast_aggregate_verify(
27	aggregate_pubkey: &PublicKeyPrepared,
28	absent_pubkeys: &Vec<PublicKeyPrepared>,
29	message: H256,
30	signature: &Signature,
31) -> Result<(), BlsError> {
32	let agg_sig = prepare_aggregate_signature(signature)?;
33	let agg_key = prepare_aggregate_pubkey_from_absent(aggregate_pubkey, absent_pubkeys)?;
34	fast_aggregate_verify_pre_aggregated(agg_sig, agg_key, message)
35}
36
37/// Decompress one public key into a point in G1.
38pub fn prepare_milagro_pubkey(pubkey: &PublicKey) -> Result<PublicKeyPrepared, BlsError> {
39	PublicKeyPrepared::from_bytes_unchecked(&pubkey.0).map_err(|_| BlsError::InvalidPublicKey)
40}
41
42/// Prepare for G1 public keys.
43pub fn prepare_g1_pubkeys(pubkeys: &[PublicKey]) -> Result<Vec<PublicKeyPrepared>, BlsError> {
44	pubkeys
45		.iter()
46		// Deserialize one public key from compressed bytes
47		.map(prepare_milagro_pubkey)
48		.collect::<Result<Vec<PublicKeyPrepared>, BlsError>>()
49}
50
51/// Prepare for G1 AggregatePublicKey.
52pub fn prepare_aggregate_pubkey(
53	pubkeys: &[PublicKeyPrepared],
54) -> Result<AggregatePublicKey, BlsError> {
55	AggregatePublicKey::into_aggregate(pubkeys).map_err(|_| BlsError::InvalidPublicKey)
56}
57
58/// Prepare for G1 AggregatePublicKey.
59pub fn prepare_aggregate_pubkey_from_absent(
60	aggregate_key: &PublicKeyPrepared,
61	absent_pubkeys: &Vec<PublicKeyPrepared>,
62) -> Result<AggregatePublicKey, BlsError> {
63	let mut aggregate_pubkey = AggregatePublicKey::from_public_key(aggregate_key);
64	if !absent_pubkeys.is_empty() {
65		let absent_aggregate_key = prepare_aggregate_pubkey(absent_pubkeys)?;
66		aggregate_pubkey.point.sub(&absent_aggregate_key.point);
67	}
68	Ok(AggregatePublicKey { point: aggregate_pubkey.point })
69}
70
71/// Prepare for G2 AggregateSignature, normally more expensive than G1 operation.
72pub fn prepare_aggregate_signature(signature: &Signature) -> Result<AggregateSignature, BlsError> {
73	Ok(AggregateSignature::from_signature(
74		&SignaturePrepared::from_bytes(&signature.0).map_err(|_| BlsError::InvalidSignature)?,
75	))
76}
77
78/// fast_aggregate_verify_pre_aggregated which is the most expensive call in beacon light client.
79pub fn fast_aggregate_verify_pre_aggregated(
80	agg_sig: AggregateSignature,
81	aggregate_key: AggregatePublicKey,
82	message: H256,
83) -> Result<(), BlsError> {
84	ensure!(
85		agg_sig.fast_aggregate_verify_pre_aggregated(&message[..], &aggregate_key),
86		BlsError::SignatureVerificationFailed
87	);
88	Ok(())
89}

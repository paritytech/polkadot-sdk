184/// Prepared G1 public key of sync committee as it is stored in the runtime storage.
185#[derive(Clone, PartialEq, Eq, Encode, Decode, TypeInfo, MaxEncodedLen)]
186pub struct SyncCommitteePrepared<const COMMITTEE_SIZE: usize> {
187	pub root: H256,
188	pub pubkeys: Box<[PublicKeyPrepared; COMMITTEE_SIZE]>,
189	pub aggregate_pubkey: PublicKeyPrepared,
190}
191
192impl<const COMMITTEE_SIZE: usize> Default for SyncCommitteePrepared<COMMITTEE_SIZE> {
193	fn default() -> Self {
194		let pubkeys: Vec<PublicKeyPrepared> =
195			repeat(PublicKeyPrepared::default()).take(COMMITTEE_SIZE).collect();
196		let pubkeys: Box<[PublicKeyPrepared; COMMITTEE_SIZE]> =
197			Box::new(pubkeys.try_into().map_err(|_| ()).expect("checked statically; qed"));
198
199		SyncCommitteePrepared {
200			root: H256::default(),
201			pubkeys,
202			aggregate_pubkey: PublicKeyPrepared::default(),
203		}
204	}
205}
206
207impl<const COMMITTEE_SIZE: usize> TryFrom<&SyncCommittee<COMMITTEE_SIZE>>
208	for SyncCommitteePrepared<COMMITTEE_SIZE>
209{
210	type Error = BlsError;
211
212	fn try_from(sync_committee: &SyncCommittee<COMMITTEE_SIZE>) -> Result<Self, Self::Error> {
213		let g1_pubkeys = prepare_g1_pubkeys(&sync_committee.pubkeys)?;
214		let sync_committee_root = sync_committee.hash_tree_root().expect("checked statically; qed");
215
216		Ok(SyncCommitteePrepared::<COMMITTEE_SIZE> {
217			pubkeys: g1_pubkeys.try_into().map_err(|_| ()).expect("checked statically; qed"),
218			aggregate_pubkey: prepare_milagro_pubkey(&sync_committee.aggregate_pubkey)?,
219			root: sync_committee_root,
220		})
221	}
222}

301	#[cfg_attr(feature = "std", serde(deserialize_with = "crate::serde_utils::from_hex_to_bytes"))]
302	pub sync_committee_bits: Vec<u8>,

305
306#[cfg(feature = "std")]
307impl<const COMMITTEE_SIZE: usize, const COMMITTEE_BITS_SIZE: usize>
308	TryFrom<IntermediateSyncAggregate> for SyncAggregate<COMMITTEE_SIZE, COMMITTEE_BITS_SIZE>
309{
310	type Error = String;
311
312	fn try_from(other: IntermediateSyncAggregate) -> Result<Self, Self::Error> {
313		Ok(Self {
314			sync_committee_bits: other
315				.sync_committee_bits
316				.try_into()
317				.map_err(|_| "unexpected length".to_owned())?,
318			sync_committee_signature: other.sync_committee_signature,
319		})
320	}

392impl VersionedExecutionPayloadHeader {
393	pub fn hash_tree_root(&self) -> Result<H256, SimpleSerializeError> {
394		match self {
395			VersionedExecutionPayloadHeader::Capella(execution_payload_header) =>
396				hash_tree_root::<SSZExecutionPayloadHeader>(
397					execution_payload_header.clone().try_into()?,
398				),
399			VersionedExecutionPayloadHeader::Deneb(execution_payload_header) =>
400				hash_tree_root::<crate::ssz::deneb::SSZExecutionPayloadHeader>(
401					execution_payload_header.clone().try_into()?,
402				),
403		}
404	}
405
406	pub fn block_hash(&self) -> H256 {
407		match self {
408			VersionedExecutionPayloadHeader::Capella(execution_payload_header) =>
409				execution_payload_header.block_hash,
410			VersionedExecutionPayloadHeader::Deneb(execution_payload_header) =>
411				execution_payload_header.block_hash,
412		}
413	}
414
415	pub fn block_number(&self) -> u64 {
416		match self {
417			VersionedExecutionPayloadHeader::Capella(execution_payload_header) =>
418				execution_payload_header.block_number,
419			VersionedExecutionPayloadHeader::Deneb(execution_payload_header) =>
420				execution_payload_header.block_number,
421		}
422	}
423
424	pub fn receipts_root(&self) -> H256 {
425		match self {
426			VersionedExecutionPayloadHeader::Capella(execution_payload_header) =>
427				execution_payload_header.receipts_root,
428			VersionedExecutionPayloadHeader::Deneb(execution_payload_header) =>
429				execution_payload_header.receipts_root,
430		}
431	}

184/// Prepared G1 public key of sync committee as it is stored in the runtime storage.
185#[derive(Clone, PartialEq, Eq, Encode, Decode, TypeInfo, MaxEncodedLen)]
186pub struct SyncCommitteePrepared<const COMMITTEE_SIZE: usize> {
187	pub root: H256,
188	pub pubkeys: Box<[PublicKeyPrepared; COMMITTEE_SIZE]>,
189	pub aggregate_pubkey: PublicKeyPrepared,
190}
191
192impl<const COMMITTEE_SIZE: usize> Default for SyncCommitteePrepared<COMMITTEE_SIZE> {
193	fn default() -> Self {
194		let pubkeys: Vec<PublicKeyPrepared> =
195			repeat(PublicKeyPrepared::default()).take(COMMITTEE_SIZE).collect();
196		let pubkeys: Box<[PublicKeyPrepared; COMMITTEE_SIZE]> =
197			Box::new(pubkeys.try_into().map_err(|_| ()).expect("checked statically; qed"));
198
199		SyncCommitteePrepared {
200			root: H256::default(),
201			pubkeys,
202			aggregate_pubkey: PublicKeyPrepared::default(),
203		}
204	}
205}
206
207impl<const COMMITTEE_SIZE: usize> TryFrom<&SyncCommittee<COMMITTEE_SIZE>>
208	for SyncCommitteePrepared<COMMITTEE_SIZE>
209{
210	type Error = BlsError;
211
212	fn try_from(sync_committee: &SyncCommittee<COMMITTEE_SIZE>) -> Result<Self, Self::Error> {
213		let g1_pubkeys = prepare_g1_pubkeys(&sync_committee.pubkeys)?;
214		let sync_committee_root = sync_committee.hash_tree_root().expect("checked statically; qed");
215
216		Ok(SyncCommitteePrepared::<COMMITTEE_SIZE> {
217			pubkeys: g1_pubkeys.try_into().map_err(|_| ()).expect("checked statically; qed"),
218			aggregate_pubkey: prepare_milagro_pubkey(&sync_committee.aggregate_pubkey)?,
219			root: sync_committee_root,
220		})
221	}
222}

306struct IntermediateSyncAggregate {
307	#[cfg_attr(feature = "std", serde(deserialize_with = "crate::serde_utils::from_hex_to_bytes"))]
308	pub sync_committee_bits: Vec<u8>,
309	pub sync_committee_signature: Signature,
310}
311
312#[cfg(feature = "std")]
313impl<const COMMITTEE_SIZE: usize, const COMMITTEE_BITS_SIZE: usize>
314	TryFrom<IntermediateSyncAggregate> for SyncAggregate<COMMITTEE_SIZE, COMMITTEE_BITS_SIZE>
315{
316	type Error = String;
317
318	fn try_from(other: IntermediateSyncAggregate) -> Result<Self, Self::Error> {
319		Ok(Self {
320			sync_committee_bits: other
321				.sync_committee_bits
322				.try_into()
323				.map_err(|_| "unexpected length".to_owned())?,
324			sync_committee_signature: other.sync_committee_signature,
325		})
326	}

412impl VersionedExecutionPayloadHeader {
413	pub fn hash_tree_root(&self) -> Result<H256, SimpleSerializeError> {
414		match self {
415			VersionedExecutionPayloadHeader::Capella(execution_payload_header) =>
416				hash_tree_root::<SSZExecutionPayloadHeader>(
417					execution_payload_header.clone().try_into()?,
418				),
419			VersionedExecutionPayloadHeader::Deneb(execution_payload_header) =>
420				hash_tree_root::<crate::ssz::deneb::SSZExecutionPayloadHeader>(
421					execution_payload_header.clone().try_into()?,
422				),
423		}
424	}
425
426	pub fn block_hash(&self) -> H256 {
427		match self {
428			VersionedExecutionPayloadHeader::Capella(execution_payload_header) =>
429				execution_payload_header.block_hash,
430			VersionedExecutionPayloadHeader::Deneb(execution_payload_header) =>
431				execution_payload_header.block_hash,
432		}
433	}
434
435	pub fn block_number(&self) -> u64 {
436		match self {
437			VersionedExecutionPayloadHeader::Capella(execution_payload_header) =>
438				execution_payload_header.block_number,
439			VersionedExecutionPayloadHeader::Deneb(execution_payload_header) =>
440				execution_payload_header.block_number,
441		}
442	}
443
444	pub fn receipts_root(&self) -> H256 {
445		match self {
446			VersionedExecutionPayloadHeader::Capella(execution_payload_header) =>
447				execution_payload_header.receipts_root,
448			VersionedExecutionPayloadHeader::Deneb(execution_payload_header) =>
449				execution_payload_header.receipts_root,
450		}
451	}
452}

3use snowbridge_ethereum::{mpt, Decodable, ReceiptEnvelope};
4use sp_core::H256;
5use sp_io::hashing::keccak_256;
6use sp_std::prelude::*;
7
8pub fn verify_receipt_proof(receipts_root: H256, values: &[Vec<u8>]) -> Option<ReceiptEnvelope> {
9	match apply_merkle_proof(values) {
10		Some((root, data)) if root == receipts_root =>
11			ReceiptEnvelope::decode(&mut data.as_slice()).ok(),
12		Some((_, _)) => None,
13		None => None,
14	}
15}
16
17fn apply_merkle_proof(proof: &[Vec<u8>]) -> Option<(H256, Vec<u8>)> {
18	let mut iter = proof.iter().rev();
19	let first_bytes = match iter.next() {
20		Some(b) => b,
21		None => return None,
22	};
23	let item_to_prove: mpt::ShortNode = rlp::decode(first_bytes).ok()?;
24
25	let final_hash: Option<[u8; 32]> = iter.try_fold(keccak_256(first_bytes), |acc, x| {
26		let node: Box<dyn mpt::Node> = x.as_slice().try_into().ok()?;
27		if (*node).contains_hash(acc.into()) {
28			return Some(keccak_256(x))
29		}
30		None
31	});
32
33	final_hash.map(|hash| (hash.into(), item_to_prove.value))
34}

43	#[test]
44	fn rc_session_change_reported_to_ah() {
45		// sets up AH chain with current and active era.
46		shared::put_ah_state(ah::ExtBuilder::default().build());
47		shared::put_rc_state(rc::ExtBuilder::default().build());
48		// shared::RC_STATE.with(|state| *state.get_mut() = rc::ExtBuilder::default().build());
49
50		// initial state of ah
51		shared::in_ah(|| {
52			assert_eq!(frame_system::Pallet::<ah::Runtime>::block_number(), 1);
53
54			assert_eq!(pallet_staking_async::CurrentEra::<ah::Runtime>::get(), Some(0));
55			assert_eq!(
56				ActiveEra::<ah::Runtime>::get(),
57				Some(ActiveEraInfo { index: 0, start: Some(0) })
58			);
59		});
60
61		shared::in_rc(|| {
62			// initial state of rc
63			assert_eq!(ah_client::Mode::<rc::Runtime>::get(), OperatingMode::Active);
64			// go to session 1 in RC and test.
65			// when
66			assert!(frame_system::Pallet::<rc::Runtime>::block_number() == 1);
67
68			// given end session 0, start session 1, plan 2
69			rc::roll_until_matches(
70				|| pallet_session::CurrentIndex::<rc::Runtime>::get() == 1,
71				true,
72			);
73
74			// then
75			assert_eq!(frame_system::Pallet::<rc::Runtime>::block_number(), rc::Period::get());
76		});
77
78		shared::in_ah(|| {
79			// ah's rc-client has also progressed some blocks, equal to 1 session
80			assert_eq!(frame_system::Pallet::<ah::Runtime>::block_number(), 30);
81			// election is ongoing, and has just started
82			assert_eq!(pallet_staking_async::CurrentEra::<ah::Runtime>::get(), Some(1));
83			assert!(matches!(
84				multi_block::CurrentPhase::<ah::Runtime>::get(),
85				multi_block::Phase::Snapshot(_)
86			));
87		});
88
89		shared::in_rc(|| {
90			// roll a few more sessions
91			rc::roll_until_matches(
92				|| pallet_session::CurrentIndex::<rc::Runtime>::get() == 4,
93				true,
94			);
95		});
96
97		shared::in_ah(|| {
98			// ah's rc-client has also progressed some blocks, equal to 4 sessions
99			assert_eq!(frame_system::Pallet::<ah::Runtime>::block_number(), 120);
100
101			// Election is done
102			assert_eq!(multi_block::CurrentPhase::<ah::Runtime>::get(), multi_block::Phase::Off);
103
104			// Validator set is queued
105			assert!(rc_client::OutgoingValidatorSet::<ah::Runtime>::exists());
106		});
107
108		// go to session 5 in rc, and forward AH too.
109		shared::in_rc(|| {
110			rc::roll_until_matches(
111				|| pallet_session::CurrentIndex::<rc::Runtime>::get() == 5,
112				true,
113			);
114		});
115
116		// ah has bumped the current era, but not the active era
117		shared::in_ah(|| {
118			assert_eq!(pallet_staking_async::CurrentEra::<ah::Runtime>::get(), Some(1));
119			assert_eq!(
120				ActiveEra::<ah::Runtime>::get(),
121				Some(ActiveEraInfo { index: 0, start: Some(0) })
122			);
123		});
124
125		// go to session 6 in rc, and forward AH too.
126		shared::in_rc(|| {
127			rc::roll_until_matches(
128				|| pallet_session::CurrentIndex::<rc::Runtime>::get() == 6,
129				true,
130			);
131		});
132	}

672		shared::in_ah(|| {
673			// roll another block just to strongly prove election is not kicked off at the end of

675			ah::roll_next();

678			// (when election is kicked off, current_era = active_era + 1)

686		// overdue. The next session change should trigger an election.

692				frame_system::Pallet::<rc::Runtime>::block_number();
693
694			// all the buffered validators points are flushed
695			assert_eq!(ah_client::ValidatorPoints::<rc::Runtime>::iter().count(), 0,);
696		});

698		// AH receives the session report.
699		assert_eq!(shared::CounterRCAHSessionReport::get(), 1);
700		shared::in_ah(|| {
701			assert_eq!(pallet_staking_async::ActiveEra::<ah::Runtime>::get().unwrap().index, 1);
702			assert_eq!(pallet_staking_async::CurrentEra::<ah::Runtime>::get(), Some(1 + 1));
703
704			// by now one session report should have been received in staking
705			assert_eq!(
706				ah::rc_client_events_since_last_call(),
707				vec![rc_client::Event::SessionReportReceived {
708					end_index: 14,
709					activation_timestamp: None,

711					leftover: false
712				}]
713			);

715			assert_eq!(
716				staking_events_since_last_call(),

719					active_era: 1,
720					planned_era: 2
721				}]
722			);
723
724			// all expected era reward points are here
725			assert_eq!(
726				pallet_staking_async::ErasRewardPoints::<ah::Runtime>::get(1).total,

728				// --- ^^ these were buffered in ah-client
729					post_migration_era_reward_points // --- ^^ these were migrated as part of AHM
730			);
731
732			// ensure new validator is sent once election is complete.
733			ah::roll_until_matches(|| shared::CounterAHRCValidatorSet::get() == 1, true);
734
735			assert_eq!(
736				ah::staking_events_since_last_call(),
737				vec![
738					pallet_staking_async::Event::PagedElectionProceeded { page: 2, result: Ok(4) },
739					pallet_staking_async::Event::PagedElectionProceeded { page: 1, result: Ok(0) },
740					pallet_staking_async::Event::PagedElectionProceeded { page: 0, result: Ok(0) }
741				]
742			);
743		});
744
745		shared::in_rc(|| {
746			assert_eq!(
747				rc::ah_client_events_since_last_call(),
748				vec![ah_client::Event::ValidatorSetReceived {
749					id: 2,
750					new_validator_set_count: 4,
751					prune_up_to: None,
752					leftover: false
753				}]
754			);
755
756			let (planned_era, next_validator_set) =
757				ah_client::ValidatorSet::<rc::Runtime>::get().unwrap();
758
759			assert_eq!(planned_era, 2);
760			assert!(next_validator_set.len() >= rc::MinimumValidatorSetSize::get() as usize);
761		});
762
763		shared::in_ah(|| {
764			assert_eq!(pallet_staking_async::ActiveEra::<ah::Runtime>::get().unwrap().index, 1);
765			// at next session, the validator set is queued but not applied yet.
766			ah::roll_until_matches(|| shared::CounterRCAHSessionReport::get() == 2, true);
767			// active era is still 1.
768			assert_eq!(pallet_staking_async::ActiveEra::<ah::Runtime>::get().unwrap().index, 1);
769			// the following session, the validator set is applied.
770			ah::roll_until_matches(|| shared::CounterRCAHSessionReport::get() == 3, true);
771			assert_eq!(pallet_staking_async::ActiveEra::<ah::Runtime>::get().unwrap().index, 2);
772		});
773	}

888					activation_timestamp: Some((0, 0)),
889					leftover: false,
890					validator_points: (0..num_validator_points)

895				// Note: the real encoded size of the message will be a few bytes more, due to call

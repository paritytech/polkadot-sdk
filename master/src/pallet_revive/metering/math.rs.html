37	/// Create a transaction-level (root) meter for Substrate-style execution.
38	///
39	/// This constructs the root resource meter that enforces explicit weight and
40	/// storage-deposit limits for the whole transaction. The returned `TransactionMeter`:
41	/// - charges weight via `WeightMeter` with the provided `weight_limit`,
42	/// - accounts storage deposit via `RootStorageMeter` with the provided `deposit_limit`,
43	/// - records that the transaction's limit mode is `WeightAndDeposit`.
44	pub fn new_root<T: Config>(
45		weight_limit: Weight,
46		deposit_limit: BalanceOf<T>,
47	) -> Result<TransactionMeter<T>, DispatchError> {
48		Ok(TransactionMeter {
49			weight: WeightMeter::new(Some(weight_limit), None),
50			deposit: RootStorageMeter::new(Some(deposit_limit)),
51			// ignore max total gas for Substrate executions
52			max_total_gas: Default::default(),
53			total_consumed_weight_before: Default::default(),
54			total_consumed_deposit_before: Default::default(),
55			transaction_limits: TransactionLimits::WeightAndDeposit { weight_limit, deposit_limit },
56			_phantom: PhantomData,
57		})
58	}

60	/// Create a nested (frame) meter derived from a parent `ResourceMeter`.
61	///
62	/// This produces a frame-local meter that enforces the resource limits for
63	/// a nested call. It computes how much of the parent's remaining resources are available
64	/// to the nested frame by:
65	/// - collecting the parent's own consumed amounts (`self_consumed_*`),
66	/// - deriving the total consumed amounts up to this point,
67	/// - applying the requested `CallResources` (no limits, ethereum gas conversion, or explicit
68	///   weight+deposit) to derive per-frame limits.
69	///
70	/// Returns `Err(Error::OutOfGas)` when weight is exhausted, or
71	/// `Err(Error::StorageDepositLimitExhausted)` when deposit bookkeeping forbids further storage.
72	pub fn new_nested_meter<T: Config, S: State>(
73		meter: &ResourceMeter<T, S>,
74		limit: &CallResources<T>,
75	) -> Result<FrameMeter<T>, DispatchError> {
76		let self_consumed_weight = meter.weight.weight_consumed();
77		let self_consumed_deposit = meter.deposit.consumed();
78
79		let total_consumed_weight =
80			meter.total_consumed_weight_before.saturating_add(self_consumed_weight);
81		let total_consumed_deposit =
82			meter.total_consumed_deposit_before.saturating_add(&self_consumed_deposit);
83
84		let weight_left = meter
85			.weight
86			.weight_limit
87			.expect(
88				"Weight limits are always defined for `ResourceMeter` in Substrate \
89				execution mode (i.e., when its `transaction_limits` are `WeightAndDeposit`); qed",
90			)
91			.checked_sub(&self_consumed_weight)
92			.ok_or(<Error<T>>::OutOfGas)?;
93
94		let deposit_limit = meter.deposit.limit.expect(
95			"Deposit limits are always defined for `ResourceMeter` in Substrate \
96				execution mode (i.e., when its `transaction_limits` are `WeightAndDeposit`); qed",
97		);
98		let deposit_left = self_consumed_deposit
99			.available(&deposit_limit)
100			.ok_or(<Error<T>>::StorageDepositLimitExhausted)?;
101
102		let (nested_weight_limit, nested_deposit_limit, stipend) = {
103			match limit {
104				CallResources::NoLimits => (weight_left, deposit_left, None),
105
106				CallResources::Ethereum { gas, add_stipend } => {
107					// Convert leftover weight and deposit to an ethereum-gas equivalent,
108					// then cap that gas by the requested `gas`. Distribute the capped gas
109					// back into weight and deposit portions using the same ratio so that
110					// the nested frame receives proportional limits.
111					let weight_gas_left = SignedGas::<T>::from_weight_fee(
112						T::FeeInfo::weight_to_fee_average(&weight_left),
113					);
114					let deposit_gas_left = SignedGas::<T>::from_adjusted_deposit_charge(
115						&StorageDeposit::Charge(deposit_left),
116					);
117					let Some(remaining_gas) =
118						(weight_gas_left.saturating_add(&deposit_gas_left)).to_ethereum_gas()
119					else {
120						return Err(<Error<T>>::OutOfGas.into());
121					};
122
123					let gas_limit = remaining_gas.min(*gas);
124
125					let ratio = if remaining_gas.is_zero() {
126						FixedU128::one()
127					} else {
128						FixedU128::from_rational(
129							gas_limit.saturated_into(),
130							remaining_gas.saturated_into(),
131						)
132					};
133
134					let mut weight_limit = Weight::from_parts(
135						ratio.saturating_mul_int(weight_left.ref_time()),
136						ratio.saturating_mul_int(weight_left.proof_size()),
137					);
138					let deposit_limit = ratio.saturating_mul_int(deposit_left);
139
140					let stipend = if *add_stipend {
141						let weight_stipend = determine_call_stipend::<T>();
142						if weight_left.any_lt(weight_stipend) {
143							return Err(<Error<T>>::OutOfGas.into())
144						}
145
146						weight_limit.saturating_accrue(weight_stipend);
147
148						Some(weight_stipend)
149					} else {
150						None
151					};
152
153					(weight_left.min(weight_limit), deposit_left.min(deposit_limit), stipend)
154				},
155
156				CallResources::WeightDeposit { weight, deposit_limit } =>
157				// when explicit weight+deposit requested, take the minimum of parent's left
158				// and the requested per-call limits.
159					(weight_left.min(*weight), deposit_left.min(*deposit_limit), None),
160			}
161		};
162
163		Ok(FrameMeter::<T> {
164			weight: WeightMeter::new(Some(nested_weight_limit), stipend),
165			deposit: meter.deposit.nested(Some(nested_deposit_limit)),
166			max_total_gas: Default::default(),
167			total_consumed_weight_before: total_consumed_weight,
168			total_consumed_deposit_before: total_consumed_deposit,
169			transaction_limits: meter.transaction_limits.clone(),
170			_phantom: PhantomData,
171		})
172	}

174	/// Compute the remaining ethereum-gas-equivalent for a Substrate-style transaction.
175	///
176	/// Converts the remaining weight and deposit into their gas-equivalents (via `FeeInfo`) and
177	/// returns the sum. Returns `None` if either component does not have enough left.
178	pub fn gas_left<T: Config, S: State>(meter: &ResourceMeter<T, S>) -> Option<SignedGas<T>> {
179		match (weight_left(meter), deposit_left(meter)) {
180			(Some(weight_left), Some(deposit_left)) => {
181				let weight_gas_left = SignedGas::<T>::from_weight_fee(
182					T::FeeInfo::weight_to_fee_average(&weight_left),
183				);
184				let deposit_gas_left = SignedGas::<T>::from_adjusted_deposit_charge(
185					&StorageDeposit::Charge(deposit_left),
186				);
187
188				Some(weight_gas_left.saturating_add(&deposit_gas_left))
189			},
190			_ => None,
191		}
192	}

194	/// Return remaining weight available in the given meter.
195	///
196	/// Subtracts the weight already consumed in the current frame from the configured limit.
197	pub fn weight_left<T: Config, S: State>(meter: &ResourceMeter<T, S>) -> Option<Weight> {
198		let weight_limit = meter.weight.weight_limit.expect(
199			"Weight limits are always defined for `ResourceMeter` in Substrate \
200				execution mode (i.e., when its `transaction_limits` are `WeightAndDeposit`); qed",
201		);
202		weight_limit.checked_sub(&meter.weight.weight_consumed())
203	}
204
205	/// Return remaining deposit available to the given meter.
206	///
207	/// Subtracts the storage deposit already consumed in the current frame from the configured
208	/// limit.
209	pub fn deposit_left<T: Config, S: State>(meter: &ResourceMeter<T, S>) -> Option<BalanceOf<T>> {
210		let deposit_limit = meter.deposit.limit.expect(
211			"Deposit limits are always defined for `ResourceMeter` in Substrate \
212				execution mode (i.e., when its `transaction_limits` are `WeightAndDeposit`); qed",
213		);
214		meter.deposit.consumed().available(&deposit_limit)
215	}

217	/// Compute the total consumed gas (signed) for Substrate-style execution.
218	///
219	/// This returns a `SignedGas` as the consumed gas can be negative (when there are major storage
220	/// deposit refunds)
221	pub fn total_consumed_gas<T: Config, S: State>(meter: &ResourceMeter<T, S>) -> SignedGas<T> {
222		let self_consumed_weight = meter.weight.weight_consumed();
223		let self_consumed_deposit = meter.deposit.consumed();
224
225		let total_consumed_weight =
226			meter.total_consumed_weight_before.saturating_add(self_consumed_weight);
227		let total_consumed_deposit =
228			meter.total_consumed_deposit_before.saturating_add(&self_consumed_deposit);
229
230		let consumed_weight_gas =
231			SignedGas::from_weight_fee(T::FeeInfo::weight_to_fee_average(&total_consumed_weight));
232		let consumed_deposit_gas = SignedGas::from_adjusted_deposit_charge(&total_consumed_deposit);
233
234		consumed_deposit_gas.saturating_add(&consumed_weight_gas)
235	}

237	/// Compute the gas (signed) during the lifetime of this meter for Substrate-style execution.
238	pub fn eth_gas_consumed<T: Config, S: State>(meter: &ResourceMeter<T, S>) -> SignedGas<T> {
239		let self_consumed_weight = meter.weight.weight_consumed();
240		let self_consumed_deposit = meter.deposit.consumed();
241
242		let total_consumed_weight =
243			meter.total_consumed_weight_before.saturating_add(self_consumed_weight);
244
245		let consumed_weight_gas_before = SignedGas::from_weight_fee(
246			T::FeeInfo::weight_to_fee_average(&meter.total_consumed_weight_before),
247		);
248		let consumed_weight_gas =
249			SignedGas::from_weight_fee(T::FeeInfo::weight_to_fee_average(&total_consumed_weight));
250
251		let self_consumed_weight_gas =
252			consumed_weight_gas.saturating_sub(&consumed_weight_gas_before);
253
254		let self_consumed_deposit_gas =
255			SignedGas::from_adjusted_deposit_charge(&self_consumed_deposit);
256
257		self_consumed_deposit_gas.saturating_add(&self_consumed_weight_gas)
258	}

264	/// Create a transaction-level (root) meter for Ethereum-style execution.
265	///
266	/// This constructs a root `TransactionMeter` where the global limit is an
267	/// ethereum-gas budget (`max_total_gas`). Weight and deposit meters are left unbounded
268	/// (None). The function validates that there is positive gas left after initialization,
269	/// otherwise it returns an error.
270	pub fn new_root<T: Config>(
271		eth_gas_limit: BalanceOf<T>,
272		maybe_weight_limit: Option<Weight>,
273		eth_tx_info: EthTxInfo<T>,
274	) -> Result<TransactionMeter<T>, DispatchError> {
275		let meter = TransactionMeter {
276			weight: WeightMeter::new(maybe_weight_limit, None),
277			deposit: RootStorageMeter::new(None),
278			max_total_gas: SignedGas::from_ethereum_gas(eth_gas_limit),
279			total_consumed_weight_before: Default::default(),
280			total_consumed_deposit_before: Default::default(),
281			transaction_limits: TransactionLimits::EthereumGas {
282				eth_gas_limit,
283				maybe_weight_limit,
284				eth_tx_info,
285			},
286			_phantom: PhantomData,
287		};
288
289		if meter.eth_gas_left().is_some() {
290			Ok(meter)
291		} else {
292			return Err(<Error<T>>::OutOfGas.into());
293		}
294	}

296	/// Create a nested (frame) meter for an Ethereum-style execution.
297	///
298	/// - computes the gas already consumed by the transaction and determines how much gas is left,
299	/// - if the parent is in a simple gas-only mode, returns a child meter that is limited only by
300	///   gas (no per-frame weight/deposit limits),
301	/// - otherwise computes concrete nested weight/deposit limits derived from the remaining
302	///   ethereum gas
303	///
304	/// The function ensures the nested frame's derived gas+resources remain within the parent's
305	/// remaining budget and returns `Err(Error::OutOfGas)` when the derived limits would exhaust
306	/// available resources.
307	pub fn new_nested_meter<T: Config, S: State>(
308		meter: &ResourceMeter<T, S>,
309		limit: &CallResources<T>,
310		eth_tx_info: &EthTxInfo<T>,
311	) -> Result<FrameMeter<T>, DispatchError> {
312		let self_consumed_weight = meter.weight.weight_consumed();
313		let self_consumed_deposit = meter.deposit.consumed();
314
315		let total_consumed_weight =
316			meter.total_consumed_weight_before.saturating_add(self_consumed_weight);
317		let total_consumed_deposit =
318			meter.total_consumed_deposit_before.saturating_add(&self_consumed_deposit);
319
320		let total_gas_consumption =
321			eth_tx_info.gas_consumption(&total_consumed_weight, &total_consumed_deposit);
322
323		let remaining_gas = meter.max_total_gas.saturating_sub(&total_gas_consumption);
324
325		let weight_left = {
326			let unbounded_weight_left = eth_tx_info
327				.weight_remaining(
328					&meter.max_total_gas,
329					&total_consumed_weight,
330					&total_consumed_deposit,
331				)
332				.ok_or(<Error<T>>::OutOfGas)?;
333
334			match meter.weight.weight_limit {
335				Some(weight_limit) => unbounded_weight_left.min(
336					weight_limit.checked_sub(&self_consumed_weight).ok_or(<Error<T>>::OutOfGas)?,
337				),
338				None => unbounded_weight_left,
339			}
340		};
341
342		let deposit_left = {
343			let Some(unbounded_deposit_left) = remaining_gas.to_adjusted_deposit_charge() else {
344				return Err(<Error<T>>::OutOfGas.into());
345			};
346
347			match meter.deposit.limit {
348				Some(deposit_limit) => unbounded_deposit_left.min(
349					self_consumed_deposit
350						.available(&deposit_limit)
351						.ok_or(<Error<T>>::StorageDepositLimitExhausted)?,
352				),
353				None => unbounded_deposit_left,
354			}
355		};
356
357		let (nested_gas_limit, nested_weight_limit, nested_deposit_limit, stipend) = {
358			match limit {
359				CallResources::NoLimits => (
360					remaining_gas,
361					if meter.weight.weight_limit.is_none() { None } else { Some(weight_left) },
362					if meter.deposit.limit.is_none() { None } else { Some(deposit_left) },
363					None,
364				),
365
366				CallResources::Ethereum { gas, add_stipend } => {
367					let gas_limit = SignedGas::from_ethereum_gas(*gas);
368
369					let (gas_limit, stipend) = if *add_stipend {
370						let weight_stipend = determine_call_stipend::<T>();
371						if weight_left.any_lt(weight_stipend) {
372							return Err(<Error<T>>::OutOfGas.into())
373						}
374
375						(
376							gas_limit.saturating_add(&SignedGas::<T>::from_weight_fee(
377								T::FeeInfo::weight_to_fee(&weight_stipend),
378							)),
379							Some(weight_stipend),
380						)
381					} else {
382						(gas_limit, None)
383					};
384
385					(
386						remaining_gas.min(&gas_limit),
387						if meter.weight.weight_limit.is_none() { None } else { Some(weight_left) },
388						if meter.deposit.limit.is_none() { None } else { Some(deposit_left) },
389						stipend,
390					)
391				},
392
393				CallResources::WeightDeposit { weight, deposit_limit } => {
394					let nested_weight_limit = weight_left.min(*weight);
395					let nested_deposit_limit = deposit_left.min(*deposit_limit);
396
397					let new_max_total_gas = eth_tx_info.gas_consumption(
398						&total_consumed_weight.saturating_add(nested_weight_limit),
399						&total_consumed_deposit
400							.saturating_add(&StorageDeposit::Charge(nested_deposit_limit)),
401					);
402
403					let gas_limit = new_max_total_gas.saturating_sub(&total_gas_consumption);
404
405					(
406						remaining_gas.min(&gas_limit),
407						Some(nested_weight_limit),
408						Some(nested_deposit_limit),
409						None,
410					)
411				},
412			}
413		};
414
415		let nested_max_total_gas = total_gas_consumption.saturating_add(&nested_gas_limit);
416
417		Ok(FrameMeter::<T> {
418			weight: WeightMeter::new(nested_weight_limit, stipend),
419			deposit: meter.deposit.nested(nested_deposit_limit),
420			max_total_gas: nested_max_total_gas,
421			total_consumed_weight_before: total_consumed_weight,
422			total_consumed_deposit_before: total_consumed_deposit,
423			transaction_limits: meter.transaction_limits.clone(),
424			_phantom: PhantomData,
425		})
426	}

429	pub fn gas_left<T: Config, S: State>(
430		meter: &ResourceMeter<T, S>,
431		eth_tx_info: &EthTxInfo<T>,
432	) -> Option<SignedGas<T>> {
433		let self_consumed_weight = meter.weight.weight_consumed();
434		let self_consumed_deposit = meter.deposit.consumed();
435
436		let total_consumed_weight =
437			meter.total_consumed_weight_before.saturating_add(self_consumed_weight);
438		let total_consumed_deposit =
439			meter.total_consumed_deposit_before.saturating_add(&self_consumed_deposit);

441		let total_gas_consumption =
442			eth_tx_info.gas_consumption(&total_consumed_weight, &total_consumed_deposit);
443
444		Some(meter.max_total_gas.saturating_sub(&total_gas_consumption))
445	}

447	/// Return the remaining weight available to a nested frame under Ethereum-style execution.
448	pub fn weight_left<T: Config, S: State>(
449		meter: &ResourceMeter<T, S>,
450		eth_tx_info: &EthTxInfo<T>,
451	) -> Option<Weight> {
452		let self_consumed_weight = meter.weight.weight_consumed();
453		let self_consumed_deposit = meter.deposit.consumed();
454
455		let total_consumed_weight =
456			meter.total_consumed_weight_before.saturating_add(self_consumed_weight);
457		let total_consumed_deposit =
458			meter.total_consumed_deposit_before.saturating_add(&self_consumed_deposit);
459
460		let weight_left = eth_tx_info.weight_remaining(
461			&meter.max_total_gas,
462			&total_consumed_weight,
463			&total_consumed_deposit,
464		)?;
465
466		Some(match meter.weight.weight_limit {
467			Some(weight_limit) => weight_left.min(weight_limit.checked_sub(&self_consumed_weight)?),
468			None => weight_left,
469		})
470	}

472	/// Return remaining deposit available to a nested frame under Ethereum-style execution.
473	pub fn deposit_left<T: Config, S: State>(
474		meter: &ResourceMeter<T, S>,
475		eth_tx_info: &EthTxInfo<T>,
476	) -> Option<BalanceOf<T>> {
477		let deposit_left = gas_left(meter, eth_tx_info)?.to_adjusted_deposit_charge()?;
478
479		Some(match meter.deposit.limit {
480			Some(deposit_limit) => {
481				let deposit_available = meter.deposit.consumed().available(&deposit_limit)?;
482				deposit_left.min(deposit_available)
483			},
484			None => deposit_left,
485		})
486	}

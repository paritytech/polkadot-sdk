31/// Implements the BALANCE instruction.
32///
33/// Gets the balance of the given account.
34pub fn balance<E: Ext>(interpreter: &mut Interpreter<E>) -> ControlFlow<Halt> {
35	interpreter.ext.charge_or_halt(RuntimeCosts::BalanceOf)?;
36	let ([], top) = interpreter.stack.popn_top()?;
37	*top = interpreter.ext.balance_of(&top.into_address());
38	ControlFlow::Continue(())
39}

41/// EIP-1884: Repricing for trie-size-dependent opcodes
42pub fn selfbalance<E: Ext>(interpreter: &mut Interpreter<E>) -> ControlFlow<Halt> {
43	interpreter.ext.charge_or_halt(RuntimeCosts::Balance)?;
44	let balance = interpreter.ext.balance();
45	interpreter.stack.push(balance)
46}

48/// Implements the EXTCODESIZE instruction.
49///
50/// Gets the size of an account's code.
51pub fn extcodesize<E: Ext>(interpreter: &mut Interpreter<E>) -> ControlFlow<Halt> {
52	let ([], top) = interpreter.stack.popn_top()?;

56	ControlFlow::Continue(())

59/// EIP-1052: EXTCODEHASH opcode
60pub fn extcodehash<E: Ext>(interpreter: &mut Interpreter<E>) -> ControlFlow<Halt> {

68/// Implements the EXTCODECOPY instruction.
69///
70/// Copies a portion of an account's code to memory.
71pub fn extcodecopy<E: Ext>(interpreter: &mut Interpreter<E>) -> ControlFlow<Halt> {
72	let [address, memory_offset, code_offset, len] = interpreter.stack.popn()?;
73	let len = as_usize_or_halt::<E::T>(len)?;
74	interpreter.ext.charge_or_halt(RuntimeCosts::ExtCodeCopy(len as u32))?;
75	if len == 0 {
76		return ControlFlow::Continue(());
77	}
78
79	let address = address.into_address();
80	let memory_offset = as_usize_or_halt::<E::T>(memory_offset)?;
81	let code_offset = as_usize_or_halt::<E::T>(code_offset)?;
82
83	interpreter.memory.resize(memory_offset, len)?;
84
85	let mut buf = interpreter.memory.slice_mut(memory_offset, len);
86	// Note: This can't panic because we resized memory to fit.
87	interpreter.ext.copy_code_slice(&mut buf, &address, code_offset);
88	ControlFlow::Continue(())
89}

91/// Implements the BLOCKHASH instruction.
92///
93/// Gets the hash of one of the 256 most recent complete blocks.
94pub fn blockhash<E: Ext>(interpreter: &mut Interpreter<E>) -> ControlFlow<Halt> {
95	interpreter.ext.charge_or_halt(RuntimeCosts::BlockHash)?;
96	let ([], number) = interpreter.stack.popn_top()?;
97
98	// blockhash should push zero if number is not within valid range.
99	if let Some(hash) = interpreter.ext.block_hash(*number) {
100		*number = U256::from_big_endian(&hash.0)
101	} else {
102		*number = U256::zero()
103	};
104	ControlFlow::Continue(())
105}

107/// Implements the SLOAD instruction.
108///
109/// Loads a word from storage.
110pub fn sload<E: Ext>(interpreter: &mut Interpreter<E>) -> ControlFlow<Halt> {
111	let ([], index) = interpreter.stack.popn_top()?;
112	// NB: SLOAD loads 32 bytes from storage (i.e. U256).
113	interpreter.ext.charge_or_halt(RuntimeCosts::GetStorage(32))?;
114	let key = Key::Fix(index.to_big_endian());
115	let value = interpreter.ext.get_storage(&key);
116
117	*index = if let Some(storage_value) = value {
118		// sload always reads a word
119		let Ok::<[u8; 32], _>(bytes) = storage_value.try_into() else {
120			log::debug!(target: crate::LOG_TARGET, "sload read invalid storage value length. Expected 32.");
121			return ControlFlow::Break(Error::<E::T>::ContractTrapped.into());
122		};
123		U256::from_big_endian(&bytes)
124	} else {
125		// the key was never written before
126		U256::zero()
127	};
128	ControlFlow::Continue(())
129}

131fn store_helper<'ext, E: Ext>(
132	interpreter: &mut Interpreter<'ext, E>,
133	cost_before: RuntimeCosts,
134	set_function: fn(&mut E, &Key, Option<Vec<u8>>, bool) -> Result<WriteOutcome, DispatchError>,
135	adjust_cost: fn(new_bytes: u32, old_bytes: u32) -> RuntimeCosts,
136) -> ControlFlow<Halt> {
137	if interpreter.ext.is_read_only() {
138		return ControlFlow::Break(Error::<E::T>::StateChangeDenied.into());
139	}
140
141	let [index, value] = interpreter.stack.popn()?;
142
143	// Charge gas before set_storage and later adjust it down to the true gas cost
144	let charged_amount = interpreter.ext.charge_or_halt(cost_before)?;
145	let key = Key::Fix(index.to_big_endian());
146	let take_old = false;
147	let Ok(write_outcome) =
148		set_function(interpreter.ext, &key, Some(value.to_big_endian().to_vec()), take_old)
149	else {
150		return ControlFlow::Break(Error::<E::T>::ContractTrapped.into());
151	};
152
153	interpreter
154		.ext
155		.gas_meter_mut()
156		.adjust_gas(charged_amount, adjust_cost(32, write_outcome.old_len()));
157
158	ControlFlow::Continue(())
159}

161/// Implements the SSTORE instruction.
162///
163/// Stores a word to storage.
164pub fn sstore<E: Ext>(interpreter: &mut Interpreter<E>) -> ControlFlow<Halt> {
165	let old_bytes = interpreter.ext.max_value_size();
166	store_helper(
167		interpreter,
168		RuntimeCosts::SetStorage { new_bytes: 32, old_bytes },
169		|ext, key, value, take_old| ext.set_storage(key, value, take_old),
170		|new_bytes, old_bytes| RuntimeCosts::SetStorage { new_bytes, old_bytes },
171	)
172}

174/// EIP-1153: Transient storage opcodes
175/// Store value to transient storage
176pub fn tstore<E: Ext>(interpreter: &mut Interpreter<E>) -> ControlFlow<Halt> {
177	let old_bytes = interpreter.ext.max_value_size();
178	store_helper(

181		|ext, key, value, take_old| ext.set_transient_storage(key, value, take_old),
182		|new_bytes, old_bytes| RuntimeCosts::SetTransientStorage { new_bytes, old_bytes },
183	)

186/// EIP-1153: Transient storage opcodes
187/// Load value from transient storage
188pub fn tload<E: Ext>(interpreter: &mut Interpreter<E>) -> ControlFlow<Halt> {
189	let ([], index) = interpreter.stack.popn_top()?;
190	interpreter.ext.charge_or_halt(RuntimeCosts::GetTransientStorage(32))?;
191
192	let key = Key::Fix(index.to_big_endian());
193	let bytes = interpreter.ext.get_transient_storage(&key);
194
195	*index = if let Some(storage_value) = bytes {
196		if storage_value.len() != 32 {
197			// tload always reads a word
198			log::debug!(target: crate::LOG_TARGET, "tload read invalid storage value length. Expected 32.");
199			return ControlFlow::Break(Error::<E::T>::ContractTrapped.into());
200		}
201
202		let Ok::<[u8; 32], _>(bytes) = storage_value.try_into() else {
203			return ControlFlow::Break(Error::<E::T>::ContractTrapped.into());
204		};
205		U256::from_big_endian(&bytes)
206	} else {
207		// the key was never written before
208		U256::zero()
209	};
210	ControlFlow::Continue(())
211}

213/// Implements the LOG0-LOG4 instructions.
214///
215/// Appends log record with N topics.
216pub fn log<'ext, const N: usize, E: Ext>(
217	interpreter: &mut Interpreter<'ext, E>,
218) -> ControlFlow<Halt> {
219	if interpreter.ext.is_read_only() {
220		return ControlFlow::Break(Error::<E::T>::StateChangeDenied.into());
221	}
222
223	let [offset, len] = interpreter.stack.popn()?;
224	let len = as_usize_or_halt::<E::T>(len)?;
225	if len as u32 > interpreter.ext.max_value_size() {
226		return ControlFlow::Break(Error::<E::T>::OutOfGas.into());
227	}
228
229	let cost = RuntimeCosts::DepositEvent { num_topic: N as u32, len: len as u32 };
230	interpreter.ext.charge_or_halt(cost)?;
231
232	let data = if len == 0 {
233		Vec::new()
234	} else {
235		let offset = as_usize_or_halt::<E::T>(offset)?;
236		interpreter.memory.resize(offset, len)?;
237		interpreter.memory.slice(offset..offset + len).to_vec()
238	};
239	if interpreter.stack.len() < N {
240		return ControlFlow::Break(Error::<E::T>::StackUnderflow.into());
241	}
242	let topics = interpreter.stack.popn::<N>()?;
243	let topics = topics.into_iter().map(|v| sp_core::H256::from(v.to_big_endian())).collect();
244
245	interpreter.ext.deposit_event(topics, data.to_vec());
246	ControlFlow::Continue(())
247}

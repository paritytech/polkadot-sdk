33/// The Keccak-256 hash of the empty string `""`.
34pub const KECCAK_EMPTY: [u8; 32] =
35	alloy_core::hex!("c5d2460186f7233c927e7db2dcc703c0e500b653ca82273b7bfad8045d85a470");
36
37/// Implements the KECCAK256 instruction.
38///
39/// Computes Keccak-256 hash of memory data.
40pub fn keccak256<E: Ext>(interpreter: &mut Interpreter<E>) -> ControlFlow<Halt> {
41	let ([offset], top) = interpreter.stack.popn_top()?;
42	let len = as_usize_or_halt::<E::T>(*top)?;
43	interpreter
44		.ext
45		.gas_meter_mut()
46		.charge_or_halt(RuntimeCosts::HashKeccak256(len as u32))?;
47
48	let hash = if len == 0 {
49		H256::from(KECCAK_EMPTY)
50	} else {
51		let from = as_usize_or_halt::<E::T>(offset)?;
52		interpreter.memory.resize(from, len)?;
53		H256::from(keccak_256(interpreter.memory.slice_len(from, len)))
54	};
55	*top = U256::from_big_endian(hash.as_ref());
56	ControlFlow::Continue(())
57}

59/// Implements the ADDRESS instruction.
60///
61/// Pushes the current contract's address onto the stack.
62pub fn address<E: Ext>(interpreter: &mut Interpreter<E>) -> ControlFlow<Halt> {
63	interpreter.ext.charge_or_halt(RuntimeCosts::Address)?;
64	let address = interpreter.ext.address();
65	interpreter.stack.push(address)
66}

68/// Implements the CALLER instruction.
69///
70/// Pushes the caller's address onto the stack.
71pub fn caller<E: Ext>(interpreter: &mut Interpreter<E>) -> ControlFlow<Halt> {
72	interpreter.ext.charge_or_halt(RuntimeCosts::Caller)?;
73	match interpreter.ext.caller().account_id() {
74		Ok(account_id) => {
75			let address = <E::T as Config>::AddressMapper::to_address(account_id);
76			interpreter.stack.push(address)
77		},
78		Err(_) => ControlFlow::Break(Error::<E::T>::ContractTrapped.into()),
79	}
80}

90/// Implements the CODECOPY instruction.
91///
92/// Copies running contract's bytecode to memory.
93pub fn codecopy<E: Ext>(interpreter: &mut Interpreter<E>) -> ControlFlow<Halt> {
94	let [memory_offset, code_offset, len] = interpreter.stack.popn()?;
95	let len = as_usize_or_halt::<E::T>(len)?;
96	let Some(memory_offset) = memory_resize(interpreter, memory_offset, len)? else {
97		return ControlFlow::Continue(())
98	};
99	let code_offset = as_usize_saturated(code_offset);
100
101	// Note: This can't panic because we resized memory.
102	interpreter.memory.set_data(
103		memory_offset,
104		code_offset,
105		len,
106		interpreter.bytecode.bytecode_slice(),
107	);
108	ControlFlow::Continue(())
109}

111/// Implements the CALLDATALOAD instruction.
112///
113/// Loads 32 bytes of input data from the specified offset.
114pub fn calldataload<E: Ext>(interpreter: &mut Interpreter<E>) -> ControlFlow<Halt> {
115	interpreter.ext.charge_or_halt(EVMGas(VERYLOW))?;
116	let ([], offset_ptr) = interpreter.stack.popn_top()?;
117	let mut word = [0u8; 32];
118	let offset = as_usize_saturated(*offset_ptr);
119	let input = &interpreter.input;
120	let input_len = input.len();
121	if offset < input_len {
122		let count = 32.min(input_len - offset);
123		word[..count].copy_from_slice(&input[offset..offset + count]);
124	}
125	*offset_ptr = U256::from_big_endian(&word);
126	ControlFlow::Continue(())
127}

146/// Implements the CALLDATACOPY instruction.
147///
148/// Copies input data to memory.
149pub fn calldatacopy<E: Ext>(interpreter: &mut Interpreter<E>) -> ControlFlow<Halt> {
150	let [memory_offset, data_offset, len] = interpreter.stack.popn()?;
151	let len = as_usize_or_halt::<E::T>(len)?;
152
153	let Some(memory_offset) = memory_resize(interpreter, memory_offset, len)? else {
154		return ControlFlow::Continue(());
155	};
156
157	let data_offset = as_usize_saturated(data_offset);
158
159	// Note: This can't panic because we resized memory.
160	interpreter.memory.set_data(memory_offset, data_offset, len, &interpreter.input);
161	ControlFlow::Continue(())
162}

164/// EIP-211: New opcodes: RETURNDATASIZE and RETURNDATACOPY
165pub fn returndatasize<E: Ext>(interpreter: &mut Interpreter<E>) -> ControlFlow<Halt> {
166	interpreter.ext.charge_or_halt(EVMGas(BASE))?;
167	let return_data_len = interpreter.ext.last_frame_output().data.len();
168	interpreter.stack.push(U256::from(return_data_len))
169}

171/// EIP-211: New opcodes: RETURNDATASIZE and RETURNDATACOPY
172pub fn returndatacopy<E: Ext>(interpreter: &mut Interpreter<E>) -> ControlFlow<Halt> {
173	let [memory_offset, offset, len] = interpreter.stack.popn()?;
174
175	let len = as_usize_or_halt::<E::T>(len)?;
176	let data_offset = as_usize_saturated(offset);
177
178	// Old legacy behavior is to panic if data_end is out of scope of return buffer.
179	let data_end = data_offset.saturating_add(len);
180	if data_end > interpreter.ext.last_frame_output().data.len() {
181		return ControlFlow::Break(Error::<E::T>::OutOfBounds.into());
182	}
183
184	let Some(memory_offset) = memory_resize(interpreter, memory_offset, len)? else {
185		return ControlFlow::Continue(())
186	};
187
188	// Note: This can't panic because we resized memory.
189	interpreter.memory.set_data(
190		memory_offset,
191		data_offset,
192		len,
193		&interpreter.ext.last_frame_output().data,
194	);
195	ControlFlow::Continue(())
196}

198/// Implements the GAS instruction.
199///
200/// Pushes the amount of remaining gas onto the stack.
201pub fn gas<E: Ext>(interpreter: &mut Interpreter<E>) -> ControlFlow<Halt> {
202	interpreter.ext.charge_or_halt(RuntimeCosts::RefTimeLeft)?;
203	let gas = interpreter.ext.gas_left();
204	interpreter.stack.push(U256::from(gas))
205}

207/// Common logic for copying data from a source buffer to the EVM's memory.
208///
209/// Handles memory expansion and gas calculation for data copy operations.
210pub fn memory_resize<'a, E: Ext>(
211	interpreter: &mut Interpreter<'a, E>,
212	memory_offset: U256,
213	len: usize,
214) -> ControlFlow<Halt, Option<usize>> {
215	if len == 0 {
216		return ControlFlow::Continue(None)
217	}
218
219	interpreter.ext.charge_or_halt(RuntimeCosts::CopyToContract(len as u32))?;
220	let memory_offset = as_usize_or_halt::<E::T>(memory_offset)?;
221	interpreter.memory.resize(memory_offset, len)?;
222	ControlFlow::Continue(Some(memory_offset))

29/// EIP-1344: ChainID opcode
30pub fn chainid<E: Ext>(interpreter: &mut Interpreter<E>) -> ControlFlow<Halt> {
31	interpreter.ext.charge_or_halt(EVMGas(BASE))?;
32	interpreter.stack.push(interpreter.ext.chain_id())?;
33	ControlFlow::Continue(())
34}
35
36/// Implements the COINBASE instruction.
37///
38/// Pushes the current block's beneficiary address onto the stack.
39pub fn coinbase<E: Ext>(interpreter: &mut Interpreter<E>) -> ControlFlow<Halt> {
40	interpreter.ext.charge_or_halt(RuntimeCosts::BlockAuthor)?;
41	let coinbase = interpreter.ext.block_author();
42	interpreter.stack.push(coinbase)?;
43	ControlFlow::Continue(())
44}

66/// Implements the DIFFICULTY/PREVRANDAO instruction.
67///
68/// Pushes the block difficulty (pre-merge) or prevrandao (post-merge) onto the stack.
69pub fn difficulty<E: Ext>(interpreter: &mut Interpreter<E>) -> ControlFlow<Halt> {
70	interpreter.ext.charge_or_halt(EVMGas(BASE))?;
71	interpreter.stack.push(U256::from(DIFFICULTY))?;
72	ControlFlow::Continue(())
73}

98/// Information for managing an account and its sub trie abstraction.
99/// This is the required info to cache for an account.
100#[derive(Encode, Decode, CloneNoBound, PartialEq, Eq, DebugNoBound, TypeInfo, MaxEncodedLen)]
101#[scale_info(skip_type_params(T))]
102pub struct ContractInfo<T: Config> {
103	/// Unique ID for the subtree encoded as a bytes vector.
104	pub trie_id: TrieId,
105	/// The code associated with a given account.
106	pub code_hash: sp_core::H256,
107	/// How many bytes of storage are accumulated in this contract's child trie.
108	storage_bytes: u32,
109	/// How many items of storage are accumulated in this contract's child trie.
110	storage_items: u32,
111	/// This records to how much deposit the accumulated `storage_bytes` amount to.
112	pub storage_byte_deposit: BalanceOf<T>,
113	/// This records to how much deposit the accumulated `storage_items` amount to.
114	storage_item_deposit: BalanceOf<T>,
115	/// This records how much deposit is put down in order to pay for the contract itself.
116	///
117	/// We need to store this information separately so it is not used when calculating any refunds
118	/// since the base deposit can only ever be refunded on contract termination.
119	storage_base_deposit: BalanceOf<T>,
120	/// The size of the immutable data of this contract.
121	immutable_data_len: u32,
122}

153impl<T: Config> AccountInfo<T> {
154	/// Returns true if the account is a contract.
155	pub fn is_contract(address: &H160) -> bool {
156		let Some(info) = <AccountInfoOf<T>>::get(address) else { return false };
157		matches!(info.account_type, AccountType::Contract(_))
158	}
159
160	/// Returns the balance of the account at the given address.
161	pub fn balance(account: AccountIdOrAddress<T>) -> BalanceWithDust<BalanceOf<T>> {
162		use frame_support::traits::{
163			fungible::Inspect,
164			tokens::{Fortitude::Polite, Preservation::Preserve},
165		};
166
167		let value = T::Currency::reducible_balance(&account.account_id(), Preserve, Polite);
168		let dust = <AccountInfoOf<T>>::get(account.address()).map(|a| a.dust).unwrap_or_default();
169		BalanceWithDust::new_unchecked::<T>(value, dust)
170	}
171
172	/// Loads the contract information for a given address.
173	pub fn load_contract(address: &H160) -> Option<ContractInfo<T>> {
174		let Some(info) = <AccountInfoOf<T>>::get(address) else { return None };
175		let AccountType::Contract(contract_info) = info.account_type else { return None };
176		Some(contract_info)
177	}
178
179	/// Insert a contract, existing dust if any will be unchanged.
180	pub fn insert_contract(address: &H160, contract: ContractInfo<T>) {
181		AccountInfoOf::<T>::mutate(address, |account| {
182			if let Some(account) = account {
183				account.account_type = contract.clone().into();
184			} else {
185				*account = Some(AccountInfo { account_type: contract.clone().into(), dust: 0 });
186			}
187		});
188	}
189}

191impl<T: Config> ContractInfo<T> {
192	/// Constructs a new contract info **without** writing it to storage.
193	///
194	/// This returns an `Err` if an contract with the supplied `account` already exists
195	/// in storage.
196	pub fn new(
197		address: &H160,
198		nonce: T::Nonce,
199		code_hash: sp_core::H256,
200	) -> Result<Self, DispatchError> {
201		if <AccountInfo<T>>::is_contract(address) {
202			return Err(Error::<T>::DuplicateContract.into());
203		}
204
205		let trie_id = {
206			let buf = ("bcontract_trie_v1", address, nonce).using_encoded(T::Hashing::hash);
207			buf.as_ref()
208				.to_vec()
209				.try_into()
210				.expect("Runtime uses a reasonable hash size. Hence sizeof(T::Hash) <= 128; qed")
211		};
212
213		let contract = Self {
214			trie_id,
215			code_hash,
216			storage_bytes: 0,
217			storage_items: 0,
218			storage_byte_deposit: Zero::zero(),
219			storage_item_deposit: Zero::zero(),
220			storage_base_deposit: Zero::zero(),
221			immutable_data_len: 0,
222		};
223
224		Ok(contract)
225	}

249	/// The read is performed from the `trie_id` only. The `address` is not necessary. If the

268	/// Update a storage entry into a contract's kv storage.
269	///
270	/// If the `new_value` is `None` then the kv pair is removed. If `take` is true
271	/// a [`WriteOutcome::Taken`] is returned instead of a [`WriteOutcome::Overwritten`].
272	///
273	/// This function also records how much storage was created or removed if a `storage_meter`
274	/// is supplied. It should only be absent for testing or benchmarking code.
275	pub fn write(
276		&self,
277		key: &Key,
278		new_value: Option<Vec<u8>>,
279		storage_meter: Option<&mut meter::NestedMeter<T>>,
280		take: bool,
281	) -> Result<WriteOutcome, DispatchError> {
282		log::trace!(target: crate::LOG_TARGET, "contract storage: writing value {:?} for key {:x?}", new_value, key);
283		let hashed_key = key.hash();
284		if_tracing(|t| {
285			let old = child::get_raw(&self.child_trie_info(), hashed_key.as_slice());
286			t.storage_write(key, old, new_value.as_deref());
287		});
288
289		self.write_raw(&hashed_key, new_value.as_deref(), storage_meter, take)
290	}

292	/// Update a storage entry into a contract's kv storage.
293	/// Function used in benchmarks, which can simulate prefix collision in keys.
294	#[cfg(feature = "runtime-benchmarks")]
295	pub fn bench_write_raw(
296		&self,
297		key: &[u8],
298		new_value: Option<Vec<u8>>,
299		take: bool,
300	) -> Result<WriteOutcome, DispatchError> {
301		self.write_raw(key, new_value.as_deref(), None, take)
302	}

304	fn write_raw(
305		&self,
306		key: &[u8],
307		new_value: Option<&[u8]>,
308		storage_meter: Option<&mut meter::NestedMeter<T>>,
309		take: bool,
310	) -> Result<WriteOutcome, DispatchError> {
311		let child_trie_info = &self.child_trie_info();
312		let (old_len, old_value) = if take {
313			let val = child::get_raw(child_trie_info, key);
314			(val.as_ref().map(|v| v.len() as u32), val)
315		} else {
316			(child::len(child_trie_info, key), None)
317		};
318
319		if let Some(storage_meter) = storage_meter {
320			let mut diff = meter::Diff::default();
321			let key_len = key.len() as u32;
322			match (old_len, new_value.as_ref().map(|v| v.len() as u32)) {
323				(Some(old_len), Some(new_len)) =>
324					if new_len > old_len {
325						diff.bytes_added = new_len - old_len;
326					} else {
327						diff.bytes_removed = old_len - new_len;
328					},
329				(None, Some(new_len)) => {
330					diff.bytes_added = new_len.saturating_add(key_len);
331					diff.items_added = 1;
332				},
333				(Some(old_len), None) => {
334					diff.bytes_removed = old_len.saturating_add(key_len);
335					diff.items_removed = 1;
336				},
337				(None, None) => (),
338			}
339			storage_meter.charge(&diff);
340		}
341
342		match &new_value {
343			Some(new_value) => child::put_raw(child_trie_info, key, new_value),
344			None => child::kill(child_trie_info, key),
345		}
346
347		Ok(match (old_len, old_value) {
348			(None, _) => WriteOutcome::New,
349			(Some(old_len), None) => WriteOutcome::Overwritten(old_len),
350			(Some(_), Some(old_value)) => WriteOutcome::Taken(old_value),
351		})
352	}

354	/// Sets and returns the contract base deposit.
355	///
356	/// The base deposit is updated when the `code_hash` of the contract changes, as it depends on
357	/// the deposit paid to upload the contract's code. It also depends on the size of immutable
358	/// storage which is also changed when the code hash of a contract is changed.
359	pub fn update_base_deposit(&mut self, code_deposit: BalanceOf<T>) -> BalanceOf<T> {
360		let contract_deposit = Diff {
361			bytes_added: (self.encoded_size() as u32).saturating_add(self.immutable_data_len),
362			items_added: if self.immutable_data_len == 0 { 1 } else { 2 },
363			..Default::default()
364		}
365		.update_contract::<T>(None)
366		.charge_or_zero();
367
368		// Instantiating the contract prevents its code to be deleted, therefore the base deposit
369		// includes a fraction (`T::CodeHashLockupDepositPercent`) of the original storage deposit
370		// to prevent abuse.
371		let code_deposit = T::CodeHashLockupDepositPercent::get().mul_ceil(code_deposit);
372
373		let deposit = contract_deposit.saturating_add(code_deposit);
374		self.storage_base_deposit = deposit;
375		deposit
376	}

385	/// Calculates the weight that is necessary to remove one key from the trie and how many
386	/// of those keys can be deleted from the deletion queue given the supplied weight limit.
387	pub fn deletion_budget(meter: &WeightMeter) -> (Weight, u32) {
388		let base_weight = T::WeightInfo::on_process_deletion_queue_batch();
389		let weight_per_key = T::WeightInfo::on_initialize_per_trie_key(1) -
390			T::WeightInfo::on_initialize_per_trie_key(0);
391
392		// `weight_per_key` being zero makes no sense and would constitute a failure to
393		// benchmark properly. We opt for not removing any keys at all in this case.
394		let key_budget = meter
395			.limit()
396			.saturating_sub(base_weight)
397			.checked_div_per_component(&weight_per_key)
398			.unwrap_or(0) as u32;
399
400		(weight_per_key, key_budget)
401	}
402
403	/// Delete as many items from the deletion queue possible within the supplied weight limit.
404	pub fn process_deletion_queue_batch(meter: &mut WeightMeter) {
405		if meter.try_consume(T::WeightInfo::on_process_deletion_queue_batch()).is_err() {
406			return
407		};
408
409		let mut queue = <DeletionQueueManager<T>>::load();
410		if queue.is_empty() {
411			return;
412		}
413
414		let (weight_per_key, budget) = Self::deletion_budget(&meter);
415		let mut remaining_key_budget = budget;
416		while remaining_key_budget > 0 {
417			let Some(entry) = queue.next() else { break };
418
419			#[allow(deprecated)]
420			let outcome = child::kill_storage(
421				&ChildInfo::new_default(&entry.trie_id),
422				Some(remaining_key_budget),
423			);
424
425			match outcome {
426				// This happens when our budget wasn't large enough to remove all keys.
427				KillStorageResult::SomeRemaining(keys_removed) => {
428					remaining_key_budget.saturating_reduce(keys_removed);
429					break
430				},
431				KillStorageResult::AllRemoved(keys_removed) => {
432					entry.remove();
433					// charge at least one key even if none were removed.
434					remaining_key_budget = remaining_key_budget.saturating_sub(keys_removed.max(1));
435				},
436			};
437		}
438
439		meter.consume(weight_per_key.saturating_mul(u64::from(budget - remaining_key_budget)))
440	}

458/// Information about what happened to the pre-existing value when calling [`ContractInfo::write`].
459#[derive(Clone, Eq, PartialEq, Encode, Decode, RuntimeDebug, TypeInfo)]
460pub enum WriteOutcome {
461	/// No value existed at the specified key.
462	New,
463	/// A value of the returned length was overwritten.
464	Overwritten(u32),
465	/// The returned value was taken out of storage before being overwritten.
466	///
467	/// This is only returned when specifically requested because it causes additional work
468	/// depending on the size of the pre-existing value. When not requested [`Self::Overwritten`]
469	/// is returned instead.
470	Taken(Vec<u8>),
471}
472
473impl WriteOutcome {
474	/// Extracts the size of the overwritten value or `0` if there
475	/// was no value in storage.
476	pub fn old_len(&self) -> u32 {
477		match self {
478			Self::New => 0,
479			Self::Overwritten(len) => *len,
480			Self::Taken(value) => value.len() as u32,
481		}
482	}
483
484	/// Extracts the size of the overwritten value or `SENTINEL` if there
485	/// was no value in storage.
486	///
487	/// # Note
488	///
489	/// We cannot use `0` as sentinel value because there could be a zero sized
490	/// storage entry which is different from a non existing one.
491	pub fn old_len_with_sentinel(&self) -> u32 {
492		match self {
493			Self::New => SENTINEL,
494			Self::Overwritten(len) => *len,
495			Self::Taken(value) => value.len() as u32,
496		}
497	}
498}

500/// Manage the removal of contracts storage that are marked for deletion.
501///
502/// When a contract is deleted by calling `seal_terminate` it becomes inaccessible
503/// immediately, but the deletion of the storage items it has accumulated is performed
504/// later by pulling the contract from the queue in the `on_idle` hook.
505#[derive(Encode, Decode, TypeInfo, MaxEncodedLen, DefaultNoBound, Clone)]
506#[scale_info(skip_type_params(T))]
507pub struct DeletionQueueManager<T: Config> {
508	/// Counter used as a key for inserting a new deleted contract in the queue.
509	/// The counter is incremented after each insertion.
510	insert_counter: u32,
511	/// The index used to read the next element to be deleted in the queue.
512	/// The counter is incremented after each deletion.
513	delete_counter: u32,
514
515	_phantom: PhantomData<T>,
516}
517
518/// View on a contract that is marked for deletion.
519struct DeletionQueueEntry<'a, T: Config> {
520	/// the trie id of the contract to delete.
521	trie_id: TrieId,
522
523	/// A mutable reference on the queue so that the contract can be removed, and none can be added
524	/// or read in the meantime.
525	queue: &'a mut DeletionQueueManager<T>,
526}
527
528impl<'a, T: Config> DeletionQueueEntry<'a, T> {
529	/// Remove the contract from the deletion queue.
530	fn remove(self) {
531		<DeletionQueue<T>>::remove(self.queue.delete_counter);
532		self.queue.delete_counter = self.queue.delete_counter.wrapping_add(1);
533		<DeletionQueueCounter<T>>::set(self.queue.clone());
534	}
535}
536
537impl<T: Config> DeletionQueueManager<T> {
538	/// Load the `DeletionQueueCounter`, so we can perform read or write operations on the
539	/// DeletionQueue storage.
540	fn load() -> Self {
541		<DeletionQueueCounter<T>>::get()
542	}
543
544	/// Returns `true` if the queue contains no elements.
545	fn is_empty(&self) -> bool {
546		self.insert_counter.wrapping_sub(self.delete_counter) == 0
547	}
548
549	/// Insert a contract in the deletion queue.
550	fn insert(&mut self, trie_id: TrieId) {
551		<DeletionQueue<T>>::insert(self.insert_counter, trie_id);
552		self.insert_counter = self.insert_counter.wrapping_add(1);
553		<DeletionQueueCounter<T>>::set(self.clone());
554	}
555
556	/// Fetch the next contract to be deleted.
557	///
558	/// Note:
559	/// we use the delete counter to get the next value to read from the queue and thus don't pay
560	/// the cost of an extra call to `sp_io::storage::next_key` to lookup the next entry in the map
561	fn next(&mut self) -> Option<DeletionQueueEntry<'_, T>> {
562		if self.is_empty() {
563			return None
564		}
565
566		let entry = <DeletionQueue<T>>::get(self.delete_counter);
567		entry.map(|trie_id| DeletionQueueEntry { trie_id, queue: self })
568	}
569}

50/// Result of an Ethereum context call execution.
51pub(crate) struct EthereumCallResult {
52	/// Receipt gas information.
53	pub receipt_gas_info: ReceiptGasInfo,
54	/// The dispatch result with post-dispatch information.
55	pub result: DispatchResultWithPostInfo,
56}
57
58impl EthereumCallResult {
59	/// Create a new `EthereumCallResult` from contract execution details.
60	///
61	/// # Parameters
62	///
63	/// - `signer`: The signer of the transaction
64	/// - `output`: The execution result
65	/// - `gas_consumed`: The weight consumed during execution
66	/// - `base_call_weight`: The base call weight
67	/// - `encoded_len`: The length of the encoded transaction in bytes
68	/// - `info`: Dispatch information used for fee computation
69	/// - `effective_gas_price`: The EVM gas price
70	pub(crate) fn new<T: Config>(
71		signer: AccountIdOf<T>,
72		mut output: ContractResult<ExecReturnValue, BalanceOf<T>>,
73		base_call_weight: Weight,
74		encoded_len: u32,
75		info: &DispatchInfo,
76		effective_gas_price: U256,
77	) -> Self {
78		let effective_gas_price = effective_gas_price.max(Pallet::<T>::evm_base_fee());
79
80		if let Ok(retval) = &output.result {
81			if retval.did_revert() {
82				output.result = Err(<Error<T>>::ContractReverted.into());
83			}
84		}
85
86		// Refund pre-charged revert event weight if the call succeeds.
87		if output.result.is_ok() {
88			output
89				.weight_consumed
90				.saturating_reduce(T::WeightInfo::deposit_eth_extrinsic_revert_event())
91		}
92
93		let result = dispatch_result(output.result, output.weight_consumed, base_call_weight);
94		let native_fee = T::FeeInfo::compute_actual_fee(encoded_len, &info, &result);
95		let result = T::FeeInfo::ensure_not_overdrawn(native_fee, result);
96
97		let fee = Pallet::<T>::convert_native_to_evm(match output.storage_deposit {
98			StorageDeposit::Refund(refund) => native_fee.saturating_sub(refund),
99			StorageDeposit::Charge(amount) => native_fee.saturating_add(amount),
100		});
101
102		let (mut gas_used, rest) = fee.div_mod(effective_gas_price);
103		if !rest.is_zero() {
104			gas_used = gas_used.saturating_add(1_u32.into());
105		}
106
107		let tx_cost = gas_used.saturating_mul(effective_gas_price);
108		if tx_cost > fee {
109			let round_up_fee = BalanceWithDust::<BalanceOf<T>>::from_value::<T>(tx_cost - fee)
110				.expect("value fits into BalanceOf<T>; qed");
111			log::debug!(target: LOG_TARGET, "Collecting round_up fee from {signer:?}: {round_up_fee:?}");
112			let _ = burn_with_dust::<T>(&signer, round_up_fee)
113					.inspect_err(|e| log::debug!(target: LOG_TARGET, "Failed to collect round up fee {round_up_fee:?} from {signer:?}: {e:?}"));
114		}
115
116		Self { receipt_gas_info: ReceiptGasInfo { gas_used, effective_gas_price }, result }
117	}
118}

120/// Capture the Ethereum log for the current transaction.
121///
122/// This method does nothing if called from outside of the ethereum context.
123pub fn capture_ethereum_log(contract: &H160, data: &[u8], topics: &[H256]) {
124	receipt::with(|receipt| {
125		receipt.add_log(contract, data, topics);
126	});
127}

129/// Get the receipt details of the current transaction.
130///
131/// This method returns `None` if and only if the function is called
132/// from outside of the ethereum context.
133pub fn get_receipt_details() -> Option<(Vec<u8>, LogsBloom)> {
134	receipt::with(|receipt| {
135		let encoding = core::mem::take(&mut receipt.encoding);
136		let bloom = core::mem::take(&mut receipt.bloom);
137		(encoding, bloom)
138	})
139}

147/// Execute the Ethereum call, and write the block storage transaction details.
148///
149/// # Parameters
150/// - transaction_encoded: The RLP encoded transaction bytes.
151/// - call: A closure that executes the transaction logic and returns an `EthereumCallResult`.
152pub fn with_ethereum_context<T: Config>(
153	transaction_encoded: Vec<u8>,
154	call: impl FnOnce() -> EthereumCallResult,
155) -> DispatchResultWithPostInfo {
156	receipt::using(&mut AccumulateReceipt::new(), || {
157		let (err, receipt_gas_info, post_info) =
158			with_transaction(|| -> TransactionOutcome<Result<_, DispatchError>> {
159				let EthereumCallResult { receipt_gas_info, result } = call();
160				match result {
161					Ok(post_info) =>
162						TransactionOutcome::Commit(Ok((None, receipt_gas_info, post_info))),
163					Err(err) => TransactionOutcome::Rollback(Ok((
164						Some(err.error),
165						receipt_gas_info,
166						err.post_info,
167					))),
168				}
169			})?;
170
171		if let Some(dispatch_error) = err {
172			deposit_eth_extrinsic_revert_event::<T>(dispatch_error);
173			crate::block_storage::process_transaction::<T>(
174				transaction_encoded,
175				false,
176				receipt_gas_info,
177			);
178			Ok(post_info)
179		} else {
180			// deposit a dummy event in benchmark mode
181			#[cfg(feature = "runtime-benchmarks")]
182			deposit_eth_extrinsic_revert_event::<T>(crate::Error::<T>::BenchmarkingError.into());
183
184			crate::block_storage::process_transaction::<T>(
185				transaction_encoded,
186				true,
187				receipt_gas_info,
188			);
189			Ok(post_info)
190		}
191	})
192}

194fn deposit_eth_extrinsic_revert_event<T: Config>(dispatch_error: DispatchError) {
195	Pallet::<T>::deposit_event(Event::<T>::EthExtrinsicRevert { dispatch_error });

198/// Clear the storage used to capture the block hash related data.
199pub fn on_initialize<T: Config>() {
200	ReceiptInfoData::<T>::kill();
201	EthereumBlock::<T>::kill();
202}

204/// Build the ethereum block and store it into the pallet storage.
205pub fn on_finalize_build_eth_block<T: Config>(block_number: BlockNumberFor<T>) {
206	let block_builder_ir = EthBlockBuilderIR::<T>::get();
207	EthBlockBuilderIR::<T>::kill();
208
209	let (block, receipt_data) =
210		EthereumBlockBuilder::<T>::from_ir(block_builder_ir).build_block(block_number);
211
212	// Put the block hash into storage.
213	BlockHash::<T>::insert(block_number, block.hash);
214
215	// Prune older block hashes.
216	let block_hash_count = BLOCK_HASH_COUNT;
217	let to_remove = block_number.saturating_sub(block_hash_count.into()).saturating_sub(One::one());
218	if !Zero::is_zero(&to_remove) {
219		<BlockHash<T>>::remove(to_remove);
220	}
221	// Store the ETH block into the last block.
222	EthereumBlock::<T>::put(block);
223	// Store the receipt info data for offchain reconstruction.
224	ReceiptInfoData::<T>::put(receipt_data);
225}

227/// Process a transaction payload with extra details.
228/// This stores the RLP encoded transaction and receipt details into storage.
229///
230/// The data is used during the `on_finalize` hook to reconstruct the ETH block.
231pub fn process_transaction<T: Config>(
232	transaction_encoded: Vec<u8>,
233	success: bool,
234	receipt_gas_info: ReceiptGasInfo,
235) {
236	// Method returns `None` only when called from outside of the ethereum context.
237	// This is not the case here, since this is called from within the
238	// ethereum context.
239	let (encoded_logs, bloom) = get_receipt_details().unwrap_or_default();
240
241	let block_builder_ir = EthBlockBuilderIR::<T>::get();
242	let mut block_builder = EthereumBlockBuilder::<T>::from_ir(block_builder_ir);
243
244	block_builder.process_transaction(
245		transaction_encoded,
246		success,
247		receipt_gas_info,
248		encoded_logs,
249		bloom,
250	);
251
252	EthBlockBuilderIR::<T>::put(block_builder.to_ir());
253}

255// The `EthereumBlockBuilder` builds the Ethereum-compatible block by maintaining
256// two incremental hash builders. Each builder accumulates entries until the trie
257// is finalized:
258//  1. `transactions_root` - builds the Merkle root of transaction payloads
259//  2. `receipts_root` - builds the Merkle root of transaction receipts (event logs)
260//
261// The `EthereumBlockBuilder` is serialized and deserialized to and from storage
262// on every transaction via the `EthereumBlockBuilderIR` object. This is needed until
263// the runtime exposes a better API to preserve the state between transactions (ie,
264// the global `environment!` is wiped because each transaction will instantiate a new
265// WASM instance).
266//
267// For this reason, we need to account for the memory used by the `EthereumBlockBuilder`
268// and for the pallet storage consumed by the `EthereumBlockBuilderIR`.
269//
270// ## Memory Usage Analysis
271//
272// The incremental hash builder accumulates entries until the trie is finalized.
273// The last added entry value is kept in memory until it can be hashed.
274// The keys are always ordered and the hashing happens when the next entry is added to
275// the trie. The common prefix of the current and previous keys forms the path into the
276// trie, and together with the value of the previous entry, a hash of 32 bytes is
277// computed.
278//
279// For this reason, the memory usage of the incremental hash builder is no greater
280// than two entries of maximum size, plus some marginal book-keeping overhead
281// (ignored to simplify calculations).
282//
283// `IncrementalHashBuilder = 2 * maximum size of the entry`
284//
285// Additionally, the block builder caches the first entry for each incremental hash.
286// The entry is loaded from storage into RAM when either:
287// - The block is finalized, OR
288// - After 127 transactions.
289// Therefore, an additional entry of maximum size is needed in memory.
290//
291// That gives us 3 items of maximum size per each hash builder.
292//
293// `EthereumBlockBuilder = 3 * (max size of transactions + max size of receipts)`
294// The maximum size of a transaction is limited by
295// `limits::MAX_TRANSACTION_PAYLOAD_SIZE`, while the maximum size of a receipt is
296// limited by `limits::EVENT_BYTES`.
297//
298// Similarly, this is the amount of pallet storage consumed by the
299// `EthereumBlockBuilderIR` object, plus a marginal book-keeping overhead.
300pub fn block_builder_bytes_usage(max_events_size: u32) -> u32 {
301	// A block builder requires 3 times the maximum size of the entry.
302	const MEMORY_COEFFICIENT: u32 = 3;
303
304	// Because events are not capped, and the builder cannot exceed the
305	// number of bytes received, the actual memory usage for receipts is:
306	// `receipts_hash_builder = min(events_per_tx * 3, max_events_size)`
307	// where `max_events_size` can be consumed by a single transaction.
308	// Since we don't know in advance the `events_per_tx`, we'll assume the
309	// worst case scenario.
310	let receipts_hash_builder = max_events_size;
311
312	// `transactions_root` hash builder
313	let transactions_hash_builder =
314		limits::MAX_TRANSACTION_PAYLOAD_SIZE.saturating_mul(MEMORY_COEFFICIENT);
315
316	receipts_hash_builder.saturating_add(transactions_hash_builder)
317}

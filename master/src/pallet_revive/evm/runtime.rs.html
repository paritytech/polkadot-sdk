115impl<LookupSource, Signature, E, Lookup> Checkable<Lookup>
116	for UncheckedExtrinsic<LookupSource, Signature, E>
117where
118	E: EthExtra,
119	Self: Encode,
120	<E::Config as frame_system::Config>::Nonce: TryFrom<U256>,
121	CallOf<E::Config>: SetWeightLimit,
122	// required by Checkable for `generic::UncheckedExtrinsic`
123	generic::UncheckedExtrinsic<LookupSource, CallOf<E::Config>, Signature, E::Extension>:
124		Checkable<
125			Lookup,

131	fn check(self, lookup: &Lookup) -> Result<Self::Checked, TransactionValidityError> {
132		if !self.0.is_signed() {
133			if let Some(crate::Call::eth_transact { payload }) = self.0.function.is_sub_type() {
134				let checked = E::try_into_checked_extrinsic(payload.to_vec(), self.encoded_size())?;
135				return Ok(checked)
136			};
137		}
138		self.0.check(lookup)
139	}

229/// EthExtra convert an unsigned [`crate::Call::eth_transact`] into a [`CheckedExtrinsic`].
230pub trait EthExtra {
231	/// The Runtime configuration.
232	type Config: Config + TxConfig;
233
234	/// The Runtime's transaction extension.
235	/// It should include at least:
236	/// - [`frame_system::CheckNonce`] to ensure that the nonce from the Ethereum transaction is
237	///   correct.
238	type Extension: TransactionExtension<CallOf<Self::Config>>;
239
240	/// Get the transaction extension to apply to an unsigned [`crate::Call::eth_transact`]
241	/// extrinsic.
242	///
243	/// # Parameters
244	/// - `nonce`: The nonce extracted from the Ethereum transaction.
245	/// - `tip`: The transaction tip calculated from the Ethereum transaction.
246	fn get_eth_extension(
247		nonce: <Self::Config as frame_system::Config>::Nonce,
248		tip: BalanceOf<Self::Config>,
249	) -> Self::Extension;
250
251	/// Convert the unsigned [`crate::Call::eth_transact`] into a [`CheckedExtrinsic`].
252	/// and ensure that the fees from the Ethereum transaction correspond to the fees computed from
253	/// the encoded_len, the injected gas_limit and storage_deposit_limit.
254	///
255	/// # Parameters
256	/// - `payload`: The RLP-encoded Ethereum transaction.
257	/// - `encoded_len`: The encoded length of the extrinsic.
258	fn try_into_checked_extrinsic(
259		payload: Vec<u8>,
260		encoded_len: usize,
261	) -> Result<
262		CheckedExtrinsic<AccountIdOf<Self::Config>, CallOf<Self::Config>, Self::Extension>,
263		InvalidTransaction,
264	>
265	where
266		<Self::Config as frame_system::Config>::Nonce: TryFrom<U256>,
267		CallOf<Self::Config>: SetWeightLimit,
268	{
269		let tx = TransactionSigned::decode(&payload).map_err(|err| {
270			log::debug!(target: LOG_TARGET, "Failed to decode transaction: {err:?}");
271			InvalidTransaction::Call
272		})?;
273
274		// Check transaction type and reject unsupported transaction types
275		match &tx {
276			crate::evm::api::TransactionSigned::Transaction1559Signed(_) |
277			crate::evm::api::TransactionSigned::Transaction2930Signed(_) |
278			crate::evm::api::TransactionSigned::TransactionLegacySigned(_) => {
279				// Supported transaction types, continue processing
280			},
281			crate::evm::api::TransactionSigned::Transaction7702Signed(_) => {
282				log::debug!(target: LOG_TARGET, "EIP-7702 transactions are not supported");
283				return Err(InvalidTransaction::Call);
284			},
285			crate::evm::api::TransactionSigned::Transaction4844Signed(_) => {
286				log::debug!(target: LOG_TARGET, "EIP-4844 transactions are not supported");
287				return Err(InvalidTransaction::Call);
288			},
289		}
290
291		let signer_addr = tx.recover_eth_address().map_err(|err| {
292			log::debug!(target: LOG_TARGET, "Failed to recover signer: {err:?}");
293			InvalidTransaction::BadProof
294		})?;
295		let signer = <Self::Config as Config>::AddressMapper::to_fallback_account_id(&signer_addr);
296		let base_fee = <Pallet<Self::Config>>::evm_base_fee();
297		let tx = GenericTransaction::from_signed(tx, base_fee, None);
298		let nonce = tx.nonce.unwrap_or_default().try_into().map_err(|_| {
299			log::debug!(target: LOG_TARGET, "Failed to convert nonce");
300			InvalidTransaction::Call
301		})?;
302		let call_info = create_call::<Self::Config>(tx, Some(encoded_len as u32))?;
303		let storage_credit = <Self::Config as Config>::Currency::withdraw(
304					&signer,
305					call_info.storage_deposit,
306					Precision::Exact,
307					Preservation::Preserve,
308					Fortitude::Polite,
309		).map_err(|_| {
310			log::debug!(target: LOG_TARGET, "Not enough balance to hold additional storage deposit of {:?}", call_info.storage_deposit);
311			InvalidTransaction::Payment
312		})?;
313		<Self::Config as Config>::FeeInfo::deposit_txfee(storage_credit);
314
315		crate::tracing::if_tracing(|tracer| {
316			tracer.watch_address(&Pallet::<Self::Config>::block_author().unwrap_or_default());
317			tracer.watch_address(&signer_addr);
318		});
319
320		log::debug!(target: LOG_TARGET, "\
321			Created checked Ethereum transaction with: \
322			weight_limit={} \
323			additional_storage_deposit_held={:?} \
324			nonce={nonce:?}
325			",
326			call_info.weight_limit,
327			call_info.storage_deposit,
328		);
329
330		// We can't calculate a tip because it needs to be based on the actual gas used which we
331		// cannot know pre-dispatch. Hence we never supply a tip here or it would be way too high.
332		Ok(CheckedExtrinsic {
333			format: ExtrinsicFormat::Signed(
334				signer.into(),
335				Self::get_eth_extension(nonce, Zero::zero()),
336			),
337			function: call_info.call,
338		})
339	}

380	impl UncheckedExtrinsicBuilder {
381		/// Create a new builder with default values.
382		fn new() -> Self {

385					from: Some(Account::default().address()),

406		fn estimate_gas(&mut self) {

435			builder

438		/// Set before_validate function.
439		fn before_validate(mut self, f: impl Fn() + Send + Sync + 'static) -> Self {
440			self.before_validate = Some(std::sync::Arc::new(f));
441			self
442		}

444		fn check(
445			self,
446		) -> Result<
447			(u32, RuntimeCall, SignedExtra, GenericTransaction, Weight),
448			TransactionValidityError,
449		> {
450			self.mutate_estimate_and_check(Box::new(|_| ()))
451		}
452
453		/// Call `check` on the unchecked extrinsic, and `pre_dispatch` on the signed extension.
454		fn mutate_estimate_and_check(
455			mut self,
456			f: Box<dyn FnOnce(&mut GenericTransaction) -> ()>,
457		) -> Result<
458			(u32, RuntimeCall, SignedExtra, GenericTransaction, Weight),
459			TransactionValidityError,
460		> {
461			ExtBuilder::default().build().execute_with(|| self.estimate_gas());
462			ExtBuilder::default().build().execute_with(|| {
463				f(&mut self.tx);
464				let UncheckedExtrinsicBuilder { tx, before_validate, .. } = self.clone();
465
466				// Fund the account.
467				let account = Account::default();
468				Self::fund_account(&account);
469
470				let payload = account
471					.sign_transaction(tx.clone().try_into_unsigned().unwrap())
472					.signed_payload();
473				let call = RuntimeCall::Contracts(crate::Call::eth_transact { payload });
474
475				let uxt: UncheckedExtrinsic = generic::UncheckedExtrinsic::new_bare(call).into();
476				let encoded_len = uxt.encoded_size();
477				let result: CheckedExtrinsic<_, _, _> = uxt.check(&TestContext {})?;
478				let (account_id, extra): (AccountId32, SignedExtra) = match result.format {
479					ExtrinsicFormat::Signed(signer, extra) => (signer, extra),
480					_ => unreachable!(),
481				};
482
483				before_validate.map(|f| f());
484				extra.clone().validate_and_prepare(
485					RuntimeOrigin::signed(account_id),
486					&result.function,
487					&result.function.get_dispatch_info(),
488					encoded_len,
489					0,
490				)?;
491
492				Ok((
493					encoded_len as u32,
494					result.function,
495					extra,
496					tx,
497					self.dry_run.unwrap().gas_required,
498				))
499			})
500		}
501	}

58impl ReceiptInfo {
59	/// Initialize a new Receipt
60	pub fn new(
61		block_hash: H256,
62		block_number: U256,
63		contract_address: Option<Address>,
64		from: Address,
65		logs: Vec<Log>,
66		to: Option<Address>,
67		effective_gas_price: U256,
68		gas_used: U256,
69		success: bool,
70		transaction_hash: H256,
71		transaction_index: U256,
72		r#type: Byte,
73	) -> Self {
74		let logs_bloom = Self::logs_bloom(&logs);
75		ReceiptInfo {
76			block_hash,
77			block_number,
78			contract_address,
79			from,
80			logs,
81			logs_bloom,
82			to,
83			effective_gas_price,
84			gas_used,
85			status: Some(if success { U256::one() } else { U256::zero() }),
86			transaction_hash,
87			transaction_index,
88			r#type: Some(r#type),
89			..Default::default()
90		}
91	}
92
93	/// Returns `true` if the transaction was successful.
94	pub fn is_success(&self) -> bool {
95		self.status.map_or(false, |status| status == U256::one())
96	}
97
98	/// Calculate receipt logs bloom.
99	fn logs_bloom(logs: &[Log]) -> Bytes256 {
100		let mut bloom = [0u8; 256];
101		for log in logs {
102			m3_2048(&mut bloom, &log.address.as_ref());
103			for topic in &log.topics {
104				m3_2048(&mut bloom, topic.as_ref());
105			}
106		}
107		bloom.into()
108	}
109}
110/// Specialised Bloom filter that sets three bits out of 2048, given an
111/// arbitrary byte sequence.
112///
113/// See Section 4.4.1 "Transaction Receipt" of the [Ethereum Yellow Paper][ref].
114///
115/// [ref]: https://ethereum.github.io/yellowpaper/paper.pdf
116fn m3_2048(bloom: &mut [u8; 256], bytes: &[u8]) {
117	let hash = sp_core::keccak_256(bytes);
118	for i in [0, 2, 4] {
119		let bit = (hash[i + 1] as usize + ((hash[i] as usize) << 8)) & 0x7FF;
120		bloom[256 - 1 - bit / 8] |= 1 << (bit % 8);
121	}
122}

124#[test]
125fn can_deserialize_input_or_data_field_from_generic_transaction() {
126	let cases = [
127		("with input", r#"{"input": "0x01"}"#),
128		("with data", r#"{"data": "0x01"}"#),
129		("with both", r#"{"data": "0x01", "input": "0x01"}"#),
130	];
131
132	for (name, json) in cases {
133		let tx = serde_json::from_str::<GenericTransaction>(json).unwrap();
134		assert_eq!(tx.input.to_vec(), vec![1u8], "{}", name);
135	}
136
137	let err = serde_json::from_str::<GenericTransaction>(r#"{"data": "0x02", "input": "0x01"}"#)
138		.unwrap_err();
139	assert!(
140		err.to_string().starts_with(
141		"Both \"data\" and \"input\" are set and not equal. Please use \"input\" to pass transaction call data"
142		)
143	);
144}

148	let val: BlockNumberOrTagOrHash = serde_json::from_str("\"latest\"").unwrap();
149	assert_eq!(val, BlockTag::Latest.into());

158		r#"{ "blockHash": "0xaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaa" }"#,
159	] {
160		let val: BlockNumberOrTagOrHash = serde_json::from_str(s).unwrap();
161		assert_eq!(val, BlockNumberOrTagOrHash::BlockHash(H256([0xaau8; 32])));
162	}

206impl GenericTransaction {
207	/// Create a new [`GenericTransaction`] from a signed transaction.
208	pub fn from_signed(tx: TransactionSigned, base_gas_price: U256, from: Option<H160>) -> Self {
209		Self::from_unsigned(tx.into(), base_gas_price, from)
210	}

212	/// The gas price that is actually paid (including priority fee).
213	pub fn effective_gas_price(&self, base_gas_price: U256) -> Option<U256> {
214		let effective_gas_price = if let Some(prio_price) = self.max_priority_fee_per_gas {
215			let max_price = self.max_fee_per_gas?;
216			Some(max_price.min(base_gas_price.saturating_add(prio_price)))
217		} else {
218			self.gas_price
219		};
220
221		// we do not implement priority fee as it does not map to tip well
222		// hence the effective gas price cannot be higher than the base price
223		effective_gas_price.map(|e| e.min(base_gas_price))
224	}

249				chain_id: Some(tx.chain_id),
250				input: tx.input.into(),
251				nonce: Some(tx.nonce),
252				value: Some(tx.value),
253				to: Some(tx.to),
254				gas: Some(tx.gas),
255				access_list: Some(tx.access_list),

259				max_priority_fee_per_gas: Some(tx.max_priority_fee_per_gas),
260				..Default::default()
261			},
262			Transaction1559Unsigned(tx) => GenericTransaction {
263				from,

274				..Default::default()
275			},

277				from,
278				r#type: Some(tx.r#type.as_byte()),

290				from,

309	/// Convert to a [`TransactionUnsigned`].
310	pub fn try_into_unsigned(self) -> Result<TransactionUnsigned, ()> {
311		match self.r#type.unwrap_or_default().0 {
312			TYPE_LEGACY => Ok(TransactionLegacyUnsigned {
313				r#type: TypeLegacy {},
314				chain_id: self.chain_id,
315				input: self.input.to_bytes(),
316				nonce: self.nonce.unwrap_or_default(),
317				value: self.value.unwrap_or_default(),
318				to: self.to,
319				gas: self.gas.unwrap_or_default(),
320				gas_price: self.gas_price.unwrap_or_default(),
321			}
322			.into()),
323			TYPE_EIP1559 => Ok(Transaction1559Unsigned {
324				r#type: TypeEip1559 {},
325				chain_id: self.chain_id.unwrap_or_default(),
326				input: self.input.to_bytes(),
327				nonce: self.nonce.unwrap_or_default(),
328				value: self.value.unwrap_or_default(),
329				to: self.to,

332				access_list: self.access_list.unwrap_or_default(),
333				max_fee_per_gas: self.max_fee_per_gas.unwrap_or_default(),
334				max_priority_fee_per_gas: self.max_priority_fee_per_gas.unwrap_or_default(),
335			}

337			TYPE_EIP2930 => Ok(Transaction2930Unsigned {

340				input: self.input.to_bytes(),
341				nonce: self.nonce.unwrap_or_default(),
342				value: self.value.unwrap_or_default(),
343				to: self.to,
344				gas: self.gas.unwrap_or_default(),
345				gas_price: self.gas_price.unwrap_or_default(),
346				access_list: self.access_list.unwrap_or_default(),
347			}
348			.into()),
349			TYPE_EIP4844 => Ok(Transaction4844Unsigned {
350				r#type: TypeEip4844 {},
351				chain_id: self.chain_id.unwrap_or_default(),
352				input: self.input.to_bytes(),
353				nonce: self.nonce.unwrap_or_default(),
354				value: self.value.unwrap_or_default(),
355				to: self.to.unwrap_or_default(),
356				gas: self.gas.unwrap_or_default(),
357				max_fee_per_gas: self.max_fee_per_gas.unwrap_or_default(),
358				max_fee_per_blob_gas: self.max_fee_per_blob_gas.unwrap_or_default(),
359				max_priority_fee_per_gas: self.max_priority_fee_per_gas.unwrap_or_default(),
360				access_list: self.access_list.unwrap_or_default(),
361				blob_versioned_hashes: self.blob_versioned_hashes,
362			}
363			.into()),
364			TYPE_EIP7702 => Ok(Transaction7702Unsigned {
365				r#type: TypeEip7702 {},
366				chain_id: self.chain_id.unwrap_or_default(),
367				input: self.input.to_bytes(),
368				nonce: self.nonce.unwrap_or_default(),
369				value: self.value.unwrap_or_default(),
370				to: self.to,
371				gas: self.gas.unwrap_or_default(),

378			.into()),
379			_ => Err(()),
380		}
381	}

384#[test]
385fn from_unsigned_works_for_legacy() {
386	let base_gas_price = U256::from(10);
387	let tx = TransactionUnsigned::from(TransactionLegacyUnsigned {
388		chain_id: Some(U256::from(1)),
389		input: Bytes::from(vec![1u8]),
390		nonce: U256::from(1),
391		value: U256::from(1),
392		to: Some(H160::zero()),
393		gas: U256::from(1),
394		gas_price: U256::from(10),
395		..Default::default()
396	});
397
398	let generic = GenericTransaction::from_unsigned(tx.clone(), base_gas_price, None);
399	assert_eq!(generic.gas_price, Some(U256::from(10)));
400
401	let tx2 = generic.try_into_unsigned().unwrap();
402	assert_eq!(tx, tx2);
403}

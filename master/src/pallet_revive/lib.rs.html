661		fn build(&self) {
662			use crate::{exec::Key, vm::ContractBlob};
663			use frame_support::traits::fungible::Mutate;
664
665			if !System::<T>::account_exists(&Pallet::<T>::account_id()) {
666				let _ = T::Currency::mint_into(
667					&Pallet::<T>::account_id(),
668					T::Currency::minimum_balance(),
669				);
670			}
671
672			for id in &self.mapped_accounts {
673				if let Err(err) = T::AddressMapper::map_no_deposit(id) {
674					log::error!(target: LOG_TARGET, "Failed to map account {id:?}: {err:?}");
675				}
676			}
677
678			let owner = Pallet::<T>::account_id();
679
680			for genesis::Account { address, balance, nonce, contract_data } in &self.accounts {
681				let account_id = T::AddressMapper::to_account_id(address);
682
683				frame_system::Account::<T>::mutate(&account_id, |info| {
684					info.nonce = (*nonce).into();
685				});
686
687				match contract_data {
688					None => {
689						AccountInfoOf::<T>::insert(
690							address,
691							AccountInfo { account_type: AccountType::EOA, dust: 0 },
692						);
693					},
694					Some(genesis::ContractData { code, storage }) => {
695						let blob = if code.starts_with(&polkavm_common::program::BLOB_MAGIC) {
696							ContractBlob::<T>::from_pvm_code(   code.clone(), owner.clone()).inspect_err(|err| {
697								log::error!(target: LOG_TARGET, "Failed to create PVM ContractBlob for {address:?}: {err:?}");
698							})
699						} else {
700							ContractBlob::<T>::from_evm_runtime_code(code.clone(), account_id).inspect_err(|err| {
701								log::error!(target: LOG_TARGET, "Failed to create EVM ContractBlob for {address:?}: {err:?}");
702							})
703						};
704
705						let Ok(blob) = blob else {
706							continue;
707						};
708
709						let code_hash = *blob.code_hash();
710						let Ok(info) = <ContractInfo<T>>::new(&address, 0u32.into(), code_hash)
711							.inspect_err(|err| {
712								log::error!(target: LOG_TARGET, "Failed to create ContractInfo for {address:?}: {err:?}");
713							})
714						else {
715							continue;
716						};
717
718						AccountInfoOf::<T>::insert(
719							address,
720							AccountInfo { account_type: info.clone().into(), dust: 0 },
721						);
722
723						<PristineCode<T>>::insert(blob.code_hash(), code);
724						<CodeInfoOf<T>>::insert(blob.code_hash(), blob.code_info().clone());
725						for (k, v) in storage {
726							let _ = info.write(&Key::from_fixed(k.0), Some(v.0.to_vec()), None, false).inspect_err(|err| {
727								log::error!(target: LOG_TARGET, "Failed to write genesis storage for {address:?} at key {k:?}: {err:?}");
728							});
729						}
730					},
731				}
732
733				let _ = Pallet::<T>::set_evm_balance(address, *balance).inspect_err(|err| {
734					log::error!(target: LOG_TARGET, "Failed to set EVM balance for {address:?}: {err:?}");
735				});
736			}
737
738			// Set debug settings.
739			if let Some(settings) = self.debug_settings.as_ref() {
740				settings.write_to_storage::<T>()
741			}
742		}

759		fn integrity_test() {
760			assert!(T::ChainId::get() > 0, "ChainId must be greater than 0");
761
762			T::FeeInfo::integrity_test();
763
764			// The memory available in the block building runtime
765			let max_runtime_mem: u32 = T::RuntimeMemory::get();
766
767			// We only allow 50% of the runtime memory to be utilized by the contracts call
768			// stack, keeping the rest for other facilities, such as PoV, etc.
769			const TOTAL_MEMORY_DEVIDER: u32 = 2;
770
771			// Check that the configured memory limits fit into runtime memory.
772			//
773			// Dynamic allocations are not available, yet. Hence they are not taken into
774			// consideration here.
775			let memory_left = i64::from(max_runtime_mem)
776				.saturating_div(TOTAL_MEMORY_DEVIDER.into())
777				.saturating_sub(limits::MEMORY_REQUIRED.into());
778
779			log::debug!(target: LOG_TARGET, "Integrity check: memory_left={} KB", memory_left / 1024);
780
781			assert!(
782				memory_left >= 0,
783				"Runtime does not have enough memory for current limits. Additional runtime memory required: {} KB",
784				memory_left.saturating_mul(TOTAL_MEMORY_DEVIDER.into()).abs() / 1024
785			);
786
787			// Validators are configured to be able to use more memory than block builders. This is
788			// because in addition to `max_runtime_mem` they need to hold additional data in
789			// memory: PoV in multiple copies (1x encoded + 2x decoded) and all storage which
790			// includes emitted events. The assumption is that storage/events size
791			// can be a maximum of half of the validator runtime memory - max_runtime_mem.
792			let max_block_ref_time = T::BlockWeights::get()
793				.get(DispatchClass::Normal)
794				.max_total
795				.unwrap_or_else(|| T::BlockWeights::get().max_block)
796				.ref_time();
797			let max_payload_size = limits::PAYLOAD_BYTES;
798			let max_key_size =
799				Key::try_from_var(alloc::vec![0u8; limits::STORAGE_KEY_BYTES as usize])
800					.expect("Key of maximal size shall be created")
801					.hash()
802					.len() as u32;
803
804			let max_immutable_key_size = T::AccountId::max_encoded_len() as u32;
805			let max_immutable_size: u32 = ((max_block_ref_time /
806				(<RuntimeCosts as gas::Token<T>>::weight(&RuntimeCosts::SetImmutableData(
807					limits::IMMUTABLE_BYTES,
808				))
809				.ref_time()))
810			.saturating_mul(limits::IMMUTABLE_BYTES.saturating_add(max_immutable_key_size) as u64))
811			.try_into()
812			.expect("Immutable data size too big");
813
814			// We can use storage to store items using the available block ref_time with the
815			// `set_storage` host function.
816			let max_storage_size: u32 = ((max_block_ref_time /
817				(<RuntimeCosts as gas::Token<T>>::weight(&RuntimeCosts::SetStorage {
818					new_bytes: max_payload_size,
819					old_bytes: 0,
820				})
821				.ref_time()))
822			.saturating_mul(max_payload_size.saturating_add(max_key_size) as u64))
823			.saturating_add(max_immutable_size.into())
824			.try_into()
825			.expect("Storage size too big");
826
827			let max_pvf_mem: u32 = T::PVFMemory::get();
828			let storage_size_limit = max_pvf_mem.saturating_sub(max_runtime_mem) / 2;
829
830			assert!(
831				max_storage_size < storage_size_limit,
832				"Maximal storage size {} exceeds the storage limit {}",
833				max_storage_size,
834				storage_size_limit
835			);
836
837			// We can use storage to store events using the available block ref_time with the
838			// `deposit_event` host function. The overhead of stored events, which is around 100B,
839			// is not taken into account to simplify calculations, as it does not change much.
840			let max_events_size: u32 = ((max_block_ref_time /
841				(<RuntimeCosts as gas::Token<T>>::weight(&RuntimeCosts::DepositEvent {
842					num_topic: 0,
843					len: max_payload_size,
844				})
845				.saturating_add(<RuntimeCosts as gas::Token<T>>::weight(&RuntimeCosts::HostFn))
846				.ref_time()))
847			.saturating_mul(max_payload_size as u64))
848			.try_into()
849			.expect("Events size too big");
850
851			assert!(
852				max_events_size < storage_size_limit,
853				"Maximal events size {} exceeds the events limit {}",
854				max_events_size,
855				storage_size_limit
856			);
857		}

1137		/// If the code does not already exist a deposit is reserved from the caller
1138		/// The size of the reserve depends on the size of the supplied `code`.
1139		///
1140		/// # Note
1141		///
1142		/// Anyone can instantiate a contract from any uploaded code and thus prevent its removal.
1143		/// To avoid this situation a constructor could employ access control so that it can
1144		/// only be instantiated by permissioned entities. The same is true when uploading
1145		/// through [`Self::instantiate_with_code`].

1148		/// references this code, the code will be removed automatically.

1153			code: Vec<u8>,
1154			#[pallet::compact] storage_deposit_limit: BalanceOf<T>,
1155		) -> DispatchResult {

1157		}

1178		/// an error if either the `code_hash` or `dest` do not exist.
1179		///
1180		/// # Note
1181		///
1182		/// This does **not** change the address of the contract in question. This means
1183		/// that the contract address is no longer derived from its code hash after calling
1184		/// this dispatchable.

1186		#[pallet::weight(<T as Config>::WeightInfo::set_code())]

1189			dest: H160,
1190			code_hash: sp_core::H256,
1191		) -> DispatchResult {
1192			ensure_root(origin)?;
1193			<AccountInfoOf<T>>::try_mutate(&dest, |account| {
1194				let Some(account) = account else {
1195					return Err(<Error<T>>::ContractNotFound.into());
1196				};
1197
1198				let AccountType::Contract(ref mut contract) = account.account_type else {
1199					return Err(<Error<T>>::ContractNotFound.into());
1200				};

1202				<CodeInfo<T>>::increment_refcount(code_hash)?;
1203				let _ = <CodeInfo<T>>::decrement_refcount(contract.code_hash)?;
1204				contract.code_hash = code_hash;

1206				Ok(())
1207			})
1208		}

1274	/// A generalized version of [`Self::call`].
1275	///
1276	/// Identical to [`Self::call`] but tailored towards being called by other code within the
1277	/// runtime as opposed to from an extrinsic. It returns more information and allows the
1278	/// enablement of features that are not suitable for an extrinsic (debugging, event
1279	/// collection).
1280	pub fn bare_call(
1281		origin: OriginFor<T>,
1282		dest: H160,
1283		evm_value: U256,
1284		gas_limit: Weight,
1285		storage_deposit_limit: BalanceOf<T>,
1286		data: Vec<u8>,
1287		exec_config: ExecConfig,
1288	) -> ContractResult<ExecReturnValue, BalanceOf<T>> {
1289		if let Err(contract_result) = Self::ensure_non_contract_if_signed(&origin) {
1290			return contract_result;
1291		}
1292		let mut gas_meter = GasMeter::new(gas_limit);
1293		let mut storage_deposit = Default::default();
1294
1295		let try_call = || {
1296			let origin = ExecOrigin::from_runtime_origin(origin)?;
1297			let mut storage_meter = StorageMeter::new(storage_deposit_limit);
1298			let result = ExecStack::<T, ContractBlob<T>>::run_call(
1299				origin.clone(),
1300				dest,
1301				&mut gas_meter,
1302				&mut storage_meter,
1303				evm_value,
1304				data,
1305				&exec_config,
1306			)?;
1307			storage_deposit =
1308				storage_meter.try_into_deposit(&origin, &exec_config).inspect_err(|err| {
1309					log::debug!(target: LOG_TARGET, "Failed to transfer deposit: {err:?}");
1310				})?;
1311			Ok(result)
1312		};
1313		let result = Self::run_guarded(try_call);
1314		ContractResult {
1315			result: result.map_err(|r| r.error),
1316			gas_consumed: gas_meter.gas_consumed(),
1317			gas_required: gas_meter.gas_required(),
1318			storage_deposit,
1319		}
1320	}

1333	/// A generalized version of [`Self::instantiate`] or [`Self::instantiate_with_code`].
1334	///
1335	/// Identical to [`Self::instantiate`] or [`Self::instantiate_with_code`] but tailored towards
1336	/// being called by other code within the runtime as opposed to from an extrinsic. It returns
1337	/// more information to the caller useful to estimate the cost of the operation.
1338	pub fn bare_instantiate(
1339		origin: OriginFor<T>,
1340		evm_value: U256,
1341		gas_limit: Weight,
1342		mut storage_deposit_limit: BalanceOf<T>,
1343		code: Code,
1344		data: Vec<u8>,
1345		salt: Option<[u8; 32]>,
1346		exec_config: ExecConfig,
1347	) -> ContractResult<InstantiateReturnValue, BalanceOf<T>> {
1348		// Enforce EIP-3607 for top-level signed origins: deny signed contract addresses.
1349		if let Err(contract_result) = Self::ensure_non_contract_if_signed(&origin) {
1350			return contract_result;
1351		}
1352		let mut gas_meter = GasMeter::new(gas_limit);
1353		let mut storage_deposit = Default::default();
1354		let try_instantiate = || {
1355			let instantiate_account = T::InstantiateOrigin::ensure_origin(origin.clone())?;
1356
1357			if_tracing(|t| t.instantiate_code(&code, salt.as_ref()));
1358			let (executable, upload_deposit) = match code {
1359				Code::Upload(code) if code.starts_with(&polkavm_common::program::BLOB_MAGIC) => {
1360					let upload_account = T::UploadOrigin::ensure_origin(origin)?;
1361					let (executable, upload_deposit) = Self::try_upload_pvm_code(
1362						upload_account,
1363						code,
1364						storage_deposit_limit,
1365						&exec_config,
1366					)?;
1367					storage_deposit_limit.saturating_reduce(upload_deposit);
1368					(executable, upload_deposit)
1369				},
1370				Code::Upload(code) =>
1371					if T::AllowEVMBytecode::get() {
1372						let origin = T::UploadOrigin::ensure_origin(origin)?;
1373						let executable = ContractBlob::from_evm_init_code(code, origin)?;
1374						(executable, Default::default())
1375					} else {
1376						return Err(<Error<T>>::CodeRejected.into())
1377					},
1378				Code::Existing(code_hash) =>
1379					(ContractBlob::from_storage(code_hash, &mut gas_meter)?, Default::default()),
1380			};
1381			let instantiate_origin = ExecOrigin::from_account_id(instantiate_account.clone());
1382			let mut storage_meter = StorageMeter::new(storage_deposit_limit);
1383			let result = ExecStack::<T, ContractBlob<T>>::run_instantiate(
1384				instantiate_account,
1385				executable,
1386				&mut gas_meter,
1387				&mut storage_meter,
1388				evm_value,
1389				data,
1390				salt.as_ref(),
1391				&exec_config,
1392			);
1393			storage_deposit = storage_meter
1394				.try_into_deposit(&instantiate_origin, &exec_config)?
1395				.saturating_add(&StorageDeposit::Charge(upload_deposit));
1396			result
1397		};
1398		let output = Self::run_guarded(try_instantiate);
1399		ContractResult {
1400			result: output
1401				.map(|(addr, result)| InstantiateReturnValue { result, addr })
1402				.map_err(|e| e.error),
1403			gas_consumed: gas_meter.gas_consumed(),
1404			gas_required: gas_meter.gas_required(),
1405			storage_deposit,
1406		}
1407	}

1409	/// Dry-run Ethereum calls.
1410	///
1411	/// # Parameters
1412	///
1413	/// - `tx`: The Ethereum transaction to simulate.
1414	pub fn dry_run_eth_transact(
1415		mut tx: GenericTransaction,
1416	) -> Result<EthTransactInfo<BalanceOf<T>>, EthTransactError>
1417	where
1418		T::Nonce: Into<U256>,
1419		CallOf<T>: SetWeightLimit,
1420	{
1421		log::debug!(target: LOG_TARGET, "dry_run_eth_transact: {tx:?}");
1422
1423		let origin = T::AddressMapper::to_account_id(&tx.from.unwrap_or_default());
1424		Self::prepare_dry_run(&origin);
1425
1426		let base_fee = Self::evm_base_fee();
1427		let effective_gas_price = tx.effective_gas_price(base_fee).unwrap_or(base_fee);
1428
1429		if effective_gas_price < base_fee {
1430			Err(EthTransactError::Message(format!(
1431				"Effective gas price {effective_gas_price:?} lower than base fee {base_fee:?}"
1432			)))?;
1433		}
1434
1435		if tx.nonce.is_none() {
1436			tx.nonce = Some(<System<T>>::account_nonce(&origin).into());
1437		}
1438		if tx.chain_id.is_none() {
1439			tx.chain_id = Some(T::ChainId::get().into());
1440		}
1441		if tx.gas_price.is_none() {
1442			tx.gas_price = Some(effective_gas_price);
1443		}
1444		if tx.max_priority_fee_per_gas.is_none() {
1445			tx.max_priority_fee_per_gas = Some(effective_gas_price);
1446		}
1447		if tx.max_fee_per_gas.is_none() {
1448			tx.max_fee_per_gas = Some(effective_gas_price);
1449		}
1450
1451		let gas = tx.gas;
1452		if tx.gas.is_none() {
1453			tx.gas = Some(Self::evm_block_gas_limit());
1454		}
1455		if tx.r#type.is_none() {
1456			tx.r#type = Some(TYPE_EIP1559.into());
1457		}
1458
1459		// Store values before moving the tx
1460		let value = tx.value.unwrap_or_default();
1461		let input = tx.input.clone().to_vec();
1462		let from = tx.from;
1463		let to = tx.to;
1464
1465		// we need to parse the weight from the transaction so that it is run
1466		// using the exact weight limit passed by the eth wallet
1467		let mut call_info = create_call::<T>(tx, None)
1468			.map_err(|err| EthTransactError::Message(format!("Invalid call: {err:?}")))?;
1469
1470		// the dry-run might leave out certain fields
1471		// in those cases we skip the check that the caller has enough balance
1472		// to pay for the fees
1473		let exec_config = {
1474			let base_info = T::FeeInfo::base_dispatch_info(&mut call_info.call);
1475			ExecConfig::new_eth_tx(
1476				effective_gas_price,
1477				call_info.encoded_len,
1478				base_info.total_weight(),
1479			)
1480		};
1481
1482		// emulate transaction behavior
1483		let fees = call_info.tx_fee.saturating_add(call_info.storage_deposit);
1484		if let Some(from) = &from {
1485			let fees = if gas.is_some() { fees } else { Zero::zero() };
1486			let balance = Self::evm_balance(from);
1487			if balance < Pallet::<T>::convert_native_to_evm(fees).saturating_add(value) {
1488				return Err(EthTransactError::Message(format!(
1489					"insufficient funds for gas * price + value ({fees:?}): address {from:?} have {balance:?} (supplied gas {gas:?})",
1490				)));
1491			}
1492		}
1493
1494		// the deposit is done when the transaction is transformed from an `eth_transact`
1495		// we emulate this behavior for the dry-run her
1496		T::FeeInfo::deposit_txfee(T::Currency::issue(fees));
1497
1498		let extract_error = |err| {
1499			if err == Error::<T>::StorageDepositNotEnoughFunds.into() {
1500				Err(EthTransactError::Message(format!("Not enough gas supplied: {err:?}")))
1501			} else {
1502				Err(EthTransactError::Message(format!("failed to run contract: {err:?}")))
1503			}
1504		};
1505
1506		// Dry run the call
1507		let mut dry_run = match to {
1508			// A contract call.
1509			Some(dest) => {
1510				if dest == RUNTIME_PALLETS_ADDR {
1511					let Ok(dispatch_call) = <CallOf<T>>::decode(&mut &input[..]) else {
1512						return Err(EthTransactError::Message(format!(
1513							"Failed to decode pallet-call {input:?}"
1514						)));
1515					};
1516
1517					if let Err(result) =
1518						dispatch_call.clone().dispatch(RawOrigin::Signed(origin).into())
1519					{
1520						return Err(EthTransactError::Message(format!(
1521							"Failed to dispatch call: {:?}",
1522							result.error,
1523						)));
1524					};
1525
1526					Default::default()
1527				} else {
1528					// Dry run the call.
1529					let result = crate::Pallet::<T>::bare_call(
1530						OriginFor::<T>::signed(origin),
1531						dest,
1532						value,
1533						call_info.weight_limit,
1534						BalanceOf::<T>::max_value(),
1535						input.clone(),
1536						exec_config,
1537					);
1538
1539					let data = match result.result {
1540						Ok(return_value) => {
1541							if return_value.did_revert() {
1542								return Err(EthTransactError::Data(return_value.data));
1543							}
1544							return_value.data
1545						},
1546						Err(err) => {
1547							log::debug!(target: LOG_TARGET, "Failed to execute call: {err:?}");
1548							return extract_error(err);
1549						},
1550					};
1551
1552					EthTransactInfo {
1553						gas_required: result.gas_required,
1554						storage_deposit: result.storage_deposit.charge_or_zero(),
1555						data,
1556						eth_gas: Default::default(),
1557					}
1558				}
1559			},
1560			// A contract deployment
1561			None => {
1562				// Extract code and data from the input.
1563				let (code, data) = if input.starts_with(&polkavm_common::program::BLOB_MAGIC) {
1564					extract_code_and_data(&input).unwrap_or_else(|| (input, Default::default()))
1565				} else {
1566					(input, vec![])
1567				};
1568
1569				// Dry run the call.
1570				let result = crate::Pallet::<T>::bare_instantiate(
1571					OriginFor::<T>::signed(origin),
1572					value,
1573					call_info.weight_limit,
1574					BalanceOf::<T>::max_value(),
1575					Code::Upload(code.clone()),
1576					data.clone(),
1577					None,
1578					exec_config,
1579				);
1580
1581				let returned_data = match result.result {
1582					Ok(return_value) => {
1583						if return_value.result.did_revert() {
1584							return Err(EthTransactError::Data(return_value.result.data));
1585						}
1586						return_value.result.data
1587					},
1588					Err(err) => {
1589						log::debug!(target: LOG_TARGET, "Failed to instantiate: {err:?}");
1590						return extract_error(err);
1591					},
1592				};
1593
1594				EthTransactInfo {
1595					gas_required: result.gas_required,
1596					storage_deposit: result.storage_deposit.charge_or_zero(),
1597					data: returned_data,
1598					eth_gas: Default::default(),
1599				}
1600			},
1601		};
1602
1603		// replace the weight passed in the transaction with the dry_run result
1604		call_info.call.set_weight_limit(dry_run.gas_required);
1605
1606		// we notify the wallet that the tx would not fit
1607		let total_weight = T::FeeInfo::dispatch_info(&call_info.call).total_weight();
1608		let max_weight = Self::evm_max_extrinsic_weight();
1609		if total_weight.any_gt(max_weight) {
1610			Err(EthTransactError::Message(format!(
1611				"\
1612				The transaction consumes more than the allowed weight. \
1613				needed={total_weight} \
1614				allowed={max_weight} \
1615				overweight_by={}\
1616				",
1617				total_weight.saturating_sub(max_weight),
1618			)))?;
1619		}
1620
1621		// not enough gas supplied to pay for both the tx fees and the storage deposit
1622		let transaction_fee = T::FeeInfo::tx_fee(call_info.encoded_len, &call_info.call);
1623		let available_fee = T::FeeInfo::remaining_txfee();
1624		if transaction_fee > available_fee {
1625			Err(EthTransactError::Message(format!(
1626				"Not enough gas supplied: Off by: {:?}",
1627				call_info.tx_fee.saturating_sub(available_fee),
1628			)))?;
1629		}
1630
1631		// We add `1` to account for the potential rounding error of the multiplication.
1632		// Returning a larger value here just increases the the pre-dispatch weight.
1633		let eth_gas: U256 = T::FeeInfo::next_fee_multiplier_reciprocal()
1634			.saturating_mul_int(transaction_fee.saturating_add(dry_run.storage_deposit))
1635			.saturating_add(1_u32.into())
1636			.into();
1637
1638		log::debug!(target: LOG_TARGET, "\
1639			dry_run_eth_transact: \
1640			weight_limit={:?}: \
1641			eth_gas={eth_gas:?})\
1642			",
1643			dry_run.gas_required,
1644
1645		);
1646		dry_run.eth_gas = eth_gas;
1647		Ok(dry_run)
1648	}

1978	/// Transfer a deposit from some account to another.
1979	///
1980	/// `from` is usually the transaction origin and `to` a contract or
1981	/// the pallets own account.
1982	fn charge_deposit(
1983		hold_reason: Option<HoldReason>,
1984		from: &T::AccountId,
1985		to: &T::AccountId,
1986		amount: BalanceOf<T>,
1987		exec_config: &ExecConfig,
1988	) -> DispatchResult {
1989		use frame_support::traits::tokens::{Fortitude, Precision, Preservation};
1990		match (exec_config.collect_deposit_from_hold.is_some(), hold_reason) {
1991			(true, hold_reason) => {
1992				T::FeeInfo::withdraw_txfee(amount)
1993					.ok_or(())
1994					.and_then(|credit| T::Currency::resolve(to, credit).map_err(|_| ()))
1995					.and_then(|_| {
1996						if let Some(hold_reason) = hold_reason {
1997							T::Currency::hold(&hold_reason.into(), to, amount).map_err(|_| ())?;
1998						}
1999						Ok(())
2000					})
2001					.map_err(|_| Error::<T>::StorageDepositNotEnoughFunds)?;
2002			},
2003			(false, Some(hold_reason)) => {
2004				T::Currency::transfer_and_hold(
2005					&hold_reason.into(),
2006					from,
2007					to,
2008					amount,
2009					Precision::Exact,
2010					Preservation::Preserve,
2011					Fortitude::Polite,
2012				)
2013				.map_err(|_| Error::<T>::StorageDepositNotEnoughFunds)?;
2014			},
2015			(false, None) => {
2016				T::Currency::transfer(from, to, amount, Preservation::Preserve)
2017					.map_err(|_| Error::<T>::StorageDepositNotEnoughFunds)?;
2018			},
2019		}
2020		Ok(())
2021	}

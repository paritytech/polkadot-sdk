31pub struct System<T>(PhantomData<T>);
32
33impl<T: Config> BuiltinPrecompile for System<T> {
34	type T = T;
35	type Interface = ISystem::ISystemCalls;
36	const MATCHER: BuiltinAddressMatcher =
37		BuiltinAddressMatcher::Fixed(NonZero::new(0x900).unwrap());
38	const HAS_CONTRACT_INFO: bool = false;
39
40	fn call(
41		_address: &[u8; 20],
42		input: &Self::Interface,
43		env: &mut impl Ext<T = Self::T>,
44	) -> Result<Vec<u8>, Error> {
45		use ISystem::ISystemCalls;
46		match input {
47			ISystemCalls::terminate(_) if env.is_read_only() =>
48				Err(crate::Error::<T>::StateChangeDenied.into()),
49			ISystemCalls::hashBlake256(ISystem::hashBlake256Call { input }) => {
50				env.frame_meter_mut()
51					.charge_weight_token(RuntimeCosts::HashBlake256(input.len() as u32))?;
52				let output = sp_io::hashing::blake2_256(input.as_bytes_ref());
53				Ok(output.abi_encode())
54			},
55			ISystemCalls::hashBlake128(ISystem::hashBlake128Call { input }) => {
56				env.frame_meter_mut()
57					.charge_weight_token(RuntimeCosts::HashBlake128(input.len() as u32))?;
58				let output = sp_io::hashing::blake2_128(input.as_bytes_ref());
59				Ok(output.abi_encode())
60			},
61			ISystemCalls::toAccountId(ISystem::toAccountIdCall { input }) => {
62				env.frame_meter_mut().charge_weight_token(RuntimeCosts::ToAccountId)?;
63				let account_id = env.to_account_id(&H160::from_slice(input.as_slice()));
64				Ok(account_id.encode().abi_encode())
65			},
66			ISystemCalls::callerIsOrigin(ISystem::callerIsOriginCall {}) => {
67				env.frame_meter_mut().charge_weight_token(RuntimeCosts::CallerIsOrigin)?;
68				let is_origin = env.caller_is_origin(true);
69				Ok(is_origin.abi_encode())
70			},
71			ISystemCalls::callerIsRoot(ISystem::callerIsRootCall {}) => {
72				env.frame_meter_mut().charge_weight_token(RuntimeCosts::CallerIsRoot)?;
73				let is_root = env.caller_is_root(true);
74				Ok(is_root.abi_encode())
75			},
76			ISystemCalls::ownCodeHash(ISystem::ownCodeHashCall {}) => {
77				env.frame_meter_mut().charge_weight_token(RuntimeCosts::OwnCodeHash)?;
78				let caller = env.caller();
79				let addr = T::AddressMapper::to_address(caller.account_id()?);
80				let output = env.code_hash(&addr.into()).0.abi_encode();
81				Ok(output)
82			},
83			ISystemCalls::minimumBalance(ISystem::minimumBalanceCall {}) => {
84				env.frame_meter_mut().charge_weight_token(RuntimeCosts::MinimumBalance)?;
85				let minimum_balance = env.minimum_balance();
86				Ok(minimum_balance.to_big_endian().abi_encode())
87			},
88			ISystemCalls::weightLeft(ISystem::weightLeftCall {}) => {
89				env.frame_meter_mut().charge_weight_token(RuntimeCosts::WeightLeft)?;
90				let ref_time = env.frame_meter().weight_left().unwrap_or_default().ref_time();
91				let proof_size = env.frame_meter().weight_left().unwrap_or_default().proof_size();
92				let res = (ref_time, proof_size);
93				Ok(res.abi_encode())
94			},
95			ISystemCalls::terminate(ISystem::terminateCall { beneficiary }) => {
96				// no need to adjust gas because this always deletes code
97				env.frame_meter_mut()
98					.charge_weight_token(RuntimeCosts::Terminate { code_removed: true })?;
99				let h160 = H160::from_slice(beneficiary.as_slice());
100				env.terminate_caller(&h160).map_err(Error::try_to_revert::<T>)?;
101				Ok(Vec::new())
102			},
103		}
104	}
105}

131	#[test]
132	fn test_system_precompile_unmapped_account() {
133		ExtBuilder::default().build().execute_with(|| {
134			// given
135			let mut call_setup = CallSetup::<Test>::default();
136			let (mut ext, _) = call_setup.ext();
137			let unmapped_address = H160::zero();
138
139			// when
140			let input = ISystem::ISystemCalls::toAccountId(ISystem::toAccountIdCall {
141				input: unmapped_address.0.into(),
142			});
143			let raw_data =
144				<System<Test>>::call(&<System<Test>>::MATCHER.base_address(), &input, &mut ext)
145					.unwrap();
146
147			// then
148			let expected_fallback_account_id =
149				Bytes::abi_decode(&raw_data).expect("decoding failed");
150			assert_eq!(
151				expected_fallback_account_id.0.as_ref()[20..32],
152				[0xEE; 12],
153				"no fallback suffix found where one should be"
154			);
155		})
156	}
157
158	#[test]
159	fn test_system_precompile_mapped_account() {
160		use crate::test_utils::EVE;
161		ExtBuilder::default().build().execute_with(|| {
162			// given
163			let mapped_address = {
164				<Test as pallet::Config>::Currency::set_balance(&EVE, caller_funding::<Test>());
165				let _ = <Test as pallet::Config>::AddressMapper::map(&EVE);
166				<Test as pallet::Config>::AddressMapper::to_address(&EVE)
167			};
168
169			let mut call_setup = CallSetup::<Test>::default();
170			let (mut ext, _) = call_setup.ext();
171
172			// when
173			let input = ISystem::ISystemCalls::toAccountId(ISystem::toAccountIdCall {
174				input: mapped_address.0.into(),
175			});
176			let raw_data =
177				<System<Test>>::call(&<System<Test>>::MATCHER.base_address(), &input, &mut ext)
178					.unwrap();
179
180			// then
181			let data = Bytes::abi_decode(&raw_data).expect("decoding failed");
182			assert_ne!(
183				data.0.as_ref()[20..32],
184				[0xEE; 12],
185				"fallback suffix found where none should be"
186			);
187			assert_eq!(
188				<Test as frame_system::Config>::AccountId::decode(&mut data.as_ref()),
189				Ok(EVE),
190			);
191		})
192	}

80/// The error type for the client.
81#[derive(Error, Debug)]
82pub enum ClientError {
83	/// A [`jsonrpsee::core::ClientError`] wrapper error.
84	#[error(transparent)]
85	Jsonrpsee(#[from] jsonrpsee::core::ClientError),
86	/// A [`subxt::Error`] wrapper error.
87	#[error(transparent)]
88	SubxtError(#[from] subxt::Error),
89	#[error(transparent)]
90	RpcError(#[from] subxt::ext::subxt_rpcs::Error),
91	/// A [`sqlx::Error`] wrapper error.
92	#[error(transparent)]
93	SqlxError(#[from] sqlx::Error),
94	/// A [`codec::Error`] wrapper error.
95	#[error(transparent)]
96	CodecError(#[from] codec::Error),
97	/// Transcact call failed.
98	#[error("contract reverted")]
99	TransactError(EthTransactError),
100	/// A decimal conversion failed.
101	#[error("conversion failed")]
102	ConversionFailed,
103	/// The block hash was not found.
104	#[error("hash not found")]
105	BlockNotFound,
106	/// The contract was not found.
107	#[error("Contract not found")]
108	ContractNotFound,
109	#[error("No Ethereum extrinsic found")]
110	EthExtrinsicNotFound,
111	/// The transaction fee could not be found
112	#[error("transactionFeePaid event not found")]
113	TxFeeNotFound,
114	/// Failed to decode a raw payload into a signed transaction.
115	#[error("Failed to decode a raw payload into a signed transaction")]
116	TxDecodingFailed,
117	/// Failed to recover eth address.
118	#[error("failed to recover eth address")]
119	RecoverEthAddressFailed,
120	/// Failed to filter logs.
121	#[error("Failed to filter logs")]
122	LogFilterFailed(#[from] anyhow::Error),
123}

125const REVERT_CODE: i32 = 3;
126impl From<ClientError> for ErrorObjectOwned {
127	fn from(err: ClientError) -> Self {
128		match err {
129			ClientError::SubxtError(subxt::Error::Rpc(subxt::error::RpcError::ClientError(
130				subxt::ext::subxt_rpcs::Error::User(err),
131			))) |
132			ClientError::RpcError(subxt::ext::subxt_rpcs::Error::User(err)) =>
133				ErrorObjectOwned::owned::<Vec<u8>>(err.code, err.message, None),
134			ClientError::TransactError(EthTransactError::Data(data)) => {
135				let msg = match decode_revert_reason(&data) {
136					Some(reason) => format!("execution reverted: {reason}"),
137					None => "execution reverted".to_string(),
138				};
139
140				let data = format!("0x{}", hex::encode(data));
141				ErrorObjectOwned::owned::<String>(REVERT_CODE, msg, Some(data))
142			},
143			ClientError::TransactError(EthTransactError::Message(msg)) =>
144				ErrorObjectOwned::owned::<String>(CALL_EXECUTION_FAILED_CODE, msg, None),
145			_ =>
146				ErrorObjectOwned::owned::<String>(CALL_EXECUTION_FAILED_CODE, err.to_string(), None),
147		}
148	}
149}

169/// Fetch the chain ID from the substrate chain.
170async fn chain_id(api: &OnlineClient<SrcChainConfig>) -> Result<u64, ClientError> {
171	let query = subxt_client::constants().revive().chain_id();
172	api.constants().at(&query).map_err(|err| err.into())
173}
174
175/// Fetch the max block weight from the substrate chain.
176async fn max_block_weight(api: &OnlineClient<SrcChainConfig>) -> Result<Weight, ClientError> {
177	let query = subxt_client::constants().system().block_weights();
178	let weights = api.constants().at(&query)?;
179	let max_block = weights.per_class.normal.max_extrinsic.unwrap_or(weights.max_block);
180	Ok(max_block.0)
181}

183/// Get the automine status from the node.
184async fn get_automine(rpc_client: &RpcClient) -> bool {
185	match rpc_client.request::<bool>("getAutomine", rpc_params![]).await {
186		Ok(val) => val,
187		Err(err) => {
188			log::info!(target: LOG_TARGET, "Node does not have getAutomine RPC. Defaulting to automine=false. error: {err:?}");
189			false
190		},
191	}
192}

194/// Extract the block timestamp.
195async fn extract_block_timestamp(block: &SubstrateBlock) -> Option<u64> {
196	let extrinsics = block.extrinsics().await.ok()?;
197	let ext = extrinsics
198		.find_first::<crate::subxt_client::timestamp::calls::types::Set>()
199		.ok()??;
200
201	Some(ext.value.now / 1000)
202}

204/// Connect to a node at the given URL, and return the underlying API, RPC client, and legacy RPC
205/// clients.
206pub async fn connect(
207	node_rpc_url: &str,
208) -> Result<(OnlineClient<SrcChainConfig>, RpcClient, LegacyRpcMethods<SrcChainConfig>), ClientError>
209{
210	log::info!(target: LOG_TARGET, "๐ Connecting to node at: {node_rpc_url} ...");
211	let rpc_client = ReconnectingRpcClient::builder()
212		.retry_policy(ExponentialBackoff::from_millis(100).max_delay(Duration::from_secs(10)))
213		.build(node_rpc_url.to_string())
214		.await?;
215	let rpc_client = RpcClient::new(rpc_client);
216	log::info!(target: LOG_TARGET, "๐ Connected to node at: {node_rpc_url}");
217
218	let api = OnlineClient::<SrcChainConfig>::from_rpc_client(rpc_client.clone()).await?;
219	let rpc = LegacyRpcMethods::<SrcChainConfig>::new(rpc_client.clone());
220	Ok((api, rpc_client, rpc))
221}

225	pub async fn new(
226		api: OnlineClient<SrcChainConfig>,
227		rpc_client: RpcClient,
228		rpc: LegacyRpcMethods<SrcChainConfig>,
229		block_provider: SubxtBlockInfoProvider,
230		receipt_provider: ReceiptProvider,
231	) -> Result<Self, ClientError> {
232		let (chain_id, max_block_weight, automine) =
233			tokio::try_join!(chain_id(&api), max_block_weight(&api), async {
234				Ok(get_automine(&rpc_client).await)
235			},)?;
236
237		let client = Self {
238			api,
239			rpc_client,
240			rpc,
241			receipt_provider,
242			block_provider,
243			fee_history_provider: FeeHistoryProvider::default(),
244			chain_id,
245			max_block_weight,
246			automine,
247			tx_notifier: automine.then(|| tokio::sync::broadcast::channel::<H256>(10).0),
248		};
249
250		Ok(client)
251	}

253	/// Subscribe to past blocks executing the callback for each block in `range`.
254	async fn subscribe_past_blocks<F, Fut>(
255		&self,
256		range: Range<SubstrateBlockNumber>,
257		callback: F,
258	) -> Result<(), ClientError>
259	where
260		F: Fn(Arc<SubstrateBlock>) -> Fut + Send + Sync,
261		Fut: std::future::Future<Output = Result<(), ClientError>> + Send,
262	{
263		let mut block = self
264			.block_provider
265			.block_by_number(range.end)
266			.await?
267			.ok_or(ClientError::BlockNotFound)?;
268
269		loop {
270			let block_number = block.number();
271			log::trace!(target: "eth-rpc::subscription", "Processing past block #{block_number}");
272
273			let parent_hash = block.header().parent_hash;
274			callback(block.clone()).await.inspect_err(|err| {
275				log::error!(target: "eth-rpc::subscription", "Failed to process past block #{block_number}: {err:?}");
276			})?;
277
278			if range.start < block_number {
279				block = self
280					.block_provider
281					.block_by_hash(&parent_hash)
282					.await?
283					.ok_or(ClientError::BlockNotFound)?;
284			} else {
285				return Ok(());
286			}
287		}
288	}

290	/// Subscribe to new blocks, and execute the async closure for each block.
291	async fn subscribe_new_blocks<F, Fut>(
292		&self,
293		subscription_type: SubscriptionType,
294		callback: F,
295	) -> Result<(), ClientError>
296	where
297		F: Fn(SubstrateBlock) -> Fut + Send + Sync,
298		Fut: std::future::Future<Output = Result<(), ClientError>> + Send,
299	{
300		let mut block_stream = match subscription_type {
301			SubscriptionType::BestBlocks => self.api.blocks().subscribe_best().await,
302			SubscriptionType::FinalizedBlocks => self.api.blocks().subscribe_finalized().await,
303		}
304		.inspect_err(|err| {
305			log::error!(target: LOG_TARGET, "Failed to subscribe to blocks: {err:?}");
306		})?;
307
308		while let Some(block) = block_stream.next().await {
309			let block = match block {
310				Ok(block) => block,
311				Err(err) => {
312					if err.is_disconnected_will_reconnect() {
313						log::warn!(
314							target: LOG_TARGET,
315							"The RPC connection was lost and we may have missed a few blocks ({subscription_type:?}): {err:?}"
316						);
317						continue;
318					}
319
320					log::error!(target: LOG_TARGET, "Failed to fetch block ({subscription_type:?}): {err:?}");
321					return Err(err.into());
322				},
323			};
324
325			let block_number = block.number();
326			log::trace!(target: "eth-rpc::subscription", "โณ Processing {subscription_type:?} block: {block_number}");
327			if let Err(err) = callback(block).await {
328				log::error!(target: LOG_TARGET, "Failed to process block {block_number}: {err:?}");
329			} else {
330				log::trace!(target: "eth-rpc::subscription", "โ Processed {subscription_type:?} block: {block_number}");
331			}
332		}
333
334		log::info!(target: LOG_TARGET, "Block subscription ended");
335		Ok(())
336	}

338	/// Start the block subscription, and populate the block cache.
339	pub async fn subscribe_and_cache_new_blocks(
340		&self,
341		subscription_type: SubscriptionType,
342	) -> Result<(), ClientError> {
343		log::info!(target: LOG_TARGET, "๐ Subscribing to new blocks ({subscription_type:?})");
344		self.subscribe_new_blocks(subscription_type, |block| async {
345			let (signed_txs, receipts): (Vec<_>, Vec<_>) =
346				self.receipt_provider.insert_block_receipts(&block).await?.into_iter().unzip();
347
348			let evm_block =
349				self.evm_block_from_receipts(&block, &receipts, signed_txs, false).await;
350			self.block_provider.update_latest(block, subscription_type).await;
351
352			self.fee_history_provider.update_fee_history(&evm_block, &receipts).await;
353
354			// Only broadcast for best blocks to avoid duplicate notifications.
355			match (subscription_type, &self.tx_notifier) {
356				(SubscriptionType::BestBlocks, Some(sender)) if sender.receiver_count() > 0 =>
357					for receipt in &receipts {
358						let _ = sender.send(receipt.transaction_hash);
359					},
360				_ => {},
361			}
362			Ok(())
363		})
364		.await
365	}

367	/// Cache old blocks up to the given block number.
368	pub async fn subscribe_and_cache_blocks(
369		&self,
370		index_last_n_blocks: SubstrateBlockNumber,
371	) -> Result<(), ClientError> {
372		let last = self.latest_block().await.number().saturating_sub(1);
373		let range = last.saturating_sub(index_last_n_blocks)..last;
374		log::info!(target: LOG_TARGET, "๐๏ธ Indexing past blocks in range {range:?}");
375		self.subscribe_past_blocks(range, |block| async move {
376			self.receipt_provider.insert_block_receipts(&block).await?;
377			Ok(())
378		})
379		.await?;
380
381		log::info!(target: LOG_TARGET, "๐๏ธ Finished indexing past blocks");
382		Ok(())
383	}

385	/// Get the block hash for the given block number or tag.
386	pub async fn block_hash_for_tag(
387		&self,
388		at: BlockNumberOrTagOrHash,
389	) -> Result<SubstrateBlockHash, ClientError> {
390		match at {
391			BlockNumberOrTagOrHash::BlockHash(hash) => Ok(hash),
392			BlockNumberOrTagOrHash::BlockNumber(block_number) => {
393				let n: SubstrateBlockNumber =
394					(block_number).try_into().map_err(|_| ClientError::ConversionFailed)?;
395				let hash = self.get_block_hash(n).await?.ok_or(ClientError::BlockNotFound)?;
396				Ok(hash)
397			},
398			BlockNumberOrTagOrHash::BlockTag(BlockTag::Finalized | BlockTag::Safe) => {
399				let block = self.latest_finalized_block().await;
400				Ok(block.hash())
401			},
402			BlockNumberOrTagOrHash::BlockTag(_) => {
403				let block = self.latest_block().await;
404				Ok(block.hash())
405			},
406		}
407	}

574	/// Get the transaction traces for the given block.
575	pub async fn trace_block_by_number(
576		&self,
577		at: BlockNumberOrTag,
578		config: TracerType,
579	) -> Result<Vec<TransactionTrace>, ClientError> {
580		if self.receipt_provider.is_before_earliest_block(&at) {
581			return Ok(vec![]);
582		}
583
584		let block_hash = self.block_hash_for_tag(at.into()).await?;
585		let block = self.tracing_block(block_hash).await?;
586		let parent_hash = block.header().parent_hash;
587		let runtime_api = RuntimeApi::new(self.api.runtime_api().at(parent_hash));
588		let traces = runtime_api.trace_block(block, config.clone()).await?;
589
590		let mut hashes = self
591			.receipt_provider
592			.block_transaction_hashes(&block_hash)
593			.await
594			.ok_or(ClientError::EthExtrinsicNotFound)?;
595
596		let traces = traces.into_iter().filter_map(|(index, trace)| {
597			Some(TransactionTrace { tx_hash: hashes.remove(&(index as usize))?, trace })
598		});
599
600		Ok(traces.collect())
601	}

606		transaction_hash: H256,

608	) -> Result<Trace, ClientError> {
609		let ReceiptInfo { block_hash, transaction_index, .. } = self
610			.receipt_provider
611			.receipt_by_hash(&transaction_hash)
612			.await
613			.ok_or(ClientError::EthExtrinsicNotFound)?;
614
615		let block = self.tracing_block(block_hash).await?;
616		let parent_hash = block.header.parent_hash;
617		let runtime_api = self.runtime_api(parent_hash);
618
619		runtime_api.trace_tx(block, transaction_index.as_u32(), config.clone()).await
620	}

634	/// Get the EVM block for the given Substrate block.
635	pub async fn evm_block(
636		&self,
637		block: Arc<SubstrateBlock>,
638		hydrated_transactions: bool,
639	) -> Block {
640		let (signed_txs, receipts): (Vec<_>, Vec<_>) = self
641			.receipt_provider
642			.receipts_from_block(&block)
643			.await
644			.unwrap_or_default()
645			.into_iter()
646			.unzip();
647		return self
648			.evm_block_from_receipts(&block, &receipts, signed_txs, hydrated_transactions)
649			.await
650	}

652	/// Get the EVM block for the given block and receipts.
653	pub async fn evm_block_from_receipts(
654		&self,
655		block: &SubstrateBlock,
656		receipts: &[ReceiptInfo],
657		signed_txs: Vec<TransactionSigned>,
658		hydrated_transactions: bool,
659	) -> Block {
660		let runtime_api = RuntimeApi::new(self.api.runtime_api().at(block.hash()));
661		let gas_limit = runtime_api.block_gas_limit().await.unwrap_or_default();
662
663		let header = block.header();
664		let timestamp = extract_block_timestamp(block).await.unwrap_or_default();
665		let block_author = runtime_api.block_author().await.ok().flatten().unwrap_or_default();
666
667		// TODO: remove once subxt is updated
668		let parent_hash = header.parent_hash.0.into();
669		let state_root = header.state_root.0.into();
670		let extrinsics_root = header.extrinsics_root.0.into();
671
672		let gas_used = receipts.iter().fold(U256::zero(), |acc, receipt| acc + receipt.gas_used);
673		let transactions = if hydrated_transactions {
674			signed_txs
675				.into_iter()
676				.zip(receipts.iter())
677				.map(|(signed_tx, receipt)| TransactionInfo::new(receipt, signed_tx))
678				.collect::<Vec<TransactionInfo>>()
679				.into()
680		} else {
681			receipts
682				.iter()
683				.map(|receipt| receipt.transaction_hash)
684				.collect::<Vec<_>>()
685				.into()
686		};
687
688		Block {
689			hash: block.hash(),
690			parent_hash,
691			state_root,
692			miner: block_author,
693			transactions_root: extrinsics_root,
694			number: header.number.into(),
695			timestamp: timestamp.into(),
696			base_fee_per_gas: runtime_api.gas_price().await.ok().unwrap_or_default(),
697			gas_limit,
698			gas_used,
699			receipts_root: extrinsics_root,
700			transactions,
701			..Default::default()
702		}
703	}

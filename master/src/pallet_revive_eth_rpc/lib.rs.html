56/// An EVM RPC server implementation.
57pub struct EthRpcServerImpl {
58	/// The client used to interact with the substrate node.
59	client: client::Client,
60
61	/// The accounts managed by the server.
62	accounts: Vec<Account>,
63}
64
65impl EthRpcServerImpl {
66	/// Creates a new [`EthRpcServerImpl`].
67	pub fn new(client: client::Client) -> Self {
68		Self { client, accounts: vec![] }
69	}
70
71	/// Sets the accounts managed by the server.
72	pub fn with_accounts(mut self, accounts: Vec<Account>) -> Self {
73		self.accounts = accounts;
74		self
75	}
76}

78/// The error type for the EVM RPC server.
79#[derive(Error, Debug)]
80pub enum EthRpcError {
81	/// A [`ClientError`] wrapper error.
82	#[error("Client error: {0}")]
83	ClientError(#[from] ClientError),
84	/// A [`rlp::DecoderError`] wrapper error.
85	#[error("Decoding error: {0}")]
86	RlpError(#[from] rlp::DecoderError),
87	/// A Decimals conversion error.
88	#[error("Conversion error")]
89	ConversionError,
90	/// An invalid signature error.
91	#[error("Invalid signature")]
92	InvalidSignature,
93	/// The account was not found at the given address
94	#[error("Account not found for address {0:?}")]
95	AccountNotFound(H160),
96	/// Received an invalid transaction
97	#[error("Invalid transaction")]
98	InvalidTransaction,
99	/// Received an invalid transaction
100	#[error("Invalid transaction {0:?}")]
101	TransactionTypeNotSupported(Byte),
102}
103
104// TODO use https://eips.ethereum.org/EIPS/eip-1474#error-codes
105impl From<EthRpcError> for ErrorObjectOwned {
106	fn from(value: EthRpcError) -> Self {
107		match value {
108			EthRpcError::ClientError(err) => Self::from(err),
109			_ => Self::owned::<String>(ErrorCode::InvalidRequest.code(), value.to_string(), None),
110		}
111	}
112}

165	async fn send_raw_transaction(&self, transaction: Bytes) -> RpcResult<H256> {
166		let hash = H256(keccak_256(&transaction.0));
167		let call = subxt_client::tx().revive().eth_transact(transaction.0);
168
169		// Subscribe to new block only when automine is enabled.
170		let receiver = self.client.tx_notifier().map(|sender| sender.subscribe());
171
172		// Submit the transaction
173		self.client.submit(call).await.map_err(|err| {
174			log::debug!(target: LOG_TARGET, "submit call failed: {err:?}");
175			err
176		})?;
177
178		// Wait for the transaction to be included in a block if automine is enabled
179		if let Some(mut receiver) = receiver {
180			if let Err(err) = tokio::time::timeout(Duration::from_millis(500), async {
181				loop {
182					if let Ok(mined_hash) = receiver.recv().await {
183						if mined_hash == hash {
184							log::debug!(target: LOG_TARGET, "{hash:} was included in a block");
185							break;
186						}
187					}
188				}
189			})
190			.await
191			{
192				log::debug!(target: LOG_TARGET, "timeout waiting for new block: {err:?}");
193			}
194		}
195
196		log::debug!(target: LOG_TARGET, "send_raw_transaction hash: {hash:?}");
197		Ok(hash)
198	}

200	async fn send_transaction(&self, mut transaction: GenericTransaction) -> RpcResult<H256> {
201		log::debug!(target: LOG_TARGET, "{transaction:#?}");
202
203		let Some(from) = transaction.from else {
204			log::debug!(target: LOG_TARGET, "Transaction must have a sender");
205			return Err(EthRpcError::InvalidTransaction.into());
206		};
207
208		let account = self
209			.accounts
210			.iter()
211			.find(|account| account.address() == from)
212			.ok_or(EthRpcError::AccountNotFound(from))?;
213
214		if transaction.gas.is_none() {
215			transaction.gas = Some(self.estimate_gas(transaction.clone(), None).await?);
216		}
217
218		if transaction.gas_price.is_none() {
219			transaction.gas_price = Some(self.gas_price().await?);
220		}
221
222		if transaction.nonce.is_none() {
223			transaction.nonce =
224				Some(self.get_transaction_count(from, BlockTag::Latest.into()).await?);
225		}
226
227		if transaction.chain_id.is_none() {
228			transaction.chain_id = Some(self.chain_id().await?);
229		}
230
231		let tx = transaction.try_into_unsigned().map_err(|_| EthRpcError::InvalidTransaction)?;
232		let payload = account.sign_transaction(tx).signed_payload();
233		self.send_raw_transaction(Bytes(payload)).await
234	}

336	async fn get_transaction_by_block_hash_and_index(
337		&self,
338		block_hash: H256,
339		transaction_index: U256,
340	) -> RpcResult<Option<TransactionInfo>> {
341		let Some(receipt) = self
342			.client
343			.receipt_by_hash_and_index(
344				&block_hash,
345				transaction_index.try_into().map_err(|_| EthRpcError::ConversionError)?,
346			)
347			.await
348		else {
349			return Ok(None);
350		};
351
352		let Some(signed_tx) = self.client.signed_tx_by_hash(&receipt.transaction_hash).await else {
353			return Ok(None);
354		};
355
356		Ok(Some(TransactionInfo::new(&receipt, signed_tx)))
357	}

360		&self,

362		transaction_index: U256,
363	) -> RpcResult<Option<TransactionInfo>> {
364		let Some(block) = self.client.block_by_number_or_tag(&block).await? else {
365			return Ok(None);
366		};
367		self.get_transaction_by_block_hash_and_index(block.hash(), transaction_index)
368			.await
369	}

371	async fn get_transaction_by_hash(
372		&self,
373		transaction_hash: H256,
374	) -> RpcResult<Option<TransactionInfo>> {
375		let receipt = self.client.receipt(&transaction_hash).await;
376		let signed_tx = self.client.signed_tx_by_hash(&transaction_hash).await;
377		if let (Some(receipt), Some(signed_tx)) = (receipt, signed_tx) {
378			return Ok(Some(TransactionInfo::new(&receipt, signed_tx)));
379		}
380
381		Ok(None)
382	}

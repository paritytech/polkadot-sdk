70/// Responsible for the voting strategy.
71/// It chooses which incoming votes to accept and which votes to generate.
72/// Keeps track of voting seen for current and future rounds.
73///
74/// Note: this is part of `PersistedState` so any changes here should also bump
75/// aux-db schema version.
76#[derive(Debug, Decode, Encode, PartialEq)]
77pub(crate) struct VoterOracle<B: Block, AuthorityId: AuthorityIdBound> {
78	/// Queue of known sessions. Keeps track of voting rounds (block numbers) within each session.
79	///
80	/// There are three voter states corresponding to three queue states:
81	/// 1. voter uninitialized: queue empty,
82	/// 2. up-to-date - all mandatory blocks leading up to current GRANDPA finalized: queue has ONE
83	///    element, the 'current session' where `mandatory_done == true`,
84	/// 3. lagging behind GRANDPA: queue has [1, N] elements, where all `mandatory_done == false`.
85	///    In this state, every time a session gets its mandatory block BEEFY finalized, it's
86	///    popped off the queue, eventually getting to state `2. up-to-date`.
87	sessions: VecDeque<Rounds<B, AuthorityId>>,
88	/// Min delta in block numbers between two blocks, BEEFY should vote on.
89	min_block_delta: u32,
90	/// Best block we received a GRANDPA finality for.
91	best_grandpa_block_header: <B as Block>::Header,
92	/// Best block a BEEFY voting round has been concluded for.
93	best_beefy_block: NumberFor<B>,
94	_phantom: PhantomData<fn() -> AuthorityId>,
95}
96
97impl<B: Block, AuthorityId> VoterOracle<B, AuthorityId>
98where
99	AuthorityId: AuthorityIdBound,
100{
101	/// Verify provided `sessions` satisfies requirements, then build `VoterOracle`.
102	pub fn checked_new(
103		sessions: VecDeque<Rounds<B, AuthorityId>>,
104		min_block_delta: u32,
105		grandpa_header: <B as Block>::Header,
106		best_beefy: NumberFor<B>,
107	) -> Option<Self> {
108		let mut prev_start = Zero::zero();
109		let mut prev_validator_id = None;
110		// verifies the
111		let mut validate = || -> bool {
112			let best_grandpa = *grandpa_header.number();
113			if sessions.is_empty() || best_beefy > best_grandpa {
114				return false;
115			}
116			for (idx, session) in sessions.iter().enumerate() {
117				let start = session.session_start();
118				if session.validators().is_empty() {
119					return false;
120				}
121				if start > best_grandpa || start <= prev_start {
122					return false;
123				}
124				#[cfg(not(test))]
125				if let Some(prev_id) = prev_validator_id {
126					if session.validator_set_id() <= prev_id {
127						return false;
128					}
129				}
130				if idx != 0 && session.mandatory_done() {
131					return false;
132				}
133				prev_start = session.session_start();
134				prev_validator_id = Some(session.validator_set_id());
135			}
136			true
137		};
138		if validate() {
139			Some(VoterOracle {
140				sessions,
141				// Always target at least one block better than current best beefy.
142				min_block_delta: min_block_delta.max(1),
143				best_grandpa_block_header: grandpa_header,
144				best_beefy_block: best_beefy,
145				_phantom: PhantomData,
146			})
147		} else {
148			error!(
149				target: LOG_TARGET,
150				"游볼 Invalid sessions queue: {:?}; best-beefy {:?} best-grandpa-header {:?}.",
151				sessions,
152				best_beefy,
153				grandpa_header
154			);
155			None
156		}
157	}
158
159	// Return reference to rounds pertaining to first session in the queue.
160	// Voting will always happen at the head of the queue.
161	fn active_rounds(&self) -> Result<&Rounds<B, AuthorityId>, Error> {
162		self.sessions.front().ok_or(Error::UninitSession)
163	}
164
165	// Return mutable reference to rounds pertaining to first session in the queue.
166	// Voting will always happen at the head of the queue.
167	fn active_rounds_mut(&mut self) -> Result<&mut Rounds<B, AuthorityId>, Error> {
168		self.sessions.front_mut().ok_or(Error::UninitSession)
169	}
170
171	fn current_validator_set(&self) -> Result<&ValidatorSet<AuthorityId>, Error> {
172		self.active_rounds().map(|r| r.validator_set())
173	}
174
175	// Prune the sessions queue to keep the Oracle in one of the expected three states.
176	//
177	// To be called on each BEEFY finality and on each new rounds/session addition.
178	fn try_prune(&mut self) {
179		if self.sessions.len() > 1 {
180			// when there's multiple sessions, only keep the `!mandatory_done()` ones.
181			self.sessions.retain(|s| !s.mandatory_done())
182		}
183	}
184
185	/// Check if an observed session can be added to the Oracle.
186	pub fn can_add_session(&self, session_start: NumberFor<B>) -> bool {
187		let latest_known_session_start =
188			self.sessions.back().map(|session| session.session_start());
189		Some(session_start) > latest_known_session_start
190	}
191
192	/// Add new observed session to the Oracle.
193	pub fn add_session(&mut self, rounds: Rounds<B, AuthorityId>) {
194		self.sessions.push_back(rounds);
195		// Once we add a new session we can drop/prune previous session if it's been finalized.
196		self.try_prune();
197	}
198
199	/// Finalize a particular block.
200	pub fn finalize(&mut self, block: NumberFor<B>) -> Result<(), Error> {
201		// Conclude voting round for this block.
202		self.active_rounds_mut()?.conclude(block);
203		// Prune any now "finalized" sessions from queue.
204		self.try_prune();
205		Ok(())
206	}
207
208	/// Return current pending mandatory block, if any, plus its active validator set.
209	pub fn mandatory_pending(&self) -> Option<(NumberFor<B>, ValidatorSet<AuthorityId>)> {
210		self.sessions.front().and_then(|round| {
211			if round.mandatory_done() {
212				None
213			} else {
214				Some((round.session_start(), round.validator_set().clone()))
215			}
216		})
217	}
218
219	/// Return `(A, B)` tuple representing inclusive [A, B] interval of votes to accept.
220	pub fn accepted_interval(&self) -> Result<(NumberFor<B>, NumberFor<B>), Error> {
221		let rounds = self.sessions.front().ok_or(Error::UninitSession)?;
222
223		if rounds.mandatory_done() {
224			// There's only one session active and its mandatory is done.
225			// Accept any vote for a GRANDPA finalized block in a better round.
226			Ok((
227				rounds.session_start().max(self.best_beefy_block),
228				(*self.best_grandpa_block_header.number()),
229			))
230		} else {
231			// Current session has mandatory not done.
232			// Only accept votes for the mandatory block in the front of queue.
233			Ok((rounds.session_start(), rounds.session_start()))
234		}
235	}
236
237	/// Utility function to quickly decide what to do for each round.
238	pub fn triage_round(&self, round: NumberFor<B>) -> Result<RoundAction, Error> {
239		let (start, end) = self.accepted_interval()?;
240		if start <= round && round <= end {
241			Ok(RoundAction::Process)
242		} else if round > end {
243			Ok(RoundAction::Enqueue)
244		} else {
245			Ok(RoundAction::Drop)
246		}
247	}
248
249	/// Return `Some(number)` if we should be voting on block `number`,
250	/// return `None` if there is no block we should vote on.
251	pub fn voting_target(&self) -> Option<NumberFor<B>> {
252		let rounds = self.sessions.front().or_else(|| {
253			debug!(target: LOG_TARGET, "游볼 No voting round started");
254			None
255		})?;
256		let best_grandpa = *self.best_grandpa_block_header.number();
257		let best_beefy = self.best_beefy_block;
258
259		// `target` is guaranteed > `best_beefy` since `min_block_delta` is at least `1`.
260		let target =
261			vote_target(best_grandpa, best_beefy, rounds.session_start(), self.min_block_delta);
262		trace!(
263			target: LOG_TARGET,
264			"游볼 best beefy: #{:?}, best finalized: #{:?}, current_vote_target: {:?}",
265			best_beefy,
266			best_grandpa,
267			target
268		);
269		target
270	}
271}

336	/// Handle session changes by starting new voting round for mandatory blocks.
337	pub fn init_session_at(
338		&mut self,
339		new_session_start: NumberFor<B>,
340		validator_set: ValidatorSet<AuthorityId>,
341		key_store: &BeefyKeystore<AuthorityId>,
342		metrics: &Option<VoterMetrics>,
343		is_authority: bool,
344	) {
345		debug!(target: LOG_TARGET, "游볼 New active validator set: {:?}", validator_set);
346
347		// BEEFY should finalize a mandatory block during each session.
348		if let Ok(active_session) = self.voting_oracle.active_rounds() {
349			if !active_session.mandatory_done() {
350				debug!(
351					target: LOG_TARGET,
352					"游볼 New session {} while active session {} is still lagging.",
353					validator_set.id(),
354					active_session.validator_set_id(),
355				);
356				metric_inc!(metrics, beefy_lagging_sessions);
357			}
358		}
359
360		// verify we have some BEEFY key available in keystore when role is authority.
361		if is_authority && key_store.public_keys().map_or(false, |k| k.is_empty()) {
362			error!(
363				target: LOG_TARGET,
364				"游볼 for session starting at block {:?} no BEEFY authority key found in store, \
365				you must generate valid session keys \
366				(https://wiki.polkadot.network/docs/maintain-guides-how-to-validate-polkadot#generating-the-session-keys)",
367				new_session_start,
368			);
369			metric_inc!(metrics, beefy_no_authority_found_in_store);
370		}
371
372		let id = validator_set.id();
373		self.voting_oracle.add_session(Rounds::new(new_session_start, validator_set));
374		metric_set!(metrics, beefy_validator_set_id, id);
375		info!(
376			target: LOG_TARGET,
377			"游볼 New Rounds for validator set id: {:?} with session_start {:?}",
378			id,
379			new_session_start
380		);
381	}

450	fn handle_finality_notification(
451		&mut self,
452		notification: &UnpinnedFinalityNotification<B>,
453	) -> Result<(), Error> {
454		let header = &notification.header;
455		debug!(
456			target: LOG_TARGET,
457			"游볼 Finality notification: header(number {:?}, hash {:?}) tree_route {:?}",
458			header.number(),
459			notification.hash,
460			notification.tree_route,
461		);
462
463		match self.runtime.runtime_api().beefy_genesis(notification.hash) {
464			Ok(Some(genesis)) if genesis != self.persisted_state.pallet_genesis => {
465				debug!(target: LOG_TARGET, "游볼 ConsensusReset detected. Expected genesis: {}, found genesis: {}", self.persisted_state.pallet_genesis, genesis);
466				return Err(Error::ConsensusReset)
467			},
468			Ok(_) => {},
469			Err(api_error) => {
470				// This can happen in case the block was already pruned.
471				// Mostly after warp sync when finality notifications are piled up.
472				debug!(target: LOG_TARGET, "游볼 Unable to check beefy genesis: {}", api_error);
473			},
474		}
475
476		let mut new_session_added = false;
477		if *header.number() > self.best_grandpa_block() {
478			// update best GRANDPA finalized block we have seen
479			self.persisted_state.set_best_grandpa(header.clone());
480
481			// Check all (newly) finalized blocks for new session(s).
482			let backend = self.backend.clone();
483			for header in notification
484				.tree_route
485				.iter()
486				.map(|hash| {
487					backend
488						.blockchain()
489						.expect_header(*hash)
490						.expect("just finalized block should be available; qed.")
491				})
492				.chain(std::iter::once(header.clone()))
493			{
494				if let Some(new_validator_set) = find_authorities_change::<B, AuthorityId>(&header)
495				{
496					self.init_session_at(new_validator_set, *header.number());
497					new_session_added = true;
498				}
499			}
500
501			if new_session_added {
502				crate::aux_schema::write_voter_state(&*self.backend, &self.persisted_state)
503					.map_err(|e| Error::Backend(e.to_string()))?;
504			}
505
506			// Update gossip validator votes filter.
507			if let Err(e) = self
508				.persisted_state
509				.gossip_filter_config()
510				.map(|filter| self.comms.gossip_validator.update_filter(filter))
511			{
512				error!(target: LOG_TARGET, "游볼 Voter error: {:?}", e);
513			}
514		}
515
516		Ok(())
517	}

519	/// Based on [VoterOracle] this vote is either processed here or discarded.
520	fn triage_incoming_vote(
521		&mut self,
522		vote: VoteMessage<NumberFor<B>, AuthorityId, <AuthorityId as RuntimeAppPublic>::Signature>,
523	) -> Result<(), Error>
524	where
525		<AuthorityId as RuntimeAppPublic>::Signature: Encode + Decode,
526	{
527		let block_num = vote.commitment.block_number;
528		match self.voting_oracle().triage_round(block_num)? {
529			RoundAction::Process =>
530				if let Some(finality_proof) = self.handle_vote(vote)? {
531					let gossip_proof =
532						GossipMessage::<B, AuthorityId>::FinalityProof(finality_proof);
533					let encoded_proof = gossip_proof.encode();
534					self.comms.gossip_engine.gossip_message(
535						proofs_topic::<B>(),
536						encoded_proof,
537						true,
538					);
539				},
540			RoundAction::Drop => metric_inc!(self.metrics, beefy_stale_votes),
541			RoundAction::Enqueue => error!(target: LOG_TARGET, "游볼 unexpected vote: {:?}.", vote),
542		};
543		Ok(())
544	}

546	/// Based on [VoterOracle] this justification is either processed here or enqueued for later.
547	///
548	/// Expects `justification` to be valid.
549	fn triage_incoming_justif(
550		&mut self,
551		justification: BeefyVersionedFinalityProof<B, AuthorityId>,
552	) -> Result<(), Error> {
553		let signed_commitment = match justification {
554			VersionedFinalityProof::V1(ref sc) => sc,
555		};
556		let block_num = signed_commitment.commitment.block_number;
557		match self.voting_oracle().triage_round(block_num)? {
558			RoundAction::Process => {
559				debug!(target: LOG_TARGET, "游볼 Process justification for round: {:?}.", block_num);
560				metric_inc!(self.metrics, beefy_imported_justifications);
561				self.finalize(justification)?
562			},
563			RoundAction::Enqueue => {
564				debug!(target: LOG_TARGET, "游볼 Buffer justification for round: {:?}.", block_num);
565				if self.pending_justifications.len() < MAX_BUFFERED_JUSTIFICATIONS {
566					self.pending_justifications.entry(block_num).or_insert(justification);
567					metric_inc!(self.metrics, beefy_buffered_justifications);
568				} else {
569					metric_inc!(self.metrics, beefy_buffered_justifications_dropped);
570					warn!(
571						target: LOG_TARGET,
572						"游볼 Buffer justification dropped for round: {:?}.", block_num
573					);
574				}
575			},
576			RoundAction::Drop => metric_inc!(self.metrics, beefy_stale_justifications),
577		};
578		Ok(())
579	}

581	fn handle_vote(
582		&mut self,
583		vote: VoteMessage<NumberFor<B>, AuthorityId, <AuthorityId as RuntimeAppPublic>::Signature>,
584	) -> Result<Option<BeefyVersionedFinalityProof<B, AuthorityId>>, Error> {
585		let rounds = self.persisted_state.voting_oracle.active_rounds_mut()?;
586
587		let block_number = vote.commitment.block_number;
588		match rounds.add_vote(vote) {
589			VoteImportResult::RoundConcluded(signed_commitment) => {
590				let finality_proof = VersionedFinalityProof::V1(signed_commitment);
591				debug!(
592					target: LOG_TARGET,
593					"游볼 Round #{} concluded, finality_proof: {:?}.", block_number, finality_proof
594				);
595				// We created the `finality_proof` and know to be valid.
596				// New state is persisted after finalization.
597				self.finalize(finality_proof.clone())?;
598				metric_inc!(self.metrics, beefy_good_votes_processed);
599				return Ok(Some(finality_proof));
600			},
601			VoteImportResult::Ok => {
602				// Persist state after handling mandatory block vote.
603				if self
604					.voting_oracle()
605					.mandatory_pending()
606					.map(|(mandatory_num, _)| mandatory_num == block_number)
607					.unwrap_or(false)
608				{
609					crate::aux_schema::write_voter_state(&*self.backend, &self.persisted_state)
610						.map_err(|e| Error::Backend(e.to_string()))?;
611				}
612				metric_inc!(self.metrics, beefy_good_votes_processed);
613			},
614			VoteImportResult::DoubleVoting(proof) => {
615				metric_inc!(self.metrics, beefy_equivocation_votes);
616				self.report_double_voting(proof)?;
617			},
618			VoteImportResult::Invalid => metric_inc!(self.metrics, beefy_invalid_votes),
619			VoteImportResult::Stale => metric_inc!(self.metrics, beefy_stale_votes),
620		};
621		Ok(None)
622	}

624	/// Provide BEEFY finality for block based on `finality_proof`:
625	/// 1. Prune now-irrelevant past sessions from the oracle,
626	/// 2. Set BEEFY best block,
627	/// 3. Persist voter state,
628	/// 4. Send best block hash and `finality_proof` to RPC worker.
629	///
630	/// Expects `finality proof` to be valid and for a block > current-best-beefy.
631	fn finalize(
632		&mut self,
633		finality_proof: BeefyVersionedFinalityProof<B, AuthorityId>,
634	) -> Result<(), Error> {
635		let block_num = match finality_proof {
636			VersionedFinalityProof::V1(ref sc) => sc.commitment.block_number,
637		};
638
639		if block_num <= self.persisted_state.voting_oracle.best_beefy_block {
640			// we've already finalized this round before, short-circuit.
641			return Ok(());
642		}
643
644		// Finalize inner round and update voting_oracle state.
645		self.persisted_state.voting_oracle.finalize(block_num)?;
646
647		// Set new best BEEFY block number.
648		self.persisted_state.set_best_beefy(block_num);
649		crate::aux_schema::write_voter_state(&*self.backend, &self.persisted_state)
650			.map_err(|e| Error::Backend(e.to_string()))?;
651
652		metric_set!(self.metrics, beefy_best_block, block_num);
653
654		self.comms.on_demand_justifications.cancel_requests_older_than(block_num);
655
656		if let Err(e) = self
657			.backend
658			.blockchain()
659			.expect_block_hash_from_id(&BlockId::Number(block_num))
660			.and_then(|hash| {
661				self.links
662					.to_rpc_best_block_sender
663					.notify(|| Ok::<_, ()>(hash))
664					.expect("forwards closure result; the closure always returns Ok; qed.");
665
666				self.backend
667					.append_justification(hash, (BEEFY_ENGINE_ID, finality_proof.encode()))
668			}) {
669			debug!(
670				target: LOG_TARGET,
671				"游볼 Error {:?} on appending justification: {:?}", e, finality_proof
672			);
673		}
674
675		self.links
676			.to_rpc_justif_sender
677			.notify(|| Ok::<_, ()>(finality_proof))
678			.expect("forwards closure result; the closure always returns Ok; qed.");
679
680		// Update gossip validator votes filter.
681		self.persisted_state
682			.gossip_filter_config()
683			.map(|filter| self.comms.gossip_validator.update_filter(filter))?;
684		Ok(())
685	}

687	/// Handle previously buffered justifications, that now land in the voting interval.
688	fn try_pending_justifications(&mut self) -> Result<(), Error> {
689		// Interval of blocks for which we can process justifications and votes right now.
690		let (start, end) = self.voting_oracle().accepted_interval()?;
691		// Process pending justifications.
692		if !self.pending_justifications.is_empty() {
693			// These are still pending.
694			let still_pending =
695				self.pending_justifications.split_off(&end.saturating_add(1u32.into()));
696			// These can be processed.
697			let justifs_to_process = self.pending_justifications.split_off(&start);
698			// The rest can be dropped.
699			self.pending_justifications = still_pending;
700
701			for (num, justification) in justifs_to_process.into_iter() {
702				debug!(target: LOG_TARGET, "游볼 Handle buffered justification for: {:?}.", num);
703				metric_inc!(self.metrics, beefy_imported_justifications);
704				if let Err(err) = self.finalize(justification) {
705					error!(target: LOG_TARGET, "游볼 Error finalizing block: {}", err);
706				}
707			}
708			metric_set!(
709				self.metrics,
710				beefy_buffered_justifications,
711				self.pending_justifications.len()
712			);
713		}
714		Ok(())
715	}

717	/// Decide if should vote, then vote.. or don't..
718	fn try_to_vote(&mut self) -> Result<(), Error> {
719		// Vote if there's now a new vote target.
720		if let Some(target) = self.voting_oracle().voting_target() {
721			metric_set!(self.metrics, beefy_should_vote_on, target);
722			if target > self.persisted_state.best_voted {
723				self.do_vote(target)?;
724			}
725		}
726		Ok(())
727	}

729	/// Create and gossip Signed Commitment for block number `target_number`.
730	///
731	/// Also handle this self vote by calling `self.handle_vote()` for it.
732	fn do_vote(&mut self, target_number: NumberFor<B>) -> Result<(), Error> {
733		debug!(target: LOG_TARGET, "游볼 Try voting on {}", target_number);
734
735		// Most of the time we get here, `target` is actually `best_grandpa`,
736		// avoid getting header from backend in that case.
737		let target_header = if target_number == self.best_grandpa_block() {
738			self.persisted_state.voting_oracle.best_grandpa_block_header.clone()
739		} else {
740			let hash = self
741				.backend
742				.blockchain()
743				.expect_block_hash_from_id(&BlockId::Number(target_number))
744				.map_err(|err| {
745					let err_msg = format!(
746						"Couldn't get hash for block #{:?} (error: {:?}), skipping vote..",
747						target_number, err
748					);
749					Error::Backend(err_msg)
750				})?;
751
752			self.backend.blockchain().expect_header(hash).map_err(|err| {
753				let err_msg = format!(
754					"Couldn't get header for block #{:?} ({:?}) (error: {:?}), skipping vote..",
755					target_number, hash, err
756				);
757				Error::Backend(err_msg)
758			})?
759		};
760		let target_hash = target_header.hash();
761
762		let payload = if let Some(hash) = self.payload_provider.payload(&target_header) {
763			hash
764		} else {
765			warn!(target: LOG_TARGET, "游볼 No MMR root digest found for: {:?}", target_hash);
766			return Ok(());
767		};
768
769		let rounds = self.persisted_state.voting_oracle.active_rounds_mut()?;
770		let (validators, validator_set_id) = (rounds.validators(), rounds.validator_set_id());
771
772		let authority_id = if let Some(id) = self.key_store.authority_id(validators) {
773			debug!(target: LOG_TARGET, "游볼 Local authority id: {:?}", id);
774			id
775		} else {
776			debug!(
777				target: LOG_TARGET,
778				"游볼 Missing validator id - can't vote for: {:?}", target_hash
779			);
780			return Ok(());
781		};
782
783		let commitment = Commitment { payload, block_number: target_number, validator_set_id };
784		let encoded_commitment = commitment.encode();
785
786		let signature = match self.key_store.sign(&authority_id, &encoded_commitment) {
787			Ok(sig) => sig,
788			Err(err) => {
789				warn!(target: LOG_TARGET, "游볼 Error signing commitment: {:?}", err);
790				return Ok(());
791			},
792		};
793
794		trace!(
795			target: LOG_TARGET,
796			"游볼 Produced signature using {:?}, is_valid: {:?}",
797			authority_id,
798			BeefyKeystore::verify(&authority_id, &signature, &encoded_commitment)
799		);
800
801		let vote = VoteMessage { commitment, id: authority_id, signature };
802		if let Some(finality_proof) = self.handle_vote(vote.clone()).map_err(|err| {
803			error!(target: LOG_TARGET, "游볼 Error handling self vote: {}", err);
804			err
805		})? {
806			let encoded_proof =
807				GossipMessage::<B, AuthorityId>::FinalityProof(finality_proof).encode();
808			self.comms
809				.gossip_engine
810				.gossip_message(proofs_topic::<B>(), encoded_proof, true);
811		} else {
812			metric_inc!(self.metrics, beefy_votes_sent);
813			debug!(target: LOG_TARGET, "游볼 Sent vote message: {:?}", vote);
814			let encoded_vote = GossipMessage::<B, AuthorityId>::Vote(vote).encode();
815			self.comms.gossip_engine.gossip_message(votes_topic::<B>(), encoded_vote, false);
816		}
817
818		// Persist state after vote to avoid double voting in case of voter restarts.
819		self.persisted_state.best_voted = target_number;
820		metric_set!(self.metrics, beefy_best_voted, target_number);
821		crate::aux_schema::write_voter_state(&*self.backend, &self.persisted_state)
822			.map_err(|e| Error::Backend(e.to_string()))
823	}
824
825	fn process_new_state(&mut self) {
826		// Handle pending justifications and/or votes for now GRANDPA finalized blocks.
827		if let Err(err) = self.try_pending_justifications() {
828			debug!(target: LOG_TARGET, "游볼 {}", err);
829		}
830
831		// Don't bother voting or requesting justifications during major sync.
832		if !self.sync.is_major_syncing() {
833			// There were external events, 'state' is changed, author a vote if needed/possible.
834			if let Err(err) = self.try_to_vote() {
835				debug!(target: LOG_TARGET, "游볼 {}", err);
836			}
837			// If the current target is a mandatory block,
838			// make sure there's also an on-demand justification request out for it.
839			if let Some((block, active)) = self.voting_oracle().mandatory_pending() {
840				// This only starts new request if there isn't already an active one.
841				self.comms.on_demand_justifications.request(block, active);
842			}
843		}
844	}

846	/// Main loop for BEEFY worker.
847	///
848	/// Run the main async loop which is driven by finality notifications and gossiped votes.
849	/// Should never end, returns `Error` otherwise.
850	pub(crate) async fn run(
851		mut self,
852		block_import_justif: &mut Fuse<
853			NotificationReceiver<BeefyVersionedFinalityProof<B, AuthorityId>>,
854		>,
855		finality_notifications: &mut Fuse<crate::FinalityNotifications<B>>,
856	) -> (Error, BeefyComms<B, N, AuthorityId>) {
857		info!(
858			target: LOG_TARGET,
859			"游볼 run BEEFY worker, best grandpa: #{:?}.",
860			self.best_grandpa_block()
861		);
862
863		let mut votes = Box::pin(
864			self.comms
865				.gossip_engine
866				.messages_for(votes_topic::<B>())
867				.filter_map(|notification| async move {
868					let vote =
869						GossipMessage::<B, AuthorityId>::decode_all(&mut &notification.message[..])
870							.ok()
871							.and_then(|message| message.unwrap_vote());
872					trace!(target: LOG_TARGET, "游볼 Got vote message: {:?}", vote);
873					vote
874				})
875				.fuse(),
876		);
877		let mut gossip_proofs = Box::pin(
878			self.comms
879				.gossip_engine
880				.messages_for(proofs_topic::<B>())
881				.filter_map(|notification| async move {
882					let proof =
883						GossipMessage::<B, AuthorityId>::decode_all(&mut &notification.message[..])
884							.ok()
885							.and_then(|message| message.unwrap_finality_proof());
886					trace!(target: LOG_TARGET, "游볼 Got gossip proof message: {:?}", proof);
887					proof
888				})
889				.fuse(),
890		);
891
892		self.process_new_state();
893		let error = loop {
894			// Mutable reference used to drive the gossip engine.
895			let mut gossip_engine = &mut self.comms.gossip_engine;
896
897			// Wait for, and handle external events.
898			// The branches below only change 'state', actual voting happens afterwards,
899			// based on the new resulting 'state'.
900			futures::select_biased! {
901				// Use `select_biased!` to prioritize order below.
902				// Process finality notifications first since these drive the voter.
903				notification = finality_notifications.next() => {
904					if let Some(notif) = notification {
905						if let Err(err) = self.handle_finality_notification(&notif) {
906							break err;
907						}
908					} else {
909						break Error::FinalityStreamTerminated;
910					}
911				},
912				// Make sure to pump gossip engine.
913				_ = gossip_engine => {
914					break Error::GossipEngineTerminated;
915				},
916				// Process incoming justifications as these can make some in-flight votes obsolete.
917				response_info = self.comms.on_demand_justifications.next().fuse() => {
918					match response_info {
919						ResponseInfo::ValidProof(justif, peer_report) => {
920							if let Err(err) = self.triage_incoming_justif(justif) {
921								debug!(target: LOG_TARGET, "游볼 {}", err);
922							}
923							self.comms.gossip_engine.report(peer_report.who, peer_report.cost_benefit);
924						},
925						ResponseInfo::PeerReport(peer_report) => {
926							self.comms.gossip_engine.report(peer_report.who, peer_report.cost_benefit);
927						},
928						ResponseInfo::Pending => {},
929					}
930				},
931				justif = block_import_justif.next() => {
932					if let Some(justif) = justif {
933						// Block import justifications have already been verified to be valid
934						// by `BeefyBlockImport`.
935						if let Err(err) = self.triage_incoming_justif(justif) {
936							debug!(target: LOG_TARGET, "游볼 {}", err);
937						}
938					} else {
939						break Error::BlockImportStreamTerminated;
940					}
941				},
942				justif = gossip_proofs.next() => {
943					if let Some(justif) = justif {
944						// Gossiped justifications have already been verified by `GossipValidator`.
945						if let Err(err) = self.triage_incoming_justif(justif) {
946							debug!(target: LOG_TARGET, "游볼 {}", err);
947						}
948					} else {
949						break Error::FinalityProofGossipStreamTerminated;
950					}
951				},
952				// Finally process incoming votes.
953				vote = votes.next() => {
954					if let Some(vote) = vote {
955						// Votes have already been verified to be valid by the gossip validator.
956						if let Err(err) = self.triage_incoming_vote(vote) {
957							debug!(target: LOG_TARGET, "游볼 {}", err);
958						}
959					} else {
960						break Error::VotesGossipStreamTerminated;
961					}
962				},
963			}
964
965			// Act on changed 'state'.
966			self.process_new_state();
967		};
968
969		// return error _and_ `comms` that can be reused
970		(error, self.comms)
971	}

987/// Calculate next block number to vote on.
988///
989/// Return `None` if there is no votable target yet.
990fn vote_target<N>(best_grandpa: N, best_beefy: N, session_start: N, min_delta: u32) -> Option<N>
991where
992	N: AtLeast32Bit + Copy + Debug,
993{
994	// if the mandatory block (session_start) does not have a beefy justification yet,
995	// we vote on it
996	let target = if best_beefy < session_start {
997		debug!(target: LOG_TARGET, "游볼 vote target - mandatory block: #{:?}", session_start);
998		session_start
999	} else {
1000		let diff = best_grandpa.saturating_sub(best_beefy) + 1u32.into();
1001		let diff = diff.saturated_into::<u32>() / 2;
1002		let target = best_beefy + min_delta.max(diff.next_power_of_two()).into();
1003		trace!(
1004			target: LOG_TARGET,
1005			"游볼 vote target - diff: {:?}, next_power_of_two: {:?}, target block: #{:?}",
1006			diff,
1007			diff.next_power_of_two(),
1008			target,
1009		);
1010
1011		target
1012	};
1013
1014	// Don't vote for targets until they've been finalized
1015	// (`target` can be > `best_grandpa` when `min_delta` is big enough).
1016	if target > best_grandpa {
1017		None
1018	} else {
1019		Some(target)
1020	}
1021}

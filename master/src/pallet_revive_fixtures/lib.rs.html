28/// Enum for different fixture types
29#[derive(Debug, Clone, Copy, PartialEq, Eq)]
30pub enum FixtureType {
31	/// Polkavm (compiled Rust contracts)
32	Rust,
33	/// Resolc (compiled Solidity contracts to Polkavm)
34	Resolc,
35	/// Solc (compiled Solidity contracts to EVM bytecode)
36	Solc,
37	/// Solc Runtime (compiled Solidity contracts to EVM runtime bytecode)
38	SolcRuntime,
39}
40
41#[cfg(feature = "std")]
42impl FixtureType {
43	fn file_extension(&self) -> &'static str {
44		match self {
45			Self::Rust => ".polkavm",
46			Self::Resolc => ".resolc.polkavm",
47			Self::Solc => ".sol.bin",
48			Self::SolcRuntime => ".sol.runtime.bin",
49		}
50	}
51}

53/// Load a fixture module with the specified type and return binary contents along with its hash.
54#[cfg(feature = "std")]
55pub fn compile_module_with_type(
56	fixture_name: &str,
57	fixture_type: FixtureType,
58) -> anyhow::Result<(Vec<u8>, sp_core::H256)> {
59	use anyhow::Context;
60	let out_dir: std::path::PathBuf = FIXTURE_DIR.into();
61	let fixture_path = out_dir.join(format!("{fixture_name}{}", fixture_type.file_extension()));
62	let binary = std::fs::read(&fixture_path)
63		.with_context(|| format!("Failed to load fixture {fixture_path:?}"))?;
64	let code_hash = sp_io::hashing::keccak_256(&binary);
65	Ok((binary, sp_core::H256(code_hash)))
66}
67
68/// Load a given polkavm module and returns a polkavm binary contents along with its hash.
69#[cfg(feature = "std")]
70pub fn compile_module(fixture_name: &str) -> anyhow::Result<(Vec<u8>, sp_core::H256)> {
71	compile_module_with_type(fixture_name, FixtureType::Rust)
72}

74/// Try to compile a fixture from the contracts_not_buildable directory.
75/// This replicates the build process but expects it to fail at the linking stage.
76/// Returns Err if compilation/linking fails (which is expected for invalid fixtures).
77#[cfg(feature = "std")]
78pub fn try_compile_invalid_fixture(fixture_name: &str) -> anyhow::Result<Vec<u8>> {
79	use anyhow::bail;
80	use std::{fs, path::PathBuf};
81
82	let manifest_dir = env!("CARGO_MANIFEST_DIR");
83	let fixture_path = PathBuf::from(manifest_dir)
84		.join("contracts_not_buildable")
85		.join(format!("{}.rs", fixture_name));
86
87	if !fixture_path.exists() {
88		bail!("Invalid fixture not found: {:?}", fixture_path);
89	}
90
91	// Create temporary build directory
92	let temp_dir = std::env::temp_dir().join(format!("revive_invalid_fixture_{}", fixture_name));
93	fs::create_dir_all(&temp_dir)?;
94
95	// Use the shared builder module to compile
96	let result = builder::compile_rust_to_polkavm(&fixture_path, fixture_name, &temp_dir);
97
98	// Cleanup
99	let _ = fs::remove_dir_all(&temp_dir);
100
101	result
102}

104/// Fixtures used in runtime benchmarks.
105///
106/// We explicitly include those fixtures into the binary to make them
107/// available in no-std environments (runtime benchmarks).
108pub mod bench {
109	use alloc::vec::Vec;
110	pub const DUMMY: Option<&[u8]> = fixture!("dummy");
111	pub const NOOP: Option<&[u8]> = fixture!("noop");
112
113	pub fn dummy() -> &'static [u8] {
114		DUMMY.expect("`DUMMY` fixture not available, remove `SKIP_PALLET_REVIVE_FIXTURES` env variable to compile them.")
115	}
116
117	pub fn noop() -> &'static [u8] {
118		NOOP.expect("`NOOP` fixture not available, remove `SKIP_PALLET_REVIVE_FIXTURES` env variable to compile them.")
119	}
120
121	pub fn dummy_unique(replace_with: u32) -> Vec<u8> {
122		let mut dummy = dummy().to_vec();
123		let idx = dummy
124			.windows(4)
125			.position(|w| w == &[0xDE, 0xAD, 0xBE, 0xEF])
126			.expect("Benchmark fixture contains this pattern; qed");
127		dummy[idx..idx + 4].copy_from_slice(&replace_with.to_le_bytes());
128		dummy
129	}
130}

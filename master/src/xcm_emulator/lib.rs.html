457macro_rules! __impl_test_ext_for_relay_chain {
458	// entry point: generate ext name
459	($name:ident, $network:ident, $genesis:expr, $on_init:expr, $api_version:tt) => {
460		$crate::paste::paste! {
461			$crate::__impl_test_ext_for_relay_chain!(
462				@impl $name,
463				$network,
464				$genesis,
465				$on_init,
466				[<ParachainHostV $api_version>],
467				[<LOCAL_EXT_ $name:upper>],
468				[<GLOBAL_EXT_ $name:upper>]
469			);
470		}
471	};
472	// impl
473	(@impl $name:ident, $network:ident, $genesis:expr, $on_init:expr, $api_version:ident, $local_ext:ident, $global_ext:ident) => {
474		thread_local! {
475			pub static $local_ext: $crate::RefCell<$crate::TestExternalities>
476				= $crate::RefCell::new($crate::TestExternalities::new($genesis));
477		}
478
479		pub static $global_ext: $crate::LazyLock<$crate::Mutex<$crate::RefCell<$crate::HashMap<String, $crate::TestExternalities>>>>
480			= $crate::LazyLock::new(|| $crate::Mutex::new($crate::RefCell::new($crate::HashMap::new())));
481
482		impl<$network: $crate::Network> $crate::TestExt for $name<$network> {
483			fn build_new_ext(storage: $crate::Storage) -> $crate::TestExternalities {
484				use $crate::{sp_tracing, Network, Chain, TestExternalities};
485
486				let mut ext = TestExternalities::new(storage);
487
488				ext.execute_with(|| {
489					#[allow(clippy::no_effect)]
490					$on_init;
491					sp_tracing::try_init_simple();
492
493					let mut block_number = <Self as Chain>::System::block_number();
494					block_number = std::cmp::max(1, block_number);
495					<Self as Chain>::System::set_block_number(block_number);
496				});
497				ext
498			}
499
500			fn new_ext() -> $crate::TestExternalities {
501				Self::build_new_ext($genesis)
502			}
503
504			fn move_ext_out(id: &'static str) {
505				use $crate::Deref;
506
507				// Take TestExternality from thread_local
508				let local_ext = $local_ext.with(|v| {
509					v.take()
510				});
511
512				// Get TestExternality from LazyLock
513				let global_ext_guard = $global_ext.lock().unwrap();
514
515				// Replace TestExternality in LazyLock by TestExternality from thread_local
516				global_ext_guard.deref().borrow_mut().insert(id.to_string(), local_ext);
517			}
518
519			fn move_ext_in(id: &'static str) {
520				use $crate::Deref;
521
522				let mut global_ext_unlocked = false;
523
524				// Keep the mutex unlocked until TesExternality from LazyLock
525				// has been updated
526				while !global_ext_unlocked {
527					// Get TesExternality from LazyLock
528					let global_ext_result = $global_ext.try_lock();
529
530					if let Ok(global_ext_guard) = global_ext_result {
531						// Unlock the mutex as long as the condition is not met
532						if !global_ext_guard.deref().borrow().contains_key(id) {
533							drop(global_ext_guard);
534						} else {
535							global_ext_unlocked = true;
536						}
537					}
538				}
539
540				// Now that we know that TestExt has been updated, we lock its mutex
541				let mut global_ext_guard = $global_ext.lock().unwrap();
542
543				// and set TesExternality from LazyLock into TesExternality for local_thread
544				let global_ext = global_ext_guard.deref();
545
546				$local_ext.with(|v| {
547					v.replace(global_ext.take().remove(id).unwrap());
548				});
549			}
550
551			fn reset_ext() {
552				$local_ext.with(|v| *v.borrow_mut() = Self::build_new_ext($genesis));
553			}
554
555			fn execute_with<R>(execute: impl FnOnce() -> R) -> R {
556				use $crate::{Chain, Network};
557				// Make sure the Network is initialized
558				<$network>::init();
559
560				// Execute
561				let r = $local_ext.with(|v| {
562					$crate::tracing::info!(target: "xcm::emulator::execute_with", "Executing as {}", stringify!($name));
563					v.borrow_mut().execute_with(execute)
564				});
565
566				// Send messages if needed
567				$local_ext.with(|v| {
568					v.borrow_mut().execute_with(|| {
569						use $crate::polkadot_primitives::runtime_api::runtime_decl_for_parachain_host::$api_version;
570
571						//TODO: mark sent count & filter out sent msg
572						for para_id in <$network>::para_ids() {
573							// downward messages
574							let downward_messages = <Self as $crate::Chain>::Runtime::dmq_contents(para_id.into())
575								.into_iter()
576								.map(|inbound| (inbound.sent_at, inbound.msg));
577							if downward_messages.len() == 0 {
578								continue;
579							}
580							<$network>::send_downward_messages(para_id, downward_messages.into_iter());
581
582							// Note: no need to handle horizontal messages, as the
583							// simulator directly sends them to dest (not relayed).
584						}
585
586						// log events
587						Self::events().iter().for_each(|event| {
588							$crate::tracing::info!(target: concat!("events::", stringify!($name)), ?event, "Event emitted");
589						});
590
591						// clean events
592						<Self as Chain>::System::reset_events();
593					})
594				});
595
596				<$network>::process_messages();
597
598				r
599			}
600
601			fn ext_wrapper<R>(func: impl FnOnce() -> R) -> R {
602				$local_ext.with(|v| {
603					v.borrow_mut().execute_with(|| {
604						func()
605					})
606				})
607			}
608		}
609	};

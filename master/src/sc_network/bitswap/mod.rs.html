46const LOG_TARGET: &str = "bitswap";

48// Undocumented, but according to JS the bitswap messages have a max size of 512*1024 bytes
49// https://github.com/ipfs/js-ipfs-bitswap/blob/
50// d8f80408aadab94c962f6b88f343eb9f39fa0fcc/src/decision-engine/index.js#L16
51// We set it to the same value as max substrate protocol message
52const MAX_PACKET_SIZE: u64 = MAX_RESPONSE_SIZE;
53
54/// Max number of queued responses before denying requests.
55const MAX_REQUEST_QUEUE: usize = 20;
56
57/// Max number of blocks per wantlist
58const MAX_WANTED_BLOCKS: usize = 16;
59
60/// Bitswap protocol name
61const PROTOCOL_NAME: &'static str = "/ipfs/bitswap/1.2.0";

63/// Check if a CID is supported by the bitswap protocol.
64pub fn is_cid_supported(cid: &Cid) -> bool {
65	cid.version() != CidVersion::V0 && cid.hash().size() == 32
66}
67
68/// Prefix represents all metadata of a CID, without the actual content.
69#[derive(PartialEq, Eq, Clone, Debug)]
70struct Prefix {
71	/// The version of CID.
72	pub version: CidVersion,
73	/// The codec of CID.
74	pub codec: u64,
75	/// The multihash type of CID.
76	pub mh_type: u64,
77	/// The multihash length of CID.
78	pub mh_len: u8,
79}
80
81impl Prefix {
82	/// Convert the prefix to encoded bytes.
83	pub fn to_bytes(&self) -> Vec<u8> {
84		let mut res = Vec::with_capacity(4);
85		let mut buf = varint_encode::u64_buffer();
86		let version = varint_encode::u64(self.version.into(), &mut buf);
87		res.extend_from_slice(version);
88		let mut buf = varint_encode::u64_buffer();
89		let codec = varint_encode::u64(self.codec, &mut buf);
90		res.extend_from_slice(codec);
91		let mut buf = varint_encode::u64_buffer();
92		let mh_type = varint_encode::u64(self.mh_type, &mut buf);
93		res.extend_from_slice(mh_type);
94		let mut buf = varint_encode::u64_buffer();
95		let mh_len = varint_encode::u64(self.mh_len as u64, &mut buf);
96		res.extend_from_slice(mh_len);
97		res
98	}
99}

101/// Bitswap request handler
102pub struct BitswapRequestHandler<B> {
103	client: Arc<dyn BlockBackend<B> + Send + Sync>,
104	request_receiver: async_channel::Receiver<IncomingRequest>,
105}
106
107impl<B: BlockT> BitswapRequestHandler<B> {
108	/// Create a new [`BitswapRequestHandler`].
109	pub fn new(client: Arc<dyn BlockBackend<B> + Send + Sync>) -> (Self, ProtocolConfig) {
110		let (tx, request_receiver) = async_channel::bounded(MAX_REQUEST_QUEUE);
111
112		let config = ProtocolConfig {
113			name: ProtocolName::from(PROTOCOL_NAME),
114			fallback_names: vec![],
115			max_request_size: MAX_PACKET_SIZE,
116			max_response_size: MAX_PACKET_SIZE,
117			request_timeout: Duration::from_secs(15),
118			inbound_queue: Some(tx),
119		};
120
121		(Self { client, request_receiver }, config)
122	}
123
124	/// Run [`BitswapRequestHandler`].
125	pub async fn run(mut self) {
126		while let Some(request) = self.request_receiver.next().await {
127			let IncomingRequest { peer, payload, pending_response } = request;
128
129			match self.handle_message(&peer, &payload) {
130				Ok(response) => {
131					let response = OutgoingResponse {
132						result: Ok(response),
133						reputation_changes: Vec::new(),
134						sent_feedback: None,
135					};
136
137					match pending_response.send(response) {
138						Ok(()) => {
139							trace!(target: LOG_TARGET, "Handled bitswap request from {peer}.",)
140						},
141						Err(_) => debug!(
142							target: LOG_TARGET,
143							"Failed to handle light client request from {peer}: {}",
144							BitswapError::SendResponse,
145						),
146					}
147				},
148				Err(err) => {
149					error!(target: LOG_TARGET, "Failed to process request from {peer}: {err}");
150
151					// TODO: adjust reputation?
152
153					let response = OutgoingResponse {
154						result: Err(()),
155						reputation_changes: vec![],
156						sent_feedback: None,
157					};
158
159					if pending_response.send(response).is_err() {
160						debug!(
161							target: LOG_TARGET,
162							"Failed to handle bitswap request from {peer}: {}",
163							BitswapError::SendResponse,
164						);
165					}
166				},
167			}
168		}
169	}
170
171	/// Handle received Bitswap request
172	fn handle_message(
173		&mut self,
174		peer: &PeerId,
175		payload: &Vec<u8>,
176	) -> Result<Vec<u8>, BitswapError> {
177		let request = schema::bitswap::Message::decode(&payload[..])?;
178
179		trace!(target: LOG_TARGET, "Received request: {:?} from {}", request, peer);
180
181		let mut response = BitswapMessage::default();
182
183		let wantlist = match request.wantlist {
184			Some(wantlist) => wantlist,
185			None => {
186				debug!(target: LOG_TARGET, "Unexpected bitswap message from {}", peer);
187				return Err(BitswapError::InvalidWantList)
188			},
189		};
190
191		if wantlist.entries.len() > MAX_WANTED_BLOCKS {
192			trace!(target: LOG_TARGET, "Ignored request: too many entries");
193			return Err(BitswapError::TooManyEntries)
194		}
195
196		for entry in wantlist.entries {
197			let cid = match Cid::read_bytes(entry.block.as_slice()) {
198				Ok(cid) => cid,
199				Err(e) => {
200					trace!(target: LOG_TARGET, "Bad CID {:?}: {:?}", entry.block, e);
201					continue
202				},
203			};
204
205			if !is_cid_supported(&cid) {
206				trace!(target: LOG_TARGET, "Ignoring unsupported CID {}: {}", peer, cid);
207				continue
208			}
209
210			let mut hash = B::Hash::default();
211			hash.as_mut().copy_from_slice(&cid.hash().digest()[0..32]);
212			let transaction = match self.client.indexed_transaction(hash) {
213				Ok(ex) => ex,
214				Err(e) => {
215					error!(target: LOG_TARGET, "Error retrieving transaction {}: {}", hash, e);
216					None
217				},
218			};
219
220			match transaction {
221				Some(transaction) => {
222					trace!(target: LOG_TARGET, "Found CID {:?}, hash {:?}", cid, hash);
223
224					if entry.want_type == WantType::Block as i32 {
225						let prefix = Prefix {
226							version: cid.version(),
227							codec: cid.codec(),
228							mh_type: cid.hash().code(),
229							mh_len: cid.hash().size(),
230						};
231						response
232							.payload
233							.push(MessageBlock { prefix: prefix.to_bytes(), data: transaction });
234					} else {
235						response.block_presences.push(BlockPresence {
236							r#type: BlockPresenceType::Have as i32,
237							cid: cid.to_bytes(),
238						});
239					}
240				},
241				None => {
242					trace!(target: LOG_TARGET, "Missing CID {:?}, hash {:?}", cid, hash);
243
244					if entry.send_dont_have {
245						response.block_presences.push(BlockPresence {
246							r#type: BlockPresenceType::DontHave as i32,
247							cid: cid.to_bytes(),
248						});
249					}
250				},
251			}
252		}
253
254		Ok(response.encode_to_vec())
255	}
256}

258/// Bitswap protocol error.
259#[derive(Debug, thiserror::Error)]
260pub enum BitswapError {
261	/// Protobuf decoding error.
262	#[error("Failed to decode request: {0}.")]
263	DecodeProto(#[from] prost::DecodeError),
264
265	/// Protobuf encoding error.
266	#[error("Failed to encode response: {0}.")]
267	EncodeProto(#[from] prost::EncodeError),
268
269	/// Client backend error.
270	#[error(transparent)]
271	Client(#[from] sp_blockchain::Error),
272
273	/// Error parsing CID
274	#[error(transparent)]
275	BadCid(#[from] CidError),
276
277	/// Packet read error.
278	#[error(transparent)]
279	Read(#[from] io::Error),
280
281	/// Error sending response.
282	#[error("Failed to send response.")]
283	SendResponse,
284
285	/// Message doesn't have a WANT list.
286	#[error("Invalid WANT list.")]
287	InvalidWantList,
288
289	/// Too many blocks requested.
290	#[error("Too many block entries in the request.")]
291	TooManyEntries,
292}

474	#[tokio::test]
475	async fn transaction_found() {
476		let client = TestClientBuilder::with_tx_storage(u32::MAX).build();
477		let mut block_builder = BlockBuilderBuilder::new(&client)
478			.on_parent_block(client.chain_info().genesis_hash)
479			.with_parent_block_number(0)

481			.unwrap();

484		let ext = ExtrinsicBuilder::new_indexed_call(vec![0x13, 0x37, 0x13, 0x38]).build();
485		let pattern_index = ext.encoded_size() - 4;
486
487		block_builder.push(ext.clone()).unwrap();
488		let block = block_builder.build().unwrap().block;
489
490		client.import(BlockOrigin::File, block).await.unwrap();
491
492		let (bitswap, config) = BitswapRequestHandler::new(Arc::new(client));
493
494		tokio::spawn(async move { bitswap.run().await });
495
496		let (tx, rx) = oneshot::channel();
497		config
498			.inbound_queue
499			.unwrap()
500			.send(IncomingRequest {
501				peer: PeerId::random(),
502				payload: BitswapMessage {
503					wantlist: Some(Wantlist {
504						entries: vec![Entry {
505							block: cid::Cid::new_v1(
506								0x70,
507								cid::multihash::Multihash::wrap(
508									u64::from(Code::Blake2b256),
509									&sp_crypto_hashing::blake2_256(&ext.encode()[pattern_index..]),
510								)
511								.unwrap(),
512							)
513							.to_bytes(),
514							..Default::default()
515						}],
516						full: false,
517					}),
518					..Default::default()
519				}
520				.encode_to_vec(),
521				pending_response: tx,
522			})
523			.await
524			.unwrap();
525
526		if let Ok(OutgoingResponse { result, reputation_changes, sent_feedback }) = rx.await {
527			assert_eq!(reputation_changes, Vec::new());
528			assert!(sent_feedback.is_none());
529
530			let response =
531				schema::bitswap::Message::decode(&result.expect("fetch to succeed")[..]).unwrap();
532			assert_eq!(response.payload[0].data, vec![0x13, 0x37, 0x13, 0x38]);
533		} else {
534			panic!("invalid event received");
535		}
536	}

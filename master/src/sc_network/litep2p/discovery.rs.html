67/// Kademlia query interval.
68const KADEMLIA_QUERY_INTERVAL: Duration = Duration::from_secs(5);
69
70/// mDNS query interval.
71const MDNS_QUERY_INTERVAL: Duration = Duration::from_secs(30);

73/// The minimum number of peers we expect an answer before we terminate the request.
74const GET_RECORD_REDUNDANCY_FACTOR: usize = 4;
75
76/// The maximum number of tracked external addresses we allow.
77const MAX_EXTERNAL_ADDRESSES: u32 = 32;
78
79/// Number of times observed address is received from different peers before it is confirmed as
80/// external.
81const MIN_ADDRESS_CONFIRMATIONS: usize = 3;
82
83/// Quorum threshold to interpret `PUT_VALUE` & `ADD_PROVIDER` as successful.
84///
85/// As opposed to libp2p, litep2p does not finish the query as soon as the required number of
86/// peers have reached. Instead, it tries to put the record to all target peers (typically 20) and
87/// uses the quorum setting only to determine the success of the query.
88///
89/// We set the threshold to 50% of the target peers to account for unreachable peers. The actual
90/// number of stored records may be higher.
91const QUORUM_THRESHOLD: NonZeroUsize = NonZeroUsize::new(10).expect("10 > 0; qed");

93/// Discovery events.
94#[derive(Debug)]
95pub enum DiscoveryEvent {
96	/// Ping RTT measured for peer.
97	Ping {
98		/// Remote peer ID.
99		peer: PeerId,
100
101		/// Ping round-trip time.
102		rtt: Duration,
103	},
104
105	/// Peer identified over `/ipfs/identify/1.0.0` protocol.
106	Identified {
107		/// Peer ID.
108		peer: PeerId,
109
110		/// Listen addresses.
111		listen_addresses: Vec<Multiaddr>,
112
113		/// Supported protocols.
114		supported_protocols: HashSet<ProtocolName>,
115	},
116
117	/// One or more addresses discovered.
118	///
119	/// This event is emitted when a new peer is discovered over mDNS.
120	Discovered {
121		/// Discovered addresses.
122		addresses: Vec<Multiaddr>,
123	},
124
125	/// Routing table has been updated.
126	RoutingTableUpdate {
127		/// Peers that were added to routing table.
128		peers: HashSet<PeerId>,
129	},
130
131	/// New external address discovered.
132	ExternalAddressDiscovered {
133		/// Discovered address.
134		address: Multiaddr,
135	},
136
137	/// The external address has expired.
138	///
139	/// This happens when the internal buffers exceed the maximum number of external addresses,
140	/// and this address is the oldest one.
141	ExternalAddressExpired {
142		/// Expired address.
143		address: Multiaddr,
144	},
145
146	/// `FIND_NODE` query succeeded.
147	FindNodeSuccess {
148		/// Query ID.
149		query_id: QueryId,
150
151		/// Target.
152		target: PeerId,
153
154		/// Found peers.
155		peers: Vec<(PeerId, Vec<Multiaddr>)>,
156	},
157
158	/// `GetRecord` query succeeded.
159	GetRecordSuccess {
160		/// Query ID.
161		query_id: QueryId,
162	},
163
164	/// Record was found from the DHT.
165	GetRecordPartialResult {
166		/// Query ID.
167		query_id: QueryId,
168
169		/// Record.
170		record: PeerRecord,
171	},
172
173	/// Record was successfully stored on the DHT.
174	PutRecordSuccess {
175		/// Query ID.
176		query_id: QueryId,
177	},
178
179	/// Providers were successfully retrieved.
180	GetProvidersSuccess {
181		/// Query ID.
182		query_id: QueryId,
183		/// Found providers sorted by distance to provided key.
184		providers: Vec<ContentProvider>,
185	},
186
187	/// Provider was successfully published.
188	AddProviderSuccess {
189		/// Query ID.
190		query_id: QueryId,
191		/// Provided key.
192		provided_key: RecordKey,
193	},
194
195	/// Query failed.
196	QueryFailed {
197		/// Query ID.
198		query_id: QueryId,
199	},
200
201	/// Incoming record to store.
202	IncomingRecord {
203		/// Record.
204		record: Record,
205	},
206
207	/// Started a random Kademlia query.
208	RandomKademliaStarted,
209}

261/// Legacy (fallback) Kademlia protocol name based on `protocol_id`.
262fn legacy_kademlia_protocol_name(id: &ProtocolId) -> ProtocolName {
263	ProtocolName::from(format!("/{}/kad", id.as_ref()))
264}
265
266/// Kademlia protocol name based on `genesis_hash` and `fork_id`.
267fn kademlia_protocol_name<Hash: AsRef<[u8]>>(
268	genesis_hash: Hash,
269	fork_id: Option<&str>,
270) -> ProtocolName {
271	let genesis_hash_hex = bytes2hex("", genesis_hash.as_ref());
272	let protocol = if let Some(fork_id) = fork_id {
273		format!("/{}/{}/kad", genesis_hash_hex, fork_id)
274	} else {
275		format!("/{}/kad", genesis_hash_hex)
276	};
277
278	ProtocolName::from(protocol)
279}

281impl Discovery {
282	/// Create new [`Discovery`].
283	///
284	/// Enables `/ipfs/ping/1.0.0` and `/ipfs/identify/1.0.0` by default and starts
285	/// the mDNS peer discovery if it was enabled.
286	pub fn new<Hash: AsRef<[u8]> + Clone>(
287		local_peer_id: litep2p::PeerId,
288		config: &NetworkConfiguration,
289		genesis_hash: Hash,
290		fork_id: Option<&str>,
291		protocol_id: &ProtocolId,
292		known_peers: HashMap<PeerId, Vec<Multiaddr>>,
293		listen_addresses: Arc<RwLock<HashSet<Multiaddr>>>,
294		_peerstore_handle: Arc<dyn PeerStoreProvider>,
295	) -> (Self, PingConfig, IdentifyConfig, KademliaConfig, Option<MdnsConfig>) {
296		let (ping_config, ping_event_stream) = PingConfig::default();
297		let user_agent = format!("{} ({}) (litep2p)", config.client_version, config.node_name);
298
299		let (identify_config, identify_event_stream) =
300			IdentifyConfig::new("/substrate/1.0".to_string(), Some(user_agent));
301
302		let (mdns_config, mdns_event_stream) = match config.transport {
303			crate::config::TransportConfig::Normal { enable_mdns, .. } => match enable_mdns {
304				true => {
305					let (mdns_config, mdns_event_stream) = MdnsConfig::new(MDNS_QUERY_INTERVAL);
306					(Some(mdns_config), Some(mdns_event_stream))
307				},
308				false => (None, None),
309			},
310			_ => panic!("memory transport not supported"),
311		};
312
313		let (kademlia_config, kademlia_handle) = {
314			let protocol_names = vec![
315				kademlia_protocol_name(genesis_hash.clone(), fork_id),
316				legacy_kademlia_protocol_name(protocol_id),
317			];
318
319			KademliaConfigBuilder::new()
320				.with_known_peers(known_peers)
321				.with_protocol_names(protocol_names)
322				.with_incoming_records_validation_mode(IncomingRecordValidationMode::Manual)
323				.with_provider_record_ttl(KADEMLIA_PROVIDER_RECORD_TTL)
324				.with_provider_refresh_interval(KADEMLIA_PROVIDER_REPUBLISH_INTERVAL)
325				.with_max_provider_keys(KADEMLIA_MAX_PROVIDER_KEYS)
326				.build()
327		};
328
329		(
330			Self {
331				local_peer_id,
332				ping_event_stream,
333				identify_event_stream,
334				mdns_event_stream,
335				kademlia_handle,
336				_peerstore_handle,
337				listen_addresses,
338				random_walk_query_id: None,
339				pending_events: VecDeque::new(),
340				duration_to_next_find_query: Duration::from_secs(1),
341				address_confirmations: LruMap::new(ByLength::new(MAX_EXTERNAL_ADDRESSES)),
342				allow_non_global_addresses: config.allow_non_globals_in_dht,
343				public_addresses: config.public_addresses.iter().cloned().map(Into::into).collect(),
344				next_kad_query: Some(Delay::new(KADEMLIA_QUERY_INTERVAL)),
345				local_protocols: HashSet::from_iter([kademlia_protocol_name(
346					genesis_hash,
347					fork_id,
348				)]),
349			},
350			ping_config,
351			identify_config,
352			kademlia_config,
353			mdns_config,
354		)
355	}

363	/// Add self-reported addresses to routing table if `peer` supports
364	/// at least one of the locally supported DHT protocol.
365	pub async fn add_self_reported_address(
366		&mut self,
367		peer: PeerId,
368		supported_protocols: HashSet<ProtocolName>,
369		addresses: Vec<Multiaddr>,
370	) {
371		if self.local_protocols.is_disjoint(&supported_protocols) {
372			log::trace!(
373				target: LOG_TARGET,
374				"Ignoring self-reported address of peer {peer} as remote node is not part of the \
375				 Kademlia DHT supported by the local node.",
376			);
377			return
378		}
379
380		let addresses = addresses
381			.into_iter()
382			.filter_map(|address| {
383				if !self.allow_non_global_addresses && !Discovery::can_add_to_dht(&address) {
384					log::trace!(
385						target: LOG_TARGET,
386						"ignoring self-reported non-global address {address} from {peer}."
387					);
388
389					return None
390				}
391
392				Some(address)
393			})
394			.collect();
395
396		log::trace!(
397			target: LOG_TARGET,
398			"add self-reported addresses for {peer:?}: {addresses:?}",
399		);
400
401		self.kademlia_handle.add_known_peer(peer, addresses).await;
402	}

489	/// Check if the observed address is a known address.
490	fn is_known_address(known: &Multiaddr, observed: &Multiaddr) -> bool {
491		let mut known = known.iter();
492		let mut observed = observed.iter();
493
494		loop {
495			match (known.next(), observed.next()) {
496				(None, None) => return true,
497				(None, Some(Protocol::P2p(_))) => return true,
498				(Some(Protocol::P2p(_)), None) => return true,
499				(known, observed) if known != observed => return false,
500				_ => {},
501			}
502		}
503	}
504
505	/// Can `address` be added to DHT.
506	fn can_add_to_dht(address: &Multiaddr) -> bool {
507		let ip = match address.iter().next() {
508			Some(Protocol::Ip4(ip)) => IpNetwork::from(ip),
509			Some(Protocol::Ip6(ip)) => IpNetwork::from(ip),
510			Some(Protocol::Dns(_)) | Some(Protocol::Dns4(_)) | Some(Protocol::Dns6(_)) =>
511				return true,
512			_ => return false,
513		};
514
515		ip.is_global()
516	}
517
518	/// Check if `address` can be considered a new external address.
519	///
520	/// If this address replaces an older address, the expired address is returned.
521	fn is_new_external_address(
522		&mut self,
523		address: &Multiaddr,
524		peer: PeerId,
525	) -> (bool, Option<Multiaddr>) {
526		log::trace!(target: LOG_TARGET, "verify new external address: {address}");
527
528		if !self.allow_non_global_addresses && !Discovery::can_add_to_dht(&address) {
529			log::trace!(
530				target: LOG_TARGET,
531				"ignoring externally reported non-global address {address} from {peer}."
532			);
533
534			return (false, None);
535		}
536
537		// is the address one of our known addresses
538		if self
539			.listen_addresses
540			.read()
541			.iter()
542			.chain(self.public_addresses.iter())
543			.any(|known_address| Discovery::is_known_address(&known_address, &address))
544		{
545			return (true, None)
546		}
547
548		match self.address_confirmations.get(address) {
549			Some(confirmations) => {
550				confirmations.insert(peer);
551
552				if confirmations.len() >= MIN_ADDRESS_CONFIRMATIONS {
553					return (true, None)
554				}
555			},
556			None => {
557				let oldest = (self.address_confirmations.len() >=
558					self.address_confirmations.limiter().max_length() as usize)
559					.then(|| {
560						self.address_confirmations.pop_oldest().map(|(address, peers)| {
561							if peers.len() >= MIN_ADDRESS_CONFIRMATIONS {
562								return Some(address)
563							} else {
564								None
565							}
566						})
567					})
568					.flatten()
569					.flatten();
570
571				self.address_confirmations.insert(address.clone(), iter::once(peer).collect());
572
573				return (false, oldest)
574			},
575		}
576
577		(false, None)
578	}
579}

581impl Stream for Discovery {
582	type Item = DiscoveryEvent;
583
584	fn poll_next(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Option<Self::Item>> {
585		let this = Pin::into_inner(self);
586
587		if let Some(event) = this.pending_events.pop_front() {
588			return Poll::Ready(Some(event))
589		}
590
591		if let Some(mut delay) = this.next_kad_query.take() {
592			match delay.poll_unpin(cx) {
593				Poll::Pending => {
594					this.next_kad_query = Some(delay);
595				},
596				Poll::Ready(()) => {
597					let peer = PeerId::random();
598
599					log::trace!(target: LOG_TARGET, "start next kademlia query for {peer:?}");
600
601					match this.kademlia_handle.try_find_node(peer) {
602						Ok(query_id) => {
603							this.random_walk_query_id = Some(query_id);
604							return Poll::Ready(Some(DiscoveryEvent::RandomKademliaStarted))
605						},
606						Err(()) => {
607							this.duration_to_next_find_query = cmp::min(
608								this.duration_to_next_find_query * 2,
609								Duration::from_secs(60),
610							);
611							this.next_kad_query =
612								Some(Delay::new(this.duration_to_next_find_query));
613						},
614					}
615				},
616			}
617		}
618
619		match Pin::new(&mut this.kademlia_handle).poll_next(cx) {
620			Poll::Pending => {},
621			Poll::Ready(None) => return Poll::Ready(None),
622			Poll::Ready(Some(KademliaEvent::FindNodeSuccess { query_id, peers, .. }))
623				if Some(query_id) == this.random_walk_query_id =>
624			{
625				// the addresses are already inserted into the DHT and in `TransportManager` so
626				// there is no need to add them again. The found peers must be registered to
627				// `Peerstore` so other protocols are aware of them through `Peerset`.
628				log::trace!(target: LOG_TARGET, "dht random walk yielded {} peers", peers.len());
629
630				this.next_kad_query = Some(Delay::new(KADEMLIA_QUERY_INTERVAL));
631
632				return Poll::Ready(Some(DiscoveryEvent::RoutingTableUpdate {
633					peers: peers.into_iter().map(|(peer, _)| peer).collect(),
634				}))
635			},
636			Poll::Ready(Some(KademliaEvent::FindNodeSuccess { query_id, target, peers })) => {
637				log::trace!(target: LOG_TARGET, "find node query yielded {} peers", peers.len());
638
639				return Poll::Ready(Some(DiscoveryEvent::FindNodeSuccess {
640					query_id,
641					target,
642					peers,
643				}))
644			},
645			Poll::Ready(Some(KademliaEvent::RoutingTableUpdate { peers })) => {
646				log::trace!(target: LOG_TARGET, "routing table update, discovered {} peers", peers.len());
647
648				return Poll::Ready(Some(DiscoveryEvent::RoutingTableUpdate {
649					peers: peers.into_iter().collect(),
650				}))
651			},
652			Poll::Ready(Some(KademliaEvent::GetRecordSuccess { query_id })) => {
653				log::trace!(
654					target: LOG_TARGET,
655					"`GET_RECORD` succeeded for {query_id:?}",
656				);
657
658				return Poll::Ready(Some(DiscoveryEvent::GetRecordSuccess { query_id }));
659			},
660			Poll::Ready(Some(KademliaEvent::GetRecordPartialResult { query_id, record })) => {
661				log::trace!(
662					target: LOG_TARGET,
663					"`GET_RECORD` intermediary succeeded for {query_id:?}: {record:?}",
664				);
665
666				return Poll::Ready(Some(DiscoveryEvent::GetRecordPartialResult {
667					query_id,
668					record,
669				}));
670			},
671			Poll::Ready(Some(KademliaEvent::PutRecordSuccess { query_id, key: _ })) =>
672				return Poll::Ready(Some(DiscoveryEvent::PutRecordSuccess { query_id })),
673			Poll::Ready(Some(KademliaEvent::QueryFailed { query_id })) => {
674				match this.random_walk_query_id == Some(query_id) {
675					true => {
676						this.random_walk_query_id = None;
677						this.duration_to_next_find_query =
678							cmp::min(this.duration_to_next_find_query * 2, Duration::from_secs(60));
679						this.next_kad_query = Some(Delay::new(this.duration_to_next_find_query));
680					},
681					false => return Poll::Ready(Some(DiscoveryEvent::QueryFailed { query_id })),
682				}
683			},
684			Poll::Ready(Some(KademliaEvent::IncomingRecord { record })) => {
685				log::trace!(
686					target: LOG_TARGET,
687					"incoming `PUT_RECORD` request with key {:?} from publisher {:?}",
688					record.key,
689					record.publisher,
690				);
691
692				return Poll::Ready(Some(DiscoveryEvent::IncomingRecord { record }))
693			},
694			Poll::Ready(Some(KademliaEvent::GetProvidersSuccess {
695				provided_key,
696				providers,
697				query_id,
698			})) => {
699				log::trace!(
700					target: LOG_TARGET,
701					"`GET_PROVIDERS` for {query_id:?} with {provided_key:?} yielded {providers:?}",
702				);
703
704				return Poll::Ready(Some(DiscoveryEvent::GetProvidersSuccess {
705					query_id,
706					providers,
707				}))
708			},
709			Poll::Ready(Some(KademliaEvent::AddProviderSuccess { query_id, provided_key })) => {
710				log::trace!(
711					target: LOG_TARGET,
712					"`ADD_PROVIDER` for {query_id:?} with {provided_key:?} succeeded",
713				);
714
715				return Poll::Ready(Some(DiscoveryEvent::AddProviderSuccess {
716					query_id,
717					provided_key,
718				}))
719			},
720			// We do not validate incoming providers.
721			Poll::Ready(Some(KademliaEvent::IncomingProvider { .. })) => {},
722		}
723
724		match Pin::new(&mut this.identify_event_stream).poll_next(cx) {
725			Poll::Pending => {},
726			Poll::Ready(None) => return Poll::Ready(None),
727			Poll::Ready(Some(IdentifyEvent::PeerIdentified {
728				peer,
729				listen_addresses,
730				supported_protocols,
731				observed_address,
732				..
733			})) => {
734				let observed_address =
735					if let Some(Protocol::P2p(peer_id)) = observed_address.iter().last() {
736						if peer_id != *this.local_peer_id.as_ref() {
737							log::warn!(
738								target: LOG_TARGET,
739								"Discovered external address for a peer that is not us: {observed_address}",
740							);
741							None
742						} else {
743							Some(observed_address)
744						}
745					} else {
746						Some(observed_address.with(Protocol::P2p(this.local_peer_id.into())))
747					};
748
749				// Ensure that an external address with a different peer ID does not have
750				// side effects of evicting other external addresses via `ExternalAddressExpired`.
751				if let Some(observed_address) = observed_address {
752					let (is_new, expired_address) =
753						this.is_new_external_address(&observed_address, peer);
754
755					if let Some(expired_address) = expired_address {
756						log::trace!(
757							target: LOG_TARGET,
758							"Removing expired external address expired={expired_address} is_new={is_new} observed={observed_address}",
759						);
760
761						this.pending_events.push_back(DiscoveryEvent::ExternalAddressExpired {
762							address: expired_address,
763						});
764					}
765
766					if is_new {
767						this.pending_events.push_back(DiscoveryEvent::ExternalAddressDiscovered {
768							address: observed_address.clone(),
769						});
770					}
771				}
772
773				return Poll::Ready(Some(DiscoveryEvent::Identified {
774					peer,
775					listen_addresses,
776					supported_protocols,
777				}));
778			},
779		}
780
781		match Pin::new(&mut this.ping_event_stream).poll_next(cx) {
782			Poll::Pending => {},
783			Poll::Ready(None) => return Poll::Ready(None),
784			Poll::Ready(Some(PingEvent::Ping { peer, ping })) =>
785				return Poll::Ready(Some(DiscoveryEvent::Ping { peer, rtt: ping })),
786		}
787
788		if let Some(ref mut mdns_event_stream) = &mut this.mdns_event_stream {
789			match Pin::new(mdns_event_stream).poll_next(cx) {
790				Poll::Pending => {},
791				Poll::Ready(None) => return Poll::Ready(None),
792				Poll::Ready(Some(MdnsEvent::Discovered(addresses))) =>
793					return Poll::Ready(Some(DiscoveryEvent::Discovered { addresses })),
794			}
795		}
796
797		Poll::Pending
798	}
799}

801#[cfg(test)]
802mod tests {
803	use super::*;
804
805	use std::sync::atomic::AtomicU32;
806
807	use crate::{

814
815	use litep2p::{
816		config::ConfigBuilder as Litep2pConfigBuilder, transport::tcp::config::Config as TcpConfig,
817		Litep2p,
818	};
819
820	#[tokio::test]
821	async fn litep2p_discovery_works() {
822		let _ = tracing_subscriber::fmt()
823			.with_env_filter(tracing_subscriber::EnvFilter::from_default_env())
824			.try_init();
825
826		let mut known_peers = HashMap::new();
827		let genesis_hash = H256::from_low_u64_be(1);
828		let fork_id = Some("test-fork-id");
829		let protocol_id = ProtocolId::from("dot");
830
831		// Build backends such that the first peer is known to all other peers.
832		let backends = (0..10)
833			.map(|i| {
834				let keypair = litep2p::crypto::ed25519::Keypair::generate();
835				let peer_id: PeerId = keypair.public().to_peer_id().into();
836
837				let listen_addresses = Arc::new(RwLock::new(HashSet::new()));
838
839				let peer_store = PeerStore::new(vec![], None);
840				let peer_store_handle: Arc<dyn PeerStoreProvider> = Arc::new(peer_store.handle());
841
842				let (discovery, ping_config, identify_config, kademlia_config, _mdns) =
843					Discovery::new(
844						peer_id,
845						&NetworkConfiguration::new_local(),
846						genesis_hash,
847						fork_id,
848						&protocol_id,
849						known_peers.clone(),
850						listen_addresses.clone(),
851						peer_store_handle,
852					);
853
854				let config = Litep2pConfigBuilder::new()
855					.with_keypair(keypair)
856					.with_tcp(TcpConfig {
857						listen_addresses: vec!["/ip6/::1/tcp/0".parse().unwrap()],
858						..Default::default()
859					})
860					.with_libp2p_ping(ping_config)
861					.with_libp2p_identify(identify_config)
862					.with_libp2p_kademlia(kademlia_config)
863					.build();
864
865				let mut litep2p = Litep2p::new(config).unwrap();
866
867				let addresses = litep2p.listen_addresses().cloned().collect::<Vec<_>>();
868				// Propagate addresses to discovery.
869				addresses.iter().for_each(|address| {
870					listen_addresses.write().insert(address.clone());
871				});
872
873				// Except the first peer, all other peers know the first peer addresses.
874				if i == 0 {
875					log::info!(target: LOG_TARGET, "First peer is {peer_id:?} with addresses {addresses:?}");
876					known_peers.insert(peer_id, addresses.clone());
877				} else {
878					let (peer, addresses) = known_peers.iter().next().unwrap();
879
880					let result = litep2p.add_known_address(*peer, addresses.into_iter().cloned());
881
882					log::info!(target: LOG_TARGET, "{peer_id:?}: Adding known peer {peer:?} with addresses {addresses:?} result={result:?}");
883
884				}
885
886				(peer_id, litep2p, discovery)
887			})
888			.collect::<Vec<_>>();
889
890		let total_peers = backends.len() as u32;
891		let remaining_peers =
892			backends.iter().map(|(peer_id, _, _)| *peer_id).collect::<HashSet<_>>();
893
894		let first_peer = *known_peers.iter().next().unwrap().0;
895
896		// Each backend must discover the whole network.
897		let mut futures = FuturesUnordered::new();
898		let num_finished = Arc::new(AtomicU32::new(0));
899
900		for (peer_id, mut litep2p, mut discovery) in backends {
901			// Remove the local peer id from the set.
902			let mut remaining_peers = remaining_peers.clone();
903			remaining_peers.remove(&peer_id);
904
905			let num_finished = num_finished.clone();
906
907			let future = async move {
908				log::info!(target: LOG_TARGET, "{peer_id:?} starting loop");
909
910				if peer_id != first_peer {
911					log::info!(target: LOG_TARGET, "{peer_id:?} dialing {first_peer:?}");
912					litep2p.dial(&first_peer).await.unwrap();
913				}
914
915				loop {
916					// We need to keep the network alive until all peers are discovered.
917					if num_finished.load(std::sync::atomic::Ordering::Relaxed) == total_peers {
918						log::info!(target: LOG_TARGET, "{peer_id:?} all peers discovered");
919						break
920					}
921
922					tokio::select! {
923						// Drive litep2p backend forward.
924						event = litep2p.next_event() => {
925							log::info!(target: LOG_TARGET, "{peer_id:?} Litep2p event: {event:?}");
926						},
927
928						// Detect discovery events.
929						event = discovery.next() => {
930							match event.unwrap() {
931								// We have discovered the peer via kademlia and established
932								// a connection on the identify protocol.
933								DiscoveryEvent::Identified { peer, .. } => {
934									log::info!(target: LOG_TARGET, "{peer_id:?} Peer {peer} identified");
935
936									remaining_peers.remove(&peer);
937
938									if remaining_peers.is_empty() {
939										log::info!(target: LOG_TARGET, "{peer_id:?} All peers discovered");
940
941										num_finished.fetch_add(1, std::sync::atomic::Ordering::AcqRel);
942									}
943								},
944
945								event => {
946									log::info!(target: LOG_TARGET, "{peer_id:?} Discovery event: {event:?}");
947								}
948							}
949						}
950					}
951				}
952			};
953
954			futures.push(future);
955		}
956
957		// Futures will exit when all peers are discovered.
958		tokio::time::timeout(Duration::from_secs(60), futures.next())
959			.await
960			.expect("All peers should finish within 60 seconds");
961	}

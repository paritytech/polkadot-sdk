18use crate::construct_runtime::Pallet;
19use proc_macro2::TokenStream;
20use quote::quote;
21use syn::Ident;
22
23pub fn expand_outer_inherent(
24	runtime: &Ident,
25	block: &TokenStream,
26	unchecked_extrinsic: &TokenStream,
27	pallet_decls: &[Pallet],
28	scrate: &TokenStream,
29) -> TokenStream {
30	let mut pallet_positions = Vec::new();
31	let mut pallet_names = Vec::new();
32	let mut pallet_attrs = Vec::new();
33	let mut query_inherent_part_macros = Vec::new();
34
35	for (pallet_pos, pallet_decl) in pallet_decls
36		.iter()
37		.filter(|pallet_decl| pallet_decl.exists_part("Inherent"))
38		.enumerate()
39	{
40		let name = &pallet_decl.name;
41		let path = &pallet_decl.path;
42		let attr = pallet_decl.get_attributes();
43
44		pallet_positions.push(pallet_pos);
45		pallet_names.push(name);
46		pallet_attrs.push(attr);
47		query_inherent_part_macros.push(quote! {
48			#path::__substrate_inherent_check::is_inherent_part_defined!(#name);
49		});
50	}
51	let pallet_count = pallet_positions.len();
52
53	quote! {
54		#( #query_inherent_part_macros )*
55
56		trait InherentDataExt {
57			fn create_extrinsics(&self) ->
58				#scrate::__private::Vec<<#block as #scrate::sp_runtime::traits::Block>::Extrinsic>;
59			fn check_extrinsics(
60				&self,
61				block: &<#block as #scrate::sp_runtime::traits::Block>::LazyBlock
62			) -> #scrate::inherent::CheckInherentsResult;
63		}
64
65		impl InherentDataExt for #scrate::inherent::InherentData {
66			fn create_extrinsics(&self) ->
67				#scrate::__private::Vec<<#block as #scrate::sp_runtime::traits::Block>::Extrinsic>
68			{
69				use #scrate::{inherent::ProvideInherent, traits::InherentBuilder};
70
71				let mut inherents = #scrate::__private::Vec::new();
72
73				#(
74					#pallet_attrs
75					if let Some(inherent) = #pallet_names::create_inherent(self) {
76						let inherent = <#unchecked_extrinsic as InherentBuilder>::new_inherent(
77							inherent.into(),
78						);
79
80						inherents.push(inherent);
81					}
82				)*
83
84				inherents
85			}
86
87			fn check_extrinsics(
88				&self,
89				block: &<#block as #scrate::sp_runtime::traits::Block>::LazyBlock
90			) ->
91				#scrate::inherent::CheckInherentsResult
92			{
93				use #scrate::inherent::{ProvideInherent, IsFatalError};
94				use #scrate::traits::IsSubType;
95				use #scrate::sp_runtime::traits::{Block as _, ExtrinsicCall, LazyBlock};
96				use #scrate::__private::{sp_inherents::Error, log};
97
98				let mut result = #scrate::inherent::CheckInherentsResult::new();
99
100				// This handle assume we abort on the first fatal error.
101				fn handle_put_error_result(res: Result<(), Error>) {
102					const LOG_TARGET: &str = "runtime::inherent";
103					match res {
104						Ok(()) => (),
105						Err(Error::InherentDataExists(id)) =>
106							log::debug!(
107								target: LOG_TARGET,
108								"Some error already reported for inherent {:?}, new non fatal \
109								error is ignored",
110								id
111							),
112						Err(Error::FatalErrorReported) =>
113							log::error!(
114								target: LOG_TARGET,
115								"Fatal error already reported, unexpected considering there is \
116								only one fatal error",
117							),
118						Err(_) =>
119							log::error!(
120								target: LOG_TARGET,
121								"Unexpected error from `put_error` operation",
122							),
123					}
124				}
125
126				let mut pallet_has_inherent = [false; #pallet_count];
127				for maybe_xt in block.extrinsics() {
128					let Ok(xt) = maybe_xt else {
129						panic!("check_extrinsics(): Unable to decode extrinsic");
130					};
131
132					// Inherents are before any other extrinsics.
133					// And signed extrinsics are not inherents.
134					if !(#scrate::sp_runtime::traits::ExtrinsicLike::is_bare(&xt)) {
135						break
136					}
137
138					let mut is_inherent = false;
139					let call = ExtrinsicCall::call(&xt);
140					#(
141						#pallet_attrs
142						{
143							if let Some(call) = IsSubType::<_>::is_sub_type(call) {
144								if #pallet_names::is_inherent(call) {
145									is_inherent = true;
146									pallet_has_inherent[#pallet_positions] = true;
147									if let Err(e) = #pallet_names::check_inherent(call, self) {
148										handle_put_error_result(result.put_error(
149											#pallet_names::INHERENT_IDENTIFIER, &e
150										));
151										if e.is_fatal_error() {
152											return result;
153										}
154									}
155								}
156							}
157						}
158					)*
159
160					// Inherents are before any other extrinsics.
161					// No module marked it as inherent, thus it is not.
162					if !is_inherent {
163						break
164					}
165				}
166
167				#(
168					#pallet_attrs
169					match #pallet_names::is_inherent_required(self) {
170						Ok(Some(e)) => {
171							if !pallet_has_inherent[#pallet_positions] {
172								handle_put_error_result(result.put_error(
173									#pallet_names::INHERENT_IDENTIFIER, &e
174								));
175								if e.is_fatal_error() {
176									return result;
177								}
178							}
179						},
180						Ok(None) => (),
181						Err(e) => {
182							handle_put_error_result(result.put_error(
183								#pallet_names::INHERENT_IDENTIFIER, &e
184							));
185							if e.is_fatal_error() {
186								return result;
187							}
188						},
189					}
190				)*
191
192				result
193			}
194		}
195
196		impl #scrate::traits::IsInherent<<#block as #scrate::sp_runtime::traits::Block>::Extrinsic> for #runtime {
197			fn is_inherent(ext: &<#block as #scrate::sp_runtime::traits::Block>::Extrinsic) -> bool {
198				use #scrate::inherent::ProvideInherent;
199				use #scrate::traits::IsSubType;
200				use #scrate::sp_runtime::traits::ExtrinsicCall;
201
202				let is_bare = #scrate::sp_runtime::traits::ExtrinsicLike::is_bare(ext);
203				if !is_bare {
204					// Inherents must be bare extrinsics.
205					return false
206				}
207
208				let call = ExtrinsicCall::call(ext);
209				#(
210					#pallet_attrs
211					{
212						if let Some(call) = IsSubType::<_>::is_sub_type(call) {
213							if <#pallet_names as ProvideInherent>::is_inherent(&call) {
214								return true;
215							}
216						}
217					}
218				)*
219				false
220			}
221		}
222	}
223}

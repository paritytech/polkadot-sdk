17//! Cross-Consensus Message format asset data structures.
18//!
19//! This encompasses four types for representing assets:
20//! - `Asset`: A description of a single asset, either an instance of a non-fungible or some amount
21//!   of a fungible.
22//! - `Assets`: A collection of `Asset`s. These are stored in a `Vec` and sorted with fungibles
23//!   first.
24//! - `Wild`: A single asset wildcard, this can either be "all" assets, or all assets of a specific
25//!   kind.
26//! - `AssetFilter`: A combination of `Wild` and `Assets` designed for efficiently filtering an XCM
27//!   holding account.
28

41/// A general identifier for an instance of a non-fungible asset class.
42#[derive(

58pub enum AssetInstance {
59	/// Undefined - used if the non-fungible asset class has only one instance.

70	Array8([u8; 8]),
71
72	/// A 16-byte fixed-length datum.
73	Array16([u8; 16]),
74
75	/// A 32-byte fixed-length datum.
76	Array32([u8; 32]),
77}

248/// Classification of whether an asset is fungible or not, along with a mandatory amount or
249/// instance.
250#[derive(
251	Clone,
252	Eq,
253	PartialEq,
254	Ord,
255	PartialOrd,
256	Debug,
257	Encode,
258	DecodeWithMemTracking,
259	TypeInfo,
260	MaxEncodedLen,
261	serde::Serialize,
262	serde::Deserialize,
263)]
264pub enum Fungibility {
265	/// A fungible asset; we record a number of units, as a `u128` in the inner item.
266	Fungible(#[codec(compact)] u128),
267	/// A non-fungible asset. We record the instance identifier in the inner item. Only one asset
268	/// of each instance identifier may ever be in existence at once.
269	NonFungible(AssetInstance),
270}
271
272#[derive(Decode)]
273enum UncheckedFungibility {
274	Fungible(#[codec(compact)] u128),
275	NonFungible(AssetInstance),
276}
277
278impl Decode for Fungibility {
279	fn decode<I: codec::Input>(input: &mut I) -> Result<Self, codec::Error> {
280		match UncheckedFungibility::decode(input)? {
281			UncheckedFungibility::Fungible(a) if a != 0 => Ok(Self::Fungible(a)),
282			UncheckedFungibility::NonFungible(i) => Ok(Self::NonFungible(i)),
283			UncheckedFungibility::Fungible(_) =>
284				Err("Fungible asset of zero amount is not allowed".into()),
285		}
286	}
287}

294	}

432/// Either an amount of a single fungible asset, or a single well-identified non-fungible asset.
433#[derive(
434	Clone,
435	Eq,
436	PartialEq,
437	Debug,
438	Encode,
439	Decode,
440	DecodeWithMemTracking,
441	TypeInfo,
442	MaxEncodedLen,
443	serde::Serialize,
444	serde::Deserialize,
445)]
446pub struct Asset {
447	/// The overall asset identity (aka *class*, in the case of a non-fungible).
448	pub id: AssetId,
449	/// The fungibility of the asset, which contains either the amount (in the case of a fungible
450	/// asset) or the *instance ID*, the secondary asset identifier.
451	pub fun: Fungibility,
452}
453
454impl PartialOrd for Asset {
455	fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
456		Some(self.cmp(other))
457	}
458}

460impl Ord for Asset {
461	fn cmp(&self, other: &Self) -> Ordering {
462		match (&self.fun, &other.fun) {
463			(Fungibility::Fungible(..), Fungibility::NonFungible(..)) => Ordering::Less,
464			(Fungibility::NonFungible(..), Fungibility::Fungible(..)) => Ordering::Greater,
465			_ => (&self.id, &self.fun).cmp(&(&other.id, &other.fun)),
466		}
467	}
468}

472		Asset { fun: fun.into(), id: id.into() }

476impl Asset {
477	pub fn is_fungible(&self, maybe_id: Option<AssetId>) -> bool {
478		use Fungibility::*;
479		matches!(self.fun, Fungible(..)) && maybe_id.map_or(true, |i| i == self.id)
480	}
481
482	pub fn is_non_fungible(&self, maybe_id: Option<AssetId>) -> bool {
483		use Fungibility::*;
484		matches!(self.fun, NonFungible(..)) && maybe_id.map_or(true, |i| i == self.id)
485	}
486
487	/// Prepend a `Location` to a concrete asset, giving it a new root location.
488	pub fn prepend_with(&mut self, prepend: &Location) -> Result<(), ()> {
489		self.id.prepend_with(prepend)
490	}
491
492	/// Returns true if `self` is a super-set of the given `inner` asset.
493	pub fn contains(&self, inner: &Asset) -> bool {
494		use Fungibility::*;
495		if self.id == inner.id {
496			match (&self.fun, &inner.fun) {
497				(Fungible(a), Fungible(i)) if a >= i => return true,
498				(NonFungible(a), NonFungible(i)) if a == i => return true,
499				_ => (),
500			}
501		}
502		false
503	}
504}

530/// A `Vec` of `Asset`s.
531///
532/// There are a number of invariants which the construction and mutation functions must ensure are
533/// maintained:
534/// - It may contain no items of duplicate asset class;
535/// - All items must be ordered;
536/// - The number of items should grow no larger than `MAX_ITEMS_IN_ASSETS`.
537#[derive(
538	Clone,
539	Eq,
540	PartialEq,
541	Ord,
542	PartialOrd,
543	Debug,
544	Encode,
545	DecodeWithMemTracking,
546	TypeInfo,
547	Default,
548	serde::Serialize,
549	serde::Deserialize,
550)]
551pub struct Assets(Vec<Asset>);
552
553/// Maximum number of items we expect in a single `Assets` value.
554/// This is enforced when decoding and provides a sensible `max_encoded_len` for `Assets`.
555pub const MAX_ITEMS_IN_ASSETS: usize = 20;
556
557impl MaxEncodedLen for Assets {
558	fn max_encoded_len() -> usize {
559		Asset::max_encoded_len() * MAX_ITEMS_IN_ASSETS
560	}
561}
562
563impl Decode for Assets {
564	fn decode<I: codec::Input>(input: &mut I) -> Result<Self, codec::Error> {
565		let mut bounded_instructions =
566			BoundedVec::<Asset, ConstU32<{ MAX_ITEMS_IN_ASSETS as u32 }>>::decode(input)?;
567
568		bounded_instructions.sort();
569
570		Self::from_sorted_and_deduplicated(bounded_instructions.into_inner())
571			.map_err(|()| "Duplicate items".into())
572	}
573}

587impl From<Vec<Asset>> for Assets {
588	fn from(mut assets: Vec<Asset>) -> Self {
589		let mut res = Vec::with_capacity(assets.len());
590		if !assets.is_empty() {
591			assets.sort();
592			let mut iter = assets.into_iter();
593			if let Some(first) = iter.next() {
594				let last = iter.fold(first, |a, b| -> Asset {
595					match (a, b) {
596						(
597							Asset { fun: Fungibility::Fungible(a_amount), id: a_id },
598							Asset { fun: Fungibility::Fungible(b_amount), id: b_id },
599						) if a_id == b_id => Asset {
600							id: a_id,
601							fun: Fungibility::Fungible(a_amount.saturating_add(b_amount)),
602						},
603						(
604							Asset { fun: Fungibility::NonFungible(a_instance), id: a_id },
605							Asset { fun: Fungibility::NonFungible(b_instance), id: b_id },
606						) if a_id == b_id && a_instance == b_instance =>
607							Asset { fun: Fungibility::NonFungible(a_instance), id: a_id },
608						(to_push, to_remember) => {
609							res.push(to_push);
610							to_remember
611						},
612					}
613				});
614				res.push(last);
615			}
616		}
617		Self(res)
618	}
619}

627impl Assets {
628	/// A new (empty) value.
629	pub fn new() -> Self {
630		Self(Vec::new())
631	}
632
633	/// Create a new instance of `Assets` from a `Vec<Asset>` whose contents are sorted
634	/// and which contain no duplicates.
635	///
636	/// Returns `Ok` if the operation succeeds and `Err` if `r` is out of order or had duplicates.
637	/// If you can't guarantee that `r` is sorted and deduplicated, then use
638	/// `From::<Vec<Asset>>::from` which is infallible.
639	pub fn from_sorted_and_deduplicated(r: Vec<Asset>) -> Result<Self, ()> {
640		if r.is_empty() {
641			return Ok(Self(Vec::new()))
642		}
643		r.iter().skip(1).try_fold(&r[0], |a, b| -> Result<&Asset, ()> {
644			if a.id < b.id || a < b && (a.is_non_fungible(None) || b.is_non_fungible(None)) {
645				Ok(b)
646			} else {
647				Err(())
648			}
649		})?;
650		Ok(Self(r))
651	}
652
653	/// Create a new instance of `Assets` from a `Vec<Asset>` whose contents are sorted
654	/// and which contain no duplicates.
655	///
656	/// In release mode, this skips any checks to ensure that `r` is correct, making it a
657	/// negligible-cost operation. Generally though you should avoid using it unless you have a
658	/// strict proof that `r` is valid.
659	#[cfg(test)]
660	pub fn from_sorted_and_deduplicated_skip_checks(r: Vec<Asset>) -> Self {
661		Self::from_sorted_and_deduplicated(r).expect("Invalid input r is not sorted/deduped")
662	}
663	/// Create a new instance of `Assets` from a `Vec<Asset>` whose contents are sorted
664	/// and which contain no duplicates.
665	///
666	/// In release mode, this skips any checks to ensure that `r` is correct, making it a
667	/// negligible-cost operation. Generally though you should avoid using it unless you have a
668	/// strict proof that `r` is valid.
669	///
670	/// In test mode, this checks anyway and panics on fail.
671	#[cfg(not(test))]
672	pub fn from_sorted_and_deduplicated_skip_checks(r: Vec<Asset>) -> Self {
673		Self(r)
674	}
675
676	/// Add some asset onto the list, saturating. This is quite a laborious operation since it
677	/// maintains the ordering.
678	pub fn push(&mut self, a: Asset) {
679		for asset in self.0.iter_mut().filter(|x| x.id == a.id) {
680			match (&a.fun, &mut asset.fun) {
681				(Fungibility::Fungible(amount), Fungibility::Fungible(balance)) => {
682					*balance = balance.saturating_add(*amount);
683					return
684				},
685				(Fungibility::NonFungible(inst1), Fungibility::NonFungible(inst2))
686					if inst1 == inst2 =>
687					return,
688				_ => (),
689			}
690		}
691		self.0.push(a);
692		self.0.sort();
693	}
694
695	/// Returns `true` if this definitely represents no asset.
696	pub fn is_none(&self) -> bool {
697		self.0.is_empty()
698	}

756/// A wildcard representing a set of assets.
757#[derive(
758	Clone,
759	Eq,
760	PartialEq,
761	Ord,
762	PartialOrd,
763	Debug,
764	Encode,
765	Decode,
766	DecodeWithMemTracking,
767	TypeInfo,
768	MaxEncodedLen,
769	serde::Serialize,
770	serde::Deserialize,
771)]
772pub enum WildAsset {
773	/// All assets in Holding.
774	All,
775	/// All assets in Holding of a given fungibility and ID.
776	AllOf { id: AssetId, fun: WildFungibility },
777	/// All assets in Holding, up to `u32` individual assets (different instances of non-fungibles
778	/// are separate assets).
779	AllCounted(#[codec(compact)] u32),
780	/// All assets in Holding of a given fungibility and ID up to `count` individual assets
781	/// (different instances of non-fungibles are separate assets).
782	AllOfCounted {
783		id: AssetId,
784		fun: WildFungibility,
785		#[codec(compact)]
786		count: u32,
787	},
788}
789
790impl TryFrom<OldWildAsset> for WildAsset {
791	type Error = ();
792	fn try_from(old: OldWildAsset) -> Result<WildAsset, ()> {
793		use OldWildAsset::*;
794		Ok(match old {
795			AllOf { id, fun } => Self::AllOf { id: id.try_into()?, fun: fun.try_into()? },
796			All => Self::All,
797			AllOfCounted { id, fun, count } =>
798				Self::AllOfCounted { id: id.try_into()?, fun: fun.try_into()?, count },
799			AllCounted(count) => Self::AllCounted(count),
800		})
801	}
802}
803
804impl WildAsset {
805	/// Returns true if `self` is a super-set of the given `inner` asset.
806	pub fn contains(&self, inner: &Asset) -> bool {
807		use WildAsset::*;
808		match self {
809			AllOfCounted { count: 0, .. } | AllCounted(0) => false,
810			AllOf { fun, id } | AllOfCounted { id, fun, .. } =>
811				inner.fun.is_kind(*fun) && &inner.id == id,
812			All | AllCounted(_) => true,
813		}
814	}
815
816	/// Returns true if the wild element of `self` matches `inner`.
817	///
818	/// Note that for `Counted` variants of wildcards, then it will disregard the count except for
819	/// always returning `false` when equal to 0.
820	#[deprecated = "Use `contains` instead"]
821	pub fn matches(&self, inner: &Asset) -> bool {
822		self.contains(inner)
823	}
824
825	/// Mutate the asset to represent the same value from the perspective of a new `target`
826	/// location. The local chain's location is provided in `context`.
827	pub fn reanchor(&mut self, target: &Location, context: &InteriorLocation) -> Result<(), ()> {
828		use WildAsset::*;
829		match self {
830			AllOf { ref mut id, .. } | AllOfCounted { ref mut id, .. } =>
831				id.reanchor(target, context),
832			All | AllCounted(_) => Ok(()),
833		}
834	}
835
836	/// Maximum count of assets allowed to match, if any.
837	pub fn count(&self) -> Option<u32> {
838		use WildAsset::*;
839		match self {
840			AllOfCounted { count, .. } | AllCounted(count) => Some(*count),
841			All | AllOf { .. } => None,
842		}
843	}
844
845	/// Explicit limit on number of assets allowed to match, if any.
846	pub fn limit(&self) -> Option<u32> {
847		self.count()
848	}
849
850	/// Consume self and return the equivalent version but counted and with the `count` set to the
851	/// given parameter.
852	pub fn counted(self, count: u32) -> Self {
853		use WildAsset::*;
854		match self {
855			AllOfCounted { fun, id, .. } | AllOf { fun, id } => AllOfCounted { fun, id, count },
856			All | AllCounted(_) => AllCounted(count),
857		}
858	}
859}

861impl<A: Into<AssetId>, B: Into<WildFungibility>> From<(A, B)> for WildAsset {
862	fn from((id, fun): (A, B)) -> WildAsset {
863		WildAsset::AllOf { fun: fun.into(), id: id.into() }
864	}
865}

867/// `Asset` collection, defined either by a number of `Assets` or a single wildcard.
868#[derive(
869	Clone,
870	Eq,
871	PartialEq,
872	Ord,
873	PartialOrd,
874	Debug,
875	Encode,
876	Decode,
877	DecodeWithMemTracking,
878	TypeInfo,
879	MaxEncodedLen,
880	serde::Serialize,
881	serde::Deserialize,
882)]
883pub enum AssetFilter {
884	/// Specify the filter as being everything contained by the given `Assets` inner.
885	Definite(Assets),
886	/// Specify the filter as the given `WildAsset` wildcard.
887	Wild(WildAsset),
888}
889
890impl<T: Into<WildAsset>> From<T> for AssetFilter {
891	fn from(x: T) -> Self {
892		Self::Wild(x.into())
893	}
894}

896impl From<Asset> for AssetFilter {
897	fn from(x: Asset) -> Self {
898		Self::Definite(vec![x].into())
899	}
900}
901
902impl From<Vec<Asset>> for AssetFilter {
903	fn from(x: Vec<Asset>) -> Self {
904		Self::Definite(x.into())
905	}
906}
907
908impl From<Assets> for AssetFilter {
909	fn from(x: Assets) -> Self {
910		Self::Definite(x)
911	}
912}

914impl AssetFilter {
915	/// Returns true if `inner` would be matched by `self`.
916	///
917	/// Note that for `Counted` variants of wildcards, then it will disregard the count except for
918	/// always returning `false` when equal to 0.
919	pub fn matches(&self, inner: &Asset) -> bool {
920		match self {
921			AssetFilter::Definite(ref assets) => assets.contains(inner),
922			AssetFilter::Wild(ref wild) => wild.contains(inner),
923		}
924	}
925
926	/// Mutate the location of the asset identifier if concrete, giving it the same location
927	/// relative to a `target` context. The local context is provided as `context`.
928	pub fn reanchor(&mut self, target: &Location, context: &InteriorLocation) -> Result<(), ()> {
929		match self {
930			AssetFilter::Definite(ref mut assets) => assets.reanchor(target, context),
931			AssetFilter::Wild(ref mut wild) => wild.reanchor(target, context),
932		}
933	}
934
935	/// Maximum count of assets it is possible to match, if known.
936	pub fn count(&self) -> Option<u32> {
937		use AssetFilter::*;
938		match self {
939			Definite(x) => Some(x.len() as u32),
940			Wild(x) => x.count(),
941		}
942	}
943
944	/// Explicit limit placed on the number of items, if any.
945	pub fn limit(&self) -> Option<u32> {
946		use AssetFilter::*;
947		match self {
948			Definite(_) => None,
949			Wild(x) => x.limit(),
950		}
951	}
952}

964/// Matches assets based on inner `AssetFilter` and tags them for a specific type of asset transfer.
965/// Please note: the transfer type is specific to each particular `(asset, source, dest)`

967#[derive(

977	TypeInfo,
978	MaxEncodedLen,
979	serde::Serialize,
980	serde::Deserialize,
981)]

985	/// reserve-transfer assets matching `AssetFilter` to a specific destination, using the local
986	/// chain as reserve

999			AssetTransferFilter::ReserveWithdraw(inner) => inner,

1001	}

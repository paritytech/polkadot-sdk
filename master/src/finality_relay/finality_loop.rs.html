17//! The loop basically reads all missing headers and their finality proofs from the source client.
18//! The proof for the best possible header is then submitted to the target node. The only exception
19//! is the mandatory headers, which we always submit to the target node. For such headers, we
20//! assume that the persistent proof either exists, or will eventually become available.

42/// Type of headers that we relay.
43#[derive(Debug, Clone, Copy, PartialEq)]
44pub enum HeadersToRelay {
45	/// Relay all headers.
46	All,
47	/// Relay only mandatory headers.
48	Mandatory,
49	/// Relay only free (including mandatory) headers.
50	Free,
51}
52
53/// Finality proof synchronization loop parameters.
54#[derive(Debug, Clone)]
55pub struct FinalitySyncParams {
56	/// Interval at which we check updates on both clients. Normally should be larger than
57	/// `min(source_block_time, target_block_time)`.
58	///
59	/// This parameter may be used to limit transactions rate. Increase the value && you'll get
60	/// infrequent updates => sparse headers => potential slow down of bridge applications, but
61	/// pallet storage won't be super large. Decrease the value to near `source_block_time` and
62	/// you'll get transaction for (almost) every block of the source chain => all source headers
63	/// will be known to the target chain => bridge applications will run faster, but pallet
64	/// storage may explode (but if pruning is there, then it's fine).
65	pub tick: Duration,
66	/// Number of finality proofs to keep in internal buffer between loop iterations.
67	///
68	/// While in "major syncing" state, we still read finality proofs from the stream. They're
69	/// stored in the internal buffer between loop iterations. When we're close to the tip of the
70	/// chain, we may meet finality delays if headers are not finalized frequently. So instead of
71	/// waiting for next finality proof to appear in the stream, we may use existing proof from
72	/// that buffer.
73	pub recent_finality_proofs_limit: usize,
74	/// Timeout before we treat our transactions as lost and restart the whole sync process.
75	pub stall_timeout: Duration,
76	/// If true, only mandatory headers are relayed.
77	pub headers_to_relay: HeadersToRelay,
78}

80/// Source client used in finality synchronization loop.
81#[async_trait]
82pub trait SourceClient<P: FinalitySyncPipeline>: SourceClientBase<P> {
83	/// Get best finalized block number.
84	async fn best_finalized_block_number(&self) -> Result<P::Number, Self::Error>;

86	/// Get canonical header and its finality proof by number.

123/// Finality sync information.
124pub struct SyncInfo<P: FinalitySyncPipeline> {
125	/// Best finalized header at the source client.
126	pub best_number_at_source: P::Number,
127	/// Best source header, known to the target client.
128	pub best_number_at_target: P::Number,
129	/// Whether the target client follows the same fork as the source client do.
130	pub is_using_same_fork: bool,
131}
132
133impl<P: FinalitySyncPipeline> SyncInfo<P> {
134	/// Checks if both clients are on the same fork.
135	async fn is_on_same_fork<SC: SourceClient<P>>(
136		source_client: &SC,
137		id_at_target: &HeaderId<P::Hash, P::Number>,
138	) -> Result<bool, SC::Error> {
139		let header_at_source = source_client.header_and_finality_proof(id_at_target.0).await?.0;
140		let header_hash_at_source = header_at_source.hash();
141		Ok(if id_at_target.1 == header_hash_at_source {
142			true
143		} else {
144			tracing::error!(
145				target: "bridge",
146				source=%P::SOURCE_NAME,
147				target=%P::TARGET_NAME,
148				height=?id_at_target.0,
149				at_source=?header_hash_at_source,
150				at_target=?id_at_target.1,
151				"Source node and pallet at target node have different headers at the same height"
152			);
153
154			false
155		})
156	}
157
158	async fn new<SC: SourceClient<P>, TC: TargetClient<P>>(
159		source_client: &SC,
160		target_client: &TC,
161	) -> Result<Self, Error<P, SC::Error, TC::Error>> {
162		let best_number_at_source =
163			source_client.best_finalized_block_number().await.map_err(Error::Source)?;
164		let best_id_at_target =
165			target_client.best_finalized_source_block_id().await.map_err(Error::Target)?;
166		let best_number_at_target = best_id_at_target.0;
167
168		let is_using_same_fork = Self::is_on_same_fork(source_client, &best_id_at_target)
169			.await
170			.map_err(Error::Source)?;
171
172		Ok(Self { best_number_at_source, best_number_at_target, is_using_same_fork })
173	}
174
175	fn update_metrics(&self, metrics_sync: &Option<SyncLoopMetrics>) {
176		if let Some(metrics_sync) = metrics_sync {
177			metrics_sync.update_best_block_at_source(self.best_number_at_source);
178			metrics_sync.update_best_block_at_target(self.best_number_at_target);
179			metrics_sync.update_using_same_fork(self.is_using_same_fork);
180		}
181	}
182
183	pub fn num_headers(&self) -> P::Number {
184		self.best_number_at_source.saturating_sub(self.best_number_at_target)
185	}
186}

188/// Information about transaction that we have submitted.
189#[derive(Debug, Clone)]
190pub struct Transaction<Tracker, Number> {
191	/// Submitted transaction tracker.
192	tracker: Tracker,
193	/// The number of the header we have submitted.
194	header_number: Number,
195}
196
197impl<Tracker: TransactionTracker, Number: Debug + PartialOrd> Transaction<Tracker, Number> {
198	pub async fn submit<
199		P: FinalitySyncPipeline<Number = Number>,
200		TC: TargetClient<P, TransactionTracker = Tracker>,
201	>(
202		target_client: &TC,
203		header: P::Header,
204		justification: P::FinalityProof,
205		is_free_execution_expected: bool,
206	) -> Result<Self, TC::Error> {
207		let header_number = header.number();
208		tracing::debug!(
209			target: "bridge",
210			source=%P::SOURCE_NAME,
211			target=%P::TARGET_NAME,
212			header=?header_number,
213			"Going to submit finality proof of header"
214		);
215
216		let tracker = target_client
217			.submit_finality_proof(header, justification, is_free_execution_expected)
218			.await?;
219		Ok(Transaction { tracker, header_number })
220	}
221
222	async fn track<
223		P: FinalitySyncPipeline<Number = Number>,
224		SC: SourceClient<P>,
225		TC: TargetClient<P>,
226	>(
227		self,
228		target_client: TC,
229	) -> Result<(), Error<P, SC::Error, TC::Error>> {
230		match self.tracker.wait().await {
231			TrackedTransactionStatus::Finalized(_) => {
232				// The transaction has been finalized, but it may have been finalized in the
233				// "failed" state. So let's check if the block number was actually updated.
234				target_client
235					.best_finalized_source_block_id()
236					.await
237					.map_err(Error::Target)
238					.and_then(|best_id_at_target| {
239						if self.header_number > best_id_at_target.0 {
240							return Err(Error::ProofSubmissionTxFailed {
241								submitted_number: self.header_number,
242								best_number_at_target: best_id_at_target.0,
243							})
244						}
245						Ok(())
246					})
247			},
248			TrackedTransactionStatus::Lost => Err(Error::ProofSubmissionTxLost),
249		}
250	}
251}

253/// Finality synchronization loop state.
254struct FinalityLoop<P: FinalitySyncPipeline, SC: SourceClient<P>, TC: TargetClient<P>> {
255	source_client: SC,
256	target_client: TC,
257
258	sync_params: FinalitySyncParams,
259	metrics_sync: Option<SyncLoopMetrics>,
260
261	progress: (Instant, Option<P::Number>),
262	retry_backoff: ExponentialBackoff,
263	finality_proofs_stream: FinalityProofsStream<P, SC>,
264	finality_proofs_buf: FinalityProofsBuf<P>,
265	best_submitted_number: Option<P::Number>,
266}
267
268impl<P: FinalitySyncPipeline, SC: SourceClient<P>, TC: TargetClient<P>> FinalityLoop<P, SC, TC> {
269	pub fn new(
270		source_client: SC,
271		target_client: TC,
272		sync_params: FinalitySyncParams,
273		metrics_sync: Option<SyncLoopMetrics>,
274	) -> Self {
275		Self {
276			source_client,
277			target_client,
278			sync_params,
279			metrics_sync,
280			progress: (Instant::now(), None),
281			retry_backoff: retry_backoff(),
282			finality_proofs_stream: FinalityProofsStream::new(),
283			finality_proofs_buf: FinalityProofsBuf::new(vec![]),
284			best_submitted_number: None,
285		}
286	}
287
288	fn update_progress(&mut self, info: &SyncInfo<P>) {
289		let (prev_time, prev_best_number_at_target) = self.progress;
290		let now = Instant::now();
291
292		let needs_update = now - prev_time > Duration::from_secs(10) ||
293			prev_best_number_at_target
294				.map(|prev_best_number_at_target| {
295					info.best_number_at_target.saturating_sub(prev_best_number_at_target) >
296						10.into()
297				})
298				.unwrap_or(true);
299
300		if !needs_update {
301			return
302		}
303
304		tracing::info!(
305			target: "bridge",
306			best_number_at_target=?info.best_number_at_target,
307			best_number_at_source=?info.best_number_at_source,
308			"Synced headers"
309		);
310
311		self.progress = (now, Some(info.best_number_at_target))
312	}
313
314	pub async fn select_header_to_submit(
315		&mut self,
316		info: &SyncInfo<P>,
317		free_headers_interval: Option<P::Number>,
318	) -> Result<Option<JustifiedHeader<P>>, Error<P, SC::Error, TC::Error>> {
319		// to see that the loop is progressing
320		tracing::trace!(
321			target: "bridge",
322			best_number_at_target=%info.best_number_at_target,
323			best_number_at_source=%info.best_number_at_source,
324			"Considering range of headers"
325		);
326
327		// read missing headers
328		let selector = JustifiedHeaderSelector::new::<SC, TC>(
329			&self.source_client,
330			info,
331			self.sync_params.headers_to_relay,
332			free_headers_interval,
333		)
334		.await?;
335		// if we see that the header schedules GRANDPA change, we need to submit it
336		if self.sync_params.headers_to_relay == HeadersToRelay::Mandatory {
337			return Ok(selector.select_mandatory())
338		}
339
340		// all headers that are missing from the target client are non-mandatory
341		// => even if we have already selected some header and its persistent finality proof,
342		// we may try to select better header by reading non-persistent proofs from the stream
343		self.finality_proofs_buf.fill(&mut self.finality_proofs_stream);
344		let maybe_justified_header = selector.select(
345			info,
346			self.sync_params.headers_to_relay,
347			free_headers_interval,
348			&self.finality_proofs_buf,
349		);
350
351		// remove obsolete 'recent' finality proofs + keep its size under certain limit
352		let oldest_finality_proof_to_keep = maybe_justified_header
353			.as_ref()
354			.map(|justified_header| justified_header.number())
355			.unwrap_or(info.best_number_at_target);
356		self.finality_proofs_buf.prune(
357			oldest_finality_proof_to_keep,
358			Some(self.sync_params.recent_finality_proofs_limit),
359		);
360
361		Ok(maybe_justified_header)
362	}
363
364	pub async fn run_iteration(
365		&mut self,
366		free_headers_interval: Option<P::Number>,
367	) -> Result<
368		Option<Transaction<TC::TransactionTracker, P::Number>>,
369		Error<P, SC::Error, TC::Error>,
370	> {
371		// read best source headers ids from source and target nodes
372		let info = SyncInfo::new(&self.source_client, &self.target_client).await?;
373		info.update_metrics(&self.metrics_sync);
374		self.update_progress(&info);
375
376		// if we have already submitted header, then we just need to wait for it
377		// if we're waiting too much, then we believe our transaction has been lost and restart sync
378		if Some(info.best_number_at_target) < self.best_submitted_number {
379			return Ok(None)
380		}
381
382		// submit new header if we have something new
383		match self.select_header_to_submit(&info, free_headers_interval).await? {
384			Some(header) => {
385				let transaction = Transaction::submit(
386					&self.target_client,
387					header.header,
388					header.proof,
389					self.sync_params.headers_to_relay == HeadersToRelay::Free,
390				)
391				.await
392				.map_err(Error::Target)?;
393				self.best_submitted_number = Some(transaction.header_number);
394				Ok(Some(transaction))
395			},
396			None => Ok(None),
397		}
398	}

400	async fn ensure_finality_proofs_stream(&mut self) -> Result<(), FailedClient> {
401		if let Err(e) = self.finality_proofs_stream.ensure_stream(&self.source_client).await {
402			if e.is_connection_error() {
403				return Err(FailedClient::Source)
404			}
405		}
406
407		Ok(())
408	}

410	/// Run finality relay loop until connection to one of nodes is lost.
411	async fn run_until_connection_lost(
412		&mut self,
413		exit_signal: impl Future<Output = ()>,
414	) -> Result<(), FailedClient> {
415		self.ensure_finality_proofs_stream().await?;
416		let proof_submission_tx_tracker = Fuse::terminated();
417		let exit_signal = exit_signal.fuse();
418		futures::pin_mut!(exit_signal, proof_submission_tx_tracker);
419
420		let free_headers_interval = free_headers_interval(&self.target_client).await?;
421
422		loop {
423			// run loop iteration
424			let next_tick = match self.run_iteration(free_headers_interval).await {
425				Ok(Some(tx)) => {
426					proof_submission_tx_tracker
427						.set(tx.track::<P, SC, _>(self.target_client.clone()).fuse());
428					self.retry_backoff.reset();
429					self.sync_params.tick
430				},
431				Ok(None) => {
432					self.retry_backoff.reset();
433					self.sync_params.tick
434				},
435				Err(error) => {
436					tracing::error!(target: "bridge", ?error, "Finality sync loop iteration has failed");
437					error.fail_if_connection_error()?;
438					self.retry_backoff
439						.next_backoff()
440						.unwrap_or(relay_utils::relay_loop::RECONNECT_DELAY)
441				},
442			};
443			self.ensure_finality_proofs_stream().await?;
444
445			// wait till exit signal, or new source block
446			select! {
447				proof_submission_result = proof_submission_tx_tracker => {
448					if let Err(e) = proof_submission_result {
449						tracing::error!(
450							target: "bridge",
451							error=?e,
452							target=%P::TARGET_NAME,
453							"Finality sync proof submission tx has failed."
454						);
455						self.best_submitted_number = None;
456						e.fail_if_connection_error()?;
457					}
458				},
459				_ = async_std::task::sleep(next_tick).fuse() => {},
460				_ = exit_signal => return Ok(()),
461			}
462		}
463	}
464
465	pub async fn run(
466		source_client: SC,
467		target_client: TC,
468		sync_params: FinalitySyncParams,
469		metrics_sync: Option<SyncLoopMetrics>,
470		exit_signal: impl Future<Output = ()>,
471	) -> Result<(), FailedClient> {
472		let mut finality_loop = Self::new(source_client, target_client, sync_params, metrics_sync);
473		finality_loop.run_until_connection_lost(exit_signal).await
474	}
475}

477async fn free_headers_interval<P: FinalitySyncPipeline>(
478	target_client: &impl TargetClient<P>,
479) -> Result<Option<P::Number>, FailedClient> {
480	match target_client.free_source_headers_interval().await {
481		Ok(Some(free_headers_interval)) if !free_headers_interval.is_zero() => {
482			tracing::trace!(
483				target: "bridge",
484				source=%P::SOURCE_NAME,
485				target=%P::TARGET_NAME,
486				?free_headers_interval,
487				"Free headers interval for headers"
488			);
489			Ok(Some(free_headers_interval))
490		},
491		Ok(Some(_free_headers_interval)) => {
492			tracing::trace!(
493				target: "bridge",
494				source=%P::SOURCE_NAME,
495				target=%P::TARGET_NAME,
496				"Free headers interval for headers is zero. Not submitting any free headers"
497			);
498			Ok(None)
499		},
500		Ok(None) => {
501			tracing::trace!(
502				target: "bridge",
503				source=%P::SOURCE_NAME,
504				target=%P::TARGET_NAME,
505				"Free headers interval for headers is None. Not submitting any free headers"
506			);
507
508			Ok(None)
509		},
510		Err(e) => {
511			tracing::error!(
512				target: "bridge",
513				error=?e,
514				source=%P::SOURCE_NAME,
515				target=%P::TARGET_NAME,
516				"Failed to read free headers interval for headers"
517			);
518			Err(FailedClient::Target)
519		},
520	}
521}

523/// Run finality proofs synchronization loop.
524pub async fn run<P: FinalitySyncPipeline>(
525	source_client: impl SourceClient<P>,
526	target_client: impl TargetClient<P>,
527	sync_params: FinalitySyncParams,
528	metrics_params: MetricsParams,
529	exit_signal: impl Future<Output = ()> + 'static + Send,
530) -> Result<(), relay_utils::Error> {
531	let exit_signal = exit_signal.shared();
532	relay_utils::relay_loop(source_client, target_client)
533		.with_metrics(metrics_params)
534		.loop_metric(SyncLoopMetrics::new(
535			Some(&metrics_prefix::<P>()),
536			"source",
537			"source_at_target",
538		)?)?
539		.expose()
540		.await?
541		.run(metrics_prefix::<P>(), move |source_client, target_client, metrics| {
542			FinalityLoop::run(
543				source_client,
544				target_client,
545				sync_params.clone(),
546				metrics,
547				exit_signal.clone(),
548			)
549		})
550		.await
551}

636	#[test]
637	fn finality_sync_loop_works() {
638		let (client_data, result) = run_sync_loop(|data| {
639			// header#7 has persistent finality proof, but it isn't mandatory => it isn't submitted,
640			// because header#8 has persistent finality proof && it is mandatory => it is submitted
641			// header#9 has persistent finality proof, but it isn't mandatory => it is submitted,
642			// because   there are no more persistent finality proofs
643			//
644			// once this ^^^ is done, we generate more blocks && read proof for blocks 12 and 14
645			// from the stream
646			if data.target_best_block_id.0 == 9 {
647				data.source_best_block_number = 14;
648				data.source_headers.insert(11, (TestSourceHeader(false, 11, 11), None));
649				data.source_headers
650					.insert(12, (TestSourceHeader(false, 12, 12), Some(TestFinalityProof(12))));
651				data.source_headers.insert(13, (TestSourceHeader(false, 13, 13), None));
652				data.source_headers
653					.insert(14, (TestSourceHeader(false, 14, 14), Some(TestFinalityProof(14))));
654			}
655			// once this ^^^ is done, we generate more blocks && read persistent proof for block 16
656			if data.target_best_block_id.0 == 14 {
657				data.source_best_block_number = 17;
658				data.source_headers.insert(15, (TestSourceHeader(false, 15, 15), None));
659				data.source_headers
660					.insert(16, (TestSourceHeader(false, 16, 16), Some(TestFinalityProof(16))));
661				data.source_headers.insert(17, (TestSourceHeader(false, 17, 17), None));
662			}
663
664			data.target_best_block_id.0 == 16
665		});
666
667		assert_eq!(result, Ok(()));
668		assert_eq!(
669			client_data.target_headers,
670			vec![
671				// before adding 11..14: finality proof for mandatory header#8
672				(TestSourceHeader(true, 8, 8), TestFinalityProof(8)),
673				// before adding 11..14: persistent finality proof for non-mandatory header#9
674				(TestSourceHeader(false, 9, 9), TestFinalityProof(9)),
675				// after adding 11..14: ephemeral finality proof for non-mandatory header#14
676				(TestSourceHeader(false, 14, 14), TestFinalityProof(14)),
677				// after adding 15..17: persistent finality proof for non-mandatory header#16
678				(TestSourceHeader(false, 16, 16), TestFinalityProof(16)),
679			],
680		);
681	}

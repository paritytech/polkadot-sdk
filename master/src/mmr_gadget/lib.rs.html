19//! # MMR offchain gadget
20//!
21//! The MMR offchain gadget is run alongside `pallet-mmr` to assist it with offchain
22//! canonicalization of finalized MMR leaves and nodes.
23//! The gadget should only be run on nodes that have Indexing API enabled (otherwise
24//! `pallet-mmr` cannot write to offchain and this gadget has nothing to do).
25//!
26//! The runtime `pallet-mmr` creates one new MMR leaf per block and all inner MMR parent nodes
27//! generated by the MMR when adding said leaf. MMR nodes are stored both in:
28//! - on-chain storage - hashes only; not full leaf content;
29//! - off-chain storage - via Indexing API, full leaf content (and all internal nodes as well) is
30//!   saved to the Off-chain DB using a key derived from `parent_hash` and node index in MMR. The
31//!   `parent_hash` is also used within the key to avoid conflicts and overwrites on forks (leaf
32//!   data is only allowed to reference data coming from parent block).
33//!
34//! This gadget is driven by block finality and in responsible for pruning stale forks from
35//! offchain db, and moving finalized forks under a "canonical" key based solely on node `pos`
36//! in the MMR.

57/// Logging target for the mmr gadget.
58pub const LOG_TARGET: &str = "mmr";
59
60/// A convenience MMR client trait that defines all the type bounds a MMR client
61/// has to satisfy and defines some helper methods.
62pub trait MmrClient<B, BE>:
63	BlockchainEvents<B> + HeaderBackend<B> + HeaderMetadata<B> + ProvideRuntimeApi<B>
64where
65	B: Block,
66	BE: Backend<B>,
67	Self::Api: MmrApi<B, MmrRootHash, NumberFor<B>>,
68{
69	/// Get the block number where the mmr pallet was added to the runtime.
70	fn first_mmr_block_num(&self, notification: &FinalityNotification<B>) -> Option<NumberFor<B>> {
71		let best_block_hash = notification.header.hash();
72		let best_block_number = *notification.header.number();
73		match self.runtime_api().mmr_leaf_count(best_block_hash) {
74			Ok(Ok(mmr_leaf_count)) => {
75				match utils::first_mmr_block_num::<B::Header>(best_block_number, mmr_leaf_count) {
76					Ok(first_mmr_block) => {
77						debug!(
78							target: LOG_TARGET,
79							"pallet-mmr detected at block {:?} with genesis at block {:?}",
80							best_block_number,
81							first_mmr_block
82						);
83						Some(first_mmr_block)
84					},
85					Err(e) => {
86						error!(
87							target: LOG_TARGET,
88							"Error calculating the first mmr block: {:?}", e
89						);
90						None
91					},
92				}
93			},
94			_ => {
95				trace!(
96					target: LOG_TARGET,
97					"pallet-mmr not detected at block {:?} ... (best finalized {:?})",
98					best_block_number,
99					notification.header.number()
100				);
101				None
102			},
103		}
104	}
105}

117struct OffchainMmrBuilder<B: Block, BE: Backend<B>, C> {
118	backend: Arc<BE>,
119	client: Arc<C>,
120	offchain_db: OffchainDb<BE::OffchainStorage>,
121	indexing_prefix: Vec<u8>,
122
123	_phantom: PhantomData<B>,
124}
125
126impl<B, BE, C> OffchainMmrBuilder<B, BE, C>
127where
128	B: Block,
129	BE: Backend<B>,
130	C: MmrClient<B, BE>,
131	C::Api: MmrApi<B, MmrRootHash, NumberFor<B>>,
132{
133	async fn try_build(
134		self,
135		finality_notifications: &mut FinalityNotifications<B>,
136	) -> Option<OffchainMmr<B, BE, C>> {
137		while let Some(notification) = finality_notifications.next().await {
138			if let Some(first_mmr_block_num) = self.client.first_mmr_block_num(&notification) {
139				let mut offchain_mmr = OffchainMmr::new(
140					self.backend,
141					self.client,
142					self.offchain_db,
143					self.indexing_prefix,
144					first_mmr_block_num,
145				)?;
146				// We need to make sure all blocks leading up to current notification
147				// have also been canonicalized.
148				offchain_mmr.canonicalize_catch_up(&notification);
149				// We have to canonicalize and prune the blocks in the finality
150				// notification that lead to building the offchain-mmr as well.
151				offchain_mmr.canonicalize_and_prune(notification);
152				return Some(offchain_mmr)
153			}
154		}
155
156		error!(
157			target: LOG_TARGET,
158			"Finality notifications stream closed unexpectedly. \
159			Couldn't build the canonicalization engine",
160		);
161		None
162	}
163}

165/// A MMR Gadget.
166pub struct MmrGadget<B: Block, BE: Backend<B>, C> {
167	finality_notifications: FinalityNotifications<B>,
168
169	_phantom: PhantomData<(B, BE, C)>,
170}
171
172impl<B, BE, C> MmrGadget<B, BE, C>
173where
174	B: Block,
175	<B::Header as Header>::Number: Into<LeafIndex>,
176	BE: Backend<B>,
177	C: MmrClient<B, BE>,
178	C::Api: MmrApi<B, MmrRootHash, NumberFor<B>>,
179{
180	async fn run(mut self, builder: OffchainMmrBuilder<B, BE, C>) {
181		let mut offchain_mmr = match builder.try_build(&mut self.finality_notifications).await {
182			Some(offchain_mmr) => offchain_mmr,
183			None => return,
184		};
185
186		while let Some(notification) = self.finality_notifications.next().await {
187			offchain_mmr.canonicalize_and_prune(notification);
188		}
189	}
190
191	/// Create and run the MMR gadget.
192	pub async fn start(client: Arc<C>, backend: Arc<BE>, indexing_prefix: Vec<u8>) {
193		let offchain_db = match backend.offchain_storage() {
194			Some(offchain_storage) => OffchainDb::new(offchain_storage),
195			None => {
196				warn!(
197					target: LOG_TARGET,
198					"Can't spawn a MmrGadget for a node without offchain storage."
199				);
200				return
201			},
202		};
203
204		let mmr_gadget = MmrGadget::<B, BE, C> {
205			finality_notifications: client.finality_notification_stream(),
206
207			_phantom: Default::default(),
208		};
209		mmr_gadget
210			.run(OffchainMmrBuilder {
211				backend,
212				client,
213				offchain_db,
214				indexing_prefix,
215				_phantom: Default::default(),
216			})
217			.await
218	}
219}

224	use sp_runtime::generic::BlockId;

227	#[test]
228	fn mmr_first_block_is_computed_correctly() {
229		// Check the case where the first block is also the first block with MMR.
230		run_test_with_mmr_gadget(|client| async move {
231			// G -> A1 -> A2
232			//      |
233			//      | -> first mmr block
234
235			let a1 = client.import_block(&BlockId::Number(0), b"a1", Some(0)).await;
236			let a2 = client.import_block(&BlockId::Hash(a1.hash()), b"a2", Some(1)).await;
237
238			client.finalize_block(a1.hash(), Some(1));
239			tokio::time::sleep(Duration::from_millis(200)).await;
240			// expected finalized heads: a1
241			client.assert_canonicalized(&[&a1]);
242			client.assert_not_pruned(&[&a2]);
243		});
244
245		// Check the case where the first block with MMR comes later.
246		run_test_with_mmr_gadget(|client| async move {
247			// G -> A1 -> A2 -> A3 -> A4 -> A5 -> A6
248			//                        |
249			//                        | -> first mmr block
250
251			let a1 = client.import_block(&BlockId::Number(0), b"a1", None).await;
252			let a2 = client.import_block(&BlockId::Hash(a1.hash()), b"a2", None).await;
253			let a3 = client.import_block(&BlockId::Hash(a2.hash()), b"a3", None).await;
254			let a4 = client.import_block(&BlockId::Hash(a3.hash()), b"a4", Some(0)).await;
255			let a5 = client.import_block(&BlockId::Hash(a4.hash()), b"a5", Some(1)).await;
256			let a6 = client.import_block(&BlockId::Hash(a5.hash()), b"a6", Some(2)).await;
257
258			client.finalize_block(a5.hash(), Some(2));
259			tokio::time::sleep(Duration::from_millis(200)).await;
260			// expected finalized heads: a4, a5
261			client.assert_canonicalized(&[&a4, &a5]);
262			client.assert_not_pruned(&[&a6]);
263		});
264	}

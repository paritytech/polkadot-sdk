40/// The structure used for parsing the runtime api implementations.
41struct RuntimeApiImpls {
42	impls: Vec<ItemImpl>,
43}
44
45impl Parse for RuntimeApiImpls {
46	fn parse(input: ParseStream) -> Result<Self> {
47		let mut impls = Vec::new();
48
49		while !input.is_empty() {
50			impls.push(ItemImpl::parse(input)?);
51		}
52
53		if impls.is_empty() {
54			Err(Error::new(Span::call_site(), "No api implementation given!"))
55		} else {
56			Ok(Self { impls })
57		}
58	}
59}

61/// Generates the call to the implementation of the requested function.
62/// The generated code includes decoding of the input arguments and encoding of the output.
63fn generate_impl_call(
64	signature: &Signature,
65	runtime: &Type,
66	input: &Ident,
67	impl_trait: &Path,
68	api_version: &ApiVersion,
69) -> Result<TokenStream> {
70	let params =
71		extract_parameter_names_types_and_borrows(signature, AllowSelfRefInParameters::No)?;
72
73	let c = generate_crate_access();
74	let fn_name = &signature.ident;
75	let fn_name_str = fn_name.to_string();
76	let pnames = params.iter().map(|v| &v.0);
77	let pnames2 = params.iter().map(|v| &v.0);
78	let ptypes = params.iter().map(|v| &v.1);
79	let pborrow = params.iter().map(|v| &v.2);
80
81	let decode_params = if params.is_empty() {
82		quote!(
83			if !#input.is_empty() {
84				panic!(
85					"Bad input data provided to {}: expected no parameters, but input buffer is not empty. Nothing bad happened: someone sent an invalid transaction to the node.",
86					#fn_name_str
87				);
88			}
89		)
90	} else {
91		let let_binding = if params.len() == 1 {
92			quote! {
93				let #( #pnames )* : #( #ptypes )*
94			}
95		} else {
96			quote! {
97				let ( #( #pnames ),* ) : ( #( #ptypes ),* )
98			}
99		};
100
101		quote!(
102			#let_binding =
103				match #c::Decode::decode(
104					&mut #input,
105				) {
106					Ok(res) => res,
107					Err(e) => panic!("Bad input data provided to {}: {}. Nothing bad happened: someone sent an invalid transaction to the node.", #fn_name_str, e),
108				};
109		)
110	};
111
112	let fn_calls = if let Some(feature_gated) = &api_version.feature_gated {
113		let pnames = pnames2;
114		let pnames2 = pnames.clone();
115		let pborrow2 = pborrow.clone();
116
117		let feature_name = &feature_gated.0;
118		let impl_trait_fg = extend_with_api_version(impl_trait.clone(), Some(feature_gated.1));
119		let impl_trait = extend_with_api_version(impl_trait.clone(), api_version.custom);
120
121		quote!(
122			#[cfg(feature = #feature_name)]
123			#[allow(deprecated)]
124			let r = <#runtime as #impl_trait_fg>::#fn_name(#( #pborrow #pnames ),*);
125
126			#[cfg(not(feature = #feature_name))]
127			#[allow(deprecated)]
128			let r = <#runtime as #impl_trait>::#fn_name(#( #pborrow2 #pnames2 ),*);
129
130			r
131		)
132	} else {
133		let pnames = pnames2;
134		let impl_trait = extend_with_api_version(impl_trait.clone(), api_version.custom);
135
136		quote!(
137			#[allow(deprecated)]
138			<#runtime as #impl_trait>::#fn_name(#( #pborrow #pnames ),*)
139		)
140	};
141
142	Ok(quote!(
143		#decode_params
144
145		#fn_calls
146	))
147}

149/// Generate all the implementation calls for the given functions.
150fn generate_impl_calls(
151	impls: &[ItemImpl],
152	input: &Ident,
153) -> Result<Vec<(Ident, Ident, TokenStream, Vec<Attribute>)>> {
154	let mut impl_calls = Vec::new();
155
156	for impl_ in impls {
157		let trait_api_ver = extract_api_version(&impl_.attrs, impl_.span())?;
158		let impl_trait_path = extract_impl_trait(impl_, RequireQualifiedTraitPath::Yes)?;
159		let impl_trait = extend_with_runtime_decl_path(impl_trait_path.clone());
160		let impl_trait_ident = &impl_trait_path
161			.segments
162			.last()
163			.ok_or_else(|| Error::new(impl_trait_path.span(), "Empty trait path not possible!"))?
164			.ident;
165
166		for item in &impl_.items {
167			if let ImplItem::Fn(method) = item {
168				let impl_call = generate_impl_call(
169					&method.sig,
170					&impl_.self_ty,
171					input,
172					&impl_trait,
173					&trait_api_ver,
174				)?;
175				let mut attrs = filter_cfg_and_allow_attrs(&impl_.attrs);
176
177				// Add any `#[cfg(feature = X)]` attributes of the method to result
178				attrs.extend(filter_cfg_and_allow_attrs(&method.attrs));
179
180				impl_calls.push((
181					impl_trait_ident.clone(),
182					method.sig.ident.clone(),
183					impl_call,
184					attrs,
185				));
186			}
187		}
188	}
189
190	Ok(impl_calls)
191}

193/// Generate the dispatch function that is used in native to call into the runtime.
194fn generate_dispatch_function(impls: &[ItemImpl]) -> Result<TokenStream> {
195	let data = Ident::new("_sp_api_input_data_", Span::call_site());
196	let c = generate_crate_access();
197	let impl_calls =
198		generate_impl_calls(impls, &data)?
199			.into_iter()
200			.map(|(trait_, fn_name, impl_, attrs)| {
201				let name = prefix_function_with_trait(&trait_, &fn_name);
202				quote!(
203					#( #attrs )*
204					#name => Some(#c::Encode::encode(&{ #impl_ })),
205				)
206			});
207
208	Ok(quote!(
209		#c::std_enabled! {
210			pub fn dispatch(method: &str, mut #data: &[u8]) -> Option<Vec<u8>> {
211				match method {
212					#( #impl_calls )*
213					_ => None,
214				}
215			}
216		}
217	))
218}

220/// Generate the interface functions that are used to call into the runtime in wasm.
221fn generate_wasm_interface(impls: &[ItemImpl]) -> Result<TokenStream> {
222	let input = Ident::new("input", Span::call_site());
223	let c = generate_crate_access();
224
225	let impl_calls =
226        generate_impl_calls(impls, &input)?
227            .into_iter()
228            .map(|(trait_, fn_name, impl_, attrs)| {
229                let fn_name =
230                    Ident::new(&prefix_function_with_trait(&trait_, &fn_name), Span::call_site());
231
232                quote!(
233                    #c::std_disabled! {
234                        #( #attrs )*
235                        #[no_mangle]
236                        #[cfg_attr(any(target_arch = "riscv32", target_arch = "riscv64"), #c::__private::polkavm_export(abi = #c::__private::polkavm_abi))]
237                        pub unsafe extern fn #fn_name(input_data: *mut u8, input_len: usize) -> u64 {
238                            let mut #input = if input_len == 0 {
239                                &[0u8; 0]
240                            } else {
241                                unsafe {
242                                    ::core::slice::from_raw_parts(input_data, input_len)
243                                }
244                            };
245
246                            #c::init_runtime_logger();
247
248                            let output = (move || { #impl_ })();
249                            #c::to_substrate_wasm_fn_return_value(&output)
250                        }
251                    }
252                )
253            });
254
255	Ok(quote!( #( #impl_calls )* ))
256}

258fn generate_runtime_api_base_structures() -> Result<TokenStream> {
259	let crate_ = generate_crate_access();
260
261	Ok(quote!(
262		pub struct RuntimeApi {}
263		#crate_::std_enabled! {
264			/// Implements all runtime apis for the client side.
265			pub struct RuntimeApiImpl<Block: #crate_::BlockT, C: #crate_::CallApiAt<Block> + 'static> {
266				call: &'static C,
267				transaction_depth: std::cell::RefCell<u16>,
268				changes: std::cell::RefCell<#crate_::OverlayedChanges<#crate_::HashingFor<Block>>>,
269				recorder: std::option::Option<#crate_::ProofRecorder<Block>>,
270				call_context: #crate_::CallContext,
271				extensions: std::cell::RefCell<#crate_::Extensions>,
272				extensions_generated_for: std::cell::RefCell<std::option::Option<Block::Hash>>,
273			}
274
275			#[automatically_derived]
276			impl<Block: #crate_::BlockT, C: #crate_::CallApiAt<Block>> #crate_::ApiExt<Block> for
277				RuntimeApiImpl<Block, C>
278			{
279				fn execute_in_transaction<F: FnOnce(&Self) -> #crate_::TransactionOutcome<R>, R>(
280					&self,
281					call: F,
282				) -> R where Self: Sized {
283					self.start_transaction();
284
285					*std::cell::RefCell::borrow_mut(&self.transaction_depth) += 1;
286					let res = call(self);
287					std::cell::RefCell::borrow_mut(&self.transaction_depth)
288						.checked_sub(1)
289						.expect("Transactions are opened and closed together; qed");
290
291					self.commit_or_rollback_transaction(
292						std::matches!(res, #crate_::TransactionOutcome::Commit(_))
293					);
294
295					res.into_inner()
296				}

302					#crate_::CallApiAt::<Block>::runtime_version_at(self.call, at)
303					.map(|v| #crate_::RuntimeVersion::has_api_with(&v, &A::ID, |v| v == A::VERSION))
304				}
305
306				fn has_api_with<A: #crate_::RuntimeApiInfo + ?Sized, P: Fn(u32) -> bool>(
307					&self,
308					at: <Block as #crate_::BlockT>::Hash,
309					pred: P,
310				) -> std::result::Result<bool, #crate_::ApiError> where Self: Sized {
311					#crate_::CallApiAt::<Block>::runtime_version_at(self.call, at)
312					.map(|v| #crate_::RuntimeVersion::has_api_with(&v, &A::ID, pred))
313				}
314
315				fn api_version<A: #crate_::RuntimeApiInfo + ?Sized>(
316					&self,
317					at: <Block as #crate_::BlockT>::Hash,
318				) -> std::result::Result<Option<u32>, #crate_::ApiError> where Self: Sized {
319					#crate_::CallApiAt::<Block>::runtime_version_at(self.call, at)
320					.map(|v| #crate_::RuntimeVersion::api_version(&v, &A::ID))
321				}
322
323				fn record_proof(&mut self) {
324					self.recorder = std::option::Option::Some(std::default::Default::default());
325				}
326
327				fn record_proof_with_recorder(&mut self, recorder: #crate_::ProofRecorder<Block>) {

337				) -> std::option::Option<#crate_::StorageProof> {
338					let recorder = std::option::Option::take(&mut self.recorder);
339					std::option::Option::map(recorder, |recorder| {
340						#crate_::ProofRecorder::<Block>::drain_storage_proof(recorder)

342				}

344				fn into_storage_changes<B: #crate_::StateBackend<#crate_::HashingFor<Block>>>(

381					call: &'a C,
382				) -> #crate_::ApiRef<'a, Self::RuntimeApi> {
383					RuntimeApiImpl {
384						call: unsafe { std::mem::transmute(call) },
385						transaction_depth: 0.into(),
386						changes: std::default::Default::default(),
387						recorder: std::default::Default::default(),
388						call_context: #crate_::CallContext::Offchain,
389						extensions: std::default::Default::default(),
390						extensions_generated_for: std::default::Default::default(),
391					}.into()
392				}
393			}
394
395			#[automatically_derived]
396			impl<Block: #crate_::BlockT, C: #crate_::CallApiAt<Block>> RuntimeApiImpl<Block, C> {
397				fn commit_or_rollback_transaction(&self, commit: bool) {
398					let proof = "\
399                    We only close a transaction when we opened one ourself.
400                    Other parts of the runtime that make use of transactions (state-machine)
401                    also balance their transactions. The runtime cannot close client initiated
402                    transactions; qed";
403
404					let res = if commit {
405						let res = if let Some(recorder) = &self.recorder {
406							#crate_::ProofRecorder::<Block>::commit_transaction(&recorder)
407						} else {
408							Ok(())
409						};
410
411						let res2 = #crate_::OverlayedChanges::commit_transaction(
412							&mut std::cell::RefCell::borrow_mut(&self.changes)
413						);
414
415						#crate_::Extensions::commit_transaction(
416							&mut std::cell::RefCell::borrow_mut(&self.extensions),
417							#crate_::TransactionType::Host,
418						);
419
420						// Will panic on an `Err` below, however we should call commit
421						// on the recorder and the changes together.
422						std::result::Result::and(res, std::result::Result::map_err(res2, drop))
423					} else {
424						let res = if let Some(recorder) = &self.recorder {
425							#crate_::ProofRecorder::<Block>::rollback_transaction(&recorder)
426						} else {
427							Ok(())
428						};
429
430						let res2 = #crate_::OverlayedChanges::rollback_transaction(
431							&mut std::cell::RefCell::borrow_mut(&self.changes)
432						);
433
434						#crate_::Extensions::rollback_transaction(
435							&mut std::cell::RefCell::borrow_mut(&self.extensions),
436							#crate_::TransactionType::Host,
437						);
438
439						// Will panic on an `Err` below, however we should call commit
440						// on the recorder and the changes together.
441						std::result::Result::and(res, std::result::Result::map_err(res2, drop))
442					};
443
444					std::result::Result::expect(res, proof);
445				}
446
447				fn start_transaction(&self) {
448					#crate_::OverlayedChanges::start_transaction(
449						&mut std::cell::RefCell::borrow_mut(&self.changes)
450					);
451					if let Some(recorder) = &self.recorder {
452						#crate_::ProofRecorder::<Block>::start_transaction(&recorder);
453					}
454
455					#crate_::Extensions::start_transaction(
456						&mut std::cell::RefCell::borrow_mut(&self.extensions),
457						#crate_::TransactionType::Host,
458					);
459				}
460			}
461		}
462	))
463}

465/// Extend the given trait path with module that contains the declaration of the trait for the
466/// runtime.
467fn extend_with_runtime_decl_path(mut trait_: Path) -> Path {
468	let runtime = {
469		let trait_name = &trait_
470			.segments
471			.last()
472			.as_ref()
473			.expect("Trait path should always contain at least one item; qed")
474			.ident;
475
476		generate_runtime_mod_name_for_trait(trait_name)
477	};
478
479	let pos = trait_.segments.len() - 1;
480	trait_.segments.insert(pos, runtime.into());
481	trait_
482}
483
484fn extend_with_api_version(mut trait_: Path, version: Option<u32>) -> Path {
485	let version = if let Some(v) = version {
486		v
487	} else {
488		// nothing to do
489		return trait_;
490	};
491
492	let trait_name = &mut trait_
493		.segments
494		.last_mut()
495		.expect("Trait path should always contain at least one item; qed")
496		.ident;
497	*trait_name = versioned_trait_name(trait_name, version);
498
499	trait_
500}

502/// Adds a feature guard to `attributes`.
503///
504/// Depending on `enable`, the feature guard either enables ('feature = "something"`) or disables
505/// (`not(feature = "something")`).
506fn add_feature_guard(attrs: &mut Vec<Attribute>, feature_name: &str, enable: bool) {
507	let attr = match enable {
508		true => parse_quote!(#[cfg(feature = #feature_name)]),
509		false => parse_quote!(#[cfg(not(feature = #feature_name))]),
510	};
511	attrs.push(attr);
512}

514/// Generates the implementations of the apis for the runtime.
515fn generate_api_impl_for_runtime(impls: &[ItemImpl]) -> Result<TokenStream> {
516	let mut impls_prepared = Vec::new();
517
518	// We put `runtime` before each trait to get the trait that is intended for the runtime and
519	// we put the `RuntimeBlock` as first argument for the trait generics.
520	for impl_ in impls.iter() {
521		let trait_api_ver = extract_api_version(&impl_.attrs, impl_.span())?;
522
523		let mut impl_ = impl_.clone();
524		impl_.attrs = filter_cfg_and_allow_attrs(&impl_.attrs);
525
526		let trait_ = extract_impl_trait(&impl_, RequireQualifiedTraitPath::Yes)?.clone();
527		let trait_ = extend_with_runtime_decl_path(trait_);
528		// If the trait api contains feature gated version - there are staging methods in it. Handle
529		// them explicitly here by adding staging implementation with `#cfg(feature = ...)` and
530		// stable implementation with `#[cfg(not(feature = ...))]`.
531		if let Some(feature_gated) = trait_api_ver.feature_gated {
532			let mut feature_gated_impl = impl_.clone();
533			add_feature_guard(&mut feature_gated_impl.attrs, &feature_gated.0, true);
534			feature_gated_impl.trait_.as_mut().unwrap().1 =
535				extend_with_api_version(trait_.clone(), Some(feature_gated.1));
536
537			impls_prepared.push(feature_gated_impl);
538
539			// Finally add `#[cfg(not(feature = ...))]` for the stable implementation (which is
540			// generated outside this if).
541			add_feature_guard(&mut impl_.attrs, &feature_gated.0, false);
542		}
543
544		// Generate stable trait implementation.
545		let trait_ = extend_with_api_version(trait_, trait_api_ver.custom);
546		impl_.trait_.as_mut().unwrap().1 = trait_;
547		impls_prepared.push(impl_);
548	}
549
550	Ok(quote!( #( #impls_prepared )* ))
551}

553/// Auxiliary data structure that is used to convert `impl Api for Runtime` to
554/// `impl Api for RuntimeApi`.
555/// This requires us to replace the runtime `Block` with the node `Block`,
556/// `impl Api for Runtime` with `impl Api for RuntimeApi` and replace the method implementations
557/// with code that calls into the runtime.
558struct ApiRuntimeImplToApiRuntimeApiImpl<'a> {
559	runtime_block: &'a TypePath,
560}
561
562impl<'a> ApiRuntimeImplToApiRuntimeApiImpl<'a> {
563	/// Process the given item implementation.
564	fn process(mut self, input: ItemImpl) -> ItemImpl {
565		let mut input = self.fold_item_impl(input);
566
567		let crate_ = generate_crate_access();
568
569		// Delete all functions, because all of them are default implemented by
570		// `decl_runtime_apis!`. We only need to implement the `__runtime_api_internal_call_api_at`
571		// function.
572		input.items.clear();
573		input.items.push(parse_quote! {
574			fn __runtime_api_internal_call_api_at(
575				&self,
576				at: <__SrApiBlock__ as #crate_::BlockT>::Hash,
577				params: std::vec::Vec<u8>,
578				fn_name: &dyn Fn(#crate_::RuntimeVersion) -> &'static str,
579			) -> std::result::Result<std::vec::Vec<u8>, #crate_::ApiError> {
580				// If we are not already in a transaction, we should create a new transaction
581				// and then commit/roll it back at the end!
582				let transaction_depth = *std::cell::RefCell::borrow(&self.transaction_depth);
583
584				if transaction_depth == 0 {
585					self.start_transaction();
586				}
587
588				let res = (|| {
589					let version = #crate_::CallApiAt::<__SrApiBlock__>::runtime_version_at(
590						self.call,
591						at,
592					)?;
593
594					match &mut *std::cell::RefCell::borrow_mut(&self.extensions_generated_for) {
595						Some(generated_for) => {
596							if *generated_for != at {
597								return std::result::Result::Err(
598									#crate_::ApiError::UsingSameInstanceForDifferentBlocks
599								)
600							}
601						},
602						generated_for @ None => {
603							#crate_::CallApiAt::<__SrApiBlock__>::initialize_extensions(
604								self.call,
605								at,
606								&mut std::cell::RefCell::borrow_mut(&self.extensions),
607							)?;
608
609							*generated_for = Some(at);
610						}
611					}
612
613					let params = #crate_::CallApiAtParams {
614						at,
615						function: (*fn_name)(version),
616						arguments: params,
617						overlayed_changes: &self.changes,
618						call_context: self.call_context,
619						recorder: &self.recorder,
620						extensions: &self.extensions,
621					};
622
623					#crate_::CallApiAt::<__SrApiBlock__>::call_api_at(
624						self.call,
625						params,
626					)
627				})();
628
629				if transaction_depth == 0 {
630					self.commit_or_rollback_transaction(std::result::Result::is_ok(&res));
631				}
632
633				res
634			}
635		});
636
637		input
638	}
639}

658		));

680/// Generate the implementations of the runtime apis for the `RuntimeApi` type.
681fn generate_api_impl_for_runtime_api(impls: &[ItemImpl]) -> Result<TokenStream> {
682	let mut result = Vec::with_capacity(impls.len());
683
684	for impl_ in impls {
685		let impl_trait_path = extract_impl_trait(impl_, RequireQualifiedTraitPath::Yes)?;
686		let runtime_block = extract_block_type_from_trait_path(impl_trait_path)?;
687		let mut runtime_mod_path = extend_with_runtime_decl_path(impl_trait_path.clone());
688		// remove the trait to get just the module path
689		runtime_mod_path.segments.pop();
690
691		let mut processed_impl =
692			ApiRuntimeImplToApiRuntimeApiImpl { runtime_block }.process(impl_.clone());
693
694		processed_impl.attrs.push(parse_quote!(#[automatically_derived]));
695
696		result.push(processed_impl);
697	}
698
699	let crate_ = generate_crate_access();
700
701	Ok(quote!( #crate_::std_enabled! { #( #result )* } ))
702}

704fn populate_runtime_api_versions(
705	result: &mut Vec<TokenStream>,
706	sections: &mut Vec<TokenStream>,
707	attrs: Vec<Attribute>,
708	id: Path,
709	version: TokenStream,
710	crate_access: &TokenStream,
711) {
712	result.push(quote!(

714			(#id, #version)

722				#[link_section = "runtime_apis"]
723				static SECTION_CONTENTS: [u8; 12] = #crate_access::serialize_runtime_api_info(#id, #version);
724			};
725		}
726	));

729/// Generates `RUNTIME_API_VERSIONS` that holds all version information about the implemented
730/// runtime apis.
731fn generate_runtime_api_versions(impls: &[ItemImpl]) -> Result<TokenStream> {
732	let mut result = Vec::<TokenStream>::with_capacity(impls.len());
733	let mut sections = Vec::<TokenStream>::with_capacity(impls.len());
734	let mut processed_traits = HashMap::new();
735
736	let c = generate_crate_access();
737
738	for impl_ in impls {
739		let versions = extract_api_version(&impl_.attrs, impl_.span())?;
740		let api_ver = versions.custom.map(|a| a as u32);
741
742		let mut path = extend_with_runtime_decl_path(
743			extract_impl_trait(impl_, RequireQualifiedTraitPath::Yes)?.clone(),
744		);
745		// Remove the trait
746		let trait_ = path
747			.segments
748			.pop()
749			.expect("extract_impl_trait already checks that this is valid; qed")
750			.into_value()
751			.ident;
752
753		let span = trait_.span();
754		if let Some(other_span) = processed_traits.insert(trait_, span) {
755			let mut error = Error::new(
756				span,
757				"Two traits with the same name detected! \
758                    The trait name is used to generate its ID. \
759                    Please rename one trait at the declaration!",
760			);
761
762			error.combine(Error::new(other_span, "First trait implementation."));
763
764			return Err(error);
765		}
766
767		let id: Path = parse_quote!( #path ID );
768		let mut attrs = filter_cfg_and_allow_attrs(&impl_.attrs);
769
770		// Handle API versioning
771		// If feature gated version is set - handle it first
772		if let Some(feature_gated) = versions.feature_gated {
773			let feature_gated_version = feature_gated.1 as u32;
774			// the attributes for the feature gated staging api
775			let mut feature_gated_attrs = attrs.clone();
776			add_feature_guard(&mut feature_gated_attrs, &feature_gated.0, true);
777			populate_runtime_api_versions(
778				&mut result,
779				&mut sections,
780				feature_gated_attrs,
781				id.clone(),
782				quote!( #feature_gated_version ),
783				&c,
784			);
785
786			// Add `#[cfg(not(feature ...))]` to the initial attributes. If the staging feature flag
787			// is not set we want to set the stable api version
788			add_feature_guard(&mut attrs, &feature_gated.0, false);
789		}
790
791		// Now add the stable api version to the versions list. If the api has got staging functions
792		// there might be a `#[cfg(not(feature ...))]` attribute attached to the stable version.
793		let base_api_version = quote!( #path VERSION );
794		let api_ver = api_ver.map(|a| quote!( #a )).unwrap_or_else(|| base_api_version);
795		populate_runtime_api_versions(&mut result, &mut sections, attrs, id, api_ver, &c);
796	}
797
798	Ok(quote!(
799		pub const RUNTIME_API_VERSIONS: #c::ApisVec = #c::create_apis_vec!([ #( #result ),* ]);
800
801		#( #sections )*
802	))
803}

805/// replaces `Self` with explicit `ItemImpl.self_ty`.
806struct ReplaceSelfImpl {
807	self_ty: Box<Type>,
808}
809
810impl ReplaceSelfImpl {
811	/// Replace `Self` with `ItemImpl.self_ty`
812	fn replace(&mut self, trait_: &mut ItemImpl) {
813		visit_mut::visit_item_impl_mut(self, trait_)
814	}
815}
816
817impl VisitMut for ReplaceSelfImpl {
818	fn visit_type_mut(&mut self, ty: &mut syn::Type) {
819		match ty {
820			Type::Path(p) if p.path.is_ident("Self") => {
821				*ty = *self.self_ty.clone();
822			},
823			ty => syn::visit_mut::visit_type_mut(self, ty),
824		}
825	}
826}
827
828/// Rename `Self` to `ItemImpl.self_ty` in all items.
829fn rename_self_in_trait_impls(impls: &mut [ItemImpl]) {
830	impls.iter_mut().for_each(|i| {
831		let mut checker = ReplaceSelfImpl { self_ty: i.self_ty.clone() };
832		checker.replace(i);
833	});
834}

836/// The implementation of the `impl_runtime_apis!` macro.
837pub fn impl_runtime_apis_impl(input: proc_macro::TokenStream) -> proc_macro::TokenStream {
838	// Parse all impl blocks
839	let RuntimeApiImpls { impls: mut api_impls } = parse_macro_input!(input as RuntimeApiImpls);
840
841	impl_runtime_apis_impl_inner(&mut api_impls)
842		.unwrap_or_else(|e| e.to_compile_error())
843		.into()
844}
845
846fn impl_runtime_apis_impl_inner(api_impls: &mut [ItemImpl]) -> Result<TokenStream> {
847	rename_self_in_trait_impls(api_impls);
848
849	let dispatch_impl = generate_dispatch_function(api_impls)?;
850	let api_impls_for_runtime = generate_api_impl_for_runtime(api_impls)?;
851	let base_runtime_api = generate_runtime_api_base_structures()?;
852	let runtime_api_versions = generate_runtime_api_versions(api_impls)?;
853	let wasm_interface = generate_wasm_interface(api_impls)?;
854	let api_impls_for_runtime_api = generate_api_impl_for_runtime_api(api_impls)?;
855
856	let runtime_metadata = crate::runtime_metadata::generate_impl_runtime_metadata(api_impls)?;
857
858	let impl_ = quote!(
859		#base_runtime_api
860
861		#api_impls_for_runtime
862
863		#api_impls_for_runtime_api
864
865		#runtime_api_versions
866
867		#runtime_metadata
868
869		pub mod api {
870			use super::*;
871
872			#dispatch_impl
873
874			#wasm_interface
875		}
876	);
877
878	let impl_ = expander::Expander::new("impl_runtime_apis")
879		.dry(std::env::var("EXPAND_MACROS").is_err())
880		.verbose(true)
881		.write_to_out_dir(impl_)
882		.expect("Does not fail because of IO in OUT_DIR; qed");
883
884	Ok(impl_)
885}

887// Filters all attributes except the cfg and allow ones.
888fn filter_cfg_and_allow_attrs(attrs: &[Attribute]) -> Vec<Attribute> {
889	attrs
890		.iter()
891		.filter(|a| a.path().is_ident("cfg") || a.path().is_ident("allow"))
892		.cloned()
893		.collect()
894}

33// Maximum number of blocks to wait for a session change.
34// If it does not arrive for whatever reason, we should not wait forever.
35const WAIT_MAX_BLOCKS_FOR_SESSION: u32 = 50;

37/// Create a batch call to assign cores to a parachain.
38pub fn create_assign_core_call(core_and_para: &[(u32, u32)]) -> DynamicPayload {
39	let mut assign_cores = vec![];
40	for (core, para_id) in core_and_para.iter() {
41		assign_cores.push(value! {
42			Coretime(assign_core { core : *core, begin: 0, assignment: ((Task(*para_id), 57600)), end_hint: None() })
43		});
44	}
45
46	zombienet_sdk::subxt::tx::dynamic(
47		"Sudo",
48		"sudo",
49		vec![value! {
50			Utility(batch { calls: assign_cores })
51		}],
52	)
53}

55/// Find an event in subxt `Events` and attempt to decode the fields fo the event.
56fn find_event_and_decode_fields<T: Decode>(
57	events: &Events<PolkadotConfig>,
58	pallet: &str,
59	variant: &str,
60) -> Result<Vec<T>, anyhow::Error> {
61	let mut result = vec![];
62	for event in events.iter() {
63		let event = event?;
64		if event.pallet_name() == pallet && event.variant_name() == variant {
65			let field_bytes = event.field_bytes().to_vec();
66			result.push(T::decode(&mut &field_bytes[..])?);
67		}
68	}
69	Ok(result)
70}

72// Helper function for asserting the throughput of parachains, after the first session change.
73//
74// The throughput is measured as total number of backed candidates in a window of relay chain
75// blocks. Relay chain blocks with session changes are generally ignores.
76pub async fn assert_para_throughput(
77	relay_client: &OnlineClient<PolkadotConfig>,
78	stop_after: u32,
79	expected_candidate_ranges: HashMap<ParaId, Range<u32>>,
80) -> Result<(), anyhow::Error> {
81	let mut blocks_sub = relay_client.blocks().subscribe_finalized().await?;
82	let mut candidate_count: HashMap<ParaId, u32> = HashMap::new();
83	let mut current_block_count = 0;
84
85	let valid_para_ids: Vec<ParaId> = expected_candidate_ranges.keys().cloned().collect();
86
87	// Wait for the first session, block production on the parachain will start after that.
88	wait_for_first_session_change(&mut blocks_sub).await?;
89
90	while let Some(block) = blocks_sub.next().await {
91		let block = block?;
92		log::debug!("Finalized relay chain block {}", block.number());
93		let events = block.events().await?;
94		let is_session_change = events.iter().any(|event| {
95			event.as_ref().is_ok_and(|event| {
96				event.pallet_name() == "Session" && event.variant_name() == "NewSession"
97			})
98		});
99
100		// Do not count blocks with session changes, no backed blocks there.
101		if is_session_change {
102			continue;
103		}
104
105		current_block_count += 1;
106
107		let receipts = find_event_and_decode_fields::<CandidateReceiptV2<H256>>(
108			&events,
109			"ParaInclusion",
110			"CandidateBacked",
111		)?;
112
113		for receipt in receipts {
114			let para_id = receipt.descriptor.para_id();
115			log::debug!("Block backed for para_id {para_id}");
116			if !valid_para_ids.contains(&para_id) {
117				return Err(anyhow!("Invalid ParaId detected: {}", para_id));
118			};
119			*(candidate_count.entry(para_id).or_default()) += 1;
120		}
121
122		if current_block_count == stop_after {
123			break;
124		}
125	}
126
127	log::info!(
128		"Reached {stop_after} finalized relay chain blocks that contain backed candidates. The per-parachain distribution is: {:#?}",
129		candidate_count.iter().map(|(para_id, count)| format!("{para_id} has {count} backed candidates")).collect::<Vec<_>>()
130	);
131
132	for (para_id, expected_candidate_range) in expected_candidate_ranges {
133		let actual = candidate_count
134			.get(&para_id)
135			.ok_or_else(|| anyhow!("ParaId did not have any backed candidates"))?;
136
137		if !expected_candidate_range.contains(actual) {
138			return Err(anyhow!(
139				"Candidate count {actual} not within range {expected_candidate_range:?}"
140			))
141		}
142	}
143
144	Ok(())
145}

147/// Wait for the first block with a session change.
148///
149/// The session change is detected by inspecting the events in the block.
150pub async fn wait_for_first_session_change(
151	blocks_sub: &mut zombienet_sdk::subxt::backend::StreamOfResults<
152		Block<PolkadotConfig, OnlineClient<PolkadotConfig>>,
153	>,
154) -> Result<(), anyhow::Error> {
155	wait_for_nth_session_change(blocks_sub, 1).await
156}
157
158/// Wait for the first block with the Nth session change.
159///
160/// The session change is detected by inspecting the events in the block.
161pub async fn wait_for_nth_session_change(
162	blocks_sub: &mut zombienet_sdk::subxt::backend::StreamOfResults<
163		Block<PolkadotConfig, OnlineClient<PolkadotConfig>>,
164	>,
165	mut sessions_to_wait: u32,
166) -> Result<(), anyhow::Error> {
167	let mut waited_block_num = 0;
168	while let Some(block) = blocks_sub.next().await {
169		let block = block?;
170		log::debug!("Finalized relay chain block {}", block.number());
171		let events = block.events().await?;
172		let is_session_change = events.iter().any(|event| {
173			event.as_ref().is_ok_and(|event| {
174				event.pallet_name() == "Session" && event.variant_name() == "NewSession"
175			})
176		});
177
178		if is_session_change {
179			sessions_to_wait -= 1;
180			if sessions_to_wait == 0 {
181				return Ok(());
182			}
183
184			waited_block_num = 0;
185		} else {
186			if waited_block_num >= WAIT_MAX_BLOCKS_FOR_SESSION {
187				return Err(anyhow::format_err!("Waited for {WAIT_MAX_BLOCKS_FOR_SESSION}, a new session should have been arrived by now."));
188			}
189
190			waited_block_num += 1;
191		}
192	}
193	Ok(())
194}

196// Helper function that asserts the maximum finality lag.
197pub async fn assert_finality_lag(
198	client: &OnlineClient<PolkadotConfig>,
199	maximum_lag: u32,
200) -> Result<(), anyhow::Error> {
201	let mut best_stream = client.blocks().subscribe_best().await?;
202	let mut fut_stream = client.blocks().subscribe_finalized().await?;
203	let (Some(Ok(best)), Some(Ok(finalized))) = join!(best_stream.next(), fut_stream.next()) else {
204		return Err(anyhow::format_err!("Unable to fetch best an finalized block!"));
205	};
206	let finality_lag = best.number() - finalized.number();
207
208	log::info!(
209		"Finality lagged by {finality_lag} blocks, maximum expected was {maximum_lag} blocks"
210	);
211
212	assert!(finality_lag <= maximum_lag, "Expected finality to lag by a maximum of {maximum_lag} blocks, but was lagging by {finality_lag} blocks.");
213	Ok(())
214}

216/// Assert that finality has not stalled.
217pub async fn assert_blocks_are_being_finalized(
218	client: &OnlineClient<PolkadotConfig>,
219) -> Result<(), anyhow::Error> {
220	let sleep_duration = Duration::from_secs(12);
221	let mut finalized_blocks = client.blocks().subscribe_finalized().await?;
222	let first_measurement = finalized_blocks
223		.next()
224		.await
225		.ok_or(anyhow::anyhow!("Can't get finalized block from stream"))??
226		.number();
227	sleep(sleep_duration).await;
228	let second_measurement = finalized_blocks
229		.next()
230		.await
231		.ok_or(anyhow::anyhow!("Can't get finalized block from stream"))??
232		.number();
233
234	log::info!(
235		"Finalized {} blocks within {sleep_duration:?}",
236		second_measurement - first_measurement
237	);
238	assert!(second_measurement > first_measurement);
239
240	Ok(())
241}

243/// Asserts that parachain blocks have the correct relay parent offset.
244///
245/// # Arguments
246///
247/// * `relay_client` - Client connected to a relay chain node
248/// * `para_client` - Client connected to a parachain node
249/// * `offset` - Expected minimum offset between relay parent and highest seen relay block
250/// * `block_limit` - Number of parachain blocks to verify before completing
251pub async fn assert_relay_parent_offset(
252	relay_client: &OnlineClient<PolkadotConfig>,
253	para_client: &OnlineClient<PolkadotConfig>,
254	offset: u32,
255	block_limit: u32,
256) -> Result<(), anyhow::Error> {
257	let mut relay_block_stream = relay_client.blocks().subscribe_all().await?;
258
259	// First parachain header #0 does not contains RSPR digest item.
260	let mut para_block_stream = para_client.blocks().subscribe_all().await?.skip(1);
261	let mut highest_relay_block_seen = 0;
262	let mut num_para_blocks_seen = 0;
263	loop {
264		tokio::select! {
265			Some(Ok(relay_block)) = relay_block_stream.next() => {
266				highest_relay_block_seen = max(relay_block.number(), highest_relay_block_seen);
267				if highest_relay_block_seen > 15 && num_para_blocks_seen == 0 {
268					return Err(anyhow!("No parachain blocks produced!"))
269				}
270			},
271			Some(Ok(para_block)) = para_block_stream.next() => {
272				let logs = &para_block.header().digest.logs;
273
274				let Some((_, relay_parent_number)): Option<(H256, u32)> = logs.iter().find_map(extract_relay_parent_storage_root) else {
275					return Err(anyhow!("No RPSR digest found in header #{}", para_block.number()));
276				};
277				log::debug!("Parachain block #{} was built on relay parent #{relay_parent_number}, highest seen was {highest_relay_block_seen}", para_block.number());
278				assert!(highest_relay_block_seen < offset || relay_parent_number <= highest_relay_block_seen.saturating_sub(offset), "Relay parent is not at the correct offset! relay_parent: #{relay_parent_number} highest_seen_relay_block: #{highest_relay_block_seen}");
279				num_para_blocks_seen += 1;
280				if num_para_blocks_seen >= block_limit {
281					log::info!("Successfully verified relay parent offset of {offset} for {num_para_blocks_seen} parachain blocks.");
282					break;
283				}
284			}
285		}
286	}
287	Ok(())
288}

290/// Extract relay parent information from the digest logs.
291fn extract_relay_parent_storage_root(
292	digest: &DigestItem,
293) -> Option<(relay_chain::Hash, relay_chain::BlockNumber)> {
294	match digest {
295		DigestItem::Consensus(id, val) if id == &RPSR_CONSENSUS_ID => {
296			let (h, n): (relay_chain::Hash, Compact<relay_chain::BlockNumber>) =
297				Decode::decode(&mut &val[..]).ok()?;
298
299			Some((h, n.0))
300		},
301		_ => None,
302	}
303}

305/// Submits the given `call` as transaction and waits for it successful finalization.
306///
307/// The transaction is send as immortal transaction.
308pub async fn submit_extrinsic_and_wait_for_finalization_success<S: Signer<PolkadotConfig>>(
309	client: &OnlineClient<PolkadotConfig>,
310	call: &DynamicPayload,
311	signer: &S,
312) -> Result<(), anyhow::Error> {
313	let extensions = PolkadotExtrinsicParamsBuilder::new().immortal().build();
314
315	let mut tx = client
316		.tx()
317		.create_signed(call, signer, extensions)
318		.await?
319		.submit_and_watch()
320		.await?;
321
322	// Below we use the low level API to replicate the `wait_for_in_block` behaviour
323	// which was removed in subxt 0.33.0. See https://github.com/paritytech/subxt/pull/1237.
324	while let Some(status) = tx.next().await {
325		let status = status?;
326		match &status {
327			TxStatus::InBestBlock(tx_in_block) | TxStatus::InFinalizedBlock(tx_in_block) => {
328				let _result = tx_in_block.wait_for_success().await?;
329				let block_status =
330					if status.as_finalized().is_some() { "Finalized" } else { "Best" };
331				log::info!("[{}] In block: {:#?}", block_status, tx_in_block.block_hash());
332			},
333			TxStatus::Error { message } |
334			TxStatus::Invalid { message } |
335			TxStatus::Dropped { message } => {
336				return Err(anyhow::format_err!("Error submitting tx: {message}"));
337			},
338			_ => continue,
339		}
340	}
341	Ok(())
342}

344/// Submits the given `call` as transaction and waits `timeout_secs` for it successful finalization.
345///
346/// If the transaction does not reach the finalized state in `timeout_secs` an error is returned.
347/// The transaction is send as immortal transaction.
348pub async fn submit_extrinsic_and_wait_for_finalization_success_with_timeout<
349	S: Signer<PolkadotConfig>,
350>(
351	client: &OnlineClient<PolkadotConfig>,
352	call: &DynamicPayload,
353	signer: &S,
354	timeout_secs: impl Into<u64>,
355) -> Result<(), anyhow::Error> {
356	let secs = timeout_secs.into();
357	let res = tokio::time::timeout(
358		Duration::from_secs(secs),
359		submit_extrinsic_and_wait_for_finalization_success(client, call, signer),
360	)
361	.await;
362
363	match res {
364		Ok(Ok(_)) => Ok(()),
365		Ok(Err(e)) => Err(anyhow!("Error waiting for metric: {}", e)),
366		// timeout
367		Err(_) => Err(anyhow!("Timeout ({secs}), waiting for extrinsic finalization")),
368	}
369}

371/// Asserts that the given `para_id` is registered at the relay chain.
372pub async fn assert_para_is_registered(
373	relay_client: &OnlineClient<PolkadotConfig>,
374	para_id: ParaId,
375	blocks_to_wait: u32,
376) -> Result<(), anyhow::Error> {
377	let mut blocks_sub = relay_client.blocks().subscribe_all().await?;
378	let para_id: u32 = para_id.into();
379
380	let keys: Vec<Value> = vec![];
381	let query = subxt::dynamic::storage("Paras", "Parachains", keys);
382
383	let mut blocks_cnt = 0;
384	while let Some(block) = blocks_sub.next().await {
385		let block = block?;
386		log::debug!("Relay block #{}, checking if para_id {para_id} is registered", block.number(),);
387		let parachains = block.storage().fetch(&query).await?;
388
389		let parachains: Vec<u32> = match parachains {
390			Some(parachains) => parachains.as_type()?,
391			None => vec![],
392		};
393
394		log::debug!("Registered para_ids: {:?}", parachains);
395
396		if parachains.iter().any(|p| para_id.eq(p)) {
397			log::debug!("para_id {para_id} registered");
398			return Ok(());
399		}
400		if blocks_cnt >= blocks_to_wait {
401			return Err(anyhow!(
402				"Parachain {para_id} not registered within {blocks_to_wait} blocks"
403			));
404		}
405		blocks_cnt += 1;
406	}
407
408	Err(anyhow!("No more blocks to check"))
409}

411pub async fn runtime_upgrade(
412	network: &Network<LocalFileSystem>,
413	node: &NetworkNode,
414	para_id: u32,
415	wasm_path: &str,
416) -> Result<(), anyhow::Error> {
417	log::info!("Performing runtime upgrade for parachain {}, wasm: {}", para_id, wasm_path);
418	let para = network.parachain(para_id).unwrap();
419
420	para.perform_runtime_upgrade(node, RuntimeUpgradeOptions::new(AssetLocation::from(wasm_path)))
421		.await
422}

424pub async fn assign_cores(
425	node: &NetworkNode,
426	para_id: u32,
427	cores: Vec<u32>,
428) -> Result<(), anyhow::Error> {
429	log::info!("Assigning {:?} cores to parachain {}", cores, para_id);
430
431	let assign_cores_call =
432		create_assign_core_call(&cores.into_iter().map(|core| (core, para_id)).collect::<Vec<_>>());
433
434	let client: OnlineClient<PolkadotConfig> = node.wait_client().await?;
435	let res = submit_extrinsic_and_wait_for_finalization_success_with_timeout(
436		&client,
437		&assign_cores_call,
438		&zombienet_sdk::subxt_signer::sr25519::dev::alice(),
439		60u64,
440	)
441	.await;
442	assert!(res.is_ok(), "Extrinsic failed to finalize: {:?}", res.unwrap_err());
443	log::info!("Cores assigned to the parachain");
444
445	Ok(())
446}

448pub async fn wait_for_upgrade(
449	client: OnlineClient<PolkadotConfig>,
450	expected_version: u32,
451) -> Result<(), anyhow::Error> {
452	let updater = client.updater();
453	let mut update_stream = updater.runtime_updates().await?;
454
455	while let Some(Ok(update)) = update_stream.next().await {
456		let version = update.runtime_version().spec_version;
457		log::info!("Update runtime spec version {version}");
458		if version == expected_version {
459			break;
460		}
461	}
462	Ok(())

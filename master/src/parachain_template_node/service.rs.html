67/// Starts a `ServiceBuilder` for a full service.
68///
69/// Use this macro if you don't actually need the full service, but just the builder in order to
70/// be able to perform chain operations.
71#[docify::export(component_instantiation)]
72pub fn new_partial(config: &Configuration) -> Result<Service, sc_service::Error> {
73	let telemetry = config
74		.telemetry_endpoints
75		.clone()
76		.filter(|x| !x.is_empty())
77		.map(|endpoints| -> Result<_, sc_telemetry::Error> {
78			let worker = TelemetryWorker::new(16)?;
79			let telemetry = worker.handle().new_telemetry(endpoints);
80			Ok((worker, telemetry))
81		})
82		.transpose()?;
83
84	let heap_pages = config
85		.executor
86		.default_heap_pages
87		.map_or(DEFAULT_HEAP_ALLOC_STRATEGY, |h| HeapAllocStrategy::Static { extra_pages: h as _ });
88
89	let executor = ParachainExecutor::builder()
90		.with_execution_method(config.executor.wasm_method)
91		.with_onchain_heap_alloc_strategy(heap_pages)
92		.with_offchain_heap_alloc_strategy(heap_pages)
93		.with_max_runtime_instances(config.executor.max_runtime_instances)
94		.with_runtime_cache_size(config.executor.runtime_cache_size)
95		.build();
96
97	let (client, backend, keystore_container, task_manager) =
98		sc_service::new_full_parts_record_import::<Block, RuntimeApi, _>(
99			config,
100			telemetry.as_ref().map(|(_, telemetry)| telemetry.handle()),
101			executor,
102			true,
103		)?;
104	let client = Arc::new(client);
105
106	let telemetry_worker_handle = telemetry.as_ref().map(|(worker, _)| worker.handle());
107
108	let telemetry = telemetry.map(|(worker, telemetry)| {
109		task_manager.spawn_handle().spawn("telemetry", None, worker.run());
110		telemetry
111	});
112
113	let transaction_pool = Arc::from(
114		sc_transaction_pool::Builder::new(
115			task_manager.spawn_essential_handle(),
116			client.clone(),
117			config.role.is_authority().into(),
118		)
119		.with_options(config.transaction_pool.clone())
120		.with_prometheus(config.prometheus_registry())
121		.build(),
122	);
123
124	let block_import = ParachainBlockImport::new(client.clone(), backend.clone());
125
126	let import_queue = build_import_queue(
127		client.clone(),
128		block_import.clone(),
129		config,
130		telemetry.as_ref().map(|telemetry| telemetry.handle()),
131		&task_manager,
132	);
133
134	Ok(PartialComponents {
135		backend,
136		client,
137		import_queue,
138		keystore_container,
139		task_manager,
140		transaction_pool,
141		select_chain: (),
142		other: (block_import, telemetry, telemetry_worker_handle),
143	})
144}

146/// Build the import queue for the parachain runtime.
147fn build_import_queue(
148	client: Arc<ParachainClient>,
149	block_import: ParachainBlockImport,
150	config: &Configuration,
151	telemetry: Option<TelemetryHandle>,
152	task_manager: &TaskManager,
153) -> sc_consensus::DefaultImportQueue<Block> {
154	cumulus_client_consensus_aura::equivocation_import_queue::fully_verifying_import_queue::<
155		sp_consensus_aura::sr25519::AuthorityPair,
156		_,
157		_,
158		_,
159		_,
160	>(
161		client,
162		block_import,
163		move |_, _| async move {
164			let timestamp = sp_timestamp::InherentDataProvider::from_system_time();
165			Ok(timestamp)
166		},
167		&task_manager.spawn_essential_handle(),
168		config.prometheus_registry(),
169		telemetry,
170	)
171}

173#[allow(clippy::too_many_arguments)]
174fn start_consensus(
175	client: Arc<ParachainClient>,
176	backend: Arc<ParachainBackend>,
177	block_import: ParachainBlockImport,
178	prometheus_registry: Option<&Registry>,
179	telemetry: Option<TelemetryHandle>,
180	task_manager: &TaskManager,
181	relay_chain_interface: Arc<dyn RelayChainInterface>,
182	transaction_pool: Arc<sc_transaction_pool::TransactionPoolHandle<Block, ParachainClient>>,
183	keystore: KeystorePtr,
184	relay_chain_slot_duration: Duration,
185	para_id: ParaId,
186	collator_key: CollatorPair,
187	collator_peer_id: PeerId,
188	overseer_handle: OverseerHandle,
189	announce_block: Arc<dyn Fn(Hash, Option<Vec<u8>>) + Send + Sync>,
190) -> Result<(), sc_service::Error> {
191	let proposer = sc_basic_authorship::ProposerFactory::new(
192		task_manager.spawn_handle(),
193		client.clone(),
194		transaction_pool,
195		prometheus_registry,
196		telemetry.clone(),
197	);
198	let collator_service = CollatorService::new(
199		client.clone(),
200		Arc::new(task_manager.spawn_handle()),
201		announce_block,
202		client.clone(),
203	);
204
205	let params = AuraParams {
206		create_inherent_data_providers: move |_, ()| async move { Ok(()) },
207		block_import,
208		para_client: client.clone(),
209		para_backend: backend,
210		relay_client: relay_chain_interface,
211		code_hash_provider: move |block_hash| {
212			client.code_at(block_hash).ok().map(|c| ValidationCode::from(c).hash())
213		},
214		keystore,
215		collator_key,
216		collator_peer_id,
217		para_id,
218		overseer_handle,
219		relay_chain_slot_duration,
220		proposer,
221		collator_service,
222		authoring_duration: Duration::from_millis(2000),
223		reinitialize: false,
224		max_pov_percentage: None,
225	};
226	let fut = aura::run::<Block, sp_consensus_aura::sr25519::AuthorityPair, _, _, _, _, _, _, _, _>(
227		params,
228	);
229	task_manager.spawn_essential_handle().spawn("aura", None, fut);
230
231	Ok(())
232}

234/// Start a node with the given parachain `Configuration` and relay chain `Configuration`.
235#[sc_tracing::logging::prefix_logs_with("Parachain")]
236pub async fn start_parachain_node(
237	parachain_config: Configuration,
238	polkadot_config: Configuration,
239	collator_options: CollatorOptions,
240	hwbench: Option<sc_sysinfo::HwBench>,
241) -> sc_service::error::Result<(TaskManager, Arc<ParachainClient>)> {
242	let parachain_config = prepare_node_config(parachain_config);
243
244	let params = new_partial(&parachain_config)?;
245	let (block_import, mut telemetry, telemetry_worker_handle) = params.other;
246
247	let prometheus_registry = parachain_config.prometheus_registry().cloned();
248	let net_config = sc_network::config::FullNetworkConfiguration::<
249		_,
250		_,
251		sc_network::NetworkWorker<Block, Hash>,
252	>::new(&parachain_config.network, prometheus_registry.clone());
253
254	let client = params.client.clone();
255	let backend = params.backend.clone();
256	let mut task_manager = params.task_manager;
257
258	let relay_chain_fork_id = polkadot_config.chain_spec.fork_id().map(ToString::to_string);
259	let parachain_fork_id = parachain_config.chain_spec.fork_id().map(ToString::to_string);
260	let advertise_non_global_ips = parachain_config.network.allow_non_globals_in_dht;
261	let parachain_public_addresses = parachain_config.network.public_addresses.clone();
262
263	let (relay_chain_interface, collator_key, relay_chain_network, paranode_rx) =
264		build_relay_chain_interface(
265			polkadot_config,
266			&parachain_config,
267			telemetry_worker_handle,
268			&mut task_manager,
269			collator_options.clone(),
270			hwbench.clone(),
271		)
272		.await
273		.map_err(|e| sc_service::Error::Application(Box::new(e) as Box<_>))?;
274
275	let validator = parachain_config.role.is_authority();
276	let transaction_pool = params.transaction_pool.clone();
277	let import_queue_service = params.import_queue.service();
278
279	// Take parachain id from runtime.
280	let best_hash = client.chain_info().best_hash;
281	let para_id = client
282		.runtime_api()
283		.parachain_id(best_hash)
284		.map_err(|_| "Failed to retrieve parachain id from runtime. Make sure you implement `cumulus_primitives_core::GetParachaiNidentity` runtime API.")?;
285
286	// NOTE: because we use Aura here explicitly, we can use `CollatorSybilResistance::Resistant`
287	// when starting the network.
288	let (network, system_rpc_tx, tx_handler_controller, sync_service) =
289		build_network(BuildNetworkParams {
290			parachain_config: &parachain_config,
291			net_config,
292			client: client.clone(),
293			transaction_pool: transaction_pool.clone(),
294			para_id,
295			spawn_handle: task_manager.spawn_handle(),
296			relay_chain_interface: relay_chain_interface.clone(),
297			import_queue: params.import_queue,
298			sybil_resistance_level: CollatorSybilResistance::Resistant, // because of Aura
299			metrics: sc_network::NetworkWorker::<Block, Hash>::register_notification_metrics(
300				parachain_config.prometheus_config.as_ref().map(|config| &config.registry),
301			),
302		})
303		.await?;
304	let collator_peer_id = network.local_peer_id();
305
306	if parachain_config.offchain_worker.enabled {
307		use futures::FutureExt;
308
309		let offchain_workers =
310			sc_offchain::OffchainWorkers::new(sc_offchain::OffchainWorkerOptions {
311				runtime_api_provider: client.clone(),
312				keystore: Some(params.keystore_container.keystore()),
313				offchain_db: backend.offchain_storage(),
314				transaction_pool: Some(OffchainTransactionPoolFactory::new(
315					transaction_pool.clone(),
316				)),
317				network_provider: Arc::new(network.clone()),
318				is_validator: parachain_config.role.is_authority(),
319				enable_http_requests: false,
320				custom_extensions: move |_| vec![],
321			})?;
322		task_manager.spawn_handle().spawn(
323			"offchain-workers-runner",
324			"offchain-work",
325			offchain_workers.run(client.clone(), task_manager.spawn_handle()).boxed(),
326		);
327	}
328
329	let rpc_builder = {
330		let client = client.clone();
331		let transaction_pool = transaction_pool.clone();
332
333		Box::new(move |_| {
334			let deps =
335				crate::rpc::FullDeps { client: client.clone(), pool: transaction_pool.clone() };
336
337			crate::rpc::create_full(deps).map_err(Into::into)
338		})
339	};
340
341	sc_service::spawn_tasks(sc_service::SpawnTasksParams {
342		rpc_builder,
343		client: client.clone(),
344		transaction_pool: transaction_pool.clone(),
345		task_manager: &mut task_manager,
346		config: parachain_config,
347		keystore: params.keystore_container.keystore(),
348		backend: backend.clone(),
349		network: network.clone(),
350		sync_service: sync_service.clone(),
351		system_rpc_tx,
352		tx_handler_controller,
353		telemetry: telemetry.as_mut(),
354		tracing_execute_block: Some(Arc::new(ParachainTracingExecuteBlock::new(client.clone()))),
355	})?;
356
357	if let Some(hwbench) = hwbench {
358		sc_sysinfo::print_hwbench(&hwbench);
359		// Here you can check whether the hardware meets your chains' requirements. Putting a link
360		// in there and swapping out the requirements for your own are probably a good idea. The
361		// requirements for a para-chain are dictated by its relay-chain.
362		match SUBSTRATE_REFERENCE_HARDWARE.check_hardware(&hwbench, false) {
363			Err(err) if validator => {
364				log::warn!(
365				"⚠️  The hardware does not meet the minimal requirements {} for role 'Authority'.",
366				err
367			);
368			},
369			_ => {},
370		}
371
372		if let Some(ref mut telemetry) = telemetry {
373			let telemetry_handle = telemetry.handle();
374			task_manager.spawn_handle().spawn(
375				"telemetry_hwbench",
376				None,
377				sc_sysinfo::initialize_hwbench_telemetry(telemetry_handle, hwbench),
378			);
379		}
380	}
381
382	let announce_block = {
383		let sync_service = sync_service.clone();
384		Arc::new(move |hash, data| sync_service.announce_block(hash, data))
385	};
386
387	let relay_chain_slot_duration = Duration::from_secs(6);
388
389	let overseer_handle = relay_chain_interface
390		.overseer_handle()
391		.map_err(|e| sc_service::Error::Application(Box::new(e)))?;
392
393	start_relay_chain_tasks(StartRelayChainTasksParams {
394		client: client.clone(),
395		announce_block: announce_block.clone(),
396		para_id,
397		relay_chain_interface: relay_chain_interface.clone(),
398		task_manager: &mut task_manager,
399		da_recovery_profile: if validator {
400			DARecoveryProfile::Collator
401		} else {
402			DARecoveryProfile::FullNode
403		},
404		import_queue: import_queue_service,
405		relay_chain_slot_duration,
406		recovery_handle: Box::new(overseer_handle.clone()),
407		sync_service: sync_service.clone(),
408		prometheus_registry: prometheus_registry.as_ref(),
409	})?;
410
411	start_bootnode_tasks(StartBootnodeTasksParams {
412		embedded_dht_bootnode: collator_options.embedded_dht_bootnode,
413		dht_bootnode_discovery: collator_options.dht_bootnode_discovery,
414		para_id,
415		task_manager: &mut task_manager,
416		relay_chain_interface: relay_chain_interface.clone(),
417		relay_chain_fork_id,
418		relay_chain_network,
419		request_receiver: paranode_rx,
420		parachain_network: network,
421		advertise_non_global_ips,
422		parachain_genesis_hash: client.chain_info().genesis_hash.encode(),
423		parachain_fork_id,
424		parachain_public_addresses,
425	});
426
427	if validator {
428		start_consensus(
429			client.clone(),
430			backend,
431			block_import,
432			prometheus_registry.as_ref(),
433			telemetry.as_ref().map(|t| t.handle()),
434			&task_manager,
435			relay_chain_interface,
436			transaction_pool,
437			params.keystore_container.keystore(),
438			relay_chain_slot_duration,
439			para_id,
440			collator_key.expect("Command line arguments do not allow this. qed"),
441			collator_peer_id,
442			overseer_handle,
443			announce_block,
444		)?;
445	}
446
447	Ok((task_manager, client))

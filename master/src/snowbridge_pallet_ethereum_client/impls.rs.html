16impl<T: Config> Verifier for Pallet<T> {
17	/// Verify a message by verifying the existence of the corresponding
18	/// Ethereum log in a block. Returns the log if successful. The execution header containing
19	/// the log is sent with the message. The beacon header containing the execution header
20	/// is also sent with the message, to check if the header is an ancestor of a finalized
21	/// header.
22	fn verify(event_log: &Log, proof: &Proof) -> Result<(), VerificationError> {
23		Self::verify_execution_proof(&proof.execution_proof)
24			.map_err(|e| InvalidExecutionProof(e.into()))?;
25
26		Self::verify_receipt_inclusion(
27			proof.execution_proof.execution_header.receipts_root(),
28			&proof.receipt_proof.1,
29			event_log,
30		)?;
31
32		Ok(())
33	}
34}

36impl<T: Config> Pallet<T> {
37	/// Verifies that the receipt encoded in `proof.data` is included in the block given by
38	/// `proof.block_hash`.
39	pub fn verify_receipt_inclusion(
40		receipts_root: H256,
41		receipt_proof: &[Vec<u8>],
42		log: &Log,
43	) -> Result<(), VerificationError> {
44		let receipt = verify_receipt_proof(receipts_root, receipt_proof).ok_or(InvalidProof)?;
45		if !receipt.logs().iter().any(|l| {
46			l.data.data.0 == log.data &&
47				l.address.0 == log.address.0 &&
48				l.topics().len() == log.topics.len()
49		}) {
50			log::error!(
51				target: "ethereum-client",
52				"๐ซ Event log not found in receipt for transaction",
53			);
54			return Err(LogNotFound)
55		}
56		Ok(())
57	}
58
59	/// Validates an execution header with ancestry_proof against a finalized checkpoint on
60	/// chain.The beacon header containing the execution header is sent, plus the execution header,
61	/// along with a proof that the execution header is rooted in the beacon header body.
62	pub(crate) fn verify_execution_proof(execution_proof: &ExecutionProof) -> DispatchResult {
63		let latest_finalized_state =
64			FinalizedBeaconState::<T>::get(LatestFinalizedBlockRoot::<T>::get())
65				.ok_or(Error::<T>::NotBootstrapped)?;
66		// Checks that the header is an ancestor of a finalized header, using slot number.
67		ensure!(
68			execution_proof.header.slot <= latest_finalized_state.slot,
69			Error::<T>::HeaderNotFinalized
70		);
71
72		let beacon_block_root: H256 = execution_proof
73			.header
74			.hash_tree_root()
75			.map_err(|_| Error::<T>::HeaderHashTreeRootFailed)?;
76
77		match &execution_proof.ancestry_proof {
78			Some(proof) => {
79				Self::verify_ancestry_proof(
80					beacon_block_root,
81					execution_proof.header.slot,
82					&proof.header_branch,
83					proof.finalized_block_root,
84				)?;
85			},
86			None => {
87				// If the ancestry proof is not provided, we expect this beacon header to be a
88				// finalized beacon header. We need to check that the header hash matches the
89				// finalized header root at the expected slot.
90				let state = <FinalizedBeaconState<T>>::get(beacon_block_root)
91					.ok_or(Error::<T>::ExpectedFinalizedHeaderNotStored)?;
92				if execution_proof.header.slot != state.slot {
93					return Err(Error::<T>::ExpectedFinalizedHeaderNotStored.into())
94				}
95			},
96		}
97
98		// Gets the hash tree root of the execution header, in preparation for the execution
99		// header proof (used to check that the execution header is rooted in the beacon
100		// header body.
101		let execution_header_root: H256 = execution_proof
102			.execution_header
103			.hash_tree_root()
104			.map_err(|_| Error::<T>::BlockBodyHashTreeRootFailed)?;
105
106		let execution_header_gindex = Self::execution_header_gindex();
107		ensure!(
108			verify_merkle_branch(
109				execution_header_root,
110				&execution_proof.execution_branch,
111				subtree_index(execution_header_gindex),
112				generalized_index_length(execution_header_gindex),
113				execution_proof.header.body_root
114			),
115			Error::<T>::InvalidExecutionHeaderProof
116		);
117		Ok(())
118	}
119
120	/// Verify that `block_root` is an ancestor of `finalized_block_root` Used to prove that
121	/// an execution header is an ancestor of a finalized header (i.e. the blocks are
122	/// on the same chain).
123	fn verify_ancestry_proof(
124		block_root: H256,
125		block_slot: u64,
126		block_root_proof: &[H256],
127		finalized_block_root: H256,
128	) -> DispatchResult {
129		let state = <FinalizedBeaconState<T>>::get(finalized_block_root)
130			.ok_or(Error::<T>::ExpectedFinalizedHeaderNotStored)?;
131
132		ensure!(block_slot < state.slot, Error::<T>::HeaderNotFinalized);
133
134		let index_in_array = block_slot % (SLOTS_PER_HISTORICAL_ROOT as u64);
135		let leaf_index = (SLOTS_PER_HISTORICAL_ROOT as u64) + index_in_array;
136
137		ensure!(
138			verify_merkle_branch(
139				block_root,
140				block_root_proof,
141				leaf_index as usize,
142				config::BLOCK_ROOT_AT_INDEX_DEPTH,
143				state.block_roots_root
144			),
145			Error::<T>::InvalidAncestryMerkleProof
146		);
147
148		Ok(())
149	}

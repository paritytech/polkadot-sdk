105/// Start necessary consensus tasks related to the relay chain.
106///
107/// Parachain nodes need to track the state of the relay chain and use the
108/// relay chain's data availability service to fetch blocks if they don't
109/// arrive via the normal p2p layer (i.e. when authors withhold their blocks deliberately).
110///
111/// This function spawns work for those side tasks.
112///
113/// It also spawns a parachain informant task that will log the relay chain state and some metrics.
114pub fn start_relay_chain_tasks<Block, Client, Backend, RCInterface>(
115	StartRelayChainTasksParams {
116		client,
117		announce_block,
118		para_id,
119		task_manager,
120		da_recovery_profile,
121		relay_chain_interface,
122		import_queue,
123		relay_chain_slot_duration,
124		recovery_handle,
125		sync_service,
126		prometheus_registry,
127	}: StartRelayChainTasksParams<Block, Client, RCInterface>,
128) -> sc_service::error::Result<()>
129where
130	Block: BlockT,
131	Client: Finalizer<Block, Backend>
132		+ UsageProvider<Block>
133		+ HeaderBackend<Block>
134		+ Send
135		+ Sync
136		+ BlockBackend<Block>
137		+ BlockchainEvents<Block>
138		+ 'static,
139	for<'a> &'a Client: BlockImport<Block>,
140	Backend: BackendT<Block> + 'static,
141	RCInterface: RelayChainInterface + Clone + 'static,
142{
143	let (recovery_chan_tx, recovery_chan_rx) = mpsc::channel(RECOVERY_CHAN_SIZE);
144
145	let consensus = cumulus_client_consensus_common::run_parachain_consensus(
146		para_id,
147		client.clone(),
148		relay_chain_interface.clone(),
149		announce_block.clone(),
150		Some(recovery_chan_tx),
151	);
152
153	task_manager
154		.spawn_essential_handle()
155		.spawn_blocking("cumulus-consensus", None, consensus);
156
157	let da_recovery_profile = match da_recovery_profile {
158		DARecoveryProfile::Collator => {
159			// We want that collators wait at maximum the relay chain slot duration before starting
160			// to recover blocks. Additionally, we wait at least half the slot time to give the
161			// relay chain the chance to increase availability.
162			RecoveryDelayRange {
163				min: relay_chain_slot_duration / 2,
164				max: relay_chain_slot_duration,
165			}
166		},
167		DARecoveryProfile::FullNode => {
168			// Full nodes should at least wait 2.5 minutes (assuming 6 seconds slot duration) and
169			// in maximum 5 minutes before starting to recover blocks. Collators should already
170			// start the recovery way before full nodes try to recover a certain block and then
171			// share the block with the network using "the normal way". Full nodes are just the
172			// "last resort" for block recovery.
173			RecoveryDelayRange {
174				min: relay_chain_slot_duration * 25,
175				max: relay_chain_slot_duration * 50,
176			}
177		},
178		DARecoveryProfile::Other(profile) => profile,
179	};
180
181	let pov_recovery = PoVRecovery::new(
182		recovery_handle,
183		da_recovery_profile,
184		client.clone(),
185		import_queue,
186		relay_chain_interface.clone(),
187		para_id,
188		recovery_chan_rx,
189		sync_service.clone(),
190	);
191
192	task_manager
193		.spawn_essential_handle()
194		.spawn("cumulus-pov-recovery", None, pov_recovery.run());
195
196	let parachain_informant = parachain_informant::<Block, _>(
197		para_id,
198		relay_chain_interface.clone(),
199		client.clone(),
200		prometheus_registry.map(ParachainInformantMetrics::new).transpose()?,
201	);
202	task_manager
203		.spawn_handle()
204		.spawn("parachain-informant", None, parachain_informant);
205
206	Ok(())
207}

209/// Prepare the parachain's node configuration
210///
211/// This function will:
212/// * Disable the default announcement of Substrate for the parachain in favor of the one of
213///   Cumulus.
214/// * Set peers needed to start warp sync to 1.
215pub fn prepare_node_config(mut parachain_config: Configuration) -> Configuration {
216	parachain_config.announce_block = false;
217	// Parachains only need 1 peer to start warp sync, because the target block is fetched from the
218	// relay chain.
219	parachain_config.network.min_peers_to_start_warp_sync = Some(1);
220
221	parachain_config
222}

259/// The expected level of collator sybil-resistance on the network. This is used to
260/// configure the type of metadata passed alongside block announcements on the network.
261pub enum CollatorSybilResistance {
262	/// There is a collator-selection protocol which provides sybil-resistance,
263	/// such as Aura. Sybil-resistant collator-selection protocols are able to
264	/// operate more efficiently.
265	Resistant,
266	/// There is no collator-selection protocol providing sybil-resistance.

269	/// validators to avoid handling unbounded numbers of blocks.
270	Unresistant,

302/// Build the network service, the network status sinks and an RPC sender.
303pub async fn build_network<'a, Block, Client, RCInterface, IQ, Network>(
304	BuildNetworkParams {
305		parachain_config,
306		net_config,
307		client,
308		transaction_pool,
309		para_id,
310		spawn_handle,
311		relay_chain_interface,
312		import_queue,
313		sybil_resistance_level,
314		metrics,
315	}: BuildNetworkParams<'a, Block, Client, Network, RCInterface, IQ>,
316) -> sc_service::error::Result<(
317	Arc<dyn NetworkService>,
318	TracingUnboundedSender<sc_rpc::system::Request<Block>>,
319	TransactionsHandlerController<Block::Hash>,
320	Arc<SyncingService<Block>>,
321)>
322where
323	Block: BlockT,
324	Client: UsageProvider<Block>
325		+ HeaderBackend<Block>
326		+ sp_consensus::block_validation::Chain<Block>
327		+ Send
328		+ Sync
329		+ BlockBackend<Block>
330		+ BlockchainEvents<Block>
331		+ ProvideRuntimeApi<Block>
332		+ HeaderMetadata<Block, Error = sp_blockchain::Error>
333		+ BlockIdTo<Block, Error = sp_blockchain::Error>
334		+ ProofProvider<Block>
335		+ 'static,
336	Client::Api: CollectCollationInfo<Block>
337		+ sp_transaction_pool::runtime_api::TaggedTransactionQueue<Block>,
338	for<'b> &'b Client: BlockImport<Block>,
339	RCInterface: RelayChainInterface + Clone + 'static,
340	IQ: ImportQueue<Block> + 'static,
341	Network: NetworkBackend<Block, <Block as BlockT>::Hash>,
342{
343	let warp_sync_config = match parachain_config.network.sync_mode {
344		SyncMode::Warp => {
345			log::debug!(target: LOG_TARGET_SYNC, "waiting for announce block...");
346
347			let target_block =
348				wait_for_finalized_para_head::<Block, _>(para_id, relay_chain_interface.clone())
349					.await
350					.inspect_err(|e| {
351						log::error!(
352							target: LOG_TARGET_SYNC,
353							"Unable to determine parachain target block {:?}",
354							e
355						);
356					})?;
357			Some(WarpSyncConfig::WithTarget(target_block))
358		},
359		_ => None,
360	};
361
362	let block_announce_validator = match sybil_resistance_level {
363		CollatorSybilResistance::Resistant => {
364			let block_announce_validator = AssumeSybilResistance::allow_seconded_messages();
365			Box::new(block_announce_validator) as Box<_>
366		},
367		CollatorSybilResistance::Unresistant => {
368			let block_announce_validator =
369				RequireSecondedInBlockAnnounce::new(relay_chain_interface, para_id);
370			Box::new(block_announce_validator) as Box<_>
371		},
372	};
373
374	sc_service::build_network(sc_service::BuildNetworkParams {
375		config: parachain_config,
376		net_config,
377		client,
378		transaction_pool,
379		spawn_handle,
380		import_queue,
381		block_announce_validator_builder: Some(Box::new(move |_| block_announce_validator)),
382		warp_sync_config,
383		block_relay: None,
384		metrics,
385	})
386}

388/// Waits for the relay chain to have finished syncing and then gets the parachain header that
389/// corresponds to the last finalized relay chain block.
390async fn wait_for_finalized_para_head<B, RCInterface>(
391	para_id: ParaId,
392	relay_chain_interface: RCInterface,
393) -> sc_service::error::Result<<B as BlockT>::Header>
394where
395	B: BlockT + 'static,
396	RCInterface: RelayChainInterface + Send + 'static,
397{
398	let mut imported_blocks = relay_chain_interface
399		.import_notification_stream()
400		.await
401		.map_err(|error| {
402			sc_service::Error::Other(format!(
403				"Relay chain import notification stream error when waiting for parachain head: \
404				{error}"
405			))
406		})?
407		.fuse();
408	while imported_blocks.next().await.is_some() {
409		let is_syncing = relay_chain_interface
410			.is_major_syncing()
411			.await
412			.map_err(|e| format!("Unable to determine sync status: {e}"))?;
413
414		if !is_syncing {
415			let relay_chain_best_hash = relay_chain_interface
416				.finalized_block_hash()
417				.await
418				.map_err(|e| Box::new(e) as Box<_>)?;
419
420			let validation_data = relay_chain_interface
421				.persisted_validation_data(
422					relay_chain_best_hash,
423					para_id,
424					OccupiedCoreAssumption::TimedOut,
425				)
426				.await
427				.map_err(|e| format!("{e:?}"))?
428				.ok_or("Could not find parachain head in relay chain")?;
429
430			let finalized_header = B::Header::decode(&mut &validation_data.parent_head.0[..])
431				.map_err(|e| format!("Failed to decode parachain head: {e}"))?;
432
433			log::info!(
434				"๐ Received target parachain header #{} ({}) from the relay chain.",
435				finalized_header.number(),
436				finalized_header.hash()
437			);
438			return Ok(finalized_header)
439		}
440	}
441
442	Err("Stopping following imported blocks. Could not determine parachain target block".into())
443}

445/// Task for logging candidate events and some related metrics.
446async fn parachain_informant<Block: BlockT, Client>(
447	para_id: ParaId,
448	relay_chain_interface: impl RelayChainInterface + Clone,
449	client: Arc<Client>,
450	metrics: Option<ParachainInformantMetrics>,
451) where
452	Client: HeaderBackend<Block> + Send + Sync + 'static,
453{
454	let mut import_notifications = match relay_chain_interface.import_notification_stream().await {
455		Ok(import_notifications) => import_notifications,
456		Err(e) => {
457			log::error!("Failed to get import notification stream: {e:?}. Parachain informant will not run!");
458			return
459		},
460	};
461	let mut last_backed_block_time: Option<Instant> = None;
462	while let Some(n) = import_notifications.next().await {
463		let candidate_events = match relay_chain_interface.candidate_events(n.hash()).await {
464			Ok(candidate_events) => candidate_events,
465			Err(e) => {
466				log::warn!("Failed to get candidate events for block {}: {e:?}", n.hash());
467				continue
468			},
469		};
470		let mut backed_candidates = Vec::new();
471		let mut included_candidates = Vec::new();
472		let mut timed_out_candidates = Vec::new();
473		for event in candidate_events {
474			match event {
475				CandidateEvent::CandidateBacked(receipt, head, _, _) => {
476					if receipt.descriptor.para_id() != para_id {
477						continue;
478					}
479					let backed_block = match Block::Header::decode(&mut &head.0[..]) {
480						Ok(header) => header,
481						Err(e) => {
482							log::warn!(
483								"Failed to decode parachain header from backed block: {e:?}"
484							);
485							continue
486						},
487					};
488					let backed_block_time = Instant::now();
489					if let Some(last_backed_block_time) = &last_backed_block_time {
490						let duration = backed_block_time.duration_since(*last_backed_block_time);
491						if let Some(metrics) = &metrics {
492							metrics.parachain_block_backed_duration.observe(duration.as_secs_f64());
493						}
494					}
495					last_backed_block_time = Some(backed_block_time);
496					backed_candidates.push(backed_block);
497				},
498				CandidateEvent::CandidateIncluded(receipt, head, _, _) => {
499					if receipt.descriptor.para_id() != para_id {
500						continue;
501					}
502					let included_block = match Block::Header::decode(&mut &head.0[..]) {
503						Ok(header) => header,
504						Err(e) => {
505							log::warn!(
506								"Failed to decode parachain header from included block: {e:?}"
507							);
508							continue
509						},
510					};
511					let unincluded_segment_size =
512						client.info().best_number.saturating_sub(*included_block.number());
513					let unincluded_segment_size: u32 = unincluded_segment_size.saturated_into();
514					if let Some(metrics) = &metrics {
515						metrics.unincluded_segment_size.observe(unincluded_segment_size.into());
516					}
517					included_candidates.push(included_block);
518				},
519				CandidateEvent::CandidateTimedOut(receipt, head, _) => {
520					if receipt.descriptor.para_id() != para_id {
521						continue;
522					}
523					let timed_out_block = match Block::Header::decode(&mut &head.0[..]) {
524						Ok(header) => header,
525						Err(e) => {
526							log::warn!(
527								"Failed to decode parachain header from timed out block: {e:?}"
528							);
529							continue
530						},
531					};
532					timed_out_candidates.push(timed_out_block);
533				},
534			}
535		}
536		let mut log_parts = Vec::new();
537		if !backed_candidates.is_empty() {
538			let backed_candidates = backed_candidates
539				.into_iter()
540				.map(|c| format!("#{} ({})", c.number(), c.hash()))
541				.collect::<Vec<_>>()
542				.join(", ");
543			log_parts.push(format!("backed: {}", backed_candidates));
544		};
545		if !included_candidates.is_empty() {
546			let included_candidates = included_candidates
547				.into_iter()
548				.map(|c| format!("#{} ({})", c.number(), c.hash()))
549				.collect::<Vec<_>>()
550				.join(", ");
551			log_parts.push(format!("included: {}", included_candidates));
552		};
553		if !timed_out_candidates.is_empty() {
554			let timed_out_candidates = timed_out_candidates
555				.into_iter()
556				.map(|c| format!("#{} ({})", c.number(), c.hash()))
557				.collect::<Vec<_>>()
558				.join(", ");
559			log_parts.push(format!("timed out: {}", timed_out_candidates));
560		};
561		if !log_parts.is_empty() {
562			log::info!(
563				"Update at relay chain block #{} ({}) - {}",
564				n.number(),
565				n.hash(),
566				log_parts.join(", ")
567			);
568		}
569	}
570}

572struct ParachainInformantMetrics {
573	/// Time between parachain blocks getting backed by the relaychain.
574	parachain_block_backed_duration: Histogram,
575	/// Number of blocks between best block and last included block.
576	unincluded_segment_size: Histogram,
577}
578
579impl ParachainInformantMetrics {
580	fn new(prometheus_registry: &Registry) -> prometheus::Result<Self> {
581		let parachain_block_authorship_duration = Histogram::with_opts(HistogramOpts::new(
582			"parachain_block_backed_duration",
583			"Time between parachain blocks getting backed by the relaychain",
584		))?;
585		prometheus_registry.register(Box::new(parachain_block_authorship_duration.clone()))?;
586
587		let unincluded_segment_size = Histogram::with_opts(
588			HistogramOpts::new(
589				"parachain_unincluded_segment_size",
590				"Number of blocks between best block and last included block",
591			)
592			.buckets((0..=24).into_iter().map(|i| i as f64).collect()),
593		)?;
594		prometheus_registry.register(Box::new(unincluded_segment_size.clone()))?;
595
596		Ok(Self {
597			parachain_block_backed_duration: parachain_block_authorship_duration,
598			unincluded_segment_size,
599		})
600	}
601}

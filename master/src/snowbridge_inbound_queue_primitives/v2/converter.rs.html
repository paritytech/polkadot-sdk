22/// Topic prefix used for generating unique identifiers for messages
23const INBOUND_QUEUE_TOPIC_PREFIX: &str = "SnowbridgeInboundQueueV2";

25/// Representation of an intermediate parsed message, before final
26/// conversion to XCM.
27#[derive(Clone, RuntimeDebug, Encode)]
28pub struct PreparedMessage {
29	/// Ethereum account that initiated this messaging operation
30	pub origin: H160,
31	/// The claimer in the case that funds get trapped.
32	pub claimer: Location,
33	/// The assets bridged from Ethereum
34	pub assets: Vec<AssetTransfer>,
35	/// The XCM to execute on the destination
36	pub remote_xcm: Xcm<()>,
37	/// Fee in Ether to cover the xcm execution on AH.
38	pub execution_fee: Asset,
39}
40
41/// An asset transfer instruction
42#[derive(Clone, RuntimeDebug, Encode)]
43pub enum AssetTransfer {
44	ReserveDeposit(Asset),
45	ReserveWithdraw(Asset),
46}

48#[derive(Clone, RuntimeDebug, Encode)]
49pub struct CreateAssetCallInfo {
50	pub call: CallIndex,
51	pub deposit: u128,
52	pub min_balance: u128,
53}

122	/// Parse the message into an intermediate form, with all fields decoded
123	/// and prepared.
124	fn prepare(message: Message) -> Result<PreparedMessage, ConvertMessageError> {
125		// ETH "asset id" is the Ethereum root location. Same location used for the "bridge owner".
126		let ether_location = Location::new(2, [GlobalConsensus(EthereumNetwork::get())]);
127		let bridge_owner = Self::bridge_owner()?;
128
129		let claimer = message
130			.claimer
131			// Get the claimer from the message,
132			.and_then(|claimer_bytes| Location::decode(&mut claimer_bytes.as_ref()).ok())
133			// or use the Snowbridge sovereign on AH as the fallback claimer.
134			.unwrap_or_else(|| {
135				Location::new(0, [AccountId32 { network: None, id: bridge_owner.clone().into() }])
136			});
137
138		let mut remote_xcm: Xcm<()> = match &message.xcm {
139			XcmPayload::Raw(raw) => Self::decode_raw_xcm(raw),
140			XcmPayload::CreateAsset { token, network } => Self::make_create_asset_xcm(
141				token,
142				*network,
143				message.value,
144				bridge_owner,
145				claimer.clone(),
146			)?,
147		};
148
149		// Asset to cover XCM execution fee
150		let execution_fee_asset: Asset = (ether_location.clone(), message.execution_fee).into();
151
152		let mut assets = vec![];
153
154		if message.value > 0 {
155			// Asset for remaining ether
156			let remaining_ether_asset: Asset = (ether_location.clone(), message.value).into();
157			assets.push(AssetTransfer::ReserveDeposit(remaining_ether_asset));
158		}
159
160		for asset in &message.assets {
161			match asset {
162				EthereumAsset::NativeTokenERC20 { token_id, value } => {
163					ensure!(*token_id != H160::zero(), ConvertMessageError::InvalidAsset);
164					let token_location: Location = Location::new(
165						2,
166						[
167							GlobalConsensus(EthereumNetwork::get()),
168							AccountKey20 { network: None, key: (*token_id).into() },
169						],
170					);
171					let asset: Asset = (token_location, *value).into();
172					assets.push(AssetTransfer::ReserveDeposit(asset));
173				},
174				EthereumAsset::ForeignTokenERC20 { token_id, value } => {
175					let asset_location = ConvertAssetId::maybe_convert(*token_id)
176						.ok_or(ConvertMessageError::InvalidAsset)?;
177					let asset_hub_from_ethereum: Location = Location::new(
178						1,
179						[
180							GlobalConsensus(LocalNetwork::get()),
181							Parachain(AssetHubParaId::get().into()),
182						],
183					);
184					let ethereum_universal: InteriorLocation =
185						[GlobalConsensus(EthereumNetwork::get())].into();
186					let reanchored_asset_location = asset_location
187						.reanchored(&asset_hub_from_ethereum, &ethereum_universal)
188						.map_err(|_| ConvertMessageError::CannotReanchor)?;
189					let asset: Asset = (reanchored_asset_location, *value).into();
190					assets.push(AssetTransfer::ReserveWithdraw(asset));
191				},
192			}
193		}
194
195		// Add SetTopic instruction if not already present as the last instruction
196		if !matches!(remote_xcm.0.last(), Some(SetTopic(_))) {
197			let topic = blake2_256(&(INBOUND_QUEUE_TOPIC_PREFIX, message.nonce).encode());
198			remote_xcm.0.push(SetTopic(topic));
199		}
200
201		let prepared_message = PreparedMessage {
202			origin: message.origin,
203			claimer,
204			assets,
205			remote_xcm,
206			execution_fee: execution_fee_asset,
207		};
208
209		Ok(prepared_message)
210	}

212	/// Get sovereign account of Ethereum on Asset Hub.
213	fn bridge_owner() -> Result<AccountId, ConvertMessageError> {
214		let account =
215			ExternalConsensusLocationsConverterFor::<
216				AssetHubUniversal<LocalNetwork, AssetHubParaId>,
217				AccountId,
218			>::convert_location(&Location::new(2, [GlobalConsensus(EthereumNetwork::get())]))
219			.ok_or(ConvertMessageError::CannotReanchor)?;
220
221		Ok(account)
222	}

224	/// Construct the remote XCM needed to create a new asset in the `ForeignAssets` pallet
225	/// on AssetHub. Polkadot is the only supported network at the moment.
226	fn make_create_asset_xcm(
227		token: &H160,
228		network: super::message::Network,
229		eth_value: u128,
230		bridge_owner: AccountId,
231		claimer: Location,
232	) -> Result<Xcm<()>, ConvertMessageError> {
233		let dot_asset = Location::new(1, Here);
234		let dot_fee: xcm::prelude::Asset = (dot_asset, CreateAssetCall::get().deposit).into();
235
236		let eth_asset: xcm::prelude::Asset =
237			(Location::new(2, [GlobalConsensus(EthereumNetwork::get())]), eth_value).into();
238
239		let create_call_index: [u8; 2] = CreateAssetCall::get().call;
240		let create_min_blance: u128 = CreateAssetCall::get().min_balance;
241
242		let asset_id = Location::new(
243			2,
244			[
245				GlobalConsensus(EthereumNetwork::get()),
246				AccountKey20 { network: None, key: (*token).into() },
247			],
248		);
249
250		match network {
251			super::message::Network::Polkadot => Ok(Self::make_create_asset_xcm_for_polkadot(
252				create_call_index,
253				create_min_blance,
254				asset_id,
255				bridge_owner,
256				dot_fee,
257				eth_asset,
258				claimer,
259			)),
260		}
261	}

263	/// Construct the asset creation XCM for the Polkdot network.
264	fn make_create_asset_xcm_for_polkadot(
265		create_call_index: [u8; 2],
266		create_min_blance: u128,
267		asset_id: Location,
268		bridge_owner: AccountId,
269		dot_fee_asset: xcm::prelude::Asset,
270		eth_asset: xcm::prelude::Asset,
271		claimer: Location,
272	) -> Xcm<()> {
273		let bridge_owner_bytes: [u8; 32] = bridge_owner.into();
274		vec![
275			// Exchange eth for dot to pay the asset creation deposit.
276			ExchangeAsset {
277				give: eth_asset.into(),
278				want: dot_fee_asset.clone().into(),
279				maximal: false,
280			},
281			// Deposit the dot deposit into the bridge sovereign account (where the asset
282			// creation fee will be deducted from).
283			DepositAsset {
284				assets: dot_fee_asset.clone().into(),
285				beneficiary: bridge_owner_bytes.into(),
286			},
287			// Call to create the asset.
288			Transact {
289				origin_kind: OriginKind::Xcm,
290				fallback_max_weight: None,
291				call: (
292					create_call_index,
293					asset_id.clone(),
294					MultiAddress::<[u8; 32], ()>::Id(bridge_owner_bytes.into()),
295					create_min_blance,
296				)
297					.encode()
298					.into(),
299			},
300			RefundSurplus,
301			// Deposit leftover funds to Snowbridge sovereign
302			DepositAsset { assets: Wild(AllCounted(2)), beneficiary: claimer },
303		]
304		.into()
305	}
306
307	/// Parse and (non-strictly) decode `raw` XCM bytes into a `Xcm<()>`.
308	/// If decoding fails, return an empty `Xcm<()>`โthus allowing the message
309	/// to proceed so assets can still be trapped on AH rather than the funds being locked on
310	/// Ethereum but not accessible on AH.
311	fn decode_raw_xcm(raw: &[u8]) -> Xcm<()> {
312		let mut data = raw;
313		if let Ok(versioned_xcm) =
314			VersionedXcm::<()>::decode_with_depth_limit(MAX_XCM_DECODE_DEPTH, &mut data)
315		{
316			if let Ok(decoded_xcm) = versioned_xcm.try_into() {
317				return decoded_xcm;
318			}
319		}
320		// Decoding failed; allow an empty XCM so the message won't fail entirely.
321		Xcm::new()
322	}
323}

325impl<
326		CreateAsset,
327		EthereumNetwork,
328		LocalNetwork,
329		GatewayProxyAddress,
330		InboundQueueLocation,
331		AssetHubParaId,
332		ConvertAssetId,
333		AccountId,
334	> ConvertMessage
335	for MessageToXcm<
336		CreateAsset,
337		EthereumNetwork,
338		LocalNetwork,
339		GatewayProxyAddress,
340		InboundQueueLocation,
341		AssetHubParaId,
342		ConvertAssetId,
343		AccountId,
344	>
345where
346	CreateAsset: Get<CreateAssetCallInfo>,
347	EthereumNetwork: Get<NetworkId>,
348	LocalNetwork: Get<NetworkId>,
349	GatewayProxyAddress: Get<H160>,
350	InboundQueueLocation: Get<InteriorLocation>,
351	AssetHubParaId: Get<ParaId>,
352	ConvertAssetId: MaybeConvert<TokenId, Location>,
353	AccountId: Into<[u8; 32]> + From<[u8; 32]> + Clone,
354{
355	fn convert(message: Message) -> Result<Xcm<()>, ConvertMessageError> {
356		let message = Self::prepare(message)?;
357
358		log::trace!(target: LOG_TARGET, "prepared message: {:?}", message);
359
360		let mut instructions = vec![
361			DescendOrigin(InboundQueueLocation::get()),
362			UniversalOrigin(GlobalConsensus(EthereumNetwork::get())),
363			ReserveAssetDeposited(message.execution_fee.clone().into()),
364		];
365
366		// Set claimer before PayFees, in case the fees are not enough. Then the claimer will be
367		// able to claim the funds still.
368		instructions.push(SetHints {
369			hints: vec![AssetClaimer { location: message.claimer }]
370				.try_into()
371				.expect("checked statically, qed"),
372		});
373
374		instructions.push(PayFees { asset: message.execution_fee.clone() });
375
376		let mut reserve_deposit_assets = vec![];
377		let mut reserve_withdraw_assets = vec![];
378
379		for asset in message.assets {
380			match asset {
381				AssetTransfer::ReserveDeposit(asset) => reserve_deposit_assets.push(asset),
382				AssetTransfer::ReserveWithdraw(asset) => reserve_withdraw_assets.push(asset),
383			};
384		}
385
386		if !reserve_deposit_assets.is_empty() {
387			instructions.push(ReserveAssetDeposited(reserve_deposit_assets.into()));
388		}
389		if !reserve_withdraw_assets.is_empty() {
390			instructions.push(WithdrawAsset(reserve_withdraw_assets.into()));
391		}
392
393		// If the message origin is not the gateway proxy contract, set the origin to
394		// the original sender on Ethereum. Important to be before the arbitrary XCM that is
395		// appended to the message on the next line.
396		if message.origin != GatewayProxyAddress::get() {
397			instructions.push(DescendOrigin(
398				AccountKey20 { key: message.origin.into(), network: None }.into(),
399			));
400		}
401
402		// Add the XCM sent in the message to the end of the xcm instruction
403		instructions.extend(message.remote_xcm.0);
404
405		Ok(instructions.into())
406	}

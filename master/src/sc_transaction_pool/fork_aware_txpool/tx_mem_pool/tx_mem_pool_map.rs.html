19//! Provides structures for storing and managing transactions in a transaction memory pool.
20//!
21//! The module includes `SizeTrackedStore`, a map designed for concurrent access, and
22//! `IndexedStorage`, which manages transaction entries by key and priority. Transactions are stored
23//! efficiently with operations to insert items based on priority and manage space utilization. This
24//! module provides core functionality for maintaining the `TxMemPool` state.

36/// Something that can report its size.
37pub(super) trait Size {
38	fn size(&self) -> usize;

41/// Trait for items with a priority and timestamp ordering.
42///
43/// `PriorityAndTimestamp` defines methods to access the priority and timestamp,
44/// facilitating sorting in structures like `SizeTrackedStore`.
45pub(super) trait PriorityAndTimestamp {
46	type Priority: Ord;
47	type Timestamp: Ord;
48
49	fn priority(&self) -> Self::Priority;
50	fn timestamp(&self) -> Self::Timestamp;
51}
52
53/// A dual-key struct for ordering by priority and timestamp.
54///
55/// `PriorityKey<U, V>` allows sorting where the primary criteria is `Priority`
56/// and ties are broken using `Timestamp`.
57#[derive(Eq, PartialEq, Ord, PartialOrd, Debug)]
58pub struct PriorityKey<U, V>(U, V);
59
60/// Composite key for sorting collections by priority keys (embedded in `PriorityKey`) and item key
61/// (typically hash).
62#[derive(Eq, PartialEq, Ord, PartialOrd, Debug)]
63struct SortKey<P, K>(P, K);
64
65impl<K, A, B> SortKey<PriorityKey<A, B>, K>
66where
67	K: Ord + Copy,
68	A: Ord,
69	B: Ord,
70{
71	/// Creates a new `SortKey` for a given item and its key, based on the item's priority and
72	/// timestamp.
73	fn new<V: PriorityAndTimestamp<Priority = A, Timestamp = B>>(key: &K, item: &V) -> Self {
74		Self(PriorityKey(item.priority(), item.timestamp()), *key)
75	}
76}

78/// Internal storage for managing TxMemPool entries.
79///
80/// `IndexedStorage` uses `HashMap` for fast access by key and `BTreeMap` for
81/// efficient priority ordering.
82#[derive(Debug)]
83struct IndexedStorage<K, S, V>
84where
85	K: Ord,
86	S: Ord,
87{
88	/// HashMap storing transactions by unique key for quick access.
89	items_by_hashes: HashMap<K, V>,
90	/// BTreeMap ordering transactions for prioritized access based on sort key.
91	items_by_priority: BTreeMap<SortKey<S, K>, V>,
92}
93
94/// Core structure for storing and managing transactions in TxMemPool.
95///
96/// `SizeTrackedStore` is a thread-safe map designed to track the size and priority
97/// of transactions, optimized for use in a transaction memory pool. The map
98/// preserves sorting order based on priority and timestamp.
99///
100/// Size reported might be slightly off and only approximately true.
101#[derive(Debug)]
102pub struct SizeTrackedStore<K, S, V>
103where
104	K: Ord,
105	S: Ord,
106{
107	/// Internal storage maintaining transaction entries.
108	index: Arc<RwLock<IndexedStorage<K, S, V>>>,
109	/// Atomic counter tracking the total size, in bytes, of all transactions.
110	bytes: AtomicIsize,
111	/// Atomic counter maintaining the count of transactions.
112	length: AtomicIsize,
113}
114

118	S: Ord,
119{
120	fn default() -> Self {
121		Self { items_by_hashes: Default::default(), items_by_priority: Default::default() }
122	}

125impl<K, S, V> IndexedStorage<K, S, V>
126where
127	K: Ord + std::hash::Hash,
128	S: Ord,
129{
130	/// Retrieves a reference to the value corresponding to the key, if present.
131	pub fn get(&self, key: &K) -> Option<&V> {
132		self.items_by_hashes.get(key)
133	}
134
135	/// Checks if the map contains the specified key.
136	pub fn contains_key(&self, key: &K) -> bool {
137		self.items_by_hashes.contains_key(key)
138	}
139
140	/// Returns an iterator over the values in the map.
141	pub fn values(&self) -> std::collections::hash_map::Values<'_, K, V> {
142		self.items_by_hashes.values()
143	}
144
145	/// Returns the number of elements in the map.
146	pub fn len(&self) -> usize {
147		debug_assert_eq!(self.items_by_hashes.len(), self.items_by_priority.len());
148		self.items_by_hashes.len()
149	}
150
151	/// Removes and returns the first entry in the priority map, if it exists (testing only).
152	#[cfg(test)]
153	pub fn pop_first(&mut self) -> Option<V> {
154		self.items_by_priority.pop_first().map(|(_, v)| v)
155	}
156
157	pub fn with_items<F, R>(&self, f: F) -> R
158	where
159		F: Fn(std::collections::hash_map::Iter<K, V>) -> R,
160	{
161		f(self.items_by_hashes.iter())
162	}
163}

165impl<K, A, B, V> IndexedStorage<K, PriorityKey<A, B>, V>
166where
167	K: Ord + std::hash::Hash + Copy,
168	A: Ord,
169	B: Ord,
170	V: Clone + PriorityAndTimestamp<Priority = A, Timestamp = B>,
171	V: std::cmp::PartialEq + std::fmt::Debug,
172{
173	/// Inserts a key-value pair into the map, ordering by priority.
174	pub fn insert(&mut self, key: K, val: V) -> Option<V> {
175		let r = self.items_by_hashes.insert(key, val.clone());
176
177		if let Some(ref removed) = r {
178			let a = self.items_by_priority.remove(&SortKey::new(&key, removed));
179			debug_assert_eq!(r, a);
180		}
181		let a = self.items_by_priority.insert(SortKey::new(&key, &val), val);
182		debug_assert!(a.is_none());
183		r
184	}
185
186	/// Removes a key-value pair from the map based on the key.
187	pub fn remove(&mut self, key: &K) -> Option<V> {
188		let r = self.items_by_hashes.remove(key);
189		let _ = r.as_ref().map(|r| {
190			let k = SortKey::new(key, r);
191			let a = self.items_by_priority.remove(&k);
192			debug_assert_eq!(r.clone(), a.expect("item should be in both maps. qed."));
193		});
194		r
195	}
196
197	/// Allows to mutate item for given key with a closure, if key exists.
198	///
199	/// Intended to mutate priority and timestamp. Changing size is not possible.
200	pub fn update_item<F>(&mut self, key: &K, f: F) -> Option<()>
201	where
202		F: FnOnce(&mut V),
203	{
204		let item = self.items_by_hashes.get_mut(key)?;
205
206		let old_key = SortKey::new(key, item);
207		f(item);
208		let new_key = SortKey::new(key, item);
209
210		if old_key != new_key {
211			self.items_by_priority.remove(&old_key);
212			self.items_by_priority.insert(new_key, item.clone());
213		}
214
215		Some(())
216	}
217}

219impl<K, A, B, V> IndexedStorage<K, PriorityKey<A, B>, V>
220where
221	K: Ord + std::hash::Hash + Copy + std::fmt::Debug,
222	A: Ord + std::fmt::Debug,
223	B: Ord + std::fmt::Debug,
224	V: Clone + PriorityAndTimestamp<Priority = A, Timestamp = B> + Size,
225	V: std::cmp::PartialEq + std::fmt::Debug,
226{
227	/// Attempts to insert an item with replacement based on free space and priority.
228	/// Returns the total size in bytes of removed items, and their keys.
229	///
230	/// Insertion always results with other item's removal, the len bound is kept elsewhere
231	///
232	/// If nothing was inserted `(None,0)` is returned.
233	pub fn try_insert_with_replacement(
234		&mut self,
235		free_bytes: usize,
236		key: K,
237		item: V,
238	) -> (Option<Vec<K>>, usize) {
239		let mut total_size_removed = 0usize;
240		let mut to_be_removed = vec![];
241
242		if item.size() == 0 {
243			return (None, 0);
244		}
245
246		if self.contains_key(&key) {
247			return (None, 0);
248		}
249
250		for (SortKey(PriorityKey(worst_priority, worst_timestamp), worst_key), worst_item) in
251			&self.items_by_priority
252		{
253			if *worst_priority > item.priority() {
254				return (None, 0);
255			}
256			if *worst_priority == item.priority() && *worst_timestamp < item.timestamp() {
257				return (None, 0);
258			}
259
260			total_size_removed += worst_item.size();
261			to_be_removed.push(*worst_key);
262
263			if free_bytes + total_size_removed >= item.size() {
264				break;
265			}
266		}
267
268		if item.size() > free_bytes + total_size_removed {
269			return (None, 0);
270		}
271
272		self.insert(key, item);
273
274		for worst_key in &to_be_removed {
275			self.remove(worst_key);
276		}
277
278		(Some(to_be_removed), total_size_removed)
279	}
280}

367pub struct SizeTrackedStoreWriteAccess<'a, K, S, V>
368where
369	K: Ord,
370	S: Ord,
371{
372	bytes: &'a AtomicIsize,
373	length: &'a AtomicIsize,
374	inner_guard: RwLockWriteGuard<'a, IndexedStorage<K, S, V>>,
375}
376
377impl<K, A, B, V> SizeTrackedStoreWriteAccess<'_, K, PriorityKey<A, B>, V>
378where
379	K: Ord + std::hash::Hash + Copy + std::fmt::Debug,
380	A: Ord + std::fmt::Debug,
381	B: Ord + std::fmt::Debug,
382	V: Clone + PriorityAndTimestamp<Priority = A, Timestamp = B> + Size,
383	V: std::cmp::PartialEq + std::fmt::Debug,
384{
385	/// Insert value and return previous (if any).
386	pub fn insert(&mut self, key: K, val: V) -> Option<V> {
387		let new_bytes = val.size();
388		self.bytes.fetch_add(new_bytes as isize, AtomicOrdering::Relaxed);
389		self.length.fetch_add(1, AtomicOrdering::Relaxed);
390		self.inner_guard.insert(key, val).inspect(|old_val| {
391			self.bytes.fetch_sub(old_val.size() as isize, AtomicOrdering::Relaxed);
392			self.length.fetch_sub(1, AtomicOrdering::Relaxed);
393		})
394	}
395
396	/// Remove value by key.
397	pub fn remove(&mut self, key: &K) -> Option<V> {
398		let val = self.inner_guard.remove(key);
399		if let Some(size) = val.as_ref().map(Size::size) {
400			self.bytes.fetch_sub(size as isize, AtomicOrdering::Relaxed);
401			self.length.fetch_sub(1, AtomicOrdering::Relaxed);
402		}
403		val
404	}
405
406	/// Refer to [`IndexedStorage::try_insert_with_replacement`]
407	pub fn try_insert_with_replacement(
408		&mut self,
409		max_total_bytes: usize,
410		key: K,
411		item: V,
412	) -> Option<Vec<K>> {
413		let item_size = item.size();
414		let current_bytes = std::cmp::max(self.bytes.load(AtomicOrdering::Relaxed), 0) as usize;
415		let free_bytes = max_total_bytes - current_bytes;
416		let (removed_keys, removed_bytes) =
417			self.inner_guard.try_insert_with_replacement(free_bytes, key, item);
418
419		if let Some(ref removed_keys) = removed_keys {
420			let delta = item_size as isize - removed_bytes as isize;
421			self.bytes.fetch_add(delta, AtomicOrdering::Relaxed);
422			self.length.fetch_sub(removed_keys.len() as isize, AtomicOrdering::Relaxed);
423			self.length.fetch_add(1, AtomicOrdering::Relaxed);
424		}
425		removed_keys
426	}
427
428	/// Allows to mutate item for given key, if exists.
429	///
430	/// Intended to mutate priority and timestamp. Changing size is not possible.
431	pub fn update_item<F>(&mut self, hash: &K, f: F) -> Option<()>
432	where
433		F: FnOnce(&mut V),
434	{
435		self.inner_guard.update_item(hash, f)
436	}
437}

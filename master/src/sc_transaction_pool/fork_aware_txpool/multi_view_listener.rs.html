47/// A side channel allowing to control the external stream instance (one per transaction) with
48/// [`ControllerCommand`].
49///
50/// Set of instances of [`Controller`] lives within the [`MultiViewListener`].
51type Controller<T> = mpsc::TracingUnboundedSender<T>;
52
53/// A receiver of [`ControllerCommand`] instances allowing to control the external stream.
54///
55/// Lives within the [`ExternalWatcherContext`] instance.
56type CommandReceiver<T> = mpsc::TracingUnboundedReceiver<T>;
57
58/// The stream of the transaction events.
59///
60/// It can represent both a single view's stream and an external watcher stream.
61pub type TxStatusStream<T> = Pin<Box<TransactionStatusStream<ExtrinsicHash<T>, BlockHash<T>>>>;
62
63/// An aggregated stream providing events for all transactions from the view.
64///
65/// This stream delivers updates for all transactions in the view, rather than for individual
66/// transactions.
67pub type ViewStatusStream<T> =
68	Pin<Box<dyn Stream<Item = TransactionStatusEvent<ExtrinsicHash<T>, BlockHash<T>>> + Send>>;
69
70/// Commands to control / drive the task of the multi view listener.
71enum ControllerCommand<ChainApi: graph::ChainApi> {
72	/// Requests transaction status updated. Sent by transaction pool implementation.
73	TransactionStatusRequest(TransactionStatusUpdate<ChainApi>),
74	/// Adds a new (aggregated) stream of transactions statuses originating in the view associated
75	/// with a specific block hash.
76	AddViewStream(BlockHash<ChainApi>, ViewStatusStream<ChainApi>),
77	/// Removes an existing view's stream associated with a specific block hash.
78	RemoveViewStream(BlockHash<ChainApi>),
79}
80
81/// Represents the transaction status update performed by transaction pool state machine. The
82/// corresponding statuses coming from the view would typically be ignored in the external watcher.
83enum TransactionStatusUpdate<ChainApi: graph::ChainApi> {
84	/// Marks a transaction as invalidated.
85	///
86	/// If all pre-conditions are met, an external invalid event will be sent out.
87	Invalidated(ExtrinsicHash<ChainApi>),
88
89	/// Notifies that a transaction was finalized in a specific block hash and transaction index.
90	///
91	/// Send out an external finalized event.
92	Finalized(ExtrinsicHash<ChainApi>, BlockHash<ChainApi>, TxIndex),
93
94	/// Notifies that a transaction was broadcasted with a list of peer addresses.
95	///
96	/// Sends out an external broadcasted event.
97	Broadcasted(ExtrinsicHash<ChainApi>, Vec<String>),
98
99	/// Notifies that a transaction was dropped from the pool.
100	///
101	/// If all preconditions are met, an external dropped event will be sent out.
102	Dropped(ExtrinsicHash<ChainApi>, DroppedReason<ExtrinsicHash<ChainApi>>),
103
104	/// Notifies that a finality watcher timed out.
105	///
106	/// An external finality timed out event will be sent out.
107	FinalityTimeout(ExtrinsicHash<ChainApi>, BlockHash<ChainApi>),
108}
109
110impl<ChainApi> TransactionStatusUpdate<ChainApi>
111where
112	ChainApi: graph::ChainApi,
113{
114	fn hash(&self) -> ExtrinsicHash<ChainApi> {
115		match self {
116			Self::Invalidated(hash) |
117			Self::Finalized(hash, _, _) |
118			Self::Broadcasted(hash, _) |
119			Self::Dropped(hash, _) => *hash,
120			Self::FinalityTimeout(hash, _) => *hash,
121		}
122	}
123}

125impl<ChainApi> Into<TransactionStatus<ExtrinsicHash<ChainApi>, BlockHash<ChainApi>>>
126	for &TransactionStatusUpdate<ChainApi>
127where
128	ChainApi: graph::ChainApi,
129{
130	fn into(self) -> TransactionStatus<ExtrinsicHash<ChainApi>, BlockHash<ChainApi>> {
131		match self {
132			TransactionStatusUpdate::Invalidated(_) => TransactionStatus::Invalid,
133			TransactionStatusUpdate::Finalized(_, hash, index) =>
134				TransactionStatus::Finalized((*hash, *index)),
135			TransactionStatusUpdate::Broadcasted(_, peers) =>
136				TransactionStatus::Broadcast(peers.clone()),
137			TransactionStatusUpdate::Dropped(_, DroppedReason::Usurped(by)) =>
138				TransactionStatus::Usurped(*by),
139			TransactionStatusUpdate::Dropped(_, DroppedReason::LimitsEnforced) =>
140				TransactionStatus::Dropped,
141			TransactionStatusUpdate::Dropped(_, DroppedReason::Invalid) =>
142				TransactionStatus::Invalid,
143			TransactionStatusUpdate::FinalityTimeout(_, block_hash) =>
144				TransactionStatus::FinalityTimeout(*block_hash),
145		}
146	}
147}

238/// This struct allows to create and control listener for multiple transactions.
239///
240/// For every view, an aggregated stream of transactions events can be added. The events are
241/// flattened and sent out to the external listener for individual transactions. (The *external*
242/// term here means that it can be exposed to [`sc_transaction_pool_api::TransactionPool`] API
243/// client e.g. over RPC.)
244///
245/// The listener allows to add and remove view's stream.
246///
247/// The listener provides a side channel that allows triggering specific events (finalized, dropped,
248/// invalid, broadcast) independently of the view's stream.
249pub struct MultiViewListener<ChainApi: graph::ChainApi> {
250	/// Provides the controller for sending control commands to the listener's task.
251	controller: Controller<ControllerCommand<ChainApi>>,
252
253	/// The map containing the sinks of the streams representing the external listeners of
254	/// the individual transactions. Hash of the transaction is used as a map's key. A map is
255	/// shared with listener's task.
256	external_controllers:
257		Arc<RwLock<HashMap<ExtrinsicHash<ChainApi>, Controller<ExternalWatcherCommand<ChainApi>>>>>,
258}
259
260/// A type representing a `MultiViewListener` task. For more details refer to
261/// [`MultiViewListener::task`].
262pub type MultiViewListenerTask = Pin<Box<dyn Future<Output = ()> + Send>>;
263
264/// The external stream unfolding context.
265///
266/// This context is used to unfold the external events stream for a individual transaction, it
267/// facilitates the logic of converting events incoming from numerous views into the external events
268/// stream.
269struct ExternalWatcherContext<ChainApi: graph::ChainApi> {
270	/// The hash of the transaction being monitored within this context.
271	tx_hash: ExtrinsicHash<ChainApi>,
272	/// A receiver for controller commands sent by [`MultiViewListener`]'s task.
273	command_receiver: CommandReceiver<ExternalWatcherCommand<ChainApi>>,
274	/// A flag indicating whether the context should terminate.
275	terminate: bool,
276	/// A flag indicating if a `Future` status has been encountered.
277	future_seen: bool,
278	/// A flag indicating if a `Ready` status has been encountered.
279	ready_seen: bool,
280	/// A hash set of block hashes from views that consider the transaction valid.
281	views_keeping_tx_valid: HashSet<BlockHash<ChainApi>>,
282	/// The set of views (represented by block hashes) currently maintained by the transaction
283	/// pool.
284	known_views: HashSet<BlockHash<ChainApi>>,
285}
286
287/// Commands to control the single external stream living within the multi view listener. These
288/// commands are sent from listener's task to [`ExternalWatcherContext`].
289enum ExternalWatcherCommand<ChainApi: graph::ChainApi> {
290	/// Command for triggering some of the transaction states, that are decided by the pool logic.
291	PoolTransactionStatus(TransactionStatusUpdate<ChainApi>),
292	/// Transaction status updates coming from the individual views.
293	ViewTransactionStatus(
294		BlockHash<ChainApi>,
295		TransactionStatus<ExtrinsicHash<ChainApi>, BlockHash<ChainApi>>,
296	),
297	/// Notification about new view being added.
298	AddView(BlockHash<ChainApi>),
299	/// Notification about view being removed.
300	RemoveView(BlockHash<ChainApi>),
301}
302
303impl<ChainApi: graph::ChainApi> ExternalWatcherContext<ChainApi>
304where
305	<<ChainApi as graph::ChainApi>::Block as BlockT>::Hash: Unpin,
306{
307	/// Creates new `ExternalWatcherContext` for particular transaction identified by `tx_hash`
308	///
309	/// The `command_receiver` is a side channel for receiving controller's
310	/// [commands][`ExternalWatcherCommand`].
311	fn new(
312		tx_hash: ExtrinsicHash<ChainApi>,
313		command_receiver: CommandReceiver<ExternalWatcherCommand<ChainApi>>,
314	) -> Self {
315		Self {
316			tx_hash,
317			command_receiver,
318			terminate: false,
319			future_seen: false,
320			ready_seen: false,
321			views_keeping_tx_valid: Default::default(),
322			known_views: Default::default(),
323		}
324	}
325
326	/// Handles transaction status updates from the pool and manages internal states based on the
327	/// input value.
328	///
329	/// Function may set the context termination flag, which will close the stream.
330	///
331	/// Returns `Some` with the `event` to be sent out or `None`.
332	fn handle_pool_transaction_status(
333		&mut self,
334		request: TransactionStatusUpdate<ChainApi>,
335	) -> Option<TransactionStatus<ExtrinsicHash<ChainApi>, BlockHash<ChainApi>>> {
336		let status = Into::<TransactionStatus<_, _>>::into(&request);
337		status.is_final().then(|| self.terminate = true);
338		return Some(status);
339	}
340
341	/// Handles various transaction status updates from individual views and manages internal states
342	/// based on the input value.
343	///
344	/// Function may set the context termination flag, which will close the stream.
345	///
346	/// Returns `Some` with the `event` to be sent out or `None`.
347	fn handle_view_transaction_status(
348		&mut self,
349		block_hash: BlockHash<ChainApi>,
350		status: TransactionStatus<ExtrinsicHash<ChainApi>, BlockHash<ChainApi>>,
351	) -> Option<TransactionStatus<ExtrinsicHash<ChainApi>, BlockHash<ChainApi>>> {
352		trace!(
353			target: LOG_TARGET,
354			tx_hash = ?self.tx_hash,
355			?block_hash,
356			?status,
357			views = ?self.known_views.iter().collect::<Vec<_>>(),
358			"mvl handle event"
359		);
360
361		match status {
362			TransactionStatus::Future => {
363				self.views_keeping_tx_valid.insert(block_hash);
364				if self.ready_seen || self.future_seen {
365					None
366				} else {
367					self.future_seen = true;
368					Some(status)
369				}
370			},
371			TransactionStatus::Ready => {
372				self.views_keeping_tx_valid.insert(block_hash);
373				if self.ready_seen {
374					None
375				} else {
376					self.ready_seen = true;
377					Some(status)
378				}
379			},
380			TransactionStatus::InBlock((..)) => {
381				self.views_keeping_tx_valid.insert(block_hash);
382				if !(self.ready_seen || self.future_seen) {
383					self.ready_seen = true;
384					Some(status)
385				} else {
386					Some(status)
387				}
388			},
389			TransactionStatus::Finalized(_) => {
390				self.terminate = true;
391				Some(status)
392			},
393			TransactionStatus::FinalityTimeout(_) |
394			TransactionStatus::Retracted(_) |
395			TransactionStatus::Broadcast(_) |
396			TransactionStatus::Usurped(_) |
397			TransactionStatus::Dropped |
398			TransactionStatus::Invalid => None,
399		}
400	}
401
402	/// Adds a new aggragted transaction status stream.
403	///
404	/// Inserts a new view's transaction status stream into the stream map. The view is represented
405	/// by `block_hash`.
406	fn add_view(&mut self, block_hash: BlockHash<ChainApi>) {
407		trace!(
408			target: LOG_TARGET,
409			tx_hash = ?self.tx_hash,
410			?block_hash,
411			views = ?self.known_views.iter().collect::<Vec<_>>(),
412			"AddView view"
413		);
414		self.known_views.insert(block_hash);
415	}
416
417	/// Removes an existing aggreagated transaction status stream.
418	///
419	/// Removes an aggregated transaction status stream associated with a specific block hash from
420	/// the stream map.
421	fn remove_view(&mut self, block_hash: BlockHash<ChainApi>) {
422		self.known_views.remove(&block_hash);
423		self.views_keeping_tx_valid.remove(&block_hash);
424		trace!(
425			target: LOG_TARGET,
426			tx_hash = ?self.tx_hash,
427			?block_hash,
428			views = ?self.known_views.iter().collect::<Vec<_>>(),
429			"RemoveView view"
430		);
431	}
432}

434impl<ChainApi> MultiViewListener<ChainApi>
435where
436	ChainApi: graph::ChainApi + 'static,
437	<<ChainApi as graph::ChainApi>::Block as BlockT>::Hash: Unpin,
438{
439	/// A worker task associated with `MultiViewListener` instance.
440	///
441	/// An asynchronous listener's task responsible for dispatching:
442	/// - stream_map containing aggregated transaction status streams from multiple views,
443	/// - view add/remove requests,
444	/// - transaction commands,
445	/// to multiple individual per-transaction external watcher contexts.
446	///
447	/// It also reports transactions statuses updates to the provided `events_metrics_collector`.
448	///
449	/// The returned future shall be polled by instantiator of `MultiViewListener`.
450	async fn task(
451		external_watchers_tx_hash_map: Arc<
452			RwLock<HashMap<ExtrinsicHash<ChainApi>, Controller<ExternalWatcherCommand<ChainApi>>>>,
453		>,
454		mut command_receiver: CommandReceiver<ControllerCommand<ChainApi>>,
455		events_metrics_collector: EventsMetricsCollector<ChainApi>,
456	) {
457		let mut aggregated_streams_map: StreamMap<BlockHash<ChainApi>, ViewStatusStream<ChainApi>> =
458			Default::default();
459
460		loop {
461			tokio::select! {
462				biased;
463				Some((view_hash, (tx_hash, status))) =  next_event(&mut aggregated_streams_map) => {
464					events_metrics_collector.report_status(tx_hash, status.clone());
465					if let Entry::Occupied(mut ctrl) = external_watchers_tx_hash_map.write().entry(tx_hash) {
466						trace!(
467							target: LOG_TARGET,
468							?tx_hash,
469							?view_hash,
470							?status,
471							"aggregated_stream_map event",
472						);
473						if let Err(error) = ctrl
474							.get_mut()
475							.unbounded_send(ExternalWatcherCommand::ViewTransactionStatus(view_hash, status))
476						{
477							trace!(target: LOG_TARGET, ?tx_hash, ?error, "send status failed");
478							ctrl.remove();
479						}
480					}
481				},
482				cmd = command_receiver.next() => {
483					match cmd {
484						Some(ControllerCommand::AddViewStream(h,stream)) => {
485							aggregated_streams_map.insert(h,stream);
486							// //todo: aysnc and join all?
487							external_watchers_tx_hash_map.write().retain(|tx_hash, ctrl| {
488								ctrl.unbounded_send(ExternalWatcherCommand::AddView(h))
489									.inspect_err(|error| {
490										trace!(target: LOG_TARGET, ?tx_hash, ?error, "add_view: send message failed");
491									})
492									.is_ok()
493							})
494						},
495						Some(ControllerCommand::RemoveViewStream(h)) => {
496							aggregated_streams_map.remove(&h);
497							//todo: aysnc and join all?
498							external_watchers_tx_hash_map.write().retain(|tx_hash, ctrl| {
499								ctrl.unbounded_send(ExternalWatcherCommand::RemoveView(h))
500									.inspect_err(|error| {
501										trace!(target: LOG_TARGET, ?tx_hash, ?error, "remove_view: send message failed");
502									})
503									.is_ok()
504							})
505						},
506
507						Some(ControllerCommand::TransactionStatusRequest(request)) => {
508							let tx_hash = request.hash();
509							events_metrics_collector.report_status(tx_hash, (&request).into());
510							if let Entry::Occupied(mut ctrl) = external_watchers_tx_hash_map.write().entry(tx_hash) {
511								if let Err(error) = ctrl
512									.get_mut()
513									.unbounded_send(ExternalWatcherCommand::PoolTransactionStatus(request))
514								{
515									trace!(target: LOG_TARGET, ?tx_hash, ?error, "send message failed");
516									ctrl.remove();
517								}
518							}
519						},
520						None =>  {}
521					}
522				},
523			};
524		}
525	}
526
527	/// Creates a new [`MultiViewListener`] instance along with its associated worker task.
528	///
529	/// This function instantiates the new `MultiViewListener` and provides the worker task that
530	/// relays messages to the external transactions listeners. The task shall be polled by caller.
531	///
532	/// The `events_metrics_collector` is an instance of `EventsMetricsCollector` that is
533	/// responsible for collecting and managing metrics related to transaction events. Newly
534	/// created instance of `MultiViewListener` will report transaction status updates and its
535	/// timestamps to the given metrics collector.
536	///
537	/// Returns a tuple containing the [`MultiViewListener`] and the
538	/// [`MultiViewListenerTask`].
539	pub fn new_with_worker(
540		events_metrics_collector: EventsMetricsCollector<ChainApi>,
541	) -> (Self, MultiViewListenerTask) {
542		let external_controllers = Arc::from(RwLock::from(HashMap::<
543			ExtrinsicHash<ChainApi>,
544			Controller<ExternalWatcherCommand<ChainApi>>,
545		>::default()));
546
547		const CONTROLLER_QUEUE_WARN_SIZE: usize = 100_000;
548		let (tx, rx) = mpsc::tracing_unbounded(
549			"txpool-multi-view-listener-task-controller",
550			CONTROLLER_QUEUE_WARN_SIZE,
551		);
552		let task = Self::task(external_controllers.clone(), rx, events_metrics_collector);
553
554		(Self { external_controllers, controller: tx }, task.boxed())
555	}
556
557	/// Creates an external tstream of events for given transaction.
558	///
559	/// This method initializes an `ExternalWatcherContext` for the provided transaction hash, sets
560	/// up the necessary communication channel with listener's task, and unfolds an external
561	/// (meaning that it can be exposed to [`sc_transaction_pool_api::TransactionPool`] API client
562	/// e.g. rpc) stream of transaction status events. If an external watcher is already present for
563	/// the given transaction, it returns `None`.
564	///
565	/// The `submit_timestamp` indicates the time at which a transaction is submitted.
566	/// It is primarily used to calculate event timings for metric collection.
567	pub(crate) fn create_external_watcher_for_tx(
568		&self,
569		tx_hash: ExtrinsicHash<ChainApi>,
570	) -> Option<TxStatusStream<ChainApi>> {
571		let external_ctx = match self.external_controllers.write().entry(tx_hash) {
572			Entry::Occupied(_) => return None,
573			Entry::Vacant(entry) => {
574				const EXT_CONTROLLER_QUEUE_WARN_THRESHOLD: usize = 128;
575				let (tx, rx) = mpsc::tracing_unbounded(
576					"txpool-multi-view-listener",
577					EXT_CONTROLLER_QUEUE_WARN_THRESHOLD,
578				);
579				entry.insert(tx);
580				ExternalWatcherContext::new(tx_hash, rx)
581			},
582		};
583
584		trace!(
585			target: LOG_TARGET,
586			?tx_hash,
587			"create_external_watcher_for_tx"
588		);
589
590		Some(
591			futures::stream::unfold(external_ctx, |mut ctx| async move {
592				if ctx.terminate {
593					trace!(target: LOG_TARGET, tx_hash = ?ctx.tx_hash, "terminate");
594					return None
595				}
596				loop {
597					tokio::select! {
598						cmd = ctx.command_receiver.next() => {
599							match cmd? {
600								ExternalWatcherCommand::ViewTransactionStatus(view_hash, status) => {
601									if let Some(new_status) = ctx.handle_view_transaction_status(view_hash, status) {
602										trace!(
603											target: LOG_TARGET,
604											tx_hash = ?ctx.tx_hash,
605											?new_status,
606											"mvl sending out"
607										);
608										return Some((new_status, ctx))
609									}
610								},
611								ExternalWatcherCommand::PoolTransactionStatus(request) => {
612									if let Some(new_status) = ctx.handle_pool_transaction_status(request) {
613										trace!(
614											target: LOG_TARGET,
615											tx_hash = ?ctx.tx_hash,
616											?new_status,
617											"mvl sending out"
618										);
619										return Some((new_status, ctx))
620									}
621								}
622								ExternalWatcherCommand::AddView(h) => {
623									ctx.add_view(h);
624								},
625								ExternalWatcherCommand::RemoveView(h) => {
626									ctx.remove_view(h);
627								},
628							}
629						},
630					};
631				}
632			})
633			.boxed(),
634		)
635	}

75/// Create a new full node.
76#[sc_tracing::logging::prefix_logs_with(custom_log_prefix.unwrap_or(config.network.node_name.as_str()))]
77pub fn new_full<OverseerGenerator: OverseerGen>(
78	config: Configuration,
79	is_parachain_node: IsParachainNode,
80	workers_path: Option<PathBuf>,
81	overseer_gen: OverseerGenerator,
82	custom_log_prefix: Option<&'static str>,
83) -> Result<NewFull, Error> {
84	let workers_path = Some(workers_path.unwrap_or_else(get_relative_workers_path_for_test));
85
86	let params = polkadot_service::NewFullParams {
87		is_parachain_node,
88		enable_beefy: true,
89		force_authoring_backoff: false,
90		telemetry_worker_handle: None,
91		node_version: None,
92		secure_validator_mode: false,
93		workers_path,
94		workers_names: None,
95		overseer_gen,
96		overseer_message_channel_capacity_override: None,
97		malus_finality_delay: None,
98		hwbench: None,
99		execute_workers_max_num: None,
100		prepare_workers_hard_max_num: None,
101		prepare_workers_soft_max_num: None,
102		keep_finalized_for: None,
103		invulnerable_ah_collators: HashSet::new(),
104		collator_protocol_hold_off: None,
105	};
106
107	match config.network.network_backend {
108		sc_network::config::NetworkBackendType::Libp2p =>
109			polkadot_service::new_full::<_, sc_network::NetworkWorker<_, _>>(config, params),
110		sc_network::config::NetworkBackendType::Litep2p =>
111			polkadot_service::new_full::<_, sc_network::Litep2pNetworkBackend>(config, params),
112	}
113}

115fn get_relative_workers_path_for_test() -> PathBuf {
116	// If no explicit worker path is passed in, we need to specify it ourselves as test binaries
117	// are in the "deps/" directory, one level below where the worker binaries are generated.
118	let mut exe_path = std::env::current_exe()
119		.expect("for test purposes it's reasonable to expect that this will not fail");
120	let _ = exe_path.pop();
121	let _ = exe_path.pop();
122	exe_path
123}

133/// Create a Polkadot `Configuration`.
134///
135/// By default a TCP socket will be used, therefore you need to provide boot
136/// nodes if you want the future node to be connected to other nodes.
137///
138/// The `storage_update_func` function will be executed in an externalities provided environment
139/// and can be used to make adjustments to the runtime genesis storage.
140pub fn node_config(
141	storage_update_func: impl Fn(),
142	tokio_handle: tokio::runtime::Handle,
143	key: Sr25519Keyring,
144	boot_nodes: Vec<MultiaddrWithPeerId>,
145	is_validator: bool,
146) -> Configuration {
147	let base_path = BasePath::new_temp_dir().expect("could not create temporary directory");
148	let root = base_path.path().join(key.to_string());
149	let role = if is_validator { Role::Authority } else { Role::Full };
150	let key_seed = key.to_seed();
151	let mut spec = polkadot_local_testnet_config();
152	let mut storage = spec.as_storage_builder().build_storage().expect("could not build storage");
153
154	BasicExternalities::execute_with_storage(&mut storage, storage_update_func);
155	spec.set_storage(storage);
156
157	let mut network_config = NetworkConfiguration::new(
158		key_seed.to_string(),
159		"network/test/0.1",
160		Default::default(),
161		None,
162	);
163
164	network_config.boot_nodes = boot_nodes;
165
166	network_config.allow_non_globals_in_dht = true;
167
168	// Libp2p needs to know the local address on which it should listen for incoming connections,
169	// while Litep2p will use `/ip4/127.0.0.1/tcp/0` by default.
170	let addr: multiaddr::Multiaddr = "/ip4/127.0.0.1/tcp/0".parse().expect("valid address; qed");
171	network_config.listen_addresses.push(addr.clone());
172	network_config.public_addresses.push(addr);
173	network_config.transport =
174		TransportConfig::Normal { enable_mdns: false, allow_private_ip: true };
175
176	Configuration {
177		impl_name: "polkadot-test-node".to_string(),
178		impl_version: "0.1".to_string(),
179		role,
180		tokio_handle,
181		transaction_pool: Default::default(),
182		network: network_config,
183		keystore: KeystoreConfig::InMemory,
184		database: DatabaseSource::RocksDb { path: root.join("db"), cache_size: 128 },
185		trie_cache_maximum_size: Some(64 * 1024 * 1024),
186		warm_up_trie_cache: None,
187		state_pruning: Default::default(),
188		blocks_pruning: BlocksPruning::KeepFinalized,
189		chain_spec: Box::new(spec),
190		executor: ExecutorConfiguration {
191			wasm_method: WasmExecutionMethod::Compiled {
192				instantiation_strategy: WasmtimeInstantiationStrategy::PoolingCopyOnWrite,
193			},
194			..ExecutorConfiguration::default()
195		},
196		wasm_runtime_overrides: Default::default(),
197		rpc: RpcConfiguration {
198			addr: Default::default(),
199			max_request_size: Default::default(),
200			max_response_size: Default::default(),
201			max_connections: Default::default(),
202			cors: None,
203			methods: Default::default(),
204			id_provider: None,
205			max_subs_per_conn: Default::default(),
206			port: 9944,
207			message_buffer_capacity: Default::default(),
208			batch_config: RpcBatchRequestConfig::Unlimited,
209			rate_limit: None,
210			rate_limit_whitelisted_ips: Default::default(),
211			rate_limit_trust_proxy_headers: Default::default(),
212		},
213		prometheus_config: None,
214		telemetry_endpoints: None,
215		offchain_worker: Default::default(),
216		force_authoring: false,
217		disable_grandpa: false,
218		dev_key_seed: Some(key_seed),
219		tracing_targets: None,
220		tracing_receiver: Default::default(),
221		announce_block: true,
222		data_path: root,
223		base_path,
224	}
225}

227/// Get the listen multiaddr from the network service.
228///
229/// The address is used to connect to the node.
230pub async fn get_listen_address(network: Arc<dyn NetworkService>) -> sc_network::Multiaddr {
231	loop {
232		// Litep2p provides instantly the listen address of the TCP protocol and
233		// ditched the `/0` port used by the `node_config` function.
234		//
235		// Libp2p backend needs to be polled in a separate tokio task a few times
236		// before the listen address is available. The address is made available
237		// through the `SwarmEvent::NewListenAddr` event.
238		let listen_addresses = network.listen_addresses();
239
240		// The network backend must produce a valid TCP port.
241		match listen_addresses.into_iter().find(|addr| {
242			addr.iter().any(|protocol| match protocol {
243				multiaddr::Protocol::Tcp(port) => port > 0,
244				_ => false,
245			})
246		}) {
247			Some(multiaddr) => return multiaddr,
248			None => {
249				tokio::time::sleep(std::time::Duration::from_millis(500)).await;
250				continue;
251			},
252		}
253	}
254}

256/// Run a test validator node that uses the test runtime and specified `config`.
257pub async fn run_validator_node(
258	config: Configuration,
259	worker_program_path: Option<PathBuf>,
260) -> PolkadotTestNode {
261	let NewFull { task_manager, client, network, rpc_handlers, overseer_handle, .. } = new_full(
262		config,
263		IsParachainNode::No,
264		worker_program_path,
265		polkadot_service::ValidatorOverseerGen,
266		None,
267	)
268	.expect("could not create Polkadot test service");
269
270	let overseer_handle = overseer_handle.expect("test node must have an overseer handle");
271	let peer_id = network.local_peer_id();
272	let multiaddr = get_listen_address(network).await;
273
274	let addr = MultiaddrWithPeerId { multiaddr, peer_id };
275
276	PolkadotTestNode { task_manager, client, overseer_handle, addr, rpc_handlers }
277}

279/// Run a test collator node that uses the test runtime.
280///
281/// The node will be using an in-memory socket, therefore you need to provide boot nodes if you
282/// want it to be connected to other nodes.
283///
284/// The `storage_update_func` function will be executed in an externalities provided environment
285/// and can be used to make adjustments to the runtime genesis storage.
286///
287/// # Note
288///
289/// The collator functionality still needs to be registered at the node! This can be done using
290/// [`PolkadotTestNode::register_collator`].
291pub async fn run_collator_node(
292	tokio_handle: tokio::runtime::Handle,
293	key: Sr25519Keyring,
294	storage_update_func: impl Fn(),
295	boot_nodes: Vec<MultiaddrWithPeerId>,
296	collator_pair: CollatorPair,
297) -> PolkadotTestNode {
298	let config = node_config(storage_update_func, tokio_handle, key, boot_nodes, false);
299	let NewFull { task_manager, client, network, rpc_handlers, overseer_handle, .. } = new_full(
300		config,
301		IsParachainNode::Collator(collator_pair),
302		None,
303		polkadot_service::CollatorOverseerGen,
304		None,
305	)
306	.expect("could not create Polkadot test service");
307
308	let overseer_handle = overseer_handle.expect("test node must have an overseer handle");
309	let peer_id = network.local_peer_id();
310
311	let multiaddr = get_listen_address(network).await;
312	let addr = MultiaddrWithPeerId { multiaddr, peer_id };
313
314	PolkadotTestNode { task_manager, client, overseer_handle, addr, rpc_handlers }
315}

332impl PolkadotTestNode {
333	/// Send a sudo call to this node.
334	async fn send_sudo(
335		&self,
336		call: impl Into<polkadot_test_runtime::RuntimeCall>,
337		caller: Sr25519Keyring,
338		nonce: u32,
339	) -> Result<(), RpcTransactionError> {
340		let sudo = SudoCall::sudo { call: Box::new(call.into()) };
341
342		let extrinsic = construct_extrinsic(&self.client, sudo, caller, nonce);
343		self.rpc_handlers.send_transaction(extrinsic.into()).await.map(drop)
344	}
345
346	/// Send an extrinsic to this node.
347	pub async fn send_extrinsic(
348		&self,
349		function: impl Into<polkadot_test_runtime::RuntimeCall>,
350		caller: Sr25519Keyring,
351	) -> Result<RpcTransactionOutput, RpcTransactionError> {
352		let extrinsic = construct_extrinsic(&self.client, function, caller, 0);
353
354		self.rpc_handlers.send_transaction(extrinsic.into()).await
355	}

357	/// Register a parachain at this relay chain.
358	pub async fn register_parachain(
359		&self,
360		id: ParaId,
361		validation_code: impl Into<ValidationCode>,
362		genesis_head: impl Into<HeadData>,
363	) -> Result<(), RpcTransactionError> {
364		let validation_code: ValidationCode = validation_code.into();
365		let call = ParasSudoWrapperCall::sudo_schedule_para_initialize {
366			id,
367			genesis: ParaGenesisArgs {
368				genesis_head: genesis_head.into(),
369				validation_code: validation_code.clone(),
370				para_kind: ParaKind::Parachain,
371			},
372		};
373
374		self.send_sudo(call, Sr25519Keyring::Alice, 0).await?;
375
376		// Bypass pvf-checking.
377		let call = ParasCall::add_trusted_validation_code { validation_code };
378		self.send_sudo(call, Sr25519Keyring::Alice, 1).await
379	}

381	/// Wait for `count` blocks to be imported in the node and then exit. This function will not
382	/// return if no blocks are ever created, thus you should restrict the maximum amount of time of
383	/// the test execution.
384	pub fn wait_for_blocks(&self, count: usize) -> impl Future<Output = ()> {
385		self.client.wait_for_blocks(count)
386	}
387
388	/// Wait for `count` blocks to be finalized and then exit. Similarly with `wait_for_blocks` this
389	/// function will not return if no block are ever finalized.
390	pub async fn wait_for_finalized_blocks(&self, count: usize) {
391		let mut import_notification_stream = self.client.finality_notification_stream();
392		let mut blocks = HashSet::new();
393
394		while let Some(notification) = import_notification_stream.next().await {
395			blocks.insert(notification.hash);
396			if blocks.len() == count {
397				break
398			}
399		}
400	}

402	/// Register the collator functionality in the overseer of this node.
403	pub async fn register_collator(
404		&mut self,
405		collator_key: CollatorPair,
406		para_id: ParaId,
407		collator: CollatorFn,
408	) {
409		let config =
410			CollationGenerationConfig { key: collator_key, collator: Some(collator), para_id };
411
412		self.overseer_handle
413			.send_msg(CollationGenerationMessage::Initialize(config), "Collator")
414			.await;
415
416		self.overseer_handle
417			.send_msg(CollatorProtocolMessage::CollateOn(para_id), "Collator")
418			.await;
419	}

422/// Construct an extrinsic that can be applied to the test runtime.
423pub fn construct_extrinsic(
424	client: &Client,
425	function: impl Into<polkadot_test_runtime::RuntimeCall>,
426	caller: Sr25519Keyring,
427	nonce: u32,
428) -> UncheckedExtrinsic {
429	let function = function.into();
430	let current_block_hash = client.info().best_hash;
431	let current_block = client.info().best_number.saturated_into();
432	let genesis_block = client.hash(0).unwrap().unwrap();
433	let period =
434		BlockHashCount::get().checked_next_power_of_two().map(|c| c / 2).unwrap_or(2) as u64;
435	let tip = 0;
436	let tx_ext: TxExtension = (
437		frame_system::AuthorizeCall::<Runtime>::new(),
438		frame_system::CheckNonZeroSender::<Runtime>::new(),
439		frame_system::CheckSpecVersion::<Runtime>::new(),
440		frame_system::CheckTxVersion::<Runtime>::new(),
441		frame_system::CheckGenesis::<Runtime>::new(),
442		frame_system::CheckEra::<Runtime>::from(generic::Era::mortal(period, current_block)),
443		frame_system::CheckNonce::<Runtime>::from(nonce),
444		frame_system::CheckWeight::<Runtime>::new(),
445		pallet_transaction_payment::ChargeTransactionPayment::<Runtime>::from(tip),
446		frame_system::WeightReclaim::<Runtime>::new(),
447	)
448		.into();
449	let raw_payload = SignedPayload::from_raw(
450		function.clone(),
451		tx_ext.clone(),
452		(
453			(),
454			(),
455			VERSION.spec_version,
456			VERSION.transaction_version,
457			genesis_block,
458			current_block_hash,
459			(),
460			(),
461			(),
462			(),
463		),
464	);
465	let signature = raw_payload.using_encoded(|e| caller.sign(e));
466	UncheckedExtrinsic::new_signed(
467		function.clone(),
468		polkadot_test_runtime::Address::Id(caller.public().into()),
469		polkadot_primitives::Signature::Sr25519(signature),
470		tx_ext.clone(),
471	)
472}

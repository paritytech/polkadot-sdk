67pub enum IsBestBlock {
68	Yes,
69	No,
70}
71
72impl IsBestBlock {
73	pub fn is_best(&self) -> bool {
74		matches!(self, Self::Yes)
75	}
76}
77
78impl From<bool> for IsBestBlock {
79	fn from(is_best: bool) -> Self {
80		if is_best {
81			Self::Yes
82		} else {
83			Self::No
84		}
85	}
86}

151	/// Push block under given number.
152	pub fn push_block(
153		&self,
154		block_number: BlockNumber,
155		xts: Vec<Extrinsic>,
156		is_best_block: bool,
157	) -> Header {
158		let parent_hash = {
159			let chain = self.chain.read();
160			block_number
161				.checked_sub(1)
162				.and_then(|num| {
163					chain.block_by_number.get(&num).map(|blocks| blocks[0].0.header.hash())
164				})
165				.unwrap_or_default()
166		};
167
168		self.push_block_with_parent(parent_hash, xts, is_best_block)
169	}
170
171	/// Push a block using the given `parent`.
172	///
173	/// Panics if `parent` does not exists.
174	pub fn push_block_with_parent(
175		&self,
176		parent: Hash,
177		xts: Vec<Extrinsic>,
178		is_best_block: bool,
179	) -> Header {
180		// `Hash::default()` is the genesis parent hash
181		let block_number = if parent == Hash::default() {
182			0
183		} else {
184			*self
185				.chain
186				.read()
187				.block_by_hash
188				.get(&parent)
189				.expect("`parent` exists")
190				.header()
191				.number() + 1
192		};
193
194		let header = Header {
195			number: block_number,
196			digest: Default::default(),
197			extrinsics_root: Hash::random(),
198			parent_hash: parent,
199			state_root: Default::default(),
200		};
201
202		self.add_block(Block::new(header.clone(), xts), is_best_block);
203
204		header
205	}

207	/// Add a block to the internal state.
208	pub fn add_block(&self, block: Block, is_best_block: bool) {
209		let hash = block.header.hash();
210		let block_number = block.header.number();
211
212		let mut chain = self.chain.write();
213		chain.block_by_hash.insert(hash, block.clone());
214
215		if *block_number > 0 {
216			// copy nonces to new block
217			let prev_nonces = chain
218				.nonces
219				.get(block.header.parent_hash())
220				.expect("there shall be nonces for parent block")
221				.clone();
222			chain.nonces.insert(hash, prev_nonces);
223		}
224
225		log::info!(
226			"add_block: {:?} {:?} {:?} nonces:{:#?}",
227			block_number,
228			hash,
229			block.header.parent_hash(),
230			chain.nonces
231		);
232
233		if is_best_block {
234			chain
235				.block_by_number
236				.entry(*block_number)
237				.or_default()
238				.iter_mut()
239				.for_each(|x| {
240					x.1 = IsBestBlock::No;
241				});
242		}
243
244		chain
245			.block_by_number
246			.entry(*block_number)
247			.or_default()
248			.push((block, is_best_block.into()));
249	}

251	fn hash_and_length_inner(ex: &Extrinsic) -> (Hash, usize) {
252		let encoded = ex.encode();

362#[async_trait]
363impl ChainApi for TestApi {
364	type Block = Block;
365	type Error = Error;
366
367	async fn validate_transaction(
368		&self,
369		at: <Self::Block as BlockT>::Hash,
370		source: TransactionSource,
371		uxt: Arc<<Self::Block as BlockT>::Extrinsic>,
372		_: ValidateTransactionPriority,
373	) -> Result<TransactionValidity, Error> {
374		self.validate_transaction_blocking(at, source, uxt)
375	}
376
377	fn validate_transaction_blocking(
378		&self,
379		at: <Self::Block as BlockT>::Hash,
380		_source: TransactionSource,
381		uxt: Arc<<Self::Block as BlockT>::Extrinsic>,
382	) -> Result<TransactionValidity, Error> {
383		let uxt = (*uxt).clone();
384		self.validation_requests.write().push(uxt.clone());
385		let block_number;
386
387		match self.block_id_to_number(&BlockId::Hash(at)) {
388			Ok(Some(number)) => {
389				let found_best = self
390					.chain
391					.read()
392					.block_by_number
393					.get(&number)
394					.map(|blocks| blocks.iter().any(|b| b.1.is_best()))
395					.unwrap_or(false);
396				block_number = Some(number);
397
398				// If there is no best block, we don't know based on which block we should validate
399				// the transaction. (This is not required for this test function, but in real
400				// environment it would fail because of this).
401				if !found_best {
402					return Ok(Err(TransactionValidityError::Invalid(InvalidTransaction::Custom(1))))
403				}
404			},
405			Ok(None) =>
406				return Ok(Err(TransactionValidityError::Invalid(InvalidTransaction::Custom(2)))),
407			Err(e) => return Err(e),
408		}
409
410		let (requires, provides) = if let Ok(transfer) = TransferData::try_from(&uxt) {
411			let chain_nonce = self
412				.chain
413				.read()
414				.nonces
415				.get(&at)
416				.expect("nonces must be there for every block")
417				.get(&transfer.from)
418				.cloned()
419				.unwrap_or(0);
420			let requires = if chain_nonce == transfer.nonce {
421				vec![]
422			} else {
423				if self.enable_stale_check {
424					vec![vec![transfer.from.tag_from(), (transfer.nonce - 1) as u8]]
425				} else {
426					vec![vec![(transfer.nonce - 1) as u8]]
427				}
428			};
429			let provides = if self.enable_stale_check {
430				vec![vec![transfer.from.tag_from(), transfer.nonce as u8]]
431			} else {
432				vec![vec![transfer.nonce as u8]]
433			};
434
435			log::info!(
436				"test_api::validate_transaction: h:{:?} n:{:?} cn:{:?} tn:{:?} r:{:?} p:{:?}",
437				at,
438				block_number,
439				chain_nonce,
440				transfer.nonce,
441				requires,
442				provides,
443			);
444
445			if self.enable_stale_check && transfer.nonce < chain_nonce {
446				log::info!("test_api::validate_transaction: invalid_transaction(stale)....");
447				return Ok(Err(TransactionValidityError::Invalid(InvalidTransaction::Stale)))
448			}
449
450			(requires, provides)
451		} else {
452			(Vec::new(), vec![uxt.encode()])
453		};
454
455		if self.chain.read().invalid_hashes.contains(&self.hash_and_length(&uxt).0) {
456			log::info!("test_api::validate_transaction: invalid_transaction....");
457			return Ok(Err(TransactionValidityError::Invalid(InvalidTransaction::Custom(0))))
458		}
459
460		let priority = self.chain.read().priorities.get(&self.hash_and_length(&uxt).0).cloned();
461		let longevity = self
462			.chain
463			.read()
464			.valid_till_blocks
465			.get(&self.hash_and_length(&uxt).0)
466			.cloned()
467			.map(|valid_till| valid_till.saturating_sub(block_number.unwrap()))
468			.unwrap_or(64);
469
470		if longevity == 0 {
471			return Ok(Err(TransactionValidityError::Invalid(InvalidTransaction::BadProof)))
472		}
473
474		let mut validity = ValidTransaction {
475			priority: priority.unwrap_or(1),
476			requires,
477			provides,
478			longevity,
479			propagate: true,
480		};
481
482		(self.valid_modifier.read())(&mut validity);
483
484		Ok(Ok(validity))
485	}

487	fn block_id_to_number(
488		&self,
489		at: &BlockId<Self::Block>,
490	) -> Result<Option<NumberFor<Self::Block>>, Error> {
491		Ok(match at {
492			generic::BlockId::Hash(x) =>
493				self.chain.read().block_by_hash.get(x).map(|b| *b.header.number()),
494			generic::BlockId::Number(num) => Some(*num),
495		})
496	}
497
498	fn block_id_to_hash(
499		&self,
500		at: &BlockId<Self::Block>,
501	) -> Result<Option<<Self::Block as BlockT>::Hash>, Error> {
502		Ok(match at {
503			generic::BlockId::Hash(x) => Some(*x),
504			generic::BlockId::Number(num) =>
505				self.chain.read().block_by_number.get(num).and_then(|blocks| {
506					blocks.iter().find(|b| b.1.is_best()).map(|b| b.0.header().hash())
507				}),
508		})
509	}

559/// Part of the test api.
560pub fn uxt(who: Sr25519Keyring, nonce: Nonce) -> Extrinsic {
561	let dummy = codec::Decode::decode(&mut TrailingZeroInput::zeroes()).unwrap();

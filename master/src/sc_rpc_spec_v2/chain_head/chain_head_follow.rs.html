52/// The maximum number of finalized blocks provided by the
53/// `Initialized` event.
54const MAX_FINALIZED_BLOCKS: usize = 16;
55
56/// Generates the events of the `chainHead_follow` method.
57pub struct ChainHeadFollower<BE: Backend<Block>, Block: BlockT, Client> {
58	/// Substrate client.
59	client: Arc<Client>,
60	/// Backend of the chain.
61	backend: Arc<BE>,
62	/// Subscriptions handle.
63	sub_handle: SubscriptionManagement<Block, BE>,
64	/// Subscription was started with the runtime updates flag.
65	with_runtime: bool,
66	/// Subscription ID.
67	sub_id: String,
68	/// The best reported block by this subscription.
69	current_best_block: Option<Block::Hash>,
70	/// LRU cache of pruned blocks.
71	pruned_blocks: LruMap<Block::Hash, ()>,
72	/// LRU cache of announced blocks.
73	announced_blocks: AnnouncedBlocks<Block>,
74	/// Stop all subscriptions if the distance between the leaves and the current finalized
75	/// block is larger than this value.
76	max_lagging_distance: usize,
77	/// The maximum number of pending messages per subscription.
78	pub subscription_buffer_cap: usize,
79}

81struct AnnouncedBlocks<Block: BlockT> {
82	/// Unfinalized blocks.
83	blocks: LruMap<Block::Hash, ()>,
84	/// Finalized blocks.
85	finalized: MostRecentFinalizedBlocks<Block>,
86}
87
88/// Wrapper over LRU to efficiently lookup hashes and remove elements as FIFO queue.
89///
90/// For the finalized blocks we use `peek` to avoid moving the block counter to the front.
91/// This effectively means that the LRU acts as a FIFO queue. Otherwise, we might
92/// end up with scenarios where the "finalized block" in the end of LRU is overwritten which
93/// may not necessarily be the oldest finalized block i.e, possible that "get" promotes an
94/// older finalized block because it was accessed more recently.
95struct MostRecentFinalizedBlocks<Block: BlockT>(LruMap<Block::Hash, ()>);
96
97impl<Block: BlockT> MostRecentFinalizedBlocks<Block> {
98	/// Insert the finalized block hash into the LRU cache.
99	fn insert(&mut self, block: Block::Hash) {
100		self.0.insert(block, ());
101	}
102
103	/// Check if the block is contained in the LRU cache.
104	fn contains(&mut self, block: &Block::Hash) -> Option<&()> {
105		self.0.peek(block)
106	}
107}
108
109impl<Block: BlockT> AnnouncedBlocks<Block> {
110	/// Creates a new `AnnouncedBlocks`.
111	fn new() -> Self {
112		Self {
113			// The total number of pinned blocks is `MAX_PINNED_BLOCKS`, ensure we don't
114			// exceed the limit.
115			blocks: LruMap::new(ByLength::new((MAX_PINNED_BLOCKS - MAX_FINALIZED_BLOCKS) as u32)),
116			// We are keeping a smaller number of announced finalized blocks in memory.
117			// This is because the `Finalized` event might be triggered before the `NewBlock` event.
118			finalized: MostRecentFinalizedBlocks(LruMap::new(ByLength::new(
119				MAX_FINALIZED_BLOCKS as u32,
120			))),
121		}
122	}
123
124	/// Insert the block into the announced blocks.
125	fn insert(&mut self, block: Block::Hash, finalized: bool) {
126		if finalized {
127			// When a block is declared as finalized, it is removed from the unfinalized blocks.
128			//
129			// Given that the finalized blocks are bounded to `MAX_FINALIZED_BLOCKS`,
130			// this ensures we keep the minimum number of blocks in memory.
131			self.blocks.remove(&block);
132			self.finalized.insert(block);
133		} else {
134			self.blocks.insert(block, ());
135		}
136	}
137
138	/// Check if the block was previously announced.
139	fn was_announced(&mut self, block: &Block::Hash) -> bool {
140		self.blocks.get(block).is_some() || self.finalized.contains(block).is_some()
141	}
142}

172/// A block notification.
173enum NotificationType<Block: BlockT> {
174	/// The initial events generated from the node's memory.
175	InitialEvents(Vec<FollowEvent<Block::Hash>>),
176	/// The new block notification obtained from `import_notification_stream`.
177	NewBlock(BlockImportNotification<Block>),
178	/// The finalized block notification obtained from `finality_notification_stream`.
179	Finalized(FinalityNotification<Block>),
180	/// The response of `chainHead` method calls.
181	MethodResponse(FollowEvent<Block::Hash>),
182}

184/// The initial blocks that should be reported or ignored by the chainHead.
185#[derive(Clone, Debug)]
186struct InitialBlocks<Block: BlockT> {
187	/// Children of the latest finalized block, for which the `NewBlock`
188	/// event must be generated.
189	///
190	/// It is a tuple of (block hash, parent hash).
191	finalized_block_descendants: Vec<(Block::Hash, Block::Hash)>,
192	/// Hashes of the last finalized blocks
193	finalized_block_hashes: VecDeque<Block::Hash>,
194	/// Blocks that should not be reported as pruned by the `Finalized` event.
195	///
196	/// Substrate database will perform the pruning of height N at
197	/// the finalization N + 1. We could have the following block tree
198	/// when the user subscribes to the `follow` method:
199	///   [A] - [A1] - [A2] - [A3]
200	///                 ^^ finalized
201	///       - [A1] - [B1]
202	///
203	/// When the A3 block is finalized, B1 is reported as pruned, however
204	/// B1 was never reported as `NewBlock` (and as such was never pinned).
205	/// This is because the `NewBlock` events are generated for children of
206	/// the finalized hash.
207	pruned_forks: HashSet<Block::Hash>,
208}

210/// The startup point from which chainHead started to generate events.
211struct StartupPoint<Block: BlockT> {
212	/// Best block hash.
213	pub best_hash: Block::Hash,
214	/// The head of the finalized chain.
215	pub finalized_hash: Block::Hash,
216	/// Last finalized block number.
217	pub finalized_number: NumberFor<Block>,
218}

227	}

241	/// Conditionally generate the runtime event of the given block.
242	fn generate_runtime_event(
243		&self,
244		block: Block::Hash,
245		parent: Option<Block::Hash>,
246	) -> Option<RuntimeEvent> {
247		// No runtime versions should be reported.
248		if !self.with_runtime {
249			return None
250		}
251
252		let block_rt = match self.client.runtime_version_at(block) {
253			Ok(rt) => rt,
254			Err(err) => return Some(err.into()),
255		};
256
257		let parent = match parent {
258			Some(parent) => parent,
259			// Nothing to compare against, always report.
260			None => return Some(RuntimeEvent::Valid(RuntimeVersionEvent { spec: block_rt.into() })),
261		};
262
263		let parent_rt = match self.client.runtime_version_at(parent) {
264			Ok(rt) => rt,
265			Err(err) => return Some(err.into()),
266		};
267
268		// Report the runtime version change.
269		if block_rt != parent_rt {
270			Some(RuntimeEvent::Valid(RuntimeVersionEvent { spec: block_rt.into() }))
271		} else {
272			None
273		}
274	}

276	/// Check the distance between the provided blocks does not exceed a
277	/// a reasonable range.
278	///
279	/// When the blocks are too far apart (potentially millions of blocks):
280	///  - Tree route is expensive to calculate.
281	///  - The RPC layer will not be able to generate the `NewBlock` events for all blocks.
282	///
283	/// This edge-case can happen for parachains where the relay chain syncs slower to
284	/// the head of the chain than the parachain node that is synced already.
285	fn distance_within_reason(
286		&self,
287		block: Block::Hash,
288		finalized: Block::Hash,
289	) -> Result<(), SubscriptionManagementError> {
290		let Some(block_num) = self.client.number(block)? else {
291			return Err(SubscriptionManagementError::BlockHashAbsent)
292		};
293		let Some(finalized_num) = self.client.number(finalized)? else {
294			return Err(SubscriptionManagementError::BlockHashAbsent)
295		};
296
297		let distance: usize = block_num.saturating_sub(finalized_num).saturated_into();
298		if distance > self.max_lagging_distance {
299			return Err(SubscriptionManagementError::BlockDistanceTooLarge);
300		}
301
302		Ok(())
303	}

305	/// Get the in-memory blocks of the client, starting from the provided finalized hash.
306	///
307	/// The reported blocks are pinned by this function.
308	fn get_init_blocks_with_forks(
309		&self,
310		finalized: Block::Hash,
311	) -> Result<InitialBlocks<Block>, SubscriptionManagementError> {
312		let blockchain = self.backend.blockchain();
313		let leaves = blockchain.leaves()?;
314		let mut pruned_forks = HashSet::new();
315		let mut finalized_block_descendants = Vec::new();
316		let mut unique_descendants = HashSet::new();
317
318		// Ensure all leaves are within a reasonable distance from the finalized block,
319		// before traversing the tree.
320		for leaf in &leaves {
321			self.distance_within_reason(*leaf, finalized)?;
322		}
323
324		for leaf in leaves {
325			let tree_route = sp_blockchain::tree_route(blockchain, finalized, leaf)?;
326
327			let blocks = tree_route.enacted().iter().map(|block| block.hash);
328			if !tree_route.retracted().is_empty() {
329				pruned_forks.extend(blocks);
330			} else {
331				// Ensure a `NewBlock` event is generated for all children of the
332				// finalized block. Describe the tree route as (child_node, parent_node)
333				// Note: the order of elements matters here.
334				let mut parent = finalized;
335				for child in blocks {
336					let pair = (child, parent);
337
338					if unique_descendants.insert(pair) {
339						// The finalized block is pinned below.
340						self.sub_handle.pin_block(&self.sub_id, child)?;
341						finalized_block_descendants.push(pair);
342					}
343
344					parent = child;
345				}
346			}
347		}
348
349		let mut current_block = finalized;
350		// The header of the finalized block must not be pruned.
351		let Some(header) = blockchain.header(current_block)? else {
352			return Err(SubscriptionManagementError::BlockHeaderAbsent);
353		};
354
355		// Report at most `MAX_FINALIZED_BLOCKS`. Note: The node might not have that many blocks.
356		let mut finalized_block_hashes = VecDeque::with_capacity(MAX_FINALIZED_BLOCKS);
357
358		// Pin the finalized block.
359		self.sub_handle.pin_block(&self.sub_id, current_block)?;
360		finalized_block_hashes.push_front(current_block);
361		current_block = *header.parent_hash();
362
363		for _ in 0..MAX_FINALIZED_BLOCKS - 1 {
364			let Ok(Some(header)) = blockchain.header(current_block) else { break };
365			// Block cannot be reported if pinning fails.
366			if self.sub_handle.pin_block(&self.sub_id, current_block).is_err() {
367				break
368			};
369
370			finalized_block_hashes.push_front(current_block);
371			current_block = *header.parent_hash();
372		}
373
374		Ok(InitialBlocks { finalized_block_descendants, finalized_block_hashes, pruned_forks })
375	}

377	/// Generate the initial events reported by the RPC `follow` method.
378	///
379	/// Returns the initial events that should be reported directly.
380	fn generate_init_events(
381		&mut self,
382		startup_point: &StartupPoint<Block>,
383	) -> Result<Vec<FollowEvent<Block::Hash>>, SubscriptionManagementError> {
384		let init = self.get_init_blocks_with_forks(startup_point.finalized_hash)?;
385
386		// The initialized event is the first one sent.
387		let initial_blocks = init.finalized_block_descendants;
388		let finalized_block_hashes = init.finalized_block_hashes;
389		// These are the pruned blocks that we should not report again.
390		for pruned in init.pruned_forks {
391			self.pruned_blocks.insert(pruned, ());
392		}
393
394		let finalized_block_hash = startup_point.finalized_hash;
395		let finalized_block_runtime = self.generate_runtime_event(finalized_block_hash, None);
396
397		for finalized in &finalized_block_hashes {
398			self.announced_blocks.insert(*finalized, true);
399		}
400
401		let initialized_event = FollowEvent::Initialized(Initialized {
402			finalized_block_hashes: finalized_block_hashes.into(),
403			finalized_block_runtime,
404			with_runtime: self.with_runtime,
405		});
406
407		let mut finalized_block_descendants = Vec::with_capacity(initial_blocks.len() + 1);
408
409		finalized_block_descendants.push(initialized_event);
410		for (child, parent) in initial_blocks.into_iter() {
411			// If the parent was not announced we have a gap currently.
412			// This can happen during a WarpSync.
413			if !self.announced_blocks.was_announced(&parent) {
414				return Err(SubscriptionManagementError::BlockHeaderAbsent);
415			}
416			self.announced_blocks.insert(child, false);
417
418			let new_runtime = self.generate_runtime_event(child, Some(parent));
419
420			let event = FollowEvent::NewBlock(NewBlock {
421				block_hash: child,
422				parent_block_hash: parent,
423				new_runtime,
424				with_runtime: self.with_runtime,
425			});
426
427			finalized_block_descendants.push(event);
428		}
429
430		// Generate a new best block event.
431		let best_block_hash = startup_point.best_hash;
432		if !self.announced_blocks.was_announced(&best_block_hash) {
433			return Err(SubscriptionManagementError::BlockHeaderAbsent);
434		}
435		self.announced_blocks.insert(best_block_hash, true);
436
437		let best_block = FollowEvent::BestBlockChanged(BestBlockChanged { best_block_hash });
438		self.current_best_block = Some(best_block_hash);
439		finalized_block_descendants.push(best_block);
440
441		Ok(finalized_block_descendants)
442	}

444	/// Generate the "NewBlock" event and potentially the "BestBlockChanged" event for the
445	/// given block hash.
446	fn generate_import_events(
447		&mut self,
448		block_hash: Block::Hash,
449		parent_block_hash: Block::Hash,
450		is_best_block: bool,
451	) -> Vec<FollowEvent<Block::Hash>> {
452		let new_runtime = self.generate_runtime_event(block_hash, Some(parent_block_hash));
453
454		let new_block = FollowEvent::NewBlock(NewBlock {
455			block_hash,
456			parent_block_hash,
457			new_runtime,
458			with_runtime: self.with_runtime,
459		});
460
461		if !is_best_block {
462			return vec![new_block]
463		}
464
465		// If this is the new best block, then we need to generate two events.
466		let best_block_event =
467			FollowEvent::BestBlockChanged(BestBlockChanged { best_block_hash: block_hash });
468
469		match self.current_best_block {
470			Some(block_cache) => {
471				// The RPC layer has not reported this block as best before.
472				// Note: This handles the race with the finalized branch.
473				if block_cache != block_hash {
474					self.current_best_block = Some(block_hash);
475					vec![new_block, best_block_event]
476				} else {
477					vec![new_block]
478				}
479			},
480			None => {
481				self.current_best_block = Some(block_hash);
482				vec![new_block, best_block_event]
483			},
484		}
485	}

487	/// Handle the import of new blocks by generating the appropriate events.
488	fn handle_import_blocks(
489		&mut self,
490		notification: BlockImportNotification<Block>,
491		startup_point: &StartupPoint<Block>,
492	) -> Result<Vec<FollowEvent<Block::Hash>>, SubscriptionManagementError> {
493		let block_hash = notification.hash;
494
495		// Ensure we are only reporting blocks after the starting point.
496		if *notification.header.number() < startup_point.finalized_number {
497			return Ok(Default::default())
498		}
499
500		// Ensure the block can be pinned before generating the events.
501		if !self.sub_handle.pin_block(&self.sub_id, block_hash)? {
502			// The block is already pinned, this is similar to the check above.
503			//
504			// The `SubscriptionManagement` ensures the block is tracked until (short lived):
505			// - 2 calls to `pin_block` are made (from `Finalized` and `NewBlock` branches).
506			// - the block is unpinned by the user
507			//
508			// This is rather a sanity checks for edge-cases (in theory), where
509			// [`MAX_FINALIZED_BLOCKS` + 1] finalized events are triggered before the `NewBlock`
510			// event of the first `Finalized` event.
511			return Ok(Default::default())
512		}
513
514		if self.announced_blocks.was_announced(&block_hash) {
515			// Block was already reported by the finalized branch.
516			return Ok(Default::default())
517		}
518
519		// Double check the parent hash. If the parent hash is not reported, we have a gap.
520		let parent_block_hash = *notification.header.parent_hash();
521		if !self.announced_blocks.was_announced(&parent_block_hash) {
522			// The parent block was not reported, we have a gap.
523			return Err(SubscriptionManagementError::Custom("Parent block was not reported".into()))
524		}
525
526		self.announced_blocks.insert(block_hash, false);
527		Ok(self.generate_import_events(block_hash, parent_block_hash, notification.is_new_best))
528	}

530	/// Generates new block events from the given finalized hashes.
531	///
532	/// It may be possible that the `Finalized` event fired before the `NewBlock`
533	/// event. Only in that case we generate:
534	/// - `NewBlock` event for all finalized hashes.
535	/// - `BestBlock` event for the last finalized hash.
536	///
537	/// This function returns an empty list if all finalized hashes were already reported
538	/// and are pinned.
539	fn generate_finalized_events(
540		&mut self,
541		finalized_block_hashes: &[Block::Hash],
542	) -> Result<Vec<FollowEvent<Block::Hash>>, SubscriptionManagementError> {
543		let mut events = Vec::new();
544
545		// Nothing to be done if no finalized hashes are provided.
546		let Some(first_hash) = finalized_block_hashes.get(0) else { return Ok(Default::default()) };
547
548		// Find the parent header.
549		let Some(first_header) = self.client.header(*first_hash)? else {
550			return Err(SubscriptionManagementError::BlockHeaderAbsent)
551		};
552
553		if !self.announced_blocks.was_announced(first_header.parent_hash()) {
554			return Err(SubscriptionManagementError::Custom(
555				"Parent block was not reported for a finalized block".into(),
556			));
557		}
558
559		let parents =
560			std::iter::once(first_header.parent_hash()).chain(finalized_block_hashes.iter());
561		for (i, (hash, parent)) in finalized_block_hashes.iter().zip(parents).enumerate() {
562			// Ensure the block is pinned before generating the events.
563			self.sub_handle.pin_block(&self.sub_id, *hash)?;
564
565			// Check if the block was already reported.
566			if self.announced_blocks.was_announced(hash) {
567				continue;
568			}
569
570			// Generate `NewBlock` events for all blocks beside the last block in the list
571			let is_last = i + 1 == finalized_block_hashes.len();
572			if !is_last {
573				// Generate only the `NewBlock` event for this block.
574				events.extend(self.generate_import_events(*hash, *parent, false));
575				self.announced_blocks.insert(*hash, true);
576				continue;
577			}
578
579			if let Some(best_block_hash) = self.current_best_block {
580				let ancestor =
581					sp_blockchain::lowest_common_ancestor(&*self.client, *hash, best_block_hash)?;
582
583				// If we end up here and the `best_block` is a descendent of the finalized block
584				// (last block in the list), it means that there were skipped notifications.
585				// Otherwise `pin_block` would had returned `false`.
586				//
587				// When the node falls out of sync and then syncs up to the tip of the chain, it can
588				// happen that we skip notifications. Then it is better to terminate the connection
589				// instead of trying to send notifications for all missed blocks.
590				if ancestor.hash == *hash {
591					return Err(SubscriptionManagementError::Custom(
592						"A descendent of the finalized block was already reported".into(),
593					))
594				}
595			}
596
597			// Let's generate the `NewBlock` and `NewBestBlock` events for the block.
598			events.extend(self.generate_import_events(*hash, *parent, true));
599			self.announced_blocks.insert(*hash, true);
600		}
601
602		Ok(events)
603	}

605	/// Get all pruned block hashes from the provided stale blocks.
606	fn get_pruned_hashes(
607		&mut self,
608		stale_blocks: &[Arc<StaleBlock<Block>>],
609	) -> Result<Vec<Block::Hash>, SubscriptionManagementError> {
610		Ok(stale_blocks
611			.iter()
612			.filter_map(|block| {
613				if self.pruned_blocks.get(&block.hash).is_some() {
614					// The block was already reported as pruned.
615					return None
616				}
617
618				self.pruned_blocks.insert(block.hash, ());
619				Some(block.hash)
620			})
621			.collect())
622	}

624	/// Handle the finalization notification by generating the `Finalized` event.
625	///
626	/// If the block of the notification was not reported yet, this method also
627	/// generates the events similar to `handle_import_blocks`.
628	fn handle_finalized_blocks(
629		&mut self,
630		notification: FinalityNotification<Block>,
631		startup_point: &StartupPoint<Block>,
632	) -> Result<Vec<FollowEvent<Block::Hash>>, SubscriptionManagementError> {
633		let last_finalized = notification.hash;
634
635		// Ensure we are only reporting blocks after the starting point.
636		if *notification.header.number() < startup_point.finalized_number {
637			return Ok(Default::default())
638		}
639
640		// The tree route contains the exclusive path from the last finalized block to the block
641		// reported by the notification. Ensure the finalized block is also reported.
642		let mut finalized_block_hashes = notification.tree_route.to_vec();
643		finalized_block_hashes.push(last_finalized);
644
645		// If the finalized hashes were not reported yet, generate the `NewBlock` events.
646		let mut events = self.generate_finalized_events(&finalized_block_hashes)?;
647
648		// Report all pruned blocks from the notification that are not
649		// part of the fork we need to ignore.
650		let pruned_block_hashes = self.get_pruned_hashes(&notification.stale_blocks)?;
651
652		for finalized in &finalized_block_hashes {
653			self.announced_blocks.insert(*finalized, true);
654		}
655
656		let finalized_event = FollowEvent::Finalized(Finalized {
657			finalized_block_hashes,
658			pruned_block_hashes: pruned_block_hashes.clone(),
659		});
660
661		if let Some(current_best_block) = self.current_best_block {
662			// We need to generate a new best block if the best block is in the pruned list.
663			let is_in_pruned_list =
664				pruned_block_hashes.iter().any(|hash| *hash == current_best_block);
665			if is_in_pruned_list {
666				self.current_best_block = Some(last_finalized);
667				events.push(FollowEvent::BestBlockChanged(BestBlockChanged {
668					best_block_hash: last_finalized,
669				}));
670			} else {
671				// The pruning logic ensures that when the finalized block is announced,
672				// all blocks on forks that have the common ancestor lower or equal
673				// to the finalized block are reported.
674				//
675				// However, we double check if the best block is a descendant of the last finalized
676				// block to ensure we don't miss any events.
677				let ancestor = sp_blockchain::lowest_common_ancestor(
678					&*self.client,
679					last_finalized,
680					current_best_block,
681				)?;
682				let is_descendant = ancestor.hash == last_finalized;
683				if !is_descendant {
684					self.current_best_block = Some(last_finalized);
685					events.push(FollowEvent::BestBlockChanged(BestBlockChanged {
686						best_block_hash: last_finalized,
687					}));
688				}
689			}
690		}
691
692		events.push(finalized_event);
693		Ok(events)
694	}

696	/// Submit the events from the provided stream to the RPC client
697	/// for as long as the `rx_stop` event was not called.
698	async fn submit_events<EventStream>(
699		&mut self,
700		startup_point: &StartupPoint<Block>,
701		stream: EventStream,
702		sink: Subscription,
703		rx_stop: oneshot::Receiver<()>,
704	) -> Result<(), SubscriptionManagementError>
705	where
706		EventStream: Stream<Item = NotificationType<Block>> + Unpin + Send,
707	{
708		let buffer_cap = self.subscription_buffer_cap;
709		// create a channel to propagate error messages
710		let mut handle_events = |event| match event {
711			NotificationType::InitialEvents(events) => Ok(events),
712			NotificationType::NewBlock(notification) =>
713				self.handle_import_blocks(notification, &startup_point),
714			NotificationType::Finalized(notification) =>
715				self.handle_finalized_blocks(notification, &startup_point),
716			NotificationType::MethodResponse(notification) => Ok(vec![notification]),
717		};
718
719		let stream = stream
720			.map(|event| handle_events(event))
721			.map_ok(|items| stream::iter(items).map(Ok))
722			.try_flatten();
723
724		tokio::pin!(stream);
725
726		let sink_future =
727			sink.pipe_from_try_stream(stream, sc_rpc::utils::BoundedVecDeque::new(buffer_cap));
728
729		let result = tokio::select! {
730			_ = rx_stop => Ok(()),
731			result = sink_future => {
732				if let Err(ref e) = result {
733					debug!(
734						target: LOG_TARGET,
735						"[follow][id={:?}] Failed to handle stream notification {:?}",
736						&self.sub_id,
737						e
738					);
739				};
740				result
741			}
742		};
743		let _ = sink.send(&FollowEvent::<String>::Stop).await;
744		result
745	}

747	/// Generate the block events for the `chainHead_follow` method.
748	pub async fn generate_events(
749		&mut self,
750		sink: Subscription,
751		sub_data: InsertedSubscriptionData<Block>,
752	) -> Result<(), SubscriptionManagementError> {
753		// Register for the new block and finalized notifications.
754		let stream_import = self
755			.client
756			.import_notification_stream()
757			.map(|notification| NotificationType::NewBlock(notification));
758
759		let stream_finalized = self
760			.client
761			.finality_notification_stream()
762			.map(|notification| NotificationType::Finalized(notification));
763
764		let stream_responses = sub_data
765			.response_receiver
766			.map(|response| NotificationType::MethodResponse(response));
767
768		let startup_point = StartupPoint::from(self.client.info());
769		let initial_events = match self.generate_init_events(&startup_point) {
770			Ok(blocks) => blocks,
771			Err(err) => {
772				debug!(
773					target: LOG_TARGET,
774					"[follow][id={:?}] Failed to generate the initial events {:?}",
775					self.sub_id,
776					err
777				);
778				let _ = sink.send(&FollowEvent::<String>::Stop).await;
779				return Err(err)
780			},
781		};
782
783		let initial = NotificationType::InitialEvents(initial_events);
784		let merged = tokio_stream::StreamExt::merge(stream_import, stream_finalized);
785		let merged = tokio_stream::StreamExt::merge(merged, stream_responses);
786		let stream = stream::once(futures::future::ready(initial)).chain(merged);
787
788		self.submit_events(&startup_point, stream.boxed(), sink, sub_data.rx_stop).await
789	}

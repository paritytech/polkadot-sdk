17//! Memory stats for preparation.
18//!
19//! Right now we gather three measurements:
20//!
21//! - `ru_maxrss` (resident set size) from `getrusage`.
22//! - `resident` memory stat provided by `tikv-malloc-ctl`.
23//! - `allocated` memory stat also from `tikv-malloc-ctl`.
24//!
25//! Currently we are only logging these for the purposes of gathering data. In the future, we may
26//! use these stats to reject PVFs during pre-checking. See
27//! <https://github.com/paritytech/polkadot/issues/6472#issuecomment-1381941762> for more
28//! background.
29
30/// Module for the memory tracker. The memory tracker runs in its own thread, where it polls memory
31/// usage at an interval.
32///
33/// NOTE: Requires jemalloc enabled.
34#[cfg(any(target_os = "linux", feature = "jemalloc-allocator"))]
35pub mod memory_tracker {
36	use crate::LOG_TARGET;
37	use polkadot_node_core_pvf_common::{
38		prepare::MemoryAllocationStats,
39		worker::{stringify_panic_payload, thread},
40	};
41	use std::{thread::JoinHandle, time::Duration};
42	use tikv_jemalloc_ctl::{epoch, stats, Error};
43
44	#[derive(Clone)]
45	struct MemoryAllocationTracker {
46		epoch: tikv_jemalloc_ctl::epoch_mib,
47		allocated: stats::allocated_mib,
48		resident: stats::resident_mib,
49	}
50
51	impl MemoryAllocationTracker {
52		pub fn new() -> Result<Self, Error> {
53			Ok(Self {
54				epoch: epoch::mib()?,
55				allocated: stats::allocated::mib()?,
56				resident: stats::resident::mib()?,
57			})
58		}
59
60		pub fn snapshot(&self) -> Result<MemoryAllocationStats, Error> {
61			// update stats by advancing the allocation epoch
62			self.epoch.advance()?;
63
64			// Convert to `u64`, as `usize` is not `Encode`able.
65			let allocated = self.allocated.read()? as u64;
66			let resident = self.resident.read()? as u64;
67			Ok(MemoryAllocationStats { allocated, resident })
68		}
69	}
70
71	/// Runs a thread in the background that observes memory statistics. The goal is to try to get
72	/// accurate stats during preparation.
73	///
74	/// # Algorithm
75	///
76	/// 1. Create the memory tracker.
77	///
78	/// 2. Sleep for some short interval. Whenever we wake up, take a snapshot by updating the
79	///    allocation epoch.
80	///
81	/// 3. When we are notified that preparation has completed, take one last snapshot and return
82	///    the maximum observed values.
83	///
84	/// # Errors
85	///
86	/// For simplicity, any errors are returned as a string. As this is not a critical component,
87	/// errors are used for informational purposes (logging) only.
88	pub fn memory_tracker_loop(condvar: thread::Cond) -> Result<MemoryAllocationStats, String> {
89		// NOTE: This doesn't need to be too fine-grained since preparation currently takes 3-10s or
90		// more. Apart from that, there is not really a science to this number.
91		const POLL_INTERVAL: Duration = Duration::from_millis(100);
92
93		let tracker = MemoryAllocationTracker::new().map_err(|err| err.to_string())?;
94		let mut max_stats = MemoryAllocationStats::default();
95
96		let mut update_stats = || -> Result<(), String> {
97			let current_stats = tracker.snapshot().map_err(|err| err.to_string())?;
98			if current_stats.resident > max_stats.resident {
99				max_stats.resident = current_stats.resident;
100			}
101			if current_stats.allocated > max_stats.allocated {
102				max_stats.allocated = current_stats.allocated;
103			}
104			Ok(())
105		};
106
107		loop {
108			// Take a snapshot and update the max stats.
109			update_stats()?;
110
111			// Sleep for the poll interval, or wake up if the condvar is triggered. Note that
112			// `wait_timeout_while` is documented as not being very precise or reliable, which is
113			// fine here -- see note above.
114			match thread::wait_for_threads_with_timeout(&condvar, POLL_INTERVAL) {
115				Some(_outcome) => {
116					update_stats()?;
117					return Ok(max_stats)
118				},
119				None => continue,
120			}
121		}
122	}
123
124	/// Helper function to get the stats from the memory tracker. Helps isolate this error handling.
125	pub fn get_memory_tracker_loop_stats(
126		thread: JoinHandle<Result<MemoryAllocationStats, String>>,
127		worker_pid: u32,
128	) -> Option<MemoryAllocationStats> {
129		match thread.join() {
130			Ok(Ok(stats)) => Some(stats),
131			Ok(Err(err)) => {
132				gum::warn!(
133					target: LOG_TARGET,
134					%worker_pid,
135					"worker: error occurred in the memory tracker thread: {}", err
136				);
137				None
138			},
139			Err(err) => {
140				gum::warn!(
141					target: LOG_TARGET,
142					%worker_pid,
143					"worker: error joining on memory tracker thread: {}", stringify_panic_payload(err)
144				);
145				None
146			},
147		}
148	}
149}

151/// Module for dealing with the `ru_maxrss` (peak resident memory) stat from `getrusage`.
152///
153/// NOTE: `getrusage` with the `RUSAGE_THREAD` parameter is only supported on Linux. `RUSAGE_SELF`
154/// works on MacOS, but we need to get the max rss only for the preparation thread. Getting it for
155/// the current process would conflate the stats of previous jobs run by the process.
156#[cfg(target_os = "linux")]
157pub mod max_rss_stat {
158	use crate::LOG_TARGET;
159	use core::mem::MaybeUninit;
160	use libc::{getrusage, rusage, RUSAGE_THREAD};
161	use std::io;
162
163	/// Get the rusage stats for the current thread.
164	fn getrusage_thread() -> io::Result<rusage> {
165		let mut result: MaybeUninit<rusage> = MaybeUninit::zeroed();
166
167		// SAFETY: `result` is a valid pointer, so calling this is safe.
168		if unsafe { getrusage(RUSAGE_THREAD, result.as_mut_ptr()) } == -1 {
169			return Err(io::Error::last_os_error())
170		}
171
172		// SAFETY: `result` was successfully initialized by `getrusage`.
173		unsafe { Ok(result.assume_init()) }
174	}
175
176	/// Gets the `ru_maxrss` for the current thread.
177	pub fn get_max_rss_thread() -> io::Result<i64> {
178		// `c_long` is either `i32` or `i64` depending on architecture. `i64::from` always works.
179		getrusage_thread().map(|rusage| i64::from(rusage.ru_maxrss))
180	}
181
182	/// Extracts the max_rss stat and logs any error.
183	pub fn extract_max_rss_stat(max_rss: io::Result<i64>, worker_pid: u32) -> Option<i64> {
184		max_rss
185			.map_err(|err| {
186				gum::warn!(
187					target: LOG_TARGET,
188					%worker_pid,
189					"error getting `ru_maxrss` in preparation thread: {}",
190					err
191				);
192				err
193			})
194			.ok()
195	}

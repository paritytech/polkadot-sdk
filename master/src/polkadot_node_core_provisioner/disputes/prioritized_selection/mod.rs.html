41/// The maximum number of disputes Provisioner will include in the inherent data.
42/// Serves as a protection not to flood the Runtime with excessive data.
43#[cfg(not(test))]
44pub const MAX_DISPUTE_VOTES_FORWARDED_TO_RUNTIME: usize = 200_000;
45#[cfg(test)]
46pub const MAX_DISPUTE_VOTES_FORWARDED_TO_RUNTIME: usize = 200;
47
48/// Controls how much dispute votes to be fetched from the `dispute-coordinator` per iteration in
49/// `fn vote_selection`. The purpose is to fetch the votes in batches until
50/// `MAX_DISPUTE_VOTES_FORWARDED_TO_RUNTIME` is reached. If all votes are fetched in single call
51/// we might fetch votes which we never use. This will create unnecessary load on
52/// `dispute-coordinator`.
53///
54/// This value should be less than `MAX_DISPUTE_VOTES_FORWARDED_TO_RUNTIME`. Increase it in case
55/// `provisioner` sends too many `QueryCandidateVotes` messages to `dispute-coordinator`.
56#[cfg(not(test))]
57const VOTES_SELECTION_BATCH_SIZE: usize = 1_100;
58#[cfg(test)]
59const VOTES_SELECTION_BATCH_SIZE: usize = 11;
60
61/// Implements the `select_disputes` function which selects dispute votes which should
62/// be sent to the Runtime.
63///
64/// # How the prioritization works
65///
66/// Generally speaking disputes can be described as:
67///   * Active vs Inactive
68///   * Known vs Unknown onchain
69///   * Offchain vs Onchain
70///   * Concluded onchain vs Unconcluded onchain
71///
72/// Provisioner fetches all disputes from `dispute-coordinator` and separates them in multiple
73/// partitions. Please refer to `struct PartitionedDisputes` for details about the actual
74/// partitions. Each partition has got a priority implicitly assigned to it and the disputes are
75/// selected based on this priority (e.g. disputes in partition 1, then if there is space - disputes
76/// from partition 2 and so on).
77///
78/// # Votes selection
79///
80/// Besides the prioritization described above the votes in each partition are filtered too.
81/// Provisioner fetches all onchain votes and filters them out from all partitions. As a result the
82/// Runtime receives only fresh votes (votes it didn't know about).
83///
84/// # How the onchain votes are fetched
85///
86/// The logic outlined above relies on `RuntimeApiRequest::Disputes` message from the Runtime. The
87/// user check the Runtime version before calling `select_disputes`. If the function is used with
88/// old runtime an error is logged and the logic will continue with empty onchain votes `HashMap`.
89pub async fn select_disputes<Sender>(
90	sender: &mut Sender,
91	metrics: &metrics::Metrics,
92	leaf: &ActivatedLeaf,
93) -> MultiDisputeStatementSet
94where
95	Sender: overseer::ProvisionerSenderTrait,
96{
97	gum::trace!(
98		target: LOG_TARGET,
99		?leaf,
100		"Selecting disputes for inherent data using prioritized selection"
101	);
102
103	// Fetch the onchain disputes. We'll do a prioritization based on them.
104	let onchain = match get_onchain_disputes(sender, leaf.hash).await {
105		Ok(r) => {
106			gum::trace!(
107				target: LOG_TARGET,
108				?leaf,
109				"Successfully fetched {} onchain disputes",
110				r.len()
111			);
112			r
113		},
114		Err(GetOnchainDisputesError::NotSupported(runtime_api_err, relay_parent)) => {
115			// Runtime version is checked before calling this method, so the error below should
116			// never happen!
117			gum::error!(
118				target: LOG_TARGET,
119				?runtime_api_err,
120				?relay_parent,
121				"Can't fetch onchain disputes, because ParachainHost runtime api version is old. Will continue with empty onchain disputes set.",
122			);
123			HashMap::new()
124		},
125		Err(GetOnchainDisputesError::Channel) => {
126			// This error usually means the node is shutting down. Log just in case.
127			gum::debug!(
128				target: LOG_TARGET,
129				"Channel error occurred while fetching onchain disputes. Will continue with empty onchain disputes set.",
130			);
131			HashMap::new()
132		},
133		Err(GetOnchainDisputesError::Execution(runtime_api_err, parent_hash)) => {
134			gum::warn!(
135				target: LOG_TARGET,
136				?runtime_api_err,
137				?parent_hash,
138				"Unexpected execution error occurred while fetching onchain votes. Will continue with empty onchain disputes set.",
139			);
140			HashMap::new()
141		},
142	};
143	metrics.on_fetched_onchain_disputes(onchain.keys().len() as u64);
144
145	gum::trace!(target: LOG_TARGET, ?leaf, "Fetching recent disputes");
146	let recent_disputes = request_disputes(sender).await;
147	gum::trace!(
148		target: LOG_TARGET,
149		?leaf,
150		"Got {} recent disputes and {} onchain disputes.",
151		recent_disputes.len(),
152		onchain.len(),
153	);
154
155	gum::trace!(target: LOG_TARGET, ?leaf, "Filtering recent disputes");
156	// Filter out unconfirmed disputes. However if the dispute is already onchain - don't skip it.
157	// In this case we'd better push as much fresh votes as possible to bring it to conclusion
158	// faster.
159	let recent_disputes = recent_disputes
160		.into_iter()
161		.filter(|(key, dispute_status)| {
162			dispute_status.is_confirmed_concluded() || onchain.contains_key(key)
163		})
164		.collect::<BTreeMap<_, _>>();
165	gum::trace!(target: LOG_TARGET, ?leaf, "Partitioning recent disputes");
166	let partitioned = partition_recent_disputes(recent_disputes, &onchain);
167	metrics.on_partition_recent_disputes(&partitioned);
168
169	if partitioned.inactive_unknown_onchain.len() > 0 {
170		gum::warn!(
171			target: LOG_TARGET,
172			?leaf,
173			"Got {} inactive unknown onchain disputes. This should not happen in normal conditions!",
174			partitioned.inactive_unknown_onchain.len()
175		);
176	}
177
178	gum::trace!(target: LOG_TARGET, ?leaf, "Vote selection for recent disputes");
179	let result = vote_selection(sender, partitioned, &onchain).await;
180
181	gum::trace!(target: LOG_TARGET, ?leaf, "Convert to multi dispute statement set");
182	make_multi_dispute_statement_set(metrics, result)
183}

185/// Selects dispute votes from `PartitionedDisputes` which should be sent to the runtime. Votes
186/// which are already onchain are filtered out. Result should be sorted by `(SessionIndex,
187/// CandidateHash)` which is enforced by the `BTreeMap`. This is a requirement from the runtime.
188async fn vote_selection<Sender>(
189	sender: &mut Sender,
190	partitioned: PartitionedDisputes,
191	onchain: &HashMap<(SessionIndex, CandidateHash), DisputeState>,
192) -> BTreeMap<(SessionIndex, CandidateHash), CandidateVotes>
193where
194	Sender: overseer::ProvisionerSenderTrait,
195{
196	// fetch in batches until there are enough votes
197	let mut disputes = partitioned.into_iter().collect::<Vec<_>>();
198	let mut total_votes_len = 0;
199	let mut result = BTreeMap::new();
200	let mut request_votes_counter = 0;
201	while !disputes.is_empty() {
202		gum::trace!(target: LOG_TARGET, "has to process {} disputes left", disputes.len());
203		let batch_size = std::cmp::min(VOTES_SELECTION_BATCH_SIZE, disputes.len());
204		let batch = Vec::from_iter(disputes.drain(0..batch_size));
205
206		// Filter votes which are already onchain
207		request_votes_counter += 1;
208		gum::trace!(target: LOG_TARGET, "requesting onchain votes",);
209		let votes = super::request_votes(sender, batch)
210			.await
211			.into_iter()
212			.map(|(session_index, candidate_hash, mut votes)| {
213				let onchain_state =
214					if let Some(onchain_state) = onchain.get(&(session_index, candidate_hash)) {
215						onchain_state
216					} else {
217						// onchain knows nothing about this dispute - add all votes
218						return (session_index, candidate_hash, votes)
219					};
220
221				votes.valid.retain(|validator_idx, (statement_kind, _)| {
222					is_vote_worth_to_keep(
223						validator_idx,
224						DisputeStatement::Valid(statement_kind.clone()),
225						&onchain_state,
226					)
227				});
228				votes.invalid.retain(|validator_idx, (statement_kind, _)| {
229					is_vote_worth_to_keep(
230						validator_idx,
231						DisputeStatement::Invalid(*statement_kind),
232						&onchain_state,
233					)
234				});
235				(session_index, candidate_hash, votes)
236			})
237			.collect::<Vec<_>>();
238		gum::trace!(target: LOG_TARGET, "got {} onchain votes after processing", votes.len());
239
240		// Check if votes are within the limit
241		for (session_index, candidate_hash, selected_votes) in votes {
242			let votes_len = selected_votes.valid.raw().len() + selected_votes.invalid.len();
243			if votes_len + total_votes_len > MAX_DISPUTE_VOTES_FORWARDED_TO_RUNTIME {
244				// we are done - no more votes can be added. Importantly, we don't add any votes for
245				// a dispute here if we can't fit them all. This gives us an important invariant,
246				// that backing votes for disputes make it into the provisioned vote set.
247				gum::trace!(
248					target: LOG_TARGET,
249					?request_votes_counter,
250					?total_votes_len,
251					"vote_selection DisputeCoordinatorMessage::QueryCandidateVotes counter",
252				);
253
254				return result
255			}
256
257			result.insert((session_index, candidate_hash), selected_votes);
258			total_votes_len += votes_len
259		}
260	}
261
262	gum::trace!(
263		target: LOG_TARGET,
264		?request_votes_counter,
265		?total_votes_len,
266		"vote_selection DisputeCoordinatorMessage::QueryCandidateVotes counter",
267	);
268
269	result
270}

272/// Contains disputes by partitions. Check the field comments for further details.
273#[derive(Default)]
274pub(crate) struct PartitionedDisputes {
275	/// Concluded and inactive disputes which are completely unknown for the Runtime.
276	/// Hopefully this should never happen.
277	/// Will be sent to the Runtime with FIRST priority.
278	pub inactive_unknown_onchain: Vec<(SessionIndex, CandidateHash)>,
279	/// Disputes which are INACTIVE locally but they are unconcluded for the Runtime.
280	/// A dispute can have enough local vote to conclude and at the same time the
281	/// Runtime knows nothing about them at treats it as unconcluded. This discrepancy
282	/// should be treated with high priority.
283	/// Will be sent to the Runtime with SECOND priority.
284	pub inactive_unconcluded_onchain: Vec<(SessionIndex, CandidateHash)>,
285	/// Active disputes completely unknown onchain.
286	/// Will be sent to the Runtime with THIRD priority.
287	pub active_unknown_onchain: Vec<(SessionIndex, CandidateHash)>,
288	/// Active disputes unconcluded onchain.
289	/// Will be sent to the Runtime with FOURTH priority.
290	pub active_unconcluded_onchain: Vec<(SessionIndex, CandidateHash)>,
291	/// Active disputes concluded onchain. New votes are not that important for
292	/// this partition.
293	/// Will be sent to the Runtime with FIFTH priority.
294	pub active_concluded_onchain: Vec<(SessionIndex, CandidateHash)>,
295	/// Inactive disputes which has concluded onchain. These are not interesting and
296	/// won't be sent to the Runtime.
297	/// Will be DROPPED
298	pub inactive_concluded_onchain: Vec<(SessionIndex, CandidateHash)>,
299}
300
301impl PartitionedDisputes {
302	fn new() -> PartitionedDisputes {
303		Default::default()
304	}
305
306	fn into_iter(self) -> impl Iterator<Item = (SessionIndex, CandidateHash)> {
307		self.inactive_unknown_onchain
308			.into_iter()
309			.chain(self.inactive_unconcluded_onchain.into_iter())
310			.chain(self.active_unknown_onchain.into_iter())
311			.chain(self.active_unconcluded_onchain.into_iter())
312			.chain(self.active_concluded_onchain.into_iter())
313		// inactive_concluded_onchain is dropped on purpose
314	}
315}

317fn secs_since_epoch() -> Timestamp {
318	match SystemTime::now().duration_since(UNIX_EPOCH) {
319		Ok(d) => d.as_secs(),
320		Err(e) => {
321			gum::warn!(
322				target: LOG_TARGET,
323				err = ?e,
324				"Error getting system time."
325			);
326			0
327		},
328	}
329}

331fn concluded_onchain(onchain_state: &DisputeState) -> bool {
332	// Check if there are enough onchain votes for or against to conclude the dispute
333	let supermajority = supermajority_threshold(onchain_state.validators_for.len());
334
335	onchain_state.validators_for.count_ones() >= supermajority ||
336		onchain_state.validators_against.count_ones() >= supermajority
337}

339fn partition_recent_disputes(
340	recent: BTreeMap<(SessionIndex, CandidateHash), DisputeStatus>,
341	onchain: &HashMap<(SessionIndex, CandidateHash), DisputeState>,
342) -> PartitionedDisputes {
343	let mut partitioned = PartitionedDisputes::new();
344	let time_now = &secs_since_epoch();
345
346	for ((session_index, candidate_hash), dispute_state) in recent {
347		let key = (session_index, candidate_hash);
348		if dispute_is_inactive(&dispute_state, time_now) {
349			match onchain.get(&key) {
350				Some(onchain_state) =>
351					if concluded_onchain(onchain_state) {
352						partitioned
353							.inactive_concluded_onchain
354							.push((session_index, candidate_hash));
355					} else {
356						partitioned
357							.inactive_unconcluded_onchain
358							.push((session_index, candidate_hash));
359					},
360				None => partitioned.inactive_unknown_onchain.push((session_index, candidate_hash)),
361			}
362		} else {
363			match onchain.get(&(session_index, candidate_hash)) {
364				Some(d) => match concluded_onchain(d) {
365					true =>
366						partitioned.active_concluded_onchain.push((session_index, candidate_hash)),
367					false =>
368						partitioned.active_unconcluded_onchain.push((session_index, candidate_hash)),
369				},
370				None => partitioned.active_unknown_onchain.push((session_index, candidate_hash)),
371			}
372		}
373	}
374
375	partitioned
376}

378/// Determines if a vote is worth to be kept, based on the onchain disputes
379fn is_vote_worth_to_keep(
380	validator_index: &ValidatorIndex,
381	dispute_statement: DisputeStatement,
382	onchain_state: &DisputeState,
383) -> bool {
384	let (offchain_vote, valid_kind) = match dispute_statement {
385		DisputeStatement::Valid(kind) => (true, Some(kind)),
386		DisputeStatement::Invalid(_) => (false, None),
387	};
388	// We want to keep all backing votes. This maximizes the number of backers
389	// punished when misbehaving.
390	if let Some(kind) = valid_kind {
391		match kind {
392			ValidDisputeStatementKind::BackingValid(_) |
393			ValidDisputeStatementKind::BackingSeconded(_) => return true,
394			_ => (),
395		}
396	}
397
398	let in_validators_for = onchain_state
399		.validators_for
400		.get(validator_index.0 as usize)
401		.as_deref()
402		.copied()
403		.unwrap_or(false);
404	let in_validators_against = onchain_state
405		.validators_against
406		.get(validator_index.0 as usize)
407		.as_deref()
408		.copied()
409		.unwrap_or(false);
410
411	if in_validators_for && in_validators_against {
412		// The validator has double voted and runtime knows about this. Ignore this vote.
413		return false
414	}
415
416	if offchain_vote && in_validators_against || !offchain_vote && in_validators_for {
417		// offchain vote differs from the onchain vote
418		// we need this vote to punish the offending validator
419		return true
420	}
421
422	// The vote is valid. Return true if it is not seen onchain.
423	!in_validators_for && !in_validators_against
424}

426/// Request disputes identified by `CandidateHash` and the `SessionIndex`.
427async fn request_disputes(
428	sender: &mut impl overseer::ProvisionerSenderTrait,
429) -> BTreeMap<(SessionIndex, CandidateHash), DisputeStatus> {
430	let (tx, rx) = oneshot::channel();
431	let msg = DisputeCoordinatorMessage::RecentDisputes(tx);
432
433	// Bounded by block production - `ProvisionerMessage::RequestInherentData`.
434	sender.send_unbounded_message(msg);
435
436	let recent_disputes = rx.await.unwrap_or_else(|err| {
437		gum::warn!(target: LOG_TARGET, err=?err, "Unable to gather recent disputes");
438		BTreeMap::new()
439	});
440	recent_disputes
441}

443// This function produces the return value for `pub fn select_disputes()`
444fn make_multi_dispute_statement_set(
445	metrics: &metrics::Metrics,
446	dispute_candidate_votes: BTreeMap<(SessionIndex, CandidateHash), CandidateVotes>,
447) -> MultiDisputeStatementSet {
448	// Transform all `CandidateVotes` into `MultiDisputeStatementSet`.
449	dispute_candidate_votes
450		.into_iter()
451		.map(|((session_index, candidate_hash), votes)| {
452			let valid_statements = votes
453				.valid
454				.into_iter()
455				.map(|(i, (s, sig))| (DisputeStatement::Valid(s), i, sig));
456
457			let invalid_statements = votes
458				.invalid
459				.into_iter()
460				.map(|(i, (s, sig))| (DisputeStatement::Invalid(s), i, sig));
461
462			metrics.inc_valid_statements_by(valid_statements.len());
463			metrics.inc_invalid_statements_by(invalid_statements.len());
464			metrics.inc_dispute_statement_sets_by(1);
465
466			DisputeStatementSet {
467				candidate_hash,
468				session: session_index,
469				statements: valid_statements.chain(invalid_statements).collect(),
470			}
471		})
472		.collect()
473}

475/// Gets the on-chain disputes at a given block number and returns them as a `HashMap` so that
476/// searching in them is cheap.
477pub async fn get_onchain_disputes<Sender>(
478	sender: &mut Sender,
479	relay_parent: Hash,
480) -> Result<HashMap<(SessionIndex, CandidateHash), DisputeState>, GetOnchainDisputesError>
481where
482	Sender: overseer::ProvisionerSenderTrait,
483{
484	gum::trace!(target: LOG_TARGET, ?relay_parent, "Fetching on-chain disputes");
485	let (tx, rx) = oneshot::channel();
486	sender
487		.send_message(RuntimeApiMessage::Request(relay_parent, RuntimeApiRequest::Disputes(tx)))
488		.await;
489
490	rx.await
491		.map_err(|_| GetOnchainDisputesError::Channel)
492		.and_then(|res| {
493			res.map_err(|e| match e {
494				RuntimeApiError::Execution { .. } =>
495					GetOnchainDisputesError::Execution(e, relay_parent),
496				RuntimeApiError::NotSupported { .. } =>
497					GetOnchainDisputesError::NotSupported(e, relay_parent),
498			})
499		})
500		.map(|v| v.into_iter().map(|e| ((e.0, e.1), e.2)).collect())
501}

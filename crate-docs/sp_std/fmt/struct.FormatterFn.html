Expand description

Implements fmt::Debug and fmt::Display using a function.

Examples

#![feature(debug_closure_helpers)]
use std::fmt;

let value = 'a';
assert_eq!(format!("{}", value), "a");
assert_eq!(format!("{:?}", value), "'a'");

let wrapped = fmt::FormatterFn(|f| write!(f, "{:?}", &value));
assert_eq!(format!("{}", wrapped), "'a'");
assert_eq!(format!("{:?}", wrapped), "'a'");
